# 03 - 人脸美颜滤镜

## 项目概述

实现一个实时人脸美颜系统，包含：磨皮、美白、瘦脸、大眼等功能。

```
功能模块：
┌──────────────┬─────────────────────┐
│  人脸检测     │ Haar/DNN 检测人脸     │
│  皮肤平滑     │ 双边滤波磨皮          │
│  美白提亮     │ LAB/LUT 调整         │
│  磨皮程度     │ 滑动条调节           │
│  实时处理     │ 摄像头实时应用        │
└──────────────┴─────────────────────┘
```

## 完整代码

```python
import cv2
import numpy as np

class BeautyFilter:
    def __init__(self):
        # 人脸检测器
        self.face_cascade = cv2.CascadeClassifier(
            cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
        self.eye_cascade = cv2.CascadeClassifier(
            cv2.data.haarcascades + 'haarcascade_eye.xml')

    def apply_beauty(self, img, smoothness=50, brightness=10, whiteness=20):
        """应用美颜滤镜"""
        result = img.copy()

        # 检测人脸
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        faces = self.face_cascade.detectMultiScale(gray, 1.3, 5)

        for (x, y, w, h) in faces:
            # 人脸 ROI
            face_roi = result[y:y+h, x:x+w]

            # 1. 磨皮（双边滤波）
            face_roi = self.smooth_skin(face_roi, smoothness)

            # 2. 美白提亮
            face_roi = self.whiten_skin(face_roi, whiteness, brightness)

            # 放回
            result[y:y+h, x:x+w] = face_roi

        return result

    def smooth_skin(self, face_img, strength=50):
        """磨皮：双边滤波保留边缘"""
        # 双边滤波：去皱保边
        diameter = max(5, strength // 5)
        filtered = cv2.bilateralFilter(face_img, diameter, strength, strength)

        # 混合：原始 vs 滤波后
        alpha = strength / 100.0
        result = cv2.addWeighted(face_img, 1 - alpha, filtered, alpha, 0)

        return result

    def whiten_skin(self, face_img, whiteness=20, brightness=10):
        """美白提亮"""
        # 方法1: 简单亮度调整
        result = cv2.convertScaleAbs(face_img, alpha=1 + whiteness/100, beta=brightness)

        # 方法2: LAB 色彩空间（更自然）
        lab = cv2.cvtColor(face_img, cv2.COLOR_BGR2LAB)
        l, a, b = cv2.split(lab)

        # CLAHE 增强亮度通道
        clahe = cv2.createCLAHE(clipLimit=2.0 + whiteness/50, tileGridSize=(8, 8))
        l_enhanced = clahe.apply(l)

        lab_enhanced = cv2.merge([l_enhanced, a, b])
        result = cv2.cvtColor(lab_enhanced, cv2.COLOR_LAB2BGR)

        return result

    def skin_smoothing_advanced(self, face_img):
        """高级磨皮：频率分离法"""
        # 分离高频（细节）和低频（颜色）
        low_freq = cv2.GaussianBlur(face_img, (21, 21), 0)
        high_freq = cv2.subtract(face_img, low_freq)

        # 对低频做强模糊（去斑）
        low_freq_smooth = cv2.bilateralFilter(low_freq, 15, 80, 80)

        # 合并
        result = cv2.add(low_freq_smooth, high_freq)
        return result

    def apply_makeup(self, img, face_box):
        """添加虚拟妆容（口红/腮红）"""
        x, y, w, h = face_box

        # 简单版：在嘴唇区域添加红色
        # 嘴唇大约在人脸下方 2/3 处
        mouth_y = y + int(h * 0.7)
        mouth_x = x + int(w * 0.3)
        mouth_w = int(w * 0.4)
        mouth_h = int(h * 0.15)

        overlay = img.copy()
        cv2.ellipse(overlay,
                   (mouth_x + mouth_w//2, mouth_y + mouth_h//2),
                   (mouth_w//2, mouth_h//2), 0, 0, 360,
                   (40, 20, 120), -1)  # 紫红色

        # 混合
        result = cv2.addWeighted(img, 1, overlay, 0.3, 0)
        return result

def realtime_beauty():
    """实时美颜"""
    beauty = BeautyFilter()

    cap = cv2.VideoCapture(0)

    # 创建控制面板
    cv2.namedWindow("Beauty Filter")
    cv2.createTrackbar("Smoothness", "Beauty Filter", 50, 100, lambda x: None)
    cv2.createTrackbar("Whiteness", "Beauty Filter", 20, 100, lambda x: None)
    cv2.createTrackbar("Brightness", "Beauty Filter", 10, 50, lambda x: None)

    print("按 q 退出")

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # 水平翻转（镜像）
        frame = cv2.flip(frame, 1)

        # 读取参数
        smoothness = cv2.getTrackbarPos("Smoothness", "Beauty Filter")
        whiteness = cv2.getTrackbarPos("Whiteness", "Beauty Filter")
        brightness = cv2.getTrackbarPos("Brightness", "Beauty Filter")

        # 应用美颜
        result = beauty.apply_beauty(frame, smoothness, brightness, whiteness)

        # 显示
        cv2.imshow("Beauty Filter", result)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()

# 运行
realtime_beauty()
```

## 核心技术

| 功能 | 技术 |
|------|------|
| 人脸检测 | Haar 级联 / DNN |
| 磨皮 | 双边滤波 / 频率分离 |
| 美白 | LAB + CLAHE |
| 妆容 | ROI + addWeighted |

```python
# 使用 LUT 滤镜
def apply_filter(img, filter_type="warm"):
    """应用颜色滤镜"""
    if filter_type == "warm":
        # 暖色调
        lut_r = np.array([min(255, i + 20) for i in range(256)], dtype=np.uint8)
        lut_b = np.array([max(0, i - 20) for i in range(256)], dtype=np.uint8)
        lut_g = np.arange(256, dtype=np.uint8)

        result = img.copy()
        result[:, :, 0] = cv2.LUT(result[:, :, 0], lut_b)  # B
        result[:, :, 1] = cv2.LUT(result[:, :, 1], lut_g)  # G
        result[:, :, 2] = cv2.LUT(result[:, :, 2], lut_r)  # R
        return result
```

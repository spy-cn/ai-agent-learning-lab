# 04 - 运动检测监控系统

## 项目概述

实现一个基于摄像头的运动检测系统：检测画面中的运动物体，绘制运动轨迹，触发告警。

```
功能：
1. 背景建模（区分前景和背景）
2. 运动区域检测
3. 运动轨迹绘制
4. 告警通知（画面/声音）
```

## 完整代码

```python
import cv2
import numpy as np
from datetime import datetime

class MotionDetector:
    def __init__(self, method="mog2"):
        """
        method: 背景建模方法
        - mog2: MOG2（高斯混合模型）
        - knn: KNN（K近邻）
        """
        self.method = method

        if method == "mog2":
            self.bg_subtractor = cv2.createBackgroundSubtractorMOG2(
                history=500,       # 历史帧数
                varThreshold=16,   # 方差阈值
                detectShadows=True # 检测阴影
            )
        elif method == "knn":
            self.bg_subtractor = cv2.createBackgroundSubtractorKNN(
                history=500,
                dist2Threshold=400,
                detectShadows=True
            )

        self.min_contour_area = 500  # 最小运动区域面积
        self.alert_active = False
        self.alert_cooldown = 0

    def process_frame(self, frame):
        """处理每一帧"""
        # 1. 背景减除
        fg_mask = self.bg_subtractor.apply(frame)

        # 2. 去除阴影（阴影值为 127）
        _, fg_mask = cv2.threshold(fg_mask, 200, 255, cv2.THRESH_BINARY)

        # 3. 形态学处理（去噪 + 填充）
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
        fg_mask = cv2.morphologyEx(fg_mask, cv2.MORPH_OPEN, kernel, iterations=2)
        fg_mask = cv2.morphologyEx(fg_mask, cv2.MORPH_CLOSE, kernel, iterations=2)

        # 4. 查找运动轮廓
        contours, _ = cv2.findContours(fg_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        detections = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if area > self.min_contour_area:
                x, y, w, h = cv2.boundingRect(contour)
                detections.append({
                    "box": (x, y, w, h),
                    "area": area,
                    "center": (x + w//2, y + h//2)
                })

        return fg_mask, detections

    def draw_detections(self, frame, detections):
        """在帧上绘制检测结果"""
        for det in detections:
            x, y, w, h = det["box"]
            area = det["area"]

            # 根据面积选择颜色
            if area > 5000:
                color = (0, 0, 255)  # 大物体 → 红色
                label = "ALERT"
            else:
                color = (0, 255, 0)  # 小物体 → 绿色
                label = "Motion"

            cv2.rectangle(frame, (x, y), (x+w, y+h), color, 2)
            cv2.putText(frame, f"{label} ({area:.0f})", (x, y-10),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 2)

        return frame

    def check_alert(self, detections):
        """检查是否需要告警"""
        large_motion = any(d["area"] > 5000 for d in detections)

        if large_motion:
            if self.alert_cooldown <= 0:
                self.alert_active = True
                self.alert_cooldown = 30  # 冷却帧数
            else:
                self.alert_active = False

        self.alert_cooldown = max(0, self.alert_cooldown - 1)
        return self.alert_active

def run_surveillance(video_source=0):
    """运行监控系统"""
    detector = MotionDetector(method="mog2")

    cap = cv2.VideoCapture(video_source)

    # 用于显示统计信息
    frame_count = 0
    motion_history = []  # 运动历史

    print("监控系统启动")
    print("按 q 退出 | r 重置背景模型")

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        frame_count += 1

        # 缩小帧（提高处理速度）
        frame = cv2.resize(frame, (640, 480))

        # 处理
        fg_mask, detections = detector.process_frame(frame)
        frame = detector.draw_detections(frame, detections)

        # 告警检查
        alert = detector.check_alert(detections)
        if alert:
            timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            cv2.putText(frame, f"!! ALERT !! {timestamp}",
                       (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2)
            # 可以在这里添加：发邮件/保存截图/播放声音
            cv2.imwrite(f"alert_{frame_count}.jpg", frame)

        # 显示信息
        info_text = f"FPS: {frame_count} | Objects: {len(detections)} | Method: {detector.method}"
        cv2.putText(frame, info_text, (10, frame.shape[0] - 10),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)

        # 显示
        cv2.imshow("Surveillance", frame)
        cv2.imshow("Foreground Mask", fg_mask)

        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'):
            break
        elif key == ord('r'):
            # 重置背景模型
            detector = MotionDetector(method=detector.method)
            print("背景模型已重置")

    cap.release()
    cv2.destroyAllWindows()

# 运行
run_surveillance(0)  # 0=摄像头，或传入视频路径
```

## 背景建模方法对比

| 方法 | 函数 | 优点 | 缺点 |
|------|------|------|------|
| MOG2 | `createBackgroundSubtractorMOG2` | 自适应阴影检测 | 计算量稍大 |
| KNN | `createBackgroundSubtractorKNN` | 对非静态背景好 | 参数敏感 |
| 帧差法 | 手动实现 | 最简单 | 无法处理静止物体 |

## 扩展功能

```python
# 1. 保存告警时的视频片段
class VideoRecorder:
    def __init__(self, output_path, fps=30):
        self.writer = cv2.VideoWriter(output_path,
            cv2.VideoWriter_fourcc(*'mp4v'), fps, (640, 480))
        self.buffer = []
        self.recording = False

    def add_frame(self, frame):
        self.buffer.append(frame)
        if len(self.buffer) > 90:  # 保存前 3 秒
            self.buffer.pop(0)
        if self.recording:
            self.writer.write(frame)

# 2. 运动轨迹追踪
class MotionTracker:
    def __init__(self):
        self.tracks = []  # 存储运动轨迹
        self.max_track_length = 30

    def update(self, detections):
        # 简单版：记录每个检测的中心点
        for det in detections:
            self.tracks.append(det["center"])
        if len(self.tracks) > self.max_track_length:
            self.tracks.pop(0)

    def draw(self, frame):
        for i in range(1, len(self.tracks)):
            cv2.line(frame, self.tracks[i-1], self.tracks[i], (0, 255, 255), 2)
```

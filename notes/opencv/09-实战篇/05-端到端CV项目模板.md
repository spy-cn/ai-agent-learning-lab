## 配置加载器

```python
# src/utils/config.py
import yaml
from pathlib import Path

class Config:
    def __init__(self, config_path="config/config.yaml"):
        with open(config_path, 'r', encoding='utf-8') as f:
            self.config = yaml.safe_load(f)

    def __getattr__(self, name):
        if name in self.config:
            return self.config[name]
        raise AttributeError(f"'{name}' not in config")

    def get(self, *keys, default=None):
        """嵌套键访问"""
        value = self.config
        for key in keys:
            if isinstance(value, dict) and key in value:
                value = value[key]
            else:
                return default
        return value

# 使用
config = Config()
print(config.get('model', 'type'))
```

## 日志模块

```python
# src/utils/logger.py
import logging
from pathlib import Path

def setup_logger(name, log_file=None, level=logging.INFO):
    """配置日志"""
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s')

    logger = logging.getLogger(name)
    logger.setLevel(level)

    # 控制台输出
    console = logging.StreamHandler()
    console.setFormatter(formatter)
    logger.addHandler(console)

    # 文件输出
    if log_file:
        Path(log_file).parent.mkdir(parents=True, exist_ok=True)
        file_handler = logging.FileHandler(log_file, encoding='utf-8')
        file_handler.setFormatter(formatter)
        logger.addHandler(file_handler)

    return logger

# 使用
logger = setup_logger("CV_Project", "outputs/app.log")
logger.info("项目启动")
```

## 主流程模板

```python
# src/pipeline.py
import cv2
import numpy as np
from pathlib import Path
from utils.config import Config
from utils.logger import setup_logger

class CVPipeline:
    def __init__(self, config_path="config/config.yaml"):
        self.config = Config(config_path)
        self.logger = setup_logger("Pipeline")
        self.logger.info("初始化流水线...")

        self._load_model()

    def _load_model(self):
        """加载模型"""
        model_type = self.config.get('model', 'type')

        if model_type == 'yolov4':
            self.net = cv2.dnn.readNet(
                self.config.get('model', 'weights'),
                self.config.get('model', 'config')
            )
            self.net.setPreferableBackend(cv2.dnn.DNN_BACKEND_OPENCV)

        self.logger.info(f"模型加载完成: {model_type}")

    def preprocess(self, img):
        """预处理"""
        target_size = self.config.get('data', 'image_size', default=[640, 480])
        return cv2.dnn.blobFromImage(img, 1/255.0, tuple(target_size), swapRB=True)

    def infer(self, img):
        """推理"""
        blob = self.preprocess(img)
        self.net.setInput(blob)
        output = self.net.forward()
        return self.postprocess(output, img.shape)

    def postprocess(self, output, original_shape):
        """后处理"""
        h, w = original_shape[:2]
        detections = []

        for detection in output.reshape(-1, 6):
            confidence = detection[4]
            if confidence > self.config.get('model', 'confidence_threshold', default=0.5):
                x1 = int(detection[0] * w)
                y1 = int(detection[1] * h)
                x2 = int(detection[2] * w)
                y2 = int(detection[3] * h)
                class_id = int(detection[5])
                detections.append({
                    "box": (x1, y1, x2-x1, y2-y1),
                    "confidence": float(confidence),
                    "class_id": class_id
                })

        return detections

    def run(self, input_path, output_path=None):
        """执行完整流程"""
        self.logger.info(f"处理: {input_path}")

        # 读取
        if str(input_path).isdigit():
            cap = cv2.VideoCapture(int(input_path))
            ret, img = cap.read()
        else:
            img = cv2.imread(str(input_path))

        if img is None:
            self.logger.error("无法读取输入")
            return None

        # 推理
        detections = self.infer(img)

        # 可视化
        for det in detections:
            x, y, w, h = det["box"]
            cv2.rectangle(img, (x, y), (x+w, y+h), (0, 255, 0), 2)
            label = f"{det['class_id']}: {det['confidence']:.2f}"
            cv2.putText(img, label, (x, y-10),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)

        # 保存
        if output_path:
            cv2.imwrite(str(output_path), img)
            self.logger.info(f"结果保存到: {output_path}")

        return img, detections

# 使用
if __name__ == "__main__":
    pipeline = CVPipeline()
    result = pipeline.run("test.jpg", "output.jpg")
```

## 可视化工具

```python
# src/utils/visualization.py
import cv2
import numpy as np

class Visualizer:
    COLORS = {
        'red': (0, 0, 255),
        'green': (0, 255, 0),
        'blue': (255, 0, 0),
        'yellow': (0, 255, 255),
        'white': (255, 255, 255),
    }

    @staticmethod
    def draw_detections(img, detections, class_names=None):
        """绘制检测结果"""
        for det in detections:
            x, y, w, h = det["box"]
            conf = det["confidence"]

            label = class_names[det["class_id"]] if class_names else str(det["class_id"])
            text = f"{label} {conf:.2f}"

            # 框
            cv2.rectangle(img, (x, y), (x+w, y+h), Visualizer.COLORS['green'], 2)

            # 标签背景
            (tw, th), _ = cv2.getTextSize(text, cv2.FONT_HERSHEY_SIMPLEX, 0.5, 1)
            cv2.rectangle(img, (x, y-th-5), (x+tw+5, y), Visualizer.COLORS['green'], -1)
            cv2.putText(img, text, (x+2, y-5),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, Visualizer.COLORS['white'], 1)

        return img

    @staticmethod
    def grid_show(images_dict, cols=3):
        """网格显示多张图像"""
        import matplotlib.pyplot as plt

        n = len(images_dict)
        rows = (n + cols - 1) // cols

        fig, axes = plt.subplots(rows, cols, figsize=(cols*4, rows*3))
        if rows == 1:
            axes = [axes]

        for ax, (name, img) in zip(axes.flat, images_dict.items()):
            if len(img.shape) == 3:
                img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
            ax.imshow(img, cmap='gray' if len(img.shape) == 2 else None)
            ax.set_title(name)
            ax.axis('off')

        plt.tight_layout()
        plt.show()
```

## Docker 部署

```dockerfile
# Dockerfile
FROM python:3.10-slim

# 安装系统依赖
RUN apt-get update && apt-get install -y \
    libgl1-mesa-glx \
    libglib2.0-0 \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# 安装依赖
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 复制代码
COPY . .

# 运行
CMD ["python", "scripts/infer.py", "--input", "data/raw/test.jpg"]
```

```yaml
# docker-compose.yml
version: '3'
services:
  cv-app:
    build: .
    volumes:
      - ./data:/app/data
      - ./outputs:/app/outputs
    environment:
      - DEVICE=cpu
```

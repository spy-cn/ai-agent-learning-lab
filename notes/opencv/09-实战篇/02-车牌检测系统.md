# 02 - 车牌检测系统

## 项目概述

基于传统图像处理的车牌检测与字符提取系统。

```
流程：
1. 车辆图像 → 车牌区域检测
2. 车牌区域 → 字符分割
3. 字符图像 → 字符识别（可选）
```

## 完整代码

```python
import cv2
import numpy as np

class LicensePlateDetector:
    def __init__(self):
        # 使用 Haar 级联检测车牌（需要训练好的分类器）
        # 也可以使用边缘检测方法
        pass

    def detect(self, image_path):
        """主检测函数"""
        img = cv2.imread(image_path)
        original = img.copy()

        # Step 1: 预处理
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

        # Step 2: 车牌定位
        plate_region = self.locate_plate(gray)

        if plate_region is None:
            print("未检测到车牌")
            return None

        x, y, w, h = plate_region
        plate_img = original[y:y+h, x:x+w]

        # Step 3: 字符分割
        chars = self.segment_characters(plate_img)

        # Step 4: 绘制结果
        cv2.rectangle(original, (x, y), (x+w, y+h), (0, 255, 0), 2)
        cv2.putText(original, "License Plate", (x, y-10),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)

        return original, plate_img, chars

    def locate_plate(self, gray):
        """车牌定位"""
        # 方法1: 基于边缘 + 形态学
        # 双边滤波（去噪保边）
        blurred = cv2.bilateralFilter(gray, 11, 17, 17)

        # Canny 边缘检测
        edges = cv2.Canny(blurred, 30, 200)

        # 形态学闭运算（连接边缘）
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (17, 5))
        closed = cv2.morphologyEx(edges, cv2.MORPH_CLOSE, kernel)

        # 查找轮廓
        contours, _ = cv2.findContours(closed, cv2.RETR_LIST, cv2.CHAIN_APPROX_SIMPLE)
        contours = sorted(contours, key=cv2.contourArea, reverse=True)[:10]

        for contour in contours:
            # 近似轮廓
            peri = cv2.arcLength(contour, True)
            approx = cv2.approxPolyDP(contour, 0.02 * peri, True)

            # 外接矩形
            x, y, w, h = cv2.boundingRect(contour)

            # 车牌宽高比通常在 2:1 到 5:1 之间
            aspect_ratio = w / float(h)
            area = w * h

            if 2.0 < aspect_ratio < 6.0 and area > 1000 and area < 100000:
                return (x, y, w, h)

        return None

    def segment_characters(self, plate_img):
        """字符分割"""
        gray = cv2.cvtColor(plate_img, cv2.COLOR_BGR2GRAY)

        # 自适应阈值
        thresh = cv2.adaptiveThreshold(
            gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY_INV, 19, 9)

        # 形态学处理
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (3, 3))
        thresh = cv2.morphologyEx(thresh, cv2.MORPH_CLOSE, kernel)

        # 查找轮廓
        contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        char_boxes = []
        for cnt in contours:
            x, y, w, h = cv2.boundingRect(cnt)

            # 过滤：字符的宽高比约在 0.2-1.0
            aspect = w / float(h)
            area = w * h

            if 0.1 < aspect < 1.0 and area > 50:
                char_boxes.append((x, y, w, h))

        # 按 x 坐标排序
        char_boxes = sorted(char_boxes, key=lambda b: b[0])

        # 提取字符图像
        chars = []
        for x, y, w, h in char_boxes:
            char_img = thresh[y:y+h, x:x+w]
            # 标准化尺寸
            char_img = cv2.resize(char_img, (20, 40))
            chars.append(char_img)

        return chars

    def visualize(self, original, plate_img, chars):
        """可视化结果"""
        cv2.imshow("Detection", original)
        cv2.imshow("Plate", plate_img)

        for i, char in enumerate(chars):
            cv2.imshow(f"Char {i}", char)

        cv2.waitKey(0)

# 使用
detector = LicensePlateDetector()
result = detector.detect("car.jpg")
if result:
    original, plate, chars = result
    detector.visualize(original, plate, chars)
    print(f"检测到 {len(chars)} 个字符")
```

## 车牌定位方法对比

| 方法 | 优点 | 缺点 |
|------|------|------|
| 边缘+形态学 | 无需训练 | 环境敏感 |
| Haar 级联 | 快速 | 需训练数据 |
| 颜色过滤 | 简单 | 仅限特定颜色 |
| 深度学习 | 鲁棒 | 需 GPU |

# 1. 颜色过滤（中国车牌为蓝色）
def filter_by_color(img):
    hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
    lower_blue = np.array([100, 80, 80])
    upper_blue = np.array([130, 255, 255])
    mask = cv2.inRange(hsv, lower_blue, upper_blue)
    return mask

# 2. 使用 SVM 进行字符识别

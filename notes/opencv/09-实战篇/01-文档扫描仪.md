# 01 - 文档扫描仪

## 项目概述

构建一个类似 CamScanner 的文档扫描应用：拍照 → 自动检测文档边缘 → 透视矫正 → 增强可读性。

```
输入：手机拍摄的文档照片（倾斜、光照不均）
处理：边缘检测 → 轮廓查找 → 透视变换 → 自适应阈值
输出：正面、清晰的扫描文档
```

## 完整代码

```python
import cv2
import numpy as np

class DocumentScanner:
    def __init__(self):
        pass

    def scan(self, image_path, output_path="scanned.jpg"):
        """主函数：扫描文档"""
        # Step 1: 读取并预处理
        img = cv2.imread(image_path)
        orig = img.copy()
        ratio = 500 / img.shape[0]  # 缩放比例
        resized = cv2.resize(img, (int(img.shape[1] * ratio), 500))

        # Step 2: 检测文档边缘
        corners = self.find_document_corners(resized)

        if corners is None:
            print("未检测到文档边缘，使用整张图")
            h, w = resized.shape[:2]
            corners = np.array([[0, 0], [w-1, 0], [w-1, h-1], [0, h-1]], dtype=np.float32)

        # Step 3: 透视矫正
        corners_orig = corners / ratio  # 恢复到原始尺寸
        warped = self.four_point_transform(orig, corners_orig)

        # Step 4: 增强文档
        enhanced = self.enhance_document(warped)

        # 保存
        cv2.imwrite(output_path, enhanced)

        # 可视化
        result = self.draw_results(orig, resized, corners_orig, warped, enhanced)
        cv2.imshow("Document Scanner", result)
        cv2.waitKey(0)

        return enhanced

    def find_document_corners(self, img):
        """检测文档的四个角点"""
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        blurred = cv2.GaussianBlur(gray, (5, 5), 0)

        # Canny 边缘检测
        edges = cv2.Canny(blurred, 50, 200)

        # 膨胀
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (3, 3))
        edges = cv2.dilate(edges, kernel, iterations=1)

        # 查找轮廓
        contours, _ = cv2.findContours(edges, cv2.RETR_LIST, cv2.CHAIN_APPROX_SIMPLE)
        contours = sorted(contours, key=cv2.contourArea, reverse=True)[:5]

        for contour in contours:
            peri = cv2.arcLength(contour, True)
            approx = cv2.approxPolyDP(contour, 0.02 * peri, True)

            # 如果近似的轮廓有 4 个点，认为找到了文档
            if len(approx) == 4:
                return approx.reshape(4, 2).astype(np.float32)

        # 备选方案：用最大轮廓的外接矩形
        if contours:
            x, y, w, h = cv2.boundingRect(contours[0])
            return np.array([[x, y], [x+w, y], [x+w, y+h], [x, y+h]], dtype=np.float32)

        return None

    def four_point_transform(self, image, pts):
        """透视变换"""
        rect = self.order_points(pts)
        tl, tr, br, bl = rect

        # 计算新图像的宽度
        widthA = np.sqrt((br[0] - bl[0])**2 + (br[1] - bl[1])**2)
        widthB = np.sqrt((tr[0] - tl[0])**2 + (tr[1] - tl[1])**2)
        maxWidth = max(int(widthA), int(widthB))

        # 计算新图像的高度
        heightA = np.sqrt((tr[0] - br[0])**2 + (tr[1] - br[1])**2)
        heightB = np.sqrt((tl[0] - bl[0])**2 + (tl[1] - bl[1])**2)
        maxHeight = max(int(heightA), int(heightB))

        # 目标点
        dst = np.array([
            [0, 0],
            [maxWidth - 1, 0],
            [maxWidth - 1, maxHeight - 1],
            [0, maxHeight - 1]
        ], dtype=np.float32)

        # 计算变换矩阵并执行
        M = cv2.getPerspectiveTransform(rect, dst)
        warped = cv2.warpPerspective(image, M, (maxWidth, maxHeight))

        return warped

    def order_points(self, pts):
        """将四个点按 左上、右上、右下、左下 排序"""
        rect = np.zeros((4, 2), dtype=np.float32)

        # 左上角的点 x+y 最小，右下角最大
        s = pts.sum(axis=1)
        rect[0] = pts[np.argmin(s)]
        rect[2] = pts[np.argmax(s)]

        # 右上角 x-y 最大，左下角 x-y 最小
        d = np.diff(pts, axis=1)
        rect[1] = pts[np.argmin(d)]
        rect[3] = pts[np.argmax(d)]

        return rect

    def enhance_document(self, img):
        """增强文档可读性"""
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

        # 方法1: 自适应阈值（适合文档）
        enhanced = cv2.adaptiveThreshold(
            gray, 255,
            cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY,
            21,  # blockSize
            10   # C
        )

        # 方法2: CLAHE（保留灰度，增强对比度）
        clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
        clahe_result = clahe.apply(gray)

        return enhanced  # 或返回 clahe_result

    def draw_results(self, orig, resized, corners, warped, enhanced):
        """绘制结果对比"""
        # 在缩小图上标记角点
        ratio = resized.shape[0] / orig.shape[0]
        marked = resized.copy()
        corners_resized = corners * ratio
        for i, pt in enumerate(corners_resized):
            cv2.circle(marked, tuple(pt.astype(int)), 8, (0, 0, 255), -1)
            cv2.putText(marked, str(i+1), tuple(pt.astype(int) + np.array([10, -10])),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)

        # 拼接结果
        row1 = np.hstack([cv2.resize(marked, (400, 300)),
                          cv2.resize(cv2.cvtColor(enhanced, cv2.COLOR_GRAY2BGR), (400, 300))])
        row2 = np.hstack([cv2.resize(warped, (400, 300)),
                          cv2.resize(self.enhance_document.__wrapped__ if hasattr(self.enhance_document, '__wrapped__') else warped, (400, 300))])

        return row1

# 使用
scanner = DocumentScanner()
scanner.scan("document_photo.jpg", "scanned_output.jpg")
```

## 关键技术点

| 技术 | 用途 |
|------|------|
| Canny + 轮廓 | 检测文档边缘 |
| approxPolyDP | 获取四边形角点 |
| getPerspectiveTransform | 透视矫正 |
| adaptiveThreshold | 文档增强 |
| CLAHE | 对比度增强 |

## 优化方向

1. **光照校正**：使用形态学顶帽操作去除不均匀光照
2. **去噪**：加入中值滤波去除扫描线噪声
3. **OCR 集成**：结合 Tesseract 进行文字识别
4. **多页处理**：批量处理多张文档图片

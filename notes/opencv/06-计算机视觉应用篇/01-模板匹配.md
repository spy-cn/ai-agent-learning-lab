# 01 - 模板匹配

## 一、模板匹配原理

模板匹配是在大图中滑动小图（模板），计算每个位置的相似度。

```
大图：
┌───────────────────────┐
│  ┌───┐                │
│  │Tpl│ ← 模板在此滑动  │
│  └───┘                │
│        最佳匹配位置 ← ★ │
└───────────────────────┘

过程：模板从左上角开始，逐像素滑动，在每个位置计算相似度。
```

## 二、基本用法

```python
import cv2
import numpy as np

img = cv2.imread("scene.jpg", 0)       # 大图（灰度）
template = cv2.imread("template.jpg", 0)  # 模板（灰度)

h, w = template.shape

# 模板匹配
result = cv2.matchTemplate(img, template, cv2.TM_CCOEFF_NORMED)

# result 是一个响应图，每个位置代表该位置的匹配度
# 形状: (img_h - h + 1, img_w - w + 1)

# 找到最佳匹配位置
min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)

# TM_CCOEFF_NORMED: 最大值 = 最佳匹配
top_left = max_loc
bottom_right = (top_left[0] + w, top_left[1] + h)

# 绘制匹配结果
img_color = cv2.cvtColor(img, cv2.COLOR_GRAY2BGR)
cv2.rectangle(img_color, top_left, bottom_right, (0, 255, 0), 2)

print(f"最佳匹配位置: {top_left}, 匹配度: {max_val:.4f}")

cv2.imshow("Template Matching", img_color)
cv2.waitKey(0)
```

## 三、6 种匹配方法

| 方法 | 说明 | 最佳值 |
|------|------|--------|
| `TM_SQDIFF` | 平方差 | 最小值 |
| `TM_SQDIFF_NORMED` | 归一化平方差 | 最小值 |
| `TM_CCORR` | 相关性 | 最大值 |
| `TM_CCORR_NORMED` | 归一化相关性 | 最大值 |
| `TM_CCOEFF` | 相关系数 | 最大值 |
| `TM_CCOEFF_NORMED` | 归一化相关系数 | **最大值（推荐）** |

```python
# 推荐使用 TM_CCOEFF_NORMED
# 匹配度范围 [-1, 1]，越接近 1 越好

methods = {
    'TM_SQDIFF_NORMED': cv2.TM_SQDIFF_NORMED,
    'TM_CCORR_NORMED': cv2.TM_CCORR_NORMED,
    'TM_CCOEFF_NORMED': cv2.TM_CCOEFF_NORMED,
}

for name, method in methods.items():
    result = cv2.matchTemplate(img, template, method)
    min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)

    if method in [cv2.TM_SQDIFF, cv2.TM_SQDIFF_NORMED]:
        location = min_loc  # 最小值位置
    else:
        location = max_loc  # 最大值位置

    print(f"{name}: location={location}")
```

## 四、多目标匹配

```python
# 查找所有匹配的位置
result = cv2.matchTemplate(img, template, cv2.TM_CCOEFF_NORMED)

# 阈值
threshold = 0.8
locations = np.where(result >= threshold)

# 合并重叠的框
for pt in zip(*locations[::-1]):
    cv2.rectangle(img_color, pt, (pt[0] + w, pt[1] + h), (0, 255, 0), 2)

# 存在大量重叠框的问题，需要 NMS
```

### 非极大值抑制（NMS）

```python
def non_max_suppression(boxes, scores, threshold=0.3):
    """非极大值抑制，去除重叠的检测框"""
    if len(boxes) == 0:
        return []

    boxes = np.array(boxes, dtype=np.float32)
    scores = np.array(scores)

    x1 = boxes[:, 0]
    y1 = boxes[:, 1]
    x2 = boxes[:, 0] + boxes[:, 2]
    y2 = boxes[:, 1] + boxes[:, 3]

    areas = (x2 - x1 + 1) * (y2 - y1 + 1)
    order = scores.argsort()[::-1]  # 按分数降序

    keep = []
    while order.size > 0:
        i = order[0]
        keep.append(i)

        xx1 = np.maximum(x1[i], x1[order[1:]])
        yy1 = np.maximum(y1[i], y1[order[1:]])
        xx2 = np.minimum(x2[i], x2[order[1:]])
        yy2 = np.minimum(y2[i], y2[order[1:]])

        w = np.maximum(0, xx2 - xx1 + 1)
        h = np.maximum(0, yy2 - yy1 + 1)
        overlap = (w * h) / areas[order[1:]]

        inds = np.where(overlap <= threshold)[0]
        order = order[inds + 1]

    return keep

# 使用 NMS
result = cv2.matchTemplate(img, template, cv2.TM_CCOEFF_NORMED)
threshold = 0.8
locations = np.where(result >= threshold)

boxes = []
scores = []
for pt in zip(*locations[::-1]):
    boxes.append([pt[0], pt[1], w, h])
    scores.append(result[pt[1], pt[0]])

keep_indices = non_max_suppression(boxes, scores, threshold=0.3)
for idx in keep_indices:
    box = boxes[idx]
    cv2.rectangle(img_color,
                  (int(box[0]), int(box[1])),
                  (int(box[0] + box[2]), int(box[1] + box[3])),
                  (0, 255, 0), 2)
```

## 五、多尺度模板匹配

```python
def multi_scale_template_matching(img, template, scales=np.linspace(0.5, 1.5, 20)):
    """多尺度模板匹配"""
    best_match = None
    best_val = -1

    for scale in scales:
        # 缩放模板
        resized = cv2.resize(template, None, fx=scale, fy=scale)
        if resized.shape[0] > img.shape[0] or resized.shape[1] > img.shape[1]:
            continue

        result = cv2.matchTemplate(img, resized, cv2.TM_CCOEFF_NORMED)
        _, max_val, _, max_loc = cv2.minMaxLoc(result)

        if max_val > best_val:
            best_val = max_val
            best_match = (max_loc, resized.shape[::-1], scale)

    return best_match, best_val

# 使用
match, score = multi_scale_template_matching(img, template)
if match:
    loc, (w, h), scale = match
    cv2.rectangle(img_color, loc, (loc[0]+w, loc[1]+h), (0, 255, 0), 2)
    print(f"最佳匹配: scale={scale:.2f}, score={score:.4f}")
```

## 六、旋转模板匹配

```python
def rotated_template_matching(img, template, angles=range(0, 360, 10)):
    """旋转模板匹配"""
    h, w = template.shape
    center = (w // 2, h // 2)
    best_val = -1
    best_match = None

    for angle in angles:
        M = cv2.getRotationMatrix2D(center, angle, 1.0)
        rotated = cv2.warpAffine(template, M, (w, h))

        result = cv2.matchTemplate(img, rotated, cv2.TM_CCOEFF_NORMED)
        _, max_val, _, max_loc = cv2.minMaxLoc(result)

        if max_val > best_val:
            best_val = max_val
            best_match = (max_loc, angle, rotated.shape)

    return best_match, best_val
```

## 七、模板匹配的局限性

```
模板匹配的限制：
✗ 不支持旋转（需多角度匹配）
✗ 不支持缩放（需多尺度匹配）
✗ 不支持形变
✗ 对光照变化敏感

对于复杂场景，建议使用：
→ 特征匹配（SIFT/ORB）
→ 深度学习目标检测（YOLO）
```

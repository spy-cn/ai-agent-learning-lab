# 03 - 光流法

## 一、什么是光流？

光流（Optical Flow）是描述相邻帧之间**像素运动**的模式。

```
帧1：              帧2：
┌──────────┐      ┌──────────┐
│  ●       │      │       ●  │  ← 物体向右移动
│          │      │          │
│        ▲ │  →   │      ▲   │  ← 物体向右下移动
└──────────┘      └──────────┘

光流场：每个像素都有一个运动向量 (dx, dy)
```

## 二、光流的类型

```
稀疏光流              密集光流
（Lucas-Kanade）      （Farneback）

只追踪特征点          追踪所有像素

┌──────────┐        ┌──────────┐
│  →       │        │ → → →    │
│     ↗    │        │ → → → →  │
│   ←      │        │ ↗ ↗ →    │
└──────────┘        └──────────┘
速度快               速度慢
```

## 三、Lucas-Kanade 稀疏光流

```python
import cv2
import numpy as np

cap = cv2.VideoCapture("video.mp4")

# 读取第一帧
ret, old_frame = cap.read()
old_gray = cv2.cvtColor(old_frame, cv2.COLOR_BGR2GRAY)

# 使用 Shi-Tomasi 检测特征点
feature_params = dict(
    maxCorners=100,
    qualityLevel=0.3,
    minDistance=7,
    blockSize=7
)
p0 = cv2.goodFeaturesToTrack(old_gray, mask=None, **feature_params)

# Lucas-Kanade 参数
lk_params = dict(
    winSize=(15, 15),
    maxLevel=2,
    criteria=(cv2.TERM_CRITERIA_EPS | cv2.TERM_CRITERIA_COUNT, 10, 0.03)
)

# 创建用于绘制的 mask
mask = np.zeros_like(old_frame)

# 颜色随机
color = np.random.randint(0, 255, (100, 3))

while True:
    ret, frame = cap.read()
    if not ret:
        break

    frame_gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

    # 计算光流
    p1, st, err = cv2.calcOpticalFlowPyrLK(
        old_gray, frame_gray, p0, None, **lk_params)

    # st: 状态标志，1 表示成功追踪
    if p1 is not None:
        good_new = p1[st == 1]
        good_old = p0[st == 1]

        # 绘制轨迹
        for i, (new, old) in enumerate(zip(good_new, good_old)):
            a, b = new.ravel()
            c, d = old.ravel()
            mask = cv2.line(mask, (int(a), int(b)), (int(c), int(d)),
                           color[i].tolist(), 2)
            frame = cv2.circle(frame, (int(a), int(b)), 5, color[i].tolist(), -1)

        result = cv2.add(frame, mask)
        cv2.imshow("Lucas-Kanade", result)

        # 更新
        old_gray = frame_gray.copy()
        p0 = good_new.reshape(-1, 1, 2)

    if cv2.waitKey(30) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

### LK 参数详解

| 参数 | 含义 |
|------|------|
| winSize | 搜索窗口大小 |
| maxLevel | 金字塔层数（0=不使用金字塔） |
| criteria | 迭代终止条件 |

## 四、Farneback 密集光流

```python
cap = cv2.VideoCapture("video.mp4")
ret, prev = cap.read()
prev_gray = cv2.cvtColor(prev, cv2.COLOR_BGR2GRAY)

while True:
    ret, frame = cap.read()
    if not ret:
        break

    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

    # 计算密集光流
    flow = cv2.calcOpticalFlowFarneback(
        prev_gray, gray,
        None,           # 输出（None表示自动创建）
        pyr_scale=0.5,  # 金字塔缩放比
        levels=3,       # 金字塔层数
        winsize=15,     # 窗口大小
        iterations=3,   # 每层迭代次数
        poly_n=5,       # 像素邻域大小
        poly_sigma=1.2, # 高斯标准差
        flags=0
    )

    # flow 形状: (H, W, 2)，每个像素的 (dx, dy)

    # ─── 可视化 ───
    # 方法1：HSV 可视化
    hsv = np.zeros_like(frame)
    hsv[..., 1] = 255

    mag, ang = cv2.cartToPolar(flow[..., 0], flow[..., 1])
    hsv[..., 0] = ang * 180 / np.pi / 2  # 角度 → 色调
    hsv[..., 2] = cv2.normalize(mag, None, 0, 255, cv2.NORM_MINMAX)  # 幅值 → 亮度

    flow_vis = cv2.cvtColor(hsv, cv2.COLOR_HSV2BGR)
    cv2.imshow("Dense Optical Flow", flow_vis)

    prev_gray = gray

    if cv2.waitKey(30) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

### 可视化说明

```
光流 HSV 可视化：
- 色调 (H)：表示运动方向
  红 = 向右
  绿 = 向上
  蓝 = 向下
  ...
- 亮度 (V)：表示运动速度（越亮越快）
```

## 五、光流应用

### 应用1：运动检测

```python
# 通过光流幅值检测运动区域
mag = np.sqrt(flow[..., 0]**2 + flow[..., 1]**2)
motion_mask = (mag > 2).astype(np.uint8) * 255  # 阈值
motion_mask = cv2.dilate(motion_mask, None, iterations=2)

contours, _ = cv2.findContours(motion_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
for cnt in contours:
    if cv2.contourArea(cnt) > 500:
        x, y, w, h = cv2.boundingRect(cnt)
        cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
```

### 应用2：运动方向统计

```python
# 统计整体运动方向
mean_dx = flow[..., 0].mean()
mean_dy = flow[..., 1].mean()
mean_mag = np.sqrt(mean_dx**2 + mean_dy**2)
mean_angle = np.degrees(np.arctan2(mean_dy, mean_dx))

print(f"平均运动: 幅值={mean_mag:.2f}, 角度={mean_angle:.1f}°")
```

## 六、稀疏 vs 密集光流对比

| 特性 | Lucas-Kanade | Farneback |
|------|-------------|-----------|
| 追踪范围 | 特征点 | 所有像素 |
| 速度 | 快 | 慢 |
| 精度 | 中 | 高 |
| 适用 | 实时跟踪 | 运动分析 |

## 小结

| 功能 | 函数 |
|------|------|
| 稀疏光流 | `cv2.calcOpticalFlowPyrLK()` |
| 密集光流 | `cv2.calcOpticalFlowFarneback()` |
| 极坐标转换 | `cv2.cartToPolar()` |

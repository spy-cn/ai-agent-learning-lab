# 04 - 目标跟踪

## 一、目标跟踪概述

目标跟踪是在视频序列中持续定位一个或多个目标的过程。

```
帧1: 初始化      帧2: 追踪      帧3: 追踪
┌──────────┐    ┌──────────┐    ┌──────────┐
│ ┌──┐     │    │  ┌──┐    │    │   ┌──┐   │
│ │★ │     │ →  │  │★ │    │ →  │   │★ │   │
│ └──┘     │    │  └──┘    │    │   └──┘   │
└──────────┘    └──────────┘    └──────────┘
用户框选目标      算法跟踪         算法跟踪
```

## 二、OpenCV 追踪器类型

OpenCV 提供 8 种追踪算法（contrib 模块）：

| 追踪器 | 速度 | 精度 | 特点 |
|--------|------|------|------|
| BOOSTING | 慢 | 低 | 最老 |
| MIL | 慢 | 中 | 比 BOOSTING 好 |
| KCF | 中 | 中 | **性价比高** |
| TLD | 变 | 中 | 可处理遮挡 |
| MEDIANFLOW | 快 | 低 | 可预测失败 |
| GOTURN | 快 | 中 | 基于深度学习 |
| MOSSE | **极快** | 中 | **速度最佳** |
| CSRT | 慢 | **高** | **精度最佳** |

## 三、使用追踪器（OpenCV 4.5+）

```python
# OpenCV 4.5+ 使用 cv2.TrackerXXX_create()
import cv2

# 创建追踪器
tracker_types = {
    'KCF': cv2.TrackerKCF_create,
    'CSRT': cv2.TrackerCSRT_create,
    'MOSSE': cv2.TrackerMOSSE_create,
}

tracker_type = 'CSRT'
tracker = tracker_types[tracker_type]()

# 读取视频
cap = cv2.VideoCapture("video.mp4")
ret, frame = cap.read()

# 选择 ROI
bbox = cv2.selectROI("Select Target", frame, False)
cv2.destroyWindow("Select Target")

# 初始化追踪器
ret = tracker.init(frame, bbox)

while True:
    ret, frame = cap.read()
    if not ret:
        break

    # 更新追踪
    ret, bbox = tracker.update(frame)

    if ret:
        # 追踪成功
        x, y, w, h = [int(v) for v in bbox]
        cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
        cv2.putText(frame, f"Tracking ({tracker_type})", (10, 30),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
    else:
        cv2.putText(frame, "Lost!", (10, 30),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2)

    cv2.imshow("Tracking", frame)
    if cv2.waitKey(30) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

## 五、追踪器选择指南

```
需要实时性（>30fps）？
├── 是 → MOSSE（最快，精度中等）
└── 否 → 需要高精度？
          ├── 是 → CSRT（精度最高，速度慢）
          └── 否 → KCF（平衡选择）
```

```python
def create_tracker(tracker_type='CSRT'):
    """创建追踪器"""
    trackers = {
        'CSRT': cv2.TrackerCSRT_create,
        'KCF': cv2.TrackerKCF_create,
        'MOSSE': cv2.TrackerMOSSE_create,
    }

    if tracker_type not in trackers:
        raise ValueError(f"不支持的追踪器: {tracker_type}")

    return trackers[tracker_type]()
```

## 六、结合检测的跟踪（Tracking-by-Detection）

```python
# 更鲁棒的方法：每隔 N 帧重新检测

class DetectionBasedTracker:
    def __init__(self, detector, tracker_type='CSRT'):
        self.detector = detector  # 面部/物体检测器
        self.tracker_type = tracker_type
        self.tracker = None
        self.detect_interval = 30  # 每30帧重新检测
        self.frame_count = 0

    def update(self, frame):
        if self.frame_count % self.detect_interval == 0 or self.tracker is None:
            # 检测阶段
            detections = self.detector(frame)
            if detections:
                best = max(detections, key=lambda x: x[2])  # 置信度最高的
                x, y, w, h = best[0]
                self.tracker = create_tracker(self.tracker_type)
                self.tracker.init(frame, (x, y, w, h))
                return (x, y, w, h)
        else:
            # 跟踪阶段
            ret, bbox = self.tracker.update(frame)
            if ret:
                return tuple(int(v) for v in bbox)

        return None
```

## 七、评估追踪效果

```python
def evaluate_tracking(tracker, video_path, ground_truth):
    """
    ground_truth: {frame_idx: (x, y, w, h), ...}
    """
    cap = cv2.VideoCapture(video_path)
    frame_idx = 0
    successes = 0
    total = 0

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        if frame_idx == 0:
            tracker.init(frame, ground_truth[0])
        else:
            ret, bbox = tracker.update(frame)
            if ret and frame_idx in ground_truth:
                # 计算 IoU
                gt = ground_truth[frame_idx]
                iou = calculate_iou(bbox, gt)
                if iou > 0.5:
                    successes += 1
                total += 1

        frame_idx += 1

    cap.release()
    return successes / total if total > 0 else 0

def calculate_iou(box1, box2):
    """计算 IoU (Intersection over Union)"""
    x1, y1, w1, h1 = box1
    x2, y2, w2, h2 = box2

    xi1 = max(x1, x2)
    yi1 = max(y1, y2)
    xi2 = min(x1 + w1, x2 + w2)
    yi2 = min(y1 + h1, y2 + h2)

    inter = max(0, xi2 - xi1) * max(0, yi2 - yi1)
    area1 = w1 * h1
    area2 = w2 * h2
    union = area1 + area2 - inter

    return inter / union if union > 0 else 0
```

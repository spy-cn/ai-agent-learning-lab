# 02 - 霍夫变换

## 一、霍夫变换原理

霍夫变换（Hough Transform）用于检测几何形状（直线、圆等）。

```
核心思想：将图像空间中的点转换到参数空间，
在参数空间中投票，峰值处即为检测到的形状。

图像空间：               参数空间（霍夫空间）：
┌───────────────┐       ┌───────────────┐
│   ╱           │       │      ●        │  ← 交点 = 直线参数
│  ╱            │  ←→   │      │        │
│ ╱             │       │  ────┼────    │
│╱              │       │      │        │
└───────────────┘       └───────────────┘
直线 y=mx+b              每个点变成一条曲线
```

## 二、直线检测（标准霍夫变换）

```python
import cv2
import numpy as np

img = cv2.imread("road.jpg")
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# Canny 边缘检测（霍夫变换的前置步骤）
edges = cv2.Canny(gray, 50, 150)

# 标准霍夫直线检测
lines = cv2.HoughLines(
    edges,
    rho=1,              # 距离分辨率（像素）
    theta=np.pi/180,    # 角度分辨率（弧度）
    threshold=100        # 累加器阈值
)

# lines 形状: (N, 1, 2)，每个是 (rho, theta)

if lines is not None:
    for line in lines:
        rho, theta = line[0]
        a = np.cos(theta)
        b = np.sin(theta)
        x0 = a * rho
        y0 = b * rho
        # 计算直线两端点
        x1 = int(x0 + 1000 * (-b))
        y1 = int(y0 + 1000 * (a))
        x2 = int(x0 - 1000 * (-b))
        y2 = int(y0 - 1000 * (a))
        cv2.line(img, (x1, y1), (x2, y2), (0, 0, 255), 2)

cv2.imshow("Hough Lines", img)
cv2.waitKey(0)
```

## 三、概率霍夫变换（推荐）

概率霍夫变换（HoughLinesP）效率更高，直接返回线段端点。

```python
lines = cv2.HoughLinesP(
    edges,
    rho=1,
    theta=np.pi/180,
    threshold=50,
    minLineLength=50,    # 最短线段长度
    maxLineGap=10        # 最大允许间隙（将断线连接）
)

# lines 形状: (N, 1, 4)，每个是 (x1, y1, x2, y2)

if lines is not None:
    for line in lines:
        x1, y1, x2, y2 = line[0]
        cv2.line(img, (x1, y1), (x2, y2), (0, 255, 0), 2)
```

### 参数调优

| 参数 | 含义 | 推荐值 |
|------|------|--------|
| threshold | 投票阈值 | 50-150 |
| minLineLength | 最短线长 | 根据场景 |
| maxLineGap | 最大间隙 | 10-30 |

## 四、圆检测

```python
# 先模糊（重要！）
blurred = cv2.medianBlur(gray, 5)

# 霍夫圆检测
circles = cv2.HoughCircles(
    blurred,
    cv2.HOUGH_GRADIENT,
    dp=1,               # 累加器分辨率与图像的比
    minDist=50,          # 圆心间最小距离
    param1=100,          # Canny 高阈值
    param2=30,           # 累加器阈值（越小检测越多）
    minRadius=10,        # 最小半径
    maxRadius=100        # 最大半径
)

if circles is not None:
    circles = np.uint16(np.around(circles))
    for circle in circles[0, :]:
        cx, cy, r = circle
        # 画圆
        cv2.circle(img, (cx, cy), r, (0, 255, 0), 2)
        # 画圆心
        cv2.circle(img, (cx, cy), 2, (0, 0, 255), 3)
```

### HoughCircles 参数详解

| 参数 | 含义 |
|------|------|
| dp | 累加器分辨率（1=同尺寸，2=半尺寸） |
| minDist | 圆心间最小距离 |
| param1 | Canny 高阈值（低阈值自动为一半） |
| param2 | 圆心累加器阈值 |
| minRadius | 最小半径 |
| maxRadius | 最大半径 |

## 五、实战：车道线检测

```python
def detect_lane_lines(img_path):
    """车道线检测"""
    img = cv2.imread(img_path)
    h, w = img.shape[:2]

    # ROI：只保留下半部分
    mask = np.zeros_like(img)
    polygon = np.array([[
        (int(w*0.1), h),
        (int(w*0.4), int(h*0.6)),
        (int(w*0.6), int(h*0.6)),
        (int(w*0.9), h)
    ]], np.int32)
    cv2.fillPoly(mask, polygon, (255, 255, 255))
    masked = cv2.bitwise_and(img, mask)

    # 边缘检测
    gray = cv2.cvtColor(masked, cv2.COLOR_BGR2GRAY)
    blurred = cv2.GaussianBlur(gray, (5, 5), 0)
    edges = cv2.Canny(blurred, 50, 150)

    # 霍夫直线
    lines = cv2.HoughLinesP(edges, 1, np.pi/180, 50,
                            minLineLength=50, maxLineGap=150)

    # 绘制
    if lines is not None:
        for line in lines:
            x1, y1, x2, y2 = line[0]
            # 计算斜率，过滤水平线
            if abs(y2 - y1) > 10:
                cv2.line(img, (x1, y1), (x2, y2), (0, 255, 0), 5)

    return img

result = detect_lane_lines("road.jpg")
cv2.imshow("Lane Detection", result)
cv2.waitKey(0)
```

## 六、实战：硬币检测

```python
def detect_coins(img_path):
    """检测圆形硬币"""
    img = cv2.imread(img_path)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    blurred = cv2.medianBlur(gray, 5)

    circles = cv2.HoughCircles(blurred, cv2.HOUGH_GRADIENT,
        dp=1, minDist=30, param1=100, param2=30,
        minRadius=15, maxRadius=50)

    count = 0
    if circles is not None:
        circles = np.uint16(np.around(circles))
        for c in circles[0, :]:
            cv2.circle(img, (c[0], c[1]), c[2], (0, 255, 0), 2)
            cv2.circle(img, (c[0], c[1]), 2, (0, 0, 255), 3)
            count += 1

    print(f"检测到 {count} 个硬币")
    return img
```

## 小结

| 功能 | 函数 | 说明 |
|------|------|------|
| 标准直线 | `cv2.HoughLines()` | 返回 rho/theta |
| 概率直线 | `cv2.HoughLinesP()` | 返回端点坐标 |
| 圆检测 | `cv2.HoughCircles()` | 返回圆心和半径 |

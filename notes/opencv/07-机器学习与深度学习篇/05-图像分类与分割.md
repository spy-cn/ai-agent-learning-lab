# 05 - 图像分类与语义分割

## 一、图像分类

使用预训练深度学习模型进行图像分类。

```python
import cv2
import numpy as np

# 加载分类模型（以 ResNet 为例）
net = cv2.dnn.readNetFromONNX("resnet50.onnx")

# ImageNet 标签
with open("classification_classes.txt") as f:
    classes = f.read().splitlines()

img = cv2.imread("dog.jpg")

# 预处理
blob = cv2.dnn.blobFromImage(img, 1/255.0, (224, 224), swapRB=True)
net.setInput(blob)

# 推理
output = net.forward()

# Softmax → 概率
probabilities = np.exp(output[0]) / np.sum(np.exp(output[0]))

# Top-5
top5 = np.argsort(probabilities)[::-1][:5]
for idx in top5:
    print(f"{classes[idx]}: {probabilities[idx]*100:.2f}%")
```

## 四、可视化分割结果

```python
# Cityscapes 数据集颜色映射
COLORS = np.array([
    (0, 0, 0),       (128, 0, 0),     (0, 128, 0),
    (128, 128, 0),   (0, 0, 128),     (128, 0, 128),
    (0, 128, 128),   (128, 128, 128), (64, 0, 0),
    # ... 更多颜色
], dtype=np.uint8)

def visualize_segmentation(seg_map, num_classes=20):
    """将分割图转为彩色可视化"""
    h, w = seg_map.shape
    colored = np.zeros((h, w, 3), dtype=np.uint8)

    for class_id in range(num_classes):
        colored[seg_map == class_id] = COLORS[class_id]

    return colored

# 可视化
seg_vis = visualize_segmentation(seg_map)

# 与原图叠加
overlay = cv2.addWeighted(img, 0.5, cv2.resize(seg_vis, (img.shape[1], img.shape[0])), 0.5, 0)
```

## 六、实时分割

```python
def real_time_segmentation(video_source=0):
    """实时语义分割"""
    cap = cv2.VideoCapture(video_source)

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        blob = cv2.dnn.blobFromImage(frame, 1/255.0, (512, 256), swapRB=True)
        net.setInput(blob)
        output = net.forward()

        seg_map = np.argmax(output[0], axis=0)
        seg_vis = visualize_segmentation(seg_map)

        # 调整到原始大小
        seg_vis = cv2.resize(seg_vis, (frame.shape[1], frame.shape[0]))

        # 并排显示
        combined = np.hstack([frame, seg_vis])
        cv2.imshow("Segmentation", combined)

        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()
```

## 八、实用工具：图像标注辅助

```python
def auto_label_image(img_path, model_path, labels_path, conf_threshold=0.5):
    """使用预训练模型自动标注图像"""
    net = cv2.dnn.readNetFromONNX(model_path)
    with open(labels_path) as f:
        labels = f.read().splitlines()

    img = cv2.imread(img_path)
    blob = cv2.dnn.blobFromImage(img, 1/255.0, (224, 224), swapRB=True)
    net.setInput(blob)
    output = net.forward()

    probs = np.exp(output[0]) / np.sum(np.exp(output[0]))
    top_label = labels[np.argmax(probs)]
    top_conf = np.max(probs)

    if top_conf > conf_threshold:
        cv2.putText(img, f"{top_label}: {top_conf:.2%}",
                    (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 0), 2)

    return img, top_label, top_conf
```

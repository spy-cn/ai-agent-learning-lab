# 04 - 人脸检测与识别

## 一、人脸检测方法概览

```
方法演进：

1. Viola-Jones (Haar级联)  ← 最快，但精度一般
2. HOG + SVM               ← 速度快，中等精度
3. DNN Face Detector        ← 精度高，推荐
4. MediaPipe / MTCNN        ← 实时高精度
```

## 二、Haar 级联分类器（经典方法）

```python
import cv2

# 加载预训练的 Haar 级联
face_cascade = cv2.CascadeClassifier(
    cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
)

# 可选的其他级联
eye_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_eye.xml')
smile_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_smile.xml')

# 检测人脸
img = cv2.imread("people.jpg")
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

faces = face_cascade.detectMultiScale(
    gray,
    scaleFactor=1.1,   # 图像缩放比例
    minNeighbors=5,     # 检测框保留的最少邻居数
    minSize=(30, 30),   # 最小人脸尺寸
    flags=cv2.CASCADE_SCALE_IMAGE
)

# 绘制
for (x, y, w, h) in faces:
    cv2.rectangle(img, (x, y), (x+w, y+h), (255, 0, 0), 2)

    # 在人脸区域内检测眼睛
    roi_gray = gray[y:y+h, x:x+w]
    roi_color = img[y:y+h, x:x+w]
    eyes = eye_cascade.detectMultiScale(roi_gray)
    for (ex, ey, ew, eh) in eyes:
        cv2.rectangle(roi_color, (ex, ey), (ex+ew, ey+eh), (0, 255, 0), 2)

cv2.imshow("Faces", img)
cv2.waitKey(0)
```

### Haar 参数调优

| 参数 | 作用 | 推荐值 |
|------|------|--------|
| scaleFactor | 缩放步长 | 1.05-1.3 |
| minNeighbors | 邻域阈值 | 3-6 |
| minSize | 最小尺寸 | (30,30) |

## 三、DNN 人脸检测（推荐）

```python
# 使用 OpenCV 内置的 DNN 人脸检测器
model_weights = "res10_300x300_ssd_iter_140000_fp16.caffemodel"
model_config = "deploy.prototxt"

net = cv2.dnn.readNetFromCaffe(model_config, model_weights)

def detect_faces_dnn(img, conf_threshold=0.7):
    """DNN 人脸检测"""
    h, w = img.shape[:2]
    blob = cv2.dnn.blobFromImage(cv2.resize(img, (300, 300)),
                                  1.0, (300, 300), (104.0, 177.0, 123.0))
    net.setInput(blob)
    detections = net.forward()

    faces = []
    for i in range(detections.shape[2]):
        confidence = detections[0, 0, i, 2]
        if confidence > conf_threshold:
            box = detections[0, 0, i, 3:7] * np.array([w, h, w, h])
            x1, y1, x2, y2 = box.astype("int")
            faces.append((x1, y1, x2-x1, y2-y1))

    return faces
```

## 四、人脸识别流程

```
人脸识别 = 检测 + 对齐 + 特征提取 + 匹配

1. 检测：找到人脸位置
2. 对齐：标准化人脸（眼睛对齐）
3. 特征提取：提取 128/512 维特征向量
4. 匹配：与数据库中的特征比对
```

### 特征脸（Eigenfaces）

```python
import os
import cv2
import numpy as np

def train_eigenfaces(data_dir, size=(100, 100)):
    """训练特征脸模型"""
    faces = []
    labels = []
    label_names = {}

    for label_id, person_name in enumerate(os.listdir(data_dir)):
        label_names[label_id] = person_name
        person_dir = os.path.join(data_dir, person_name)
        for img_name in os.listdir(person_dir):
            img_path = os.path.join(person_dir, img_name)
            img = cv2.imread(img_path, 0)  # 灰度
            img = cv2.resize(img, size)
            faces.append(img.flatten().astype(np.float32))
            labels.append(label_id)

    faces = np.array(faces)
    labels = np.array(labels)

    # 创建特征脸识别器
    recognizer = cv2.face.EigenFaceRecognizer_create()
    recognizer.train(faces, labels)

    return recognizer, label_names

# 预测
recognizer, names = train_eigenfaces("face_database/")
gray = cv2.cvtColor(face_img, cv2.COLOR_BGR2GRAY)
gray = cv2.resize(gray, (100, 100))
label, confidence = recognizer.predict(gray.flatten().astype(np.float32))
print(f"识别: {names[label]}, 置信度: {confidence:.1f}")
```

## 五、使用深度特征进行识别

```python
# 使用 DNN 提取人脸特征（如 OpenFace）
embed_net = cv2.dnn.readNetFromTorch("openface_nn4.small2.v1.t7")

def get_face_embedding(face_img):
    """提取 128 维人脸特征"""
    blob = cv2.dnn.blobFromImage(face_img, 1/255.0, (96, 96), (0, 0, 0),
                                 swapRB=True, crop=False)
    embed_net.setInput(blob)
    embedding = embed_net.forward()
    return embedding.flatten()

# 比对特征（余弦相似度）
def compare_faces(embedding1, embedding2, threshold=0.6):
    """比对两个人脸特征"""
    similarity = np.dot(embedding1, embedding2) / (
        np.linalg.norm(embedding1) * np.linalg.norm(embedding2))
    return similarity > threshold, similarity

# 使用
emb1 = get_face_embedding(face1)
emb2 = get_face_embedding(face2)
is_match, score = compare_faces(emb1, emb2)
print(f"匹配: {is_match}, 相似度: {score:.4f}")
```

## 六、完整人脸识别系统

```python
class FaceRecognitionSystem:
    def __init__(self):
        self.detector = cv2.CascadeClassifier(
            cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
        self.recognizer = cv2.face.LBPHRecognizer_create()
        self.label_names = {}
        self.trained = False

    def register_face(self, img, name):
        """注册人脸"""
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        faces = self.detector.detectMultiScale(gray, 1.3, 5)

        for (x, y, w, h) in faces:
            face = gray[y:y+h, x:x+w]
            # 存储到数据库
            self.save_to_database(face, name)

    def save_to_database(self, face_img, name):
        """保存人脸到数据库"""
        os.makedirs(f"database/{name}", exist_ok=True)
        count = len(os.listdir(f"database/{name}"))
        cv2.imwrite(f"database/{name}/face_{count}.jpg", face_img)

    def train(self):
        """训练识别器"""
        faces, labels = [], []
        names = os.listdir("database")

        for label_id, name in enumerate(names):
            self.label_names[label_id] = name
            for img_name in os.listdir(f"database/{name}"):
                img = cv2.imread(f"database/{name}/{img_name}", 0)
                faces.append(img)
                labels.append(label_id)

        self.recognizer.train(faces, np.array(labels))
        self.trained = True

    def recognize(self, img, confidence_threshold=80):
        """识别图像中的人脸"""
        if not self.trained:
            return []

        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        detected = self.detector.detectMultiScale(gray, 1.3, 5)

        results = []
        for (x, y, w, h) in detected:
            face = gray[y:y+h, x:x+w]
            label, confidence = self.recognizer.predict(face)

            if confidence < confidence_threshold:
                name = self.label_names.get(label, "Unknown")
            else:
                name = "Unknown"

            results.append({
                "name": name,
                "confidence": confidence,
                "box": (x, y, w, h)
            })

        return results
```

## 小结

| 方法 | 函数 | 特点 |
|------|------|------|
| Haar | `CascadeClassifier` | 快速，精度中等 |
| DNN | `dnn.readNetFromCaffe` | 精度高 |
| Eigenfaces | `cv2.face.EigenFaceRecognizer` | 基于PCA |
| LBPH | `cv2.face.LBPHRecognizer` | 光照鲁棒 |
| 深度特征 | DNN embedding + 余弦相似度 | 精度最高 |

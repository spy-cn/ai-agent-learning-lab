# 04 - 绘图功能

## 一、绘图基础

OpenCV 的绘图函数直接在图像（NumPy 数组）上修改像素，不会创建新图像。

```python
import cv2
import numpy as np

# 创建画布
canvas = np.zeros((500, 500, 3), dtype=np.uint8)
```

### 坐标说明

所有绘图函数使用 **(x, y)** 坐标（注意不是 NumPy 的 [y, x]）。

```
画布坐标系：
(0,0) → x
  ↓
  y
```

## 二、直线

```python
# cv2.line(img, pt1, pt2, color, thickness)
cv2.line(canvas, (0, 0), (500, 500), (0, 255, 0), 2)

# 带抗锯齿
cv2.line(canvas, (0, 500), (500, 0), (0, 0, 255), 2, cv2.LINE_AA)

# 参数：
# - pt1: 起点 (x, y)
# - pt2: 终点 (x, y)
# - color: BGR 颜色 (B, G, R)
# - thickness: 线宽（像素）
# - lineType: cv2.LINE_4 / LINE_8 / LINE_AA
```

## 三、矩形

```python
# cv2.rectangle(img, pt1, pt2, color, thickness)
# pt1 = 左上角, pt2 = 右下角
cv2.rectangle(canvas, (50, 50), (200, 150), (255, 0, 0), 2)

# 实心矩形（thickness = -1）
cv2.rectangle(canvas, (250, 50), (400, 150), (0, 255, 255), -1)

# 圆角矩形（OpenCV 4.x）
cv2.rectangle(canvas, (50, 200), (200, 300), (255, 0, 255), 2,
              cv2.LINE_AA, shift=0)
```

## 四、圆形

```python
# cv2.circle(img, center, radius, color, thickness)
cv2.circle(canvas, (100, 350), 50, (0, 255, 0), 2)

# 实心圆
cv2.circle(canvas, (250, 350), 50, (0, 0, 255), -1)

# 抗锯齿圆
cv2.circle(canvas, (400, 350), 50, (255, 0, 0), 2, cv2.LINE_AA)
```

## 五、椭圆

```python
# cv2.ellipse(img, center, axes, angle, startAngle, endAngle, color, thickness)
cv2.ellipse(canvas, (250, 250), (100, 50),   # 中心, 长短轴
            0,                                  # 旋转角度
            0, 360,                             # 起始/结束角度
            (255, 255, 0), 2)

# 旋转 45 度
cv2.ellipse(canvas, (250, 250), (100, 50),
            45, 0, 360, (255, 0, 255), 2)

# 半椭圆（0-180度）
cv2.ellipse(canvas, (250, 250), (80, 40),
            0, 0, 180, (0, 255, 255), -1)
```

## 六、多边形

```python
# ─── 画多边形线段 ───
pts = np.array([[100, 50], [200, 100], [150, 200], [50, 150]], np.int32)
cv2.polylines(canvas, [pts],      # 点集
              True,               # isClosed: 是否闭合
              (0, 255, 255), 3)   # 颜色, 线宽

# ─── 实心多边形 ───
cv2.fillPoly(canvas, [pts], (0, 255, 255))

# ─── 抗锯齿 ───
cv2.polylines(canvas, [pts], True, (255, 0, 0), 3, cv2.LINE_AA)
```

### 中文显示问题

OpenCV 的 `putText` **不支持中文**，需要借助 Pillow：

```python
from PIL import Image, ImageDraw, ImageFont

def put_chinese_text(img, text, position, text_color=(0, 255, 0), font_size=25):
    """在图像上绘制中文"""
    cv2_img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    pil_img = Image.fromarray(cv2_img)

    draw = ImageDraw.Draw(pil_img)
    font = ImageFont.truetype("msyh.ttc", font_size)  # 微软雅黑

    # text_color 是 BGR，Pillow 需要 RGB
    rgb_color = (text_color[2], text_color[1], text_color[0])
    draw.text(position, text, fill=rgb_color, font=font)

    return cv2.cvtColor(np.array(pil_img), cv2.COLOR_RGB2BGR)

# 使用
img = cv2.imread("image.jpg")
img = put_chinese_text(img, "你好，世界！", (50, 50), (0, 255, 0), 30)
```

## 九、组合绘图示例

```python
def draw_annotations(img, detections):
    """
    在图像上绘制检测框和标签
    detections: [{"box": [x1,y1,x2,y2], "label": "face", "confidence": 0.95}]
    """
    for det in detections:
        x1, y1, x2, y2 = det["box"]
        label = det["label"]
        conf = det["confidence"]

        # 画检测框
        cv2.rectangle(img, (x1, y1), (x2, y2), (0, 255, 0), 2)

        # 标签文字
        text = f"{label} {conf:.2f}"
        (tw, th), _ = cv2.getTextSize(text, cv2.FONT_HERSHEY_SIMPLEX, 0.6, 2)

        # 标签背景
        cv2.rectangle(img, (x1, y1 - th - 10), (x1 + tw + 5, y1), (0, 255, 0), -1)
        cv2.putText(img, text, (x1 + 2, y1 - 5),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 0, 0), 2)

    return img
```

## 十、实用辅助函数

```python
def draw_grid(img, grid_size=50, color=(50, 50, 50)):
    """在图像上绘制网格"""
    h, w = img.shape[:2]
    for x in range(0, w, grid_size):
        cv2.line(img, (x, 0), (x, h), color, 1)
    for y in range(0, h, grid_size):
        cv2.line(img, (0, y), (w, y), color, 1)
    return img

def draw_center_crosshair(img, color=(0, 255, 0), size=20):
    """绘制中心十字准星"""
    h, w = img.shape[:2]
    cx, cy = w // 2, h // 2
    cv2.line(img, (cx - size, cy), (cx + size, cy), color, 2)
    cv2.line(img, (cx, cy - size), (cx, cy + size), color, 2)
    cv2.circle(img, (cx, cy), size // 2, color, 1)
    return img

def draw_text_with_bg(img, text, org, font_scale=0.6,
                      text_color=(255, 255, 255), bg_color=(0, 0, 0)):
    """带背景的文字"""
    (tw, th), baseline = cv2.getTextSize(
        text, cv2.FONT_HERSHEY_SIMPLEX, font_scale, 2)
    x, y = org
    cv2.rectangle(img, (x, y - th), (x + tw + 10, y + baseline), bg_color, -1)
    cv2.putText(img, text, (x + 5, y),
                cv2.FONT_HERSHEY_SIMPLEX, font_scale, text_color, 2)
    return img
```

## 小结

| 绘图操作 | 函数 |
|----------|------|
| 直线 | `cv2.line(img, pt1, pt2, color, thickness)` |
| 矩形 | `cv2.rectangle(img, pt1, pt2, color, thickness)` |
| 圆形 | `cv2.circle(img, center, radius, color, thickness)` |
| 椭圆 | `cv2.ellipse(img, center, axes, angle, ...)` |
| 多边形 | `cv2.polylines(img, pts, isClosed, color, thickness)` |
| 文字 | `cv2.putText(img, text, org, fontFace, ...)` |
| 填充 | `thickness=-1` |

# 05 - 鼠标与滑动条交互

## 一、鼠标事件基础

OpenCV 提供了鼠标回调机制，可以响应用户的鼠标操作。

```python
import cv2
import numpy as np

# 查看所有可用事件
events = [i for i in dir(cv2) if 'EVENT' in i]
print(events)
```

### 常用鼠标事件

| 事件常量 | 含义 |
|----------|------|
| `cv2.EVENT_LBUTTONDOWN` | 左键按下 |
| `cv2.EVENT_RBUTTONDOWN` | 右键按下 |
| `cv2.EVENT_MBUTTONDOWN` | 中键按下 |
| `cv2.EVENT_LBUTTONUP` | 左键释放 |
| `cv2.EVENT_RBUTTONUP` | 右键释放 |
| `cv2.EVENT_MOUSEMOVE` | 鼠标移动 |
| `cv2.EVENT_LBUTTONDBLCLK` | 左键双击 |
| `cv2.EVENT_MOUSEWHEEL` | 滚轮滚动 |

### flags 参数

| flag | 含义 |
|------|------|
| `cv2.EVENT_FLAG_LBUTTON` | 左键按下（移动时） |
| `cv2.EVENT_FLAG_RBUTTON` | 右键按下 |
| `cv2.EVENT_FLAG_SHIFT` | Shift 键按下 |
| `cv2.EVENT_FLAG_CTRL` | Ctrl 键按下 |
| `cv2.EVENT_FLAG_ALT` | Alt 键按下 |

## 二、鼠标回调函数

```python
# 回调函数签名
def mouse_callback(event, x, y, flags, param):
    """
    event: 事件类型
    x, y: 鼠标坐标
    flags: 修饰键状态
    param: 用户传入的数据
    """
    if event == cv2.EVENT_LBUTTONDOWN:
        print(f"左键点击: ({x}, {y})")
    elif event == cv2.EVENT_RBUTTONDOWN:
        print(f"右键点击: ({x}, {y})")

# 绑定回调
cv2.namedWindow("Image")
cv2.setMouseCallback("Image", mouse_callback)

img = cv2.imread("image.jpg")
cv2.imshow("Image", img)
cv2.waitKey(0)
```

## 三、实战示例：画矩形框

```python
import cv2
import numpy as np

# 全局变量
drawing = False
ix, iy = -1, -1
img = None
img_copy = None

def draw_rectangle(event, x, y, flags, param):
    global drawing, ix, iy, img, img_copy

    if event == cv2.EVENT_LBUTTONDOWN:
        drawing = True
        ix, iy = x, y

    elif event == cv2.EVENT_MOUSEMOVE:
        if drawing:
            img = img_copy.copy()  # 恢复原图
            cv2.rectangle(img, (ix, iy), (x, y), (0, 255, 0), 2)

    elif event == cv2.EVENT_LBUTTONUP:
        drawing = False
        cv2.rectangle(img, (ix, iy), (x, y), (0, 255, 0), 2)
        print(f"矩形: ({ix},{iy}) → ({x},{y}), 大小: {abs(x-ix)}×{abs(y-iy)}")

img = cv2.imread("image.jpg")
img_copy = img.copy()

cv2.namedWindow("Draw Rectangle")
cv2.setMouseCallback("Draw Rectangle", draw_rectangle)

while True:
    cv2.imshow("Draw Rectangle", img)
    key = cv2.waitKey(1) & 0xFF
    if key == ord('q'):   # 退出
        break
    elif key == ord('r'): # 重置
        img = img_copy.copy()

cv2.destroyAllWindows()
```

## 四、实战示例：自由绘制

```python
import cv2
import numpy as np

canvas = np.zeros((500, 500, 3), dtype=np.uint8)
drawing = False
prev_pt = None
color = (0, 255, 0)
brush_size = 3

def draw(event, x, y, flags, param):
    global drawing, prev_pt, canvas

    if event == cv2.EVENT_LBUTTONDOWN:
        drawing = True
        prev_pt = (x, y)

    elif event == cv2.EVENT_MOUSEMOVE:
        if drawing:
            cv2.line(canvas, prev_pt, (x, y), color, brush_size)
            prev_pt = (x, y)

    elif event == cv2.EVENT_LBUTTONUP:
        drawing = False

cv2.namedWindow("Paint")
cv2.setMouseCallback("Paint", draw)

print("操作说明:")
print("  鼠标拖拽: 绘制")
print("  c: 清空画布")
print("  1-5: 切换颜色")
print("  +/-: 调整笔刷大小")
print("  q: 退出")

while True:
    cv2.imshow("Paint", canvas)
    key = cv2.waitKey(1) & 0xFF

    if key == ord('q'):
        break
    elif key == ord('c'):
        canvas = np.zeros((500, 500, 3), dtype=np.uint8)
    elif key == ord('1'): color = (0, 255, 0)    # 绿
    elif key == ord('2'): color = (0, 0, 255)    # 红
    elif key == ord('3'): color = (255, 0, 0)    # 蓝
    elif key == ord('4'): color = (0, 255, 255)  # 黄
    elif key == ord('5'): color = (255, 255, 255)# 白
    elif key == ord('=') or key == ord('+'): brush_size = min(brush_size + 1, 20)
    elif key == ord('-'): brush_size = max(brush_size - 1, 1)

cv2.destroyAllWindows()
```

## 五、滑动条（Trackbar）

```python
import cv2
import numpy as np

def nothing(x):
    """滑动条回调（x 是当前值）"""
    pass

# 创建窗口和滑动条
cv2.namedWindow("Trackbar Demo")
cv2.createTrackbar("R", "Trackbar Demo", 0, 255, nothing)
cv2.createTrackbar("G", "Trackbar Demo", 0, 255, nothing)
cv2.createTrackbar("B", "Trackbar Demo", 0, 255, nothing)
cv2.createTrackbar("Switch", "Trackbar Demo", 0, 1, nothing)

while True:
    # 读取滑动条值
    r = cv2.getTrackbarPos("R", "Trackbar Demo")
    g = cv2.getTrackbarPos("G", "Trackbar Demo")
    b = cv2.getTrackbarPos("B", "Trackbar Demo")
    s = cv2.getTrackbarPos("Switch", "Trackbar Demo")

    # 根据开关决定显示颜色还是黑色
    color = (b, g, r) if s else (0, 0, 0)
    img = np.zeros((300, 300, 3), np.uint8)
    img[:] = color

    cv2.imshow("Trackbar Demo", img)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cv2.destroyAllWindows()
```

## 六、综合应用：实时 Canny 边缘参数调节

```python
import cv2

def nothing(x):
    pass

img = cv2.imread("image.jpg")
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

cv2.namedWindow("Canny Edge")
cv2.createTrackbar("Threshold1", "Canny Edge", 50, 500, nothing)
cv2.createTrackbar("Threshold2", "Canny Edge", 150, 500, nothing)
cv2.createTrackbar("Blur Size", "Canny Edge", 3, 15, nothing)

while True:
    t1 = cv2.getTrackbarPos("Threshold1", "Canny Edge")
    t2 = cv2.getTrackbarPos("Threshold2", "Canny Edge")
    blur_size = cv2.getTrackbarPos("Blur Size", "Canny Edge")
    blur_size = blur_size if blur_size % 2 == 1 else blur_size + 1  # 必须为奇数

    blurred = cv2.GaussianBlur(gray, (blur_size, blur_size), 0)
    edges = cv2.Canny(blurred, t1, t2)

    cv2.imshow("Canny Edge", edges)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cv2.destroyAllWindows()
```

## 七、实战：HSV 颜色选取器

```python
import cv2
import numpy as np

cap = cv2.VideoCapture(0)

cv2.namedWindow("HSV Picker")
cv2.createTrackbar("H_Min", "HSV Picker", 0, 179, nothing)
cv2.createTrackbar("H_Max", "HSV Picker", 179, 179, nothing)
cv2.createTrackbar("S_Min", "HSV Picker", 0, 255, nothing)
cv2.createTrackbar("S_Max", "HSV Picker", 255, 255, nothing)
cv2.createTrackbar("V_Min", "HSV Picker", 0, 255, nothing)
cv2.createTrackbar("V_Max", "HSV Picker", 255, 255, nothing)

def nothing(x): pass

while True:
    ret, frame = cap.read()
    if not ret:
        break

    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)

    h_min = cv2.getTrackbarPos("H_Min", "HSV Picker")
    h_max = cv2.getTrackbarPos("H_Max", "HSV Picker")
    s_min = cv2.getTrackbarPos("S_Min", "HSV Picker")
    s_max = cv2.getTrackbarPos("S_Max", "HSV Picker")
    v_min = cv2.getTrackbarPos("V_Min", "HSV Picker")
    v_max = cv2.getTrackbarPos("V_Max", "HSV Picker")

    lower = np.array([h_min, s_min, v_min])
    upper = np.array([h_max, s_max, v_max])

    mask = cv2.inRange(hsv, lower, upper)
    result = cv2.bitwise_and(frame, frame, mask=mask)

    cv2.imshow("Original", frame)
    cv2.imshow("Mask", mask)
    cv2.imshow("Result", result)

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

# 打印最终的 HSV 范围
print(f"HSV 范围: Lower={lower}, Upper={upper}")

cap.release()
cv2.destroyAllWindows()
```

## 小结

| 功能 | 函数 |
|------|------|
| 鼠标回调 | `cv2.setMouseCallback(winname, callback)` |
| 创建滑动条 | `cv2.createTrackbar(name, winname, value, count, callback)` |
| 获取滑动条值 | `cv2.getTrackbarPos(name, winname)` |
| 常用模式 | 全局变量 + while 循环 + waitKey(1) |

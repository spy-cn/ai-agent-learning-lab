# 04 - 特征匹配

## 一、特征匹配流程

```
图像A ──→ 特征提取 ──→ 描述符A ─┐
                                  ├─→ 匹配器 ──→ 匹配结果
图像B ──→ 特征提取 ──→ 描述符B ─┘

应用场景：
- 图像拼接（全景图）
- 物体识别
- 3D 重建
- 增强现实
```

## 二、暴力匹配详解

```python
import cv2
import numpy as np

img1 = cv2.imread("object.jpg", 0)       # 模板
img2 = cv2.imread("scene.jpg", 0)        # 场景

# SIFT 特征
sift = cv2.SIFT_create()
kp1, des1 = sift.detectAndCompute(img1, None)
kp2, des2 = sift.detectAndCompute(img2, None)

# ─── 方式1：match（每个特征返回最佳匹配）───
bf = cv2.BFMatcher(cv2.NORM_L2, crossCheck=True)
matches = bf.match(des1, des2)

# ─── 方式2：knnMatch（每个特征返回 k 个匹配）───
bf = cv2.BFMatcher(cv2.NORM_L2)
matches = bf.knnMatch(des1, des2, k=2)  # 返回 2 个最近邻
```

### match 对象

```python
for m in matches[:3]:
    print(f"""

    trainIdx: {m.trainIdx}  (图B中的描述符索引)
    distance: {m.distance:.2f}  (匹配距离，越小越好)
```

## 三、比率测试（Lowe's Ratio Test）

```python
# 对每个特征点，取最近和次近的两个匹配
# 如果最近/次近 < 阈值，则认为最近匹配是可靠的

bf = cv2.BFMatcher(cv2.NORM_L2)
matches = bf.knnMatch(des1, des2, k=2)

good_matches = []
ratio_threshold = 0.75  # Lowe 推荐值

for m, n in matches:
    if m.distance < ratio_threshold * n.distance:
        good_matches.append(m)

print(f"原始匹配: {len(matches)}, 过滤后: {len(good_matches)}")
```

```
比率测试原理：

情况1（好匹配）：        情况2（模糊匹配）：
最佳: distance=50       最佳: distance=100
次佳: distance=200      次佳: distance=110
比率: 50/200=0.25       比率: 100/110=0.91
→ 通过 ✓               → 拒绝 ✗（太相似，无法区分）
```

## 四、RANSAC 过滤

```python
# 即使比率测试后，仍可能有误匹配
# RANSAC 通过几何约束进一步过滤

src_pts = np.float32([kp1[m.queryIdx].pt for m in good_matches]).reshape(-1, 1, 2)
dst_pts = np.float32([kp2[m.trainIdx].pt for m in good_matches]).reshape(-1, 1, 2)

# 计算单应矩阵
H, mask = cv2.findHomography(src_pts, dst_pts, cv2.RANSAC, 5.0)

# mask 是一个布尔数组，True 表示内点
inlier_matches = [m for i, m in enumerate(good_matches) if mask[i]]
print(f"内点匹配: {len(inlier_matches)}")
```

## 六、完整流程：物体定位

```python
def locate_object(template_path, scene_path):
    """在场景图中定位模板物体"""
    img1 = cv2.imread(template_path, cv2.COLOR_GRAY2RGB)
    img2 = cv2.imread(scene_path, cv2.COLOR_GRAY2RGB)
    gray1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
    gray2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)

    # 1. 提取 SIFT 特征
    sift = cv2.SIFT_create()
    kp1, des1 = sift.detectAndCompute(gray1, None)
    kp2, des2 = sift.detectAndCompute(gray2, None)

    # 2. FLANN 匹配
    FLANN_INDEX_KDTREE = 1
    index_params = dict(algorithm=FLANN_INDEX_KDTREE, trees=5)
    search_params = dict(checks=50)
    flann = cv2.FlannBasedMatcher(index_params, search_params)
    matches = flann.knnMatch(des1, des2, k=2)

    # 3. 比率测试
    good = []
    for m, n in matches:
        if m.distance < 0.7 * n.distance:
            good.append(m)

    if len(good) < 4:
        print("匹配点不足，无法定位")
        return None

    # 4. 计算单应矩阵
    src_pts = np.float32([kp1[m.queryIdx].pt for m in good]).reshape(-1, 1, 2)
    dst_pts = np.float32([kp2[m.trainIdx].pt for m in good]).reshape(-1, 1, 2)
    H, mask = cv2.findHomography(src_pts, dst_pts, cv2.RANSAC, 5.0)

    # 5. 在场景图中绘制模板的边界框
    h, w = gray1.shape
    corners = np.float32([[0, 0], [w-1, 0], [w-1, h-1], [0, h-1]]).reshape(-1, 1, 2)
    transformed_corners = cv2.perspectiveTransform(corners, H)
    img2 = cv2.polylines(img2, [np.int32(transformed_corners)], True, (0, 255, 0), 3)

    # 6. 绘制匹配
    result = cv2.drawMatches(img1, kp1, img2, kp2, good, None,
        flags=cv2.DrawMatchesFlags_NOT_DRAW_SINGLE_POINTS)

    return result

result = locate_object("template.jpg", "scene.jpg")
if result is not None:
    cv2.imshow("Object Located", result)
    cv2.waitKey(0)
```

## 七、ORB 特征匹配

```python
# ORB 使用 Hamming 距离
orb = cv2.ORB_create()
kp1, des1 = orb.detectAndCompute(img1, None)
kp2, des2 = orb.detectAndCompute(img2, None)

# 注意：NORM_HAMMING
bf = cv2.BFMatcher(cv2.NORM_HAMMING, crossCheck=True)
matches = bf.match(des1, des2)

# 绘制最好的 20 个
matches = sorted(matches, key=lambda x: x.distance)
result = cv2.drawMatches(img1, kp1, img2, kp2, matches[:20], None)
```

## 八、FLANN 配置指南

```python
# ─── SIFT/SURF (float 描述符) ───
FLANN_INDEX_KDTREE = 1
index_params = dict(algorithm=FLANN_INDEX_KDTREE, trees=5)
search_params = dict(checks=50)
# trees: KD树数量 (5是推荐值)
# checks: 搜索深度 (50-100)

# ─── ORB/BRIEF (binary 描述符) ───
FLANN_INDEX_LSH = 6
index_params = dict(
    algorithm=FLANN_INDEX_LSH,
    table_number=6,      # 哈希表数量
    key_size=12,         # 哈希键长度
    multi_probe_level=1  # 多探查级别
)
search_params = dict(checks=50)
```

## 小结

| 步骤 | 方法 | 说明 |
|------|------|------|
| 匹配 | BFMatcher | 精确但慢 |
| 匹配 | FLANN | 快速近似 |
| 过滤 | 比率测试 | 0.7-0.8 阈值 |
| 过滤 | RANSAC | 几何一致性 |
| 几何 | findHomography | 单应矩阵 |
| 几何 | findFundamentalMat | 基础矩阵 |

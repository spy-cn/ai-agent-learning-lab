# 05 - 轮廓检测与处理

## 一、什么是轮廓？

轮廓是具有**相同颜色或强度**的连续点构成的曲线。轮廓分析是形状分析、物体检测的基础。

```
轮廓 vs 边缘：
- 边缘：像素级的变化检测
- 轮廓：完整的封闭曲线，可用于后续形状分析
```

## 二、查找轮廓

```python
import cv2
import numpy as np

img = cv2.imread("shapes.jpg")
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 二值化（轮廓检测前必须）
ret, thresh = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)
# 或用自适应阈值 / Canny

# 查找轮廓
contours, hierarchy = cv2.findContours(thresh,
    cv2.RETR_EXTERNAL,           # 轮廓检索模式
    cv2.CHAIN_APPROX_SIMPLE)     # 轮廓近似方法

print(f"检测到 {len(contours)} 个轮廓")
```

### 轮廓检索模式

| 模式 | 说明 |
|------|------|
| `cv2.RETR_EXTERNAL` | 只检测最外层轮廓 |
| `cv2.RETR_LIST` | 检测所有轮廓（无层级关系） |
| `cv2.RETR_CCOMP` | 两级轮廓（外轮廓+内孔） |
| `cv2.RETR_TREE` | 完整层级树 |

### 轮廓近似方法

| 方法 | 说明 |
|------|------|
| `cv2.CHAIN_APPROX_NONE` | 存储所有轮廓点 |
| `cv2.CHAIN_APPROX_SIMPLE` | 压缩水平/垂直/对角线（推荐） |
| `cv2.CHAIN_APPROX_TC89_L1` | 应用 Teh-Chin 链近似 |

## 四、轮廓特征

```python
for i, cnt in enumerate(contours):
    # ─── 面积 ───

    (cx, cy), radius = cv2.minEnclosingCircle(cnt)

    # ─── 凸包
    hull = cv2.convexHull(cnt)

    # ─── 拟合椭圆（至少需要 5 个点）───
    if len(cnt) >= 5:

    # ─── 拟合直线 ───
    [vx, vy, x, y] = cv2.fitLine(cnt, cv2.DIST_L2, 0, 0.01, 0.01)
```

## 五、轮廓近似

```python
# Douglas-Peucker 算法：用更少的点近似轮廓
epsilon = 0.01 * cv2.arcLength(cnt, True)  # 近似精度（1%周长）
approx = cv2.approxPolyDP(cnt, epsilon, True)

# 不同 epsilon 的效果
for ratio in [0.1, 0.05, 0.02, 0.01]:
    eps = ratio * cv2.arcLength(cnt, True)
    approx = cv2.approxPolyDP(cnt, eps, True)
    print(f"epsilon={ratio:.2f}: {len(approx)} 个顶点")
```

```
epsilon 越小 → 顶点越多 → 越精确
epsilon 越大 → 顶点越少 → 越简化
```

## 六、形状识别

```python
def identify_shape(cnt):
    """识别基本形状"""
    epsilon = 0.04 * cv2.arcLength(cnt, True)
    approx = cv2.approxPolyDP(cnt, epsilon, True)
    num_vertices = len(approx)

    if num_vertices == 3:
        return "Triangle"
    elif num_vertices == 4:
        # 判断是矩形还是正方形
        x, y, w, h = cv2.boundingRect(approx)
        aspect_ratio = float(w) / h
        if 0.9 <= aspect_ratio <= 1.1:
            return "Square"
        else:
            return "Rectangle"
    elif num_vertices == 5:
        return "Pentagon"
    elif num_vertices == 6:
        return "Hexagon"
    else:
        # 判断是否是圆
        area = cv2.contourArea(cnt)
        perimeter = cv2.arcLength(cnt, True)
        if perimeter == 0:
            return "Unknown"
        circularity = 4 * np.pi * area / (perimeter ** 2)
        if circularity > 0.85:
            return "Circle"
        return f"Polygon ({num_vertices} vertices)"

# 应用
for cnt in contours:
    shape = identify_shape(cnt)
    M = cv2.moments(cnt)
    if M["m00"] != 0:
        cx = int(M["m10"] / M["m00"])
        cy = int(M["m01"] / M["m00"])
        cv2.putText(img, shape, (cx - 30, cy),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 0, 0), 2)
```

## 七、轮廓矩（Moments）

轮廓矩可用于计算质心、面积等特征。

```python
M = cv2.moments(cnt)

# 空间矩
print(f"面积 (m00): {M['m00']}")

# 质心
if M["m00"] != 0:
    cx = int(M["m10"] / M["m00"])
    cy = int(M["m01"] / M["m00"])
    print(f"质心: ({cx}, {cy})")

# Hu 矩（旋转/缩放/镜像不变）
hu_moments = cv2.HuMoments(M)
```

## 九、凸缺陷

```python
# 凸包与凸缺陷（用于手势识别）
cnt = contours[0]
hull = cv2.convexHull(cnt, returnPoints=False)
defects = cv2.convexityDefects(cnt, hull)

if defects is not None:
    for i in range(defects.shape[0]):
        s, e, f, d = defects[i, 0]
        start = tuple(cnt[s][0])
        end = tuple(cnt[e][0])
        far = tuple(cnt[f][0])
        depth = d / 256.0

        if depth > 10:  # 只画显著的缺陷
            cv2.line(img, start, far, (0, 255, 0), 2)
            cv2.line(img, end, far, (0, 255, 0), 2)
            cv2.circle(img, far, 5, (0, 0, 255), -1)
```

## 十、点与轮廓的关系

```python
# 判断点是否在轮廓内
point = (100, 100)
result = cv2.pointPolygonTest(cnt, point, True)
# result > 0: 点在内部
# result = 0: 点在边上
# result < 0: 点在外部
# True 返回距离值, False 返回 -1/0/1

distance = cv2.pointPolygonTest(cnt, point, True)
if distance > 0:
    print(f"点在内部，距离边界 {distance:.1f}")
```

# 01 - 图像算术与逻辑运算

## 一、图像加法

```python
import cv2
import numpy as np

img1 = cv2.imread("image1.jpg")
img2 = cv2.imread("image2.jpg")

# ─── NumPy 加法（取模运算）───
# 超过 255 的值会取模：255 + 10 = 9 (265 % 256)
result_numpy = img1 + img2

# ─── OpenCV 加法（饱和运算）───
# 超过 255 的值会截断为 255：255 + 10 = 255
result_cv = cv2.add(img1, img2)

# 对比
print(f"NumPy 加法: {img1[0,0]} + {img2[0,0]} = {result_numpy[0,0]}")
print(f"OpenCV 加法: {img1[0,0]} + {img2[0,0]} = {result_cv[0,0]}")
```

```
区别示意：
NumPy:  200 + 100 = 44  (300 % 256)
OpenCV: 200 + 100 = 255 (饱和截断)

推荐使用 cv2.add()
```

## 二、图像加权融合

```python
# cv2.addWeighted(src1, alpha, src2, beta, gamma)
# result = alpha × src1 + beta × src2 + gamma
# 要求：两图尺寸和类型必须相同

result = cv2.addWeighted(img1, 0.7, img2, 0.3, 0)

# 渐变效果：动态调整权重
for alpha in np.linspace(0, 1, 100):
    beta = 1.0 - alpha
    blend = cv2.addWeighted(img1, alpha, img2, beta, 0)
    cv2.imshow("Blend", blend)
    if cv2.waitKey(30) == 27:
        break
```

## 三、亮度与对比度调整

```python
# 数学公式：g(x) = α × f(x) + β
# α: 对比度（gain），β: 亮度（bias）

def adjust_brightness_contrast(img, alpha=1.0, beta=0):
    """
    alpha: 对比度 (1.0 = 原始, >1 增强, <1 减弱)
    beta:  亮度 (-100 ~ +100)
    """
    # 转为 float 防止溢出
    adjusted = cv2.convertScaleAbs(img, alpha=alpha, beta=beta)
    return adjusted

# 增加 30 亮度，1.5 倍对比度
bright = adjust_brightness_contrast(img, alpha=1.5, beta=30)
# 降低亮度
dark = adjust_brightness_contrast(img, alpha=0.8, beta=-30)
```

## 五、Mask（掩模）操作

Mask 是位运算最重要的应用之一。

```python
# ─── 用 mask 提取特定区域 ───
img = cv2.imread("image.jpg")

# 创建圆形 mask
mask = np.zeros(img.shape[:2], dtype=np.uint8)
cv2.circle(mask, (img.shape[1]//2, img.shape[0]//2), 200, 255, -1)

# 只保留 mask 白色区域
masked = cv2.bitwise_and(img, img, mask=mask)

cv2.imshow("Masked", masked)
cv2.waitKey(0)
```

```python
# ─── 将 Logo 叠加到图像上 ───
background = cv2.imread("background.jpg")
logo = cv2.imread("logo.png")

# 假设 logo 有 alpha 通道
if logo.shape[2] == 4:
    # 分离 alpha 通道
    b, g, r, a = cv2.split(logo)
    overlay_color = cv2.merge([b, g, r])
    mask = cv2.merge([a, a, a])  # 3通道mask
else:
    # 简化：创建矩形 mask
    mask = np.zeros(logo.shape, dtype=np.uint8)
    mask[:,:] = 255

# 放置位置
rows, cols = logo.shape[:2]
roi = background[0:rows, 0:cols]

# 融合
bg = cv2.bitwise_and(roi, cv2.bitwise_not(mask))
fg = cv2.bitwise_and(overlay_color, mask)
combined = cv2.add(bg, fg)

background[0:rows, 0:cols] = combined
```

## 六、图像相减（运动检测）

```python
# 两帧相减，提取运动区域
frame1 = cv2.imread("frame1.jpg")
frame2 = cv2.imread("frame2.jpg")

# 转灰度
g1 = cv2.cvtColor(frame1, cv2.COLOR_BGR2GRAY)
g2 = cv2.cvtColor(frame2, cv2.COLOR_BGR2GRAY)

# 绝对差值
diff = cv2.absdiff(g1, g2)

# 二值化
_, thresh = cv2.threshold(diff, 25, 255, cv2.THRESH_BINARY)

# 膨胀
kernel = np.ones((5,5), np.uint8)
dilated = cv2.dilate(thresh, kernel, iterations=3)

# 查找运动轮廓
contours, _ = cv2.findContours(dilated, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

for cnt in contours:
    if cv2.contourArea(cnt) > 500:  # 过滤小区域
        x, y, w, h = cv2.boundingRect(cnt)
        cv2.rectangle(frame2, (x, y), (x+w, y+h), (0, 255, 0), 2)

cv2.imshow("Motion Detection", frame2)
cv2.waitKey(0)
```

## 小结

| 运算 | 函数 | 说明 |
|------|------|------|
| 加法 | `cv2.add()` | 饱和运算 |
| 加权 | `cv2.addWeighted()` | 图像融合 |
| 位运算 | `cv2.bitwise_and/or/xor/not()` | 掩模操作 |
| 差值 | `cv2.absdiff()` | 运动检测 |
| 亮度对比度 | `cv2.convertScaleAbs(img, alpha, beta)` | α对比度 β亮度 |

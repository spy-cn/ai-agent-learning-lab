# 04 - 增强现实（AR）

## 一、AR 基本原理

增强现实（AR）的核心是在现实世界的图像上叠加虚拟内容。

```
AR 流程：
1. 检测标记（ArUco/二维码/特征点）
2. 估计相机位姿（位置+方向）
3. 在 3D 空间中放置虚拟物体
4. 渲染到图像上
```

## 二、ArUco 标记检测

ArUco 是一种特殊的二维码，专用于 AR。

```python
import cv2
import numpy as np

# 加载预定义字典
dictionary = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)

# 也可以用其他字典
# DICT_4X4_50, DICT_5X5_100, DICT_6X6_250, DICT_7X7_1000
# DICT_ARUCO_ORIGINAL

# 创建检测器
parameters = cv2.aruco.DetectorParameters()
detector = cv2.aruco.ArucoDetector(dictionary, parameters)

# 检测标记
img = cv2.imread("aruco_image.jpg")
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

corners, ids, rejected = detector.detectMarkers(gray)

if ids is not None:
    print(f"检测到 {len(ids)} 个标记")
    print(f"ID: {ids.flatten()}")

    # 绘制检测结果
    cv2.aruco.drawDetectedMarkers(img, corners, ids)
```

## 三、生成 ArUco 标记

```python
# 生成单个标记
marker_id = 0

marker = cv2.aruco.generateImageMarker(dictionary, marker_id, marker_size)

cv2.imwrite(f"aruco_marker_{marker_id}.png", marker)
cv2.imshow("Marker", marker)
cv2.waitKey(0)

# 生成标记板（多个标记）
# 4x4 板，每格 40mm，标记间间距 5mm
board = cv2.aruco.GridBoard_create(
    markersX=4, markersY=4,
    markerLength=0.04,  # 4cm
    markerSeparation=0.01,  # 1cm
    dictionary=dictionary
)

# 绘制板
board_img = board.generateImage((800, 800))
cv2.imwrite("aruco_board.png", board_img)
```

## 四、位姿估计

```python
# 需要相机标定参数
camera_matrix = np.array([[800, 0, 320], [0, 800, 240], [0, 0, 1]], dtype=np.float32)
dist_coeffs = np.zeros(5)

# 检测标记并估计位姿
corners, ids, _ = detector.detectMarkers(gray)

if ids is not None:
    # 估计每个标记的位姿（旋转向量 + 平移向量）
    rvecs, tvecs, _ = cv2.aruco.estimatePoseSingleMarkers(
        corners, 0.05,  # 标记实际尺寸 5cm
        camera_matrix, dist_coeffs
    )

    for i in range(len(ids)):
        # 绘制坐标轴
        cv2.aruco.drawAxis(img, camera_matrix, dist_coeffs,
                          rvecs[i], tvecs[i], 0.1)  # 轴长度 10cm

        # 位姿信息
        print(f"Marker {ids[i][0]}:")
        print(f"  旋转: {rvecs[i][0]}")
        print(f"  平移: {tvecs[i][0]}")
```

## 五、在标记上叠加 3D 物体

```python
def draw_cube(img, rvec, tvec, camera_matrix, dist_coeffs, size=0.05):
    """在标记上画 3D 立方体"""
    # 定义立方体的 3D 点（底部在标记平面上）
    half = size / 2
    height = size  # 立方体高度

    # 8 个顶点
    object_points = np.array([
        [-half, -half, 0],    # 底部4个点
        [ half, -half, 0],
        [ half,  half, 0],
        [-half,  half, 0],
        [-half, -half, -height],  # 顶部4个点
        [ half, -half, -height],
        [ half,  half, -height],
        [-half,  half, -height],
    ], dtype=np.float32)

    # 投影到图像
    img_points, _ = cv2.projectPoints(object_points, rvec, tvec,
                                       camera_matrix, dist_coeffs)
    img_points = img_points.astype(int)

    # 绘制立方体
    # 底部（红色）
    cv2.polylines(img, [img_points[:4].reshape(-1, 2)], True, (0, 0, 255), 3)
    # 顶部（绿色）
    cv2.polylines(img, [img_points[4:].reshape(-1, 2)], True, (0, 255, 0), 3)
    # 竖直边
    for i in range(4):
        cv2.line(img, tuple(img_points[i][0]), tuple(img_points[i+4][0]),
                (255, 0, 0), 3)

# 使用
for i in range(len(ids)):
    draw_cube(img, rvecs[i], tvecs[i], camera_matrix, dist_coeffs)
```

## 六、实时 AR 应用

```python
def ar_application():
    """实时 AR：在标记上叠加立方体"""
    cap = cv2.VideoCapture(0)

    # 相机标定参数（需要先标定）
    camera_matrix = np.array([[800, 0, 320], [0, 800, 240], [0, 0, 1]], dtype=np.float32)
    dist_coeffs = np.zeros(5)

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        corners, ids, _ = detector.detectMarkers(gray)

        if ids is not None:
            rvecs, tvecs, _ = cv2.aruco.estimatePoseSingleMarkers(
                corners, 0.05, camera_matrix, dist_coeffs)

            for i in range(len(ids)):
                draw_cube(frame, rvecs[i], tvecs[i], camera_matrix, dist_coeffs)

        cv2.imshow("AR", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()
```

## 七、无标记 AR（基于特征点）

```python
# 不使用 ArUco，而是基于图像特征点实现 AR
# 适合在自然场景中叠加虚拟物体

def feature_based_ar(target_img_path):
    """基于特征点的 AR"""
    target = cv2.imread(target_img_path, 0)
    sift = cv2.SIFT_create()
    kp1, des1 = sift.detectAndCompute(target, None)

    # FLANN 匹配器
    flann = cv2.FlannBasedMatcher(
        dict(algorithm=1, trees=5), dict(checks=50))

    cap = cv2.VideoCapture(0)

    while True:
        ret, frame = cap.read()
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        kp2, des2 = sift.detectAndCompute(gray, None)

        matches = flann.knnMatch(des1, des2, k=2)
        good = [m for m, n in matches if m.distance < 0.7 * n.distance]

        if len(good) > 10:
            src_pts = np.float32([kp1[m.queryIdx].pt for m in good]).reshape(-1, 1, 2)
            dst_pts = np.float32([kp2[m.trainIdx].pt for m in good]).reshape(-1, 1, 2)

            H, mask = cv2.findHomography(src_pts, dst_pts, cv2.RANSAC, 5.0)

            if H is not None:
                # 在目标区域上叠加图像
                h, w = target.shape
                corners = np.float32([[0, 0], [w, 0], [w, h], [0, h]]).reshape(-1, 1, 2)
                transformed = cv2.perspectiveTransform(corners, H)

                # 绘制边界
                cv2.polylines(frame, [np.int32(transformed)], True, (0, 255, 0), 3)

                # 叠加内容（如文字）
                center = transformed.mean(axis=0).astype(int)[0]
                cv2.putText(frame, "AR Object", tuple(center),
                           cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 0, 255), 2)

        cv2.imshow("Feature AR", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()
```

# 02 - 视频与多媒体处理

## 一、视频读写基础

### 读取视频

```python
import cv2

cap = cv2.VideoCapture("video.mp4")

# 检查是否打开
if not cap.isOpened():
    print("无法打开视频")
    exit()

# 获取视频属性
fps = cap.get(cv2.CAP_PROP_FPS)
width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
duration = total_frames / fps

print(f"分辨率: {width}×{height}")
print(f"FPS: {fps}")
print(f"总帧数: {total_frames}")
print(f"时长: {duration:.1f}秒")

while True:
    ret, frame = cap.read()
    if not ret:
        break
    # 处理 frame...
    cv2.imshow("Video", frame)
    if cv2.waitKey(int(1000/fps)) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

### 写入视频

```python
# 定义编码格式
fourcc = cv2.VideoWriter_fourcc(*'mp4v')  # 或 'XVID', 'avc1'

out = cv2.VideoWriter(
    "output.mp4",
    fourcc,
    fps,           # 帧率
    (width, height), # 尺寸
    isColor=True    # 是否彩色
)

while True:
    ret, frame = cap.read()
    if not ret:
        break

    # 处理
    processed = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    processed = cv2.cvtColor(processed, cv2.COLOR_GRAY2BGR)  # 转回3通道

    out.write(processed)

cap.release()
out.release()  # 重要：必须释放
```

### 编码格式对照

| FourCC | 格式 | 扩展名 |
|--------|------|--------|
| `mp4v` | MPEG-4 | .mp4 |
| `XVID` | MPEG-4 | .avi |
| `avc1` | H.264 | .mp4 |
| `MJPG` | Motion JPEG | .avi |
| `X264` | H.264 | .avi |

def sample_frames(video_path, num_samples=10):
    """均匀采样视频帧"""
    cap = cv2.VideoCapture(video_path)
    total = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))

    frames = []
    indices = np.linspace(0, total - 1, num_samples, dtype=int)

    for idx in indices:
        cap.set(cv2.CAP_PROP_POS_FRAMES, idx)
        ret, frame = cap.read()
        if ret:
            frames.append(frame)

    cap.release()
    return frames

## 五、实时处理优化

```python
def process_video_realtime(video_path, process_func, target_fps=30):
    """实时视频处理"""
    cap = cv2.VideoCapture(video_path)
    original_fps = cap.get(cv2.CAP_PROP_FPS)

    # 计算跳帧数
    skip = max(1, int(original_fps / target_fps))

    frame_count = 0

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        frame_count += 1

        # 跳帧处理
        if frame_count % skip != 0:
            continue

        # 处理
        processed = process_func(frame)

        cv2.imshow("Processed", processed)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()
```

## 六、创建延时摄影

```python
def create_timelapse(input_video, output_video, speed_factor=60):
    """创建延时摄影"""
    cap = cv2.VideoCapture(input_video)
    fps = cap.get(cv2.CAP_PROP_FPS)
    width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
    height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))

    # 输出帧率
    output_fps = fps * speed_factor

    fourcc = cv2.VideoWriter_fourcc(*'mp4v')
    out = cv2.VideoWriter(output_video, fourcc, output_fps, (width, height))

    frame_interval = int(speed_factor)

    frame_count = 0
    while True:
        ret, frame = cap.read()
        if not ret:
            break

        if frame_count % frame_interval == 0:
            out.write(frame)

        frame_count += 1

    cap.release()
    out.release()
    print(f"处理 {frame_count} 帧，输出 {frame_count // frame_interval} 帧")
```

## 七、视频拼接

```python
def process_multiple_videos(video_paths, output_path):
    """拼接多个视频"""
    writer = None

    for video_path in video_paths:
        cap = cv2.VideoCapture(video_path)

        if writer is None:
            fps = cap.get(cv2.CAP_PROP_FPS)
            width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
            height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
            fourcc = cv2.VideoWriter_fourcc(*'mp4v')
            writer = cv2.VideoWriter(output_path, fourcc, fps, (width, height))

        while True:
            ret, frame = cap.read()
            if not ret:
                break
            writer.write(frame)

        cap.release()

    writer.release()
```

| 缓冲 | `CAP_PROP_BUFFERSIZE` | 帧数 |

# 03 - 全景图像拼接

## 一、全景拼接原理

```
全景拼接流程：

1. 特征提取 → SIFT/ORB
2. 特征匹配 → BFMatcher/FLANN
3. 计算单应矩阵 → findHomography (RANSAC)
4. 透视变换 → warpPerspective
5. 图像融合 → 多频段融合 / 简单拼接
```

```
输入：两张有重叠的图片       输出：全景图
┌──────────┐               ┌──────────────────┐
│   A      │               │     全景          │
│     ┌────┼────┐          │                  │
│     │重叠│  B │   →      │                  │
│     └────┼────┘          │                  │
└──────────┘               └──────────────────┘
```

## 二、手动拼接（理解原理）

```python
import cv2
import numpy as np

img1 = cv2.imread("left.jpg")
img2 = cv2.imread("right.jpg")

# Step 1: 提取 SIFT 特征
sift = cv2.SIFT_create()
gray1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
gray2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)

kp1, des1 = sift.detectAndCompute(gray1, None)
kp2, des2 = sift.detectAndCompute(gray2, None)

# Step 2: 特征匹配
bf = cv2.BFMatcher()
matches = bf.knnMatch(des1, des2, k=2)

# 比率测试
good = []
for m, n in matches:
    if m.distance < 0.75 * n.distance:
        good.append(m)

print(f"好匹配数: {len(good)}")

if len(good) < 4:
    print("匹配点不足")
    exit()

# Step 3: 计算单应矩阵
src_pts = np.float32([kp1[m.queryIdx].pt for m in good]).reshape(-1, 1, 2)
dst_pts = np.float32([kp2[m.trainIdx].pt for m in good]).reshape(-1, 1, 2)

H, mask = cv2.findHomography(src_pts, dst_pts, cv2.RANSAC, 5.0)
print(f"内点数: {sum(mask.ravel())}")

# Step 4: 透视变换并拼接
h1, w1 = img1.shape[:2]
h2, w2 = img2.shape[:2]

# 计算变换后画布大小
corners1 = np.float32([[0, 0], [w1, 0], [w1, h1], [0, h1]]).reshape(-1, 1, 2)
corners2 = cv2.perspectiveTransform(corners1, H)
all_corners = np.concatenate([corners1, corners2], axis=0)

[x_min, y_min] = np.int32(all_corners.min(axis=0).ravel()) - 0.5
[x_max, y_max] = np.int32(all_corners.max(axis=0).ravel()) + 0.5

# 平移矩阵
translation_dist = [-x_min, -y_min]
H_translation = np.array([[1, 0, translation_dist[0]],
                          [0, 1, translation_dist[1]],
                          [0, 0, 1]])

# 变换 img1
result = cv2.warpPerspective(img1, H_translation @ H, (x_max - x_min, y_max - y_min))

# 将 img2 放入正确位置
result[translation_dist[1]:h2+translation_dist[1],
       translation_dist[0]:w2+translation_dist[0]] = img2

cv2.imshow("Panorama", result)
cv2.waitKey(0)
```

## 三、使用 Stitcher 类（推荐）

OpenCV 提供了高级的 `Stitcher` 类，自动完成所有步骤。

```python
import cv2

# 加载图像
images = []
for path in ["img1.jpg", "img2.jpg", "img3.jpg"]:
    img = cv2.imread(path)
    images.append(img)

# 创建 Stitcher
stitcher = cv2.Stitcher_create()

# 拼接模式
# PANORAMA_SCANS: 扫描模式（精度高）
# PANORAMA_ROTATION: 旋转模式（快速）

# 执行拼接
status, panorama = stitcher.stitch(images)

if status == cv2.Stitcher_OK:
    cv2.imshow("Panorama", panorama)
    cv2.waitKey(0)
else:
    print(f"拼接失败，错误码: {status}")
```

### Stitcher 状态码

| 状态码 | 含义 |
|--------|------|
| `Stitcher_OK` | 成功 |
| `ERR_NEED_MORE_IMGS` | 需要更多图片 |
| `ERR_HOMOGRAPHY_EST_FAIL` | 单应矩阵估计失败 |
| `ERR_CAMERA_PARAMS_ADJUST_FAIL` | 相机参数调整失败 |

## 四、多张图片全景

```python
def create_panorama(image_paths, output_path="panorama.jpg"):
    """多张图片拼接全景"""
    images = []
    for path in image_paths:
        img = cv2.imread(path)
        # 缩小图片以提高速度
        h, w = img.shape[:2]
        if max(h, w) > 1200:
            scale = 1200 / max(h, w)
            img = cv2.resize(img, (int(w*scale), int(h*scale)))
        images.append(img)

    stitcher = cv2.Stitcher_create()
    stitcher.setRegistrationResol(0.6)  # 降低分辨率提高速度
    stitcher.setSeamEstimationResol(0.1)
    stitcher.setCompositingResol(-1)  # 原分辨率

    status, panorama = stitcher.stitch(images)

    if status == cv2.Stitcher_OK:
        cv2.imwrite(output_path, panorama)
        print(f"全景图保存到: {output_path}")
        print(f"尺寸: {panorama.shape[1]}×{panorama.shape[0]}")
        return panorama
    else:
        print(f"拼接失败: {status}")
        return None

# 使用
paths = sorted(glob.glob("panorama_images/*.jpg"))
panorama = create_panorama(paths)
```

## 五、实时全景（扫描模式）

```python
def realtime_panorama():
    """摄像头实时全景扫描"""
    cap = cv2.VideoCapture(0)

    # 初始化
    stitcher = cv2.Stitcher_create()
    captured_images = []

    print("操作：")
    print("  SPACE: 拍照")
    print("  s: 拼接全景")
    print("  q: 退出")

    while True:
        ret, frame = cap.read()
        cv2.imshow("Camera", frame)

        key = cv2.waitKey(1) & 0xFF
        if key == ord(' '):
            captured_images.append(frame.copy())
            print(f"已拍摄 {len(captured_images)} 张")
        elif key == ord('s') and len(captured_images) >= 2:
            status, panorama = stitcher.stitch(captured_images)
            if status == cv2.Stitcher_OK:
                cv2.imshow("Panorama", panorama)
            else:
                print(f"拼接失败: {status}")
        elif key == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()
```

## 六、拼接质量优化

```python
# 1. 确保有足够的重叠区域（建议 25-50%）
# 2. 使用高分辨率图片
# 3. 确保良好的光照条件
# 4. 保持相机水平移动

# 检查重叠区域
def check_overlap(img1, img2):
    """通过特征匹配检查重叠"""
    sift = cv2.SIFT_create()
    kp1, des1 = sift.detectAndCompute(cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY), None)
    kp2, des2 = sift.detectAndCompute(cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY), None)

    bf = cv2.BFMatcher()
    matches = bf.knnMatch(des1, des2, k=2)
    good = [m for m, n in matches if m.distance < 0.75 * n.distance]

    overlap_ratio = len(good) / min(len(kp1), len(kp2))
    print(f"匹配率: {overlap_ratio:.2%}")

    return overlap_ratio > 0.1  # 至少 10% 匹配
```

## 一、性能优化概览

```
优化层次：
```

## 二、UMat（OpenCL 加速）

```python
import cv2
import time

img = cv2.imread("large_image.jpg")

# ─── 普通 NumPy 处理 ───
start = time.time()
result1 = cv2.GaussianBlur(img, (51, 51), 0)
print(f"CPU: {time.time()-start:.4f}s")

# ─── UMat 处理（自动使用 OpenCL）───
start = time.time()
umat = cv2.UMat(img)
result2 = cv2.GaussianBlur(umat, (51, 51), 0)
result2 = result2.get()  # 转回 NumPy
print(f"UMat: {time.time()-start:.4f}s")

# 检查 OpenCL 是否可用
print(f"OpenCL 可用: {cv2.ocl.haveOpenCL()}")
cv2.ocl.setUseOpenCL(True)
```

## 三、避免常见性能陷阱

### 陷阱1：不必要的复制

```python
# ✗ 错误：不必要的 copy
img_copy = img.copy()  # 如果不修改，不需要复制
result = process(img_copy)

# ✓ 正确：直接使用
result = process(img)

# 需要复制的情况：函数会修改输入
original = img.copy()
process_inplace(img)  # 修改 img
```

### 陷阱2：逐像素操作

```python
# ✗ 慢：Python 循环
for y in range(h):
    for x in range(w):
        img[y, x] = 255 - img[y, x]

# ✓ 快：向量化
img = 255 - img

# ✓ 最快：OpenCV 函数
img = cv2.bitwise_not(img)
```

### 陷阱3：颜色空间转换

```python
# ✗ 不必要：读彩色但只用灰度
img = cv2.imread("image.jpg")  # 默认彩色
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# ✓ 直接读灰度
gray = cv2.imread("image.jpg", 0)  # 直接灰度
```

## 四、查找表（LUT）优化

```python
# ✗ 慢：逐像素 gamma 校正
gamma = 2.2
result = np.array([255 * (x/255)**gamma for x in range(256)], dtype=np.uint8)
for y in range(h):
    for x in range(w):
        img[y, x] = result[img[y, x]]

# ✓ 快：使用 LUT
table = np.array([255 * (i/255)**gamma for i in range(256)], dtype=np.uint8)
result = cv2.LUT(img, table)
```

## 六、多线程处理

```python
import concurrent.futures

def process_single_image(img_path):
    img = cv2.imread(img_path)
    result = cv2.GaussianBlur(img, (5, 5), 0)
    return result

# 多进程处理多张图像
image_paths = ["img1.jpg", "img2.jpg", "img3.jpg", "img4.jpg"]

with concurrent.futures.ThreadPoolExecutor(max_workers=4) as executor:
    results = list(executor.map(process_single_image, image_paths))
```

## 七、内存优化

```python
# ─── 预分配数组 ───
```

## 九、降采样策略

```python
# 金字塔处理：先在小图上处理，再精化
def pyramid_process(img, levels=3):
    """金字塔层次处理"""
    current = img.copy()
    results = []

    for _ in range(levels):
        small = cv2.pyrDown(current)
        # 在小图上做处理
        result = process(small)
        results.append(result)
        current = small

    # 从粗到细精化
    for i in range(len(results)-2, -1, -1):
        results[i] = refine(results[i], results[i+1])

    return results[0]
```

## 十、性能测量工具

```python
# ─── 使用 getTickCount ───
def measure_time(func, *args, **kwargs):
    """测量函数执行时间"""
    start = cv2.getTickCount()
    result = func(*args, **kwargs)
    end = cv2.getTickCount()
    elapsed = (end - start) / cv2.getTickFrequency()
    print(f"{func.__name__}: {elapsed*1000:.2f}ms")
    return result

# 使用
result = measure_time(cv2.GaussianBlur, img, (5, 5), 0)

# ─── 使用 TickMeter ───
tm = cv2.TickMeter()
tm.start()
# ... 你的代码 ...
tm.stop()
print(f"耗时: {tm.getTimeMilli():.2f}ms")
print(f"FPS: {tm.getFPS():.1f}")
```

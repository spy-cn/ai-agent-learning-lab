# 文本数据增强

在 NLP 实践中，标注数据往往有限且昂贵。文本数据增强（Text Data Augmentation）通过在原文基础上生成变体，在不收集新数据的前提下扩充训练集，有效缓解过拟合和类别不平衡问题。

```
训练数据不足时的困境:

  ┌───────────────────────────────────────────────┐
  │  训练集: 100 条 (类别不均衡)                  │
  │  模型: 在训练集上准确率 99%                   │
  │        在测试集上准确率 65%  ← 过拟合         │
  └───────────────────────────────────────────────┘
                           │
                           ▼  数据增强
  ┌───────────────────────────────────────────────┐
  │  增强后: 1000 条 (类别均衡)                   │
  │  模型: 在训练集上准确率 95%                   │
  │        在测试集上准确率 82%  ← 泛化更好       │
  └───────────────────────────────────────────────┘
```

## 1. 为什么需要数据增强

### 数据增强的动机

```
数据增强解决三大问题:

  1. 数据不平衡 (Class Imbalance)
     ┌────────────────────────┐
     │ 正面评论: 10000 条     │  ← 多数类
     │ 负面评论:   200 条     │  ← 少数类
     │                        │
     │ → 增强少数类到 2000 条 │  ← 平衡
     └────────────────────────┘

  2. 小样本 (Few-shot / Low-resource)
     ┌────────────────────────┐
     │ 每个类别只有 10~50 条  │
     │ → 增强到 500~1000 条   │
     └────────────────────────┘

  3. 鲁棒性 (Robustness)
     ┌────────────────────────┐
     │ 原文: "这部电影很好看" │
     │ 增强: "这部影片很不错" │  ← 同义不同形
     │       "这电影挺好看的" │
     │ → 模型学会多样的表达   │
     └────────────────────────┘
```

## 2. 同义词替换

同义词替换是最简单直接的增强方法。

```
同义词替换原理:

  原文: "这部电影非常好看"
  分词: 这部 / 电影 / 非常 / 好看

  查找同义词:
    电影 → 影片
    非常 → 十分 / 特别 / 相当
    好看 → 精彩 / 出色 / 优秀

  生成变体:
    "这部影片十分精彩"
    "这部电影特别出色"
    "这部影片相当优秀"
```

### 英文：基于 WordNet

```python
# === NLTK WordNet 同义词 ===
# import nltk
# nltk.download('wordnet')
# nltk.download('omw-1.4')

from nltk.corpus import wordnet

def get_synonyms_wordnet(word, lang='eng'):
    """使用 WordNet 获取同义词"""
    synonyms = set()
    for syn in wordnet.synsets(word, lang=lang):
        for lemma in syn.lemmas(lang=lang):
            name = lemma.name()
            # 排除原词 (要 lower + 替换下划线)
            if name.lower().replace('_', ' ') != word.lower():
                synonyms.add(name.lower().replace('_', ' '))
    return list(synonyms)


def synonym_replacement_wordnet(text, n=2):
    """
    用 WordNet 做同义词替换
    
    Args:
        text: 英文文本
        n: 替换的词数
    """
    import nltk
    words = nltk.word_tokenize(text)
    # 只替换非停用词
    from nltk.corpus import stopwords
    stop_words = set(stopwords.words('english'))
    
    candidates = [(i, w) for i, w in enumerate(words)
                  if w.lower() not in stop_words and w.isalpha()]
    
    import random
    random.shuffle(candidates)
    
    replaced = 0
    for idx, word in candidates:
        synonyms = get_synonyms_wordnet(word)
        if synonyms:
            # 随机选一个同义词
            new_word = random.choice(synonyms)
            words[idx] = new_word
            replaced += 1
            if replaced >= n:
                break
    
    return ' '.join(words)


# 测试
text_en = "The movie is very interesting and exciting"
augmented = synonym_replacement_wordnet(text_en, n=2)
print(f"原文:   {text_en}")
print(f"增强后: {augmented}")
```

## 3. EDA (Easy Data Augmentation)

EDA 是一组简单但有效的增强操作。

```
EDA 四种基本操作:

  1. 同义词替换 (Synonym Replacement, SR)
     "这部电影很好看"
     → "这部影片很好看"

  2. 随机插入 (Random Insertion, RI)
     "这部电影很好看"
     → "这部电影真的很好看"

  3. 随机交换 (Random Swap, RS)
     "这部电影很好看"
     → "很好看部电影这"

  4. 随机删除 (Random Deletion, RD)
     "这部电影很好看"
     → "电影很好看"
```

```python
import random
import re


class EDA:
    """Easy Data Augmentation"""
    
    def __init__(self, language='zh'):
        self.language = language
    
    # === 1. 同义词替换 ===
    def synonym_replacement(self, words, n):
        """替换 n 个非停用词为同义词"""
        new_words = words.copy()
        random.shuffle(new_words)
        
        replaced = 0
        for word in new_words:
            synonyms = self._get_synonyms(word)
            if synonyms:
                # 在原句中替换
                synonym = random.choice(synonyms)
                words = [synonym if w == word else w for w in words]
                replaced += 1
                if replaced >= n:
                    break
        return words
    
    # === 2. 随机插入 ===
    def random_insertion(self, words, n):
        """随机插入 n 个同义词"""
        new_words = words.copy()
        for _ in range(n):
            self._add_word(new_words)
        return new_words
    
    def _add_word(self, new_words):
        synonyms = []
        # 从非停用词中选一个
        candidates = [w for w in new_words if len(w) > 1]
        if not candidates:
            return
        random.shuffle(candidates)
        for word in candidates:
            synonyms = self._get_synonyms(word)
            if synonyms:
                break
        if synonyms:
            synonym = random.choice(synonyms)
            insert_pos = random.randint(0, len(new_words))
            new_words.insert(insert_pos, synonym)
    
    # === 3. 随机交换 ===
    def random_swap(self, words, n):
        """随机交换 n 对词的位置"""
        new_words = words.copy()
        for _ in range(n):
            if len(new_words) >= 2:
                idx1, idx2 = random.sample(range(len(new_words)), 2)
                new_words[idx1], new_words[idx2] = new_words[idx2], new_words[idx1]
        return new_words
    
    # === 4. 随机删除 ===
    def random_deletion(self, words, p):
        """
        以概率 p 删除每个词
        至少保留一个词
        """
        if len(words) == 1:
            return words
        
        new_words = []
        for word in words:
            if random.random() > p:
                new_words.append(word)
        
        # 如果全删了，随机保留一个
        if len(new_words) == 0:
            return [random.choice(words)]
        
        return new_words
    
    def _get_synonyms(self, word):
        """获取同义词"""
        try:
            import synonyms
            result = synonyms.nearby(word)
            if result and len(result) >= 2:
                words_list, scores = result
                return [w for w, s in zip(words_list, scores)
                        if w != word and s > 0.7][:3]
        except ImportError:
            pass
        
        CHINESE_SYNONYMS = {
            '美丽': ['漂亮', '好看', '迷人'],
            '快乐': ['开心', '高兴', '愉快'],
            '重要': ['关键', '核心', '重大'],
            '非常': ['十分', '特别', '相当'],
            '好看': ['精彩', '出色', '优秀'],
        }
        return CHINESE_SYNONYMS.get(word, [])
    
    def augment(self, text, alpha_sr=0.1, alpha_ri=0.1,
                alpha_rs=0.1, p_rd=0.1, num_aug=4):
        """
        EDA 增强
        
        Args:
            text: 输入文本
            alpha_sr: 同义词替换比例
            alpha_ri: 随机插入比例
            alpha_rs: 随机交换比例
            p_rd: 随机删除概率
            num_aug: 生成变体数
        Returns:
            augmented_texts: 增强文本列表
        """
        if self.language == 'zh':
            import jieba
            words = jieba.lcut(text)
        else:
            words = text.split()
        
        if len(words) == 0:
            return [text]
        
        n_sr = max(1, int(alpha_sr * len(words)))
        n_ri = max(1, int(alpha_ri * len(words)))
        n_rs = max(1, int(alpha_rs * len(words)))
        
        # 生成变体
        augmented = []
        ops_per_aug = num_aug // 4 if num_aug >= 4 else 1
        
        for _ in range(ops_per_aug):
            # 同义词替换
            aug_words = self.synonym_replacement(words, n_sr)
            augmented.append(self._join(aug_words))
        
        for _ in range(ops_per_aug):
            # 随机插入
            aug_words = self.random_insertion(words, n_ri)
            augmented.append(self._join(aug_words))
        
        for _ in range(ops_per_aug):
            # 随机交换
            aug_words = self.random_swap(words, n_rs)
            augmented.append(self._join(aug_words))
        
        for _ in range(ops_per_aug):
            # 随机删除
            aug_words = self.random_deletion(words, p_rd)
            augmented.append(self._join(aug_words))
        
        # 去重
        augmented = list(dict.fromkeys(augmented))
        return augmented[:num_aug]
    
    def _join(self, words):
        """连接词列表"""
        if self.language == 'zh':
            return ''.join(words)
        else:
            return ' '.join(words)


# === 测试 EDA ===
eda = EDA(language='zh')
text = "自然语言处理是人工智能的重要分支"

print(f"原文: {text}\n")
print("EDA 增强结果:")
augmented_texts = eda.augment(text, num_aug=8)
for i, aug_text in enumerate(augmented_texts, 1):
    print(f"  变体{i}: {aug_text}")
```

### EDA 的推荐参数

```
EDA 推荐参数设置:

┌──────────────────┬──────────────────────────────┐
│  参数            │  推荐值                      │
├──────────────────┼──────────────────────────────┤
│ alpha_sr         │ 0.1 (替换10%的词)            │
│ alpha_ri         │ 0.1 (插入10%的词)            │
│ alpha_rs         │ 0.1 (交换10%的词对)          │

└──────────────────┴──────────────────────────────┘

  - 数据量 < 500:    增强到 4~8 倍
  - 数据量 500~5000: 增强到 2~4 倍
  - 数据量 > 5000:   增强到 1~2 倍或不增强
  
  数据量越少，增强效果越明显
```

## 4. 回译增强 (Back Translation)

利用翻译模型做"往返翻译"，生成同义不同形的文本。

```
回译原理:

  原文 (中文):
    "这部电影非常精彩"
         │
         │  翻译到英文
         ▼
  英文翻译:
    "This movie is very exciting"
         │
         │  翻译回中文
         ▼
  回译结果 (中文):
    "这部影片十分令人兴奋"
    
  对比:
    原文: 这部电影非常精彩
    回译: 这部影片十分令���兴奋
    → 同义但用词不同, 可作为增强样本

多语言回译 (经过多个中间语言):
  
  中文 → 英文 → 中文
  中文 → 日文 → 中文
  中文 → 法文 → 中文
  → 每条路径产生不同的变体
```

```python
# === 使用翻译 API 做回译 ===

# 方法1: 使用 translators 库
# pip install translators
try:
    import translators as ts
    
    def back_translate(text, mid_lang='en', target_lang='zh'):
        """
        回译增强
        
        Args:
            text: 原文
            mid_lang: 中间语言
            target_lang: 目标语言
        """
        # 翻译到中间语言
        translated = ts.translate_text(text, translator='google',
                                       from_language=target_lang,
                                       to_language=mid_lang)
        # 翻译回目标语言
        back_translated = ts.translate_text(translated, translator='google',
                                            from_language=mid_lang,
                                            to_language=target_lang)
        return back_translated

    # 多路径回译
    print("\n多路径回译:")
    for mid_lang in ['en', 'ja', 'fr', 'de']:
        try:
            aug = back_translate(text, mid_lang=mid_lang)
            print(f"  [{mid_lang}] {aug}")
        except:
            pass

except ImportError:
```

## 5. 基于上下文的替换 (BERT Mask)

利用 BERT 的 Masked Language Model 能力，根据上下文生成合理的替换词。

```
BERT Mask 替换原理:

  原文: "自然语言处理是人工智能的分支"
  
  步骤1: 随机选择位置并 Mask
         "自然语言处理是 [MASK] 的分支"
  
  步骤2: BERT 预测 top-k 替换词
         [MASK] → 重要 (0.35)
                → 核心 (0.20)
                → 关键 (0.15)
                → 重要 (0.10)
  
  步骤3: 选一个替换词生成变体
         "自然语言处理是重要的分支"
         "自然语言处理是核心的分支"
  
  优势: 替换词与上下文高度相关
```

```python
# === BERT Mask 替换增强 ===
# pip install transformers torch

from transformers import (
    AutoTokenizer,
    AutoModelForMaskedLM,
)
import torch
import random
import jieba


class BERTAugmentor:
    """基于 BERT MLM 的上下文替换增强"""
    
    def __init__(self, model_name='bert-base-chinese', device='cpu'):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForMaskedLM.from_pretrained(model_name).to(device)
        self.model.eval()
        self.device = device
    
    def get_mask_predictions(self, text, mask_position, top_k=5):
        """
        获取指定位置的替换候选词
        
        Args:
            text: 完整文本
            mask_position: 要替换的字符位置
            top_k: 返回前k个候选
        """
        # 将目标字符替换为 [MASK]
        chars = list(text)
        original_char = chars[mask_position]
        chars[mask_position] = self.tokenizer.mask_token
        masked_text = ''.join(chars)
        
        # 编码
        inputs = self.tokenizer(masked_text, return_tensors='pt').to(self.device)
        mask_token_index = torch.where(
            inputs['input_ids'][0] == self.tokenizer.mask_token_id
        )[0]
        
        # 预测
        with torch.no_grad():
            outputs = self.model(**inputs)
        
        # 获取 top-k 预测
        mask_logits = outputs.logits[0, mask_token_index, :]
        top_tokens = torch.topk(mask_logits, k=top_k, dim=-1)
        
        candidates = []
        for score, token_id in zip(top_tokens.values[0], top_tokens.indices[0]):
            token = self.tokenizer.convert_ids_to_tokens([token_id])[0]
            # 跳过特殊 token
            if token not in ['[PAD]', '[CLS]', '[SEP]', '[MASK]', original_char]:
                if token.startswith('##'):
                    token = token[2:]
                candidates.append((token, score.item()))
        
        return candidates
    
    def augment(self, text, replace_ratio=0.15, top_k=5):
        """
        BERT 上下文增强
        
        Args:
            text: 输入文本
            replace_ratio: 替换比例
            top_k: 每个位置候选数
        """
        chars = list(text)
        n_chars = len(chars)
        n_replace = max(1, int(n_chars * replace_ratio))
        
        # 随机选择要替换的位置
        positions = list(range(n_chars))
        # 保留标点和空格
        positions = [p for p in positions if chars[p].isalnum()]
        random.shuffle(positions)
        
        new_chars = chars.copy()
        for pos in positions[:n_replace]:
            candidates = self.get_mask_predictions(text, pos, top_k=top_k)
            if candidates:
                # 随机选一个 (或选概率最高的)
                new_char = random.choice(candidates)[0]
                new_chars[pos] = new_char
        
        return ''.join(new_chars)
    
    def augment_batch(self, text, n_augments=4, replace_ratio=0.15):
        """生成多个增强变体"""
        return [self.augment(text, replace_ratio) for _ in range(n_augments)]
```

```python
    # 2. 用原始 + 增强数据训练
    combined_texts = original_texts + augmented_texts
    combined_labels = original_labels + augmented_labels
    
    model_aug = train_fn(combined_texts, combined_labels)
    metrics_aug = evaluate_fn(model_aug, test_texts, test_labels)
    
    # 3. 对比

    print(f"{'指标':<15} {'原始数据':<12} {'增强后':<12} {'变化'}")

        print(f"{metric:<15} {orig_val:<12.4f} {aug_val:<12.4f} "

    return {
        'original': metrics_orig,
```

```python
# === 增强质量检查 ===

def check_augmentation_quality(original, augmented):
    """
    检查增强样本质量
    
    Args:
        original: 原文列表
        augmented: 增强文本列表
    Returns:
        质量报告
    """
    report = {
        'total': len(augmented),
        'empty': 0,           # 空文本
        'duplicate': 0,       # 与原文完全相同
        'too_similar': 0,     # 与原文过于相似
        'too_different': 0,   # 与原文差异过大
        'valid': 0,           # 有效样本
    }
    
    for aug in augmented:
        if not aug or len(aug.strip()) == 0:
            report['empty'] += 1
            continue
        
        # 与每条原文比较
        max_sim = 0
        for orig in original:
            sim = _text_similarity(orig, aug)
            max_sim = max(max_sim, sim)
        
        if max_sim == 1.0:
            report['duplicate'] += 1
        elif max_sim > 0.95:
            report['too_similar'] += 1
        elif max_sim < 0.3:
            report['too_different'] += 1
        else:
            report['valid'] += 1
    
    valid_rate = report['valid'] / report['total'] if report['total'] > 0 else 0
    
    print("=== 增强质量报告 ===")
    print(f"总增强数:     {report['total']}")
    print(f"空文本:       {report['empty']}")
    print(f"完全重复:     {report['duplicate']}")
    print(f"过于相似:     {report['too_similar']}")
    print(f"差异过大:     {report['too_different']}")
    print(f"有效样本:     {report['valid']} ({valid_rate:.1%})")
    
    return report


def _text_similarity(text1, text2):
    """简单的字符级 Jaccard 相似度"""
    if not text1 or not text2:
        return 0
    set1 = set(text1)
    set2 = set(text2)
    intersection = set1 & set2
    union = set1 | set2
    return len(intersection) / len(union) if union else 0

    "天气不错适合出去玩",     # 差异过大
    "NLP技术非常关键",        # 好
]

check_augmentation_quality(originals, augmented_samples)
```

## 小结

| 要点 | 内容 |
|------|------|
| **增强目的** | 解决数据不平衡、小样本、提升鲁棒性 |
| **同义词替换** | WordNet(英文)/同义词词林(中文)，简单高效 |
| **EDA** | SR+RI+RS+RD 四种操作，无需额外资源 |
| **回译增强** | 双向翻译生成变体，质量高但速度慢 |
| **BERT Mask** | 上下文感知替换，替换词与语境高度相关 |
| **生成式增强** | LLM 改写/生成，质量最高但成本高 |
| **nlpaug** | 统一接口，支持 char/word/sentence 三级增强 |
| **效果评估** | 增强前后对比验证集，检查质量(重复/相似/差异) |
| **经验** | 数据越少收益越大，增强过度可能有害 |

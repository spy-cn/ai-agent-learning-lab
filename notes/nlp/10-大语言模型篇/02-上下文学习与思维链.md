# 上下文学习与思维链

大语言模型最令人惊奇的特性之一，是不需要更新任何参数，仅通过提示词就能学习新任务。这就是**上下文学习（In-Context Learning）**。而**思维链（Chain-of-Thought）** 则进一步解锁了LLM的复杂推理能力。本章将系统讲解ICL、CoT及其衍生技术。

## In-Context Learning (ICL)

传统的机器学习需要大量标注数据来"训练"模型参数。而LLM通过**上下文学习**，只需在输入中给出几个示例，就能完成新任务，无需任何梯度更新。

```
┌──────────────────────────────────────────────────────────────────┐
│              传统微调 vs 上下文学习(ICL)                          │
│                                                                  │
│  传统微调:                        上下文学习(ICL):               │
│  ┌──────────────┐                 ┌──────────────────────────┐   │
│  │  标注数据集   │                 │  输入: 几个示例 + 新输入  │   │
│  │  (1000+条)   │                 │                          │   │
│  └──────┬───────┘                 │  "正面: 今天很开心"      │   │
│         │                         │  "负面: 天气真糟"        │   │
│         ▼                         │  "正面: 太棒了"          │   │
│  ┌──────────────┐                 │  → "这个电影真好看"      │   │
│  │  梯度下降更新 │                 └──────────┬───────────────┘   │
│  │  模型参数     │                            │                  │
│  └──────┬───────┘                            │                  │
│         │                                    ▼                  │
│         ▼                          ┌──────────────────┐         │
│  ┌──────────────┐                  │  LLM (参数不变)  │         │
│  │  微调后模型   │                  │  直接生成答案    │         │
│  │  → 预测       │                  └──────────────────┘         │
│  └──────────────┘                                              │
│                                                                  │
│  优点: 效果上限高                优点: 无需训练，灵活快速        │
│  缺点: 每个任务都需重新训练       缺点: 受上下文窗口限制         │
└──────────────────────────────────────────────────────────────────┘
```

### Zero-shot / One-shot / Few-shot

根据提供给模型的示例数量，ICL分为三种模式：

```python
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

model_name = "Qwen/Qwen2.5-0.5B"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCausalLM.from_pretrained(
    model_name, torch_dtype=torch.float16, device_map="auto"
)

def llm_generate(prompt, max_new_tokens=80):
    """统一的LLM生成接口"""
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    with torch.no_grad():
        outputs = model.generate(
            **inputs, max_new_tokens=max_new_tokens,
            temperature=0.3, do_sample=True, top_p=0.9
        )
    return tokenizer.decode(outputs[0][inputs['input_ids'].shape[1]:], 
                           skip_special_tokens=True)

# ========== Zero-shot: 不给任何示例 ==========
zero_shot_prompt = """请对以下评论进行情感分类（正面/负面）。
评论：这个手机的电池续航很差。
情感："""
print("=== Zero-shot ===")
print(llm_generate(zero_shot_prompt))

# ========== One-shot: 给1个示例 ==========
one_shot_prompt = """请对评论进行情感分类。

示例：
评论：这家餐厅的菜很好吃。
情感：正面

现在请分类：
评论：这个手机的电池续航很差。
情感："""
print("\n=== One-shot ===")
print(llm_generate(one_shot_prompt))

# ========== Few-shot: 给多个示例 ==========
few_shot_prompt = """请对评论进行情感分类。

示例1：
评论：这家餐厅的菜很好吃。
情感：正面

示例2：
评论：快递太慢了，等了一周。
情感：负面

示例3：
评论：质量不错，物有所值。
情感：正面

现在请分类：
评论：这个手机的电池续航很差。
情感："""
print("\n=== Few-shot ===")
print(llm_generate(few_shot_prompt))
```

### ICL的工作原理

```
┌──────────────────────────────────────────────────────────────┐
│                   ICL 内部机制                               │
│                                                              │
│  输入: "正面: 开心  负面: 难过  正面: 愉快  测试: 悲伤"     │
│                                                              │
│  Token序列 → Transformer Layers                              │
│  ┌────┬────┬────┬────┬────┬────┬────┬────┬────┬────┐       │
│  │正  │面  │:   │开心│负面│:   │难过│... │测试│:   │       │
│  └─┬──┴─┬──┴─┬──┴─┬──┴────┴────┴────┴────┴────┴────┘       │
│    │    │    │                                                │
│    │    │    │    Attention层在此构建示例间关联               │
│    │    │    │    ↓                                           │
│    │    │    │  ┌─────────────────────────┐                  │
│    │    │    │  │ 隐式形成"任务定义"       │                  │
│    │    │    │  │ "A: B" 模式 = 分类任务   │                  │
│    │    │    │  └─────────────────────────┘                  │
│    │    │    │                                                │
│    ▼    ▼    ▼                                                │
│  最终token位置的隐状态 → 预测下一个token                      │
│  → "负面"                                                    │
│                                                              │
│  关键: 示例的格式和顺序显著影响效果                           │
└──────────────────────────────────────────────────────────────┘
```

```python
# 演示ICL示例选择对效果的影响
import json

# 准备示例池
example_pool = {
    "正面": ["今天天气真好", "这部电影太精彩了", "产品质量很棒", 
             "服务态度很好", "非常满意"],
    "负面": ["快递太慢了", "东西坏了", "客服态度差", 
             "完全不值这个价", "太失望了"]
}

def build_few_shot_prompt(test_text, num_shots=4, strategy="random"):
    """构建few-shot提示，支持不同示例选择策略"""
    examples = []
    for label, texts in example_pool.items():
        for t in texts:
            examples.append((t, label))
    
    import random
    if strategy == "random":
        selected = random.sample(examples, min(num_shots, len(examples)))
    elif strategy == "balanced":
        # 平衡正负样本
        pos = [e for e in examples if e[1] == "正面"]
        neg = [e for e in examples if e[1] == "负面"]
        n_each = num_shots // 2
        selected = random.sample(pos, n_each) + random.sample(neg, n_each)
    elif strategy == "similar":
        # 简化版：选择与测试文本相似的示例（实际中用embedding相似度）
        selected = random.sample(examples, min(num_shots, len(examples)))
    
    # 构建提示
    prompt = "请对评论进行情感分类。\n\n"
    for text, label in selected:
        prompt += f"评论：{text}\n情感：{label}\n\n"
    prompt += f"评论：{test_text}\n情感："
    return prompt

# 测试不同策略
test_cases = ["这个产品一般般", "超��推荐！", "不值这个价"]

for strategy in ["random", "balanced"]:
    print(f"\n{'='*40}")
    print(f"策略: {strategy}")
    for test in test_cases:
        prompt = build_few_shot_prompt(test, num_shots=4, strategy=strategy)
        result = llm_generate(prompt, max_new_tokens=10)
        print(f"  '{test}' → {result.strip()}")
```

## Chain-of-Thought (CoT)

思维链是解锁LLM推理能力的关键技术。Wei等人（2022）发现：**在few-shot示例中加入中间推理步骤，能大幅提升LLM在数学、逻辑等复杂推理任务上的表现。**

```
┌──────────────────────────────────────────────────────────────────┐
│             标准Prompt vs 思维链(CoT) Prompt                      │
│                                                                  │
│  标准Few-shot:                              CoT Few-shot:        │
│  Q: 罗杰有5个球，买了2盒，                   Q: 罗杰有5个球，    │
│     每盒6个。他有多少球？                          买了2盒，每    │
│  A: 17                                      盒6个。他有多少球？  │
│                                             A: 罗杰开始有5个球。 │
│  Q: 食堂有23个苹果，用了20个                  2盒每盒6个 = 12个。 │
│     午餐又买了6个。有多少苹果？              5 + 12 = 17个球。    │
│  A: 27                                           ↑ 推理步骤      │
│                                             Q: 食堂有23个苹果， │
│  ✗ 直接给答案，推理任务失败                       用了20个午餐    │
│                                               又买了6个。有    │
│                                               多少苹果？       │
│                                             A: 食堂有23个苹果。 │
│                                               用了20个：       │
│                                               23-20=3。        │
│                                               又买了6个：       │
│                                               3+6=9。          │
│                                                   ↑ 带推理      │
│                                             ✓ 推理正确          │
└──────────────────────────────────────────────────────────────────┘
```

### Zero-shot CoT

更惊人的是，仅需加一句魔法咒语 "Let's think step by step"，就能激发LLM的推理能力：

```python
# Zero-shot CoT 的神奇效果
standard_prompt = """Q: 一个班级有32名学生，其中女生比男生多4人。女生有多少人？
A:"""

cot_prompt = """Q: 一个班级有32名学生，其中女生比男生多4人。女生有多少人？
A: 让我们一步一步来思考。"""

print("=== 标准Prompt ===")
print(llm_generate(standard_prompt, max_new_tokens=50))

print("\n=== Zero-shot CoT ===")
print(llm_generate(cot_prompt, max_new_tokens=150))

# CoT推理过程的可视化
print("\n" + "="*60)
print("CoT推理过程图解:")
print("""
问题: 班级32人，女生比男生多4人，女生多少人？

标准推理 (直接给答案):
  32人, 多4 → 18           ← 可能出错

CoT推理 (分步推理):
  步骤1: 设男生x人，则女生(x+4)人
  步骤2: x + (x+4) = 32
  步骤3: 2x + 4 = 32
  步骤4: 2x = 28
  步骤5: x = 14 (男生)
  步骤6: 女生 = 14 + 4 = 18  ← 正确！

CoT的优势:
  ✓ 分解复杂问题为简单步骤
  ✓ 中间步骤可检查
  ✓ 减少跳跃性错误
  ✓ 适合数学/逻辑/常识推理
""")
```

## Self-Consistency

Self-Consistency（自洽性）是对CoT的增强：**对同一问题多次采样不同的CoT推理路径，然后通过多数投票选择最终答案**。

```
┌──────────────────────────────────────────────────────────────┐
│              Self-Consistency 流程                           │
│                                                              │
│              ┌──────────────┐                               │
│              │  同一个问题   │                               │
│              └──────┬───────┘                               │
│                     │                                       │
│          ┌──────────┼──────────┐                            │
│          ▼          ▼          ▼                            │
│    ┌─────────┐┌─────────┐┌─────────┐                       │
│    │ CoT路径1││ CoT路径2││ CoT路径3│  ... 多次采样          │
│    │ temp=0.7││ temp=0.7││ temp=0.7│                       │
│    └────┬────┘└────┬────┘└────┬────┘                       │
│         │         │         │                               │
│         ▼         ▼         ▼                               │
│    ┌─────────┐┌─────────┐┌─────────┐                       │
│    │ 答案: 18││ 答案: 18││ 答案: 14│                       │
│    └────┬────┘└────┬────┘└────┬────┘                       │
│         │         │         │                               │
│         └─────────┼─────────┘                               │
│                   ▼                                         │
│            ┌─────────────┐                                  │
│            │  多数投票   │                                  │
│            │  18: 2票 ✓  │                                  │
│            │  14: 1票   │                                  │
│            └──────┬──────┘                                  │
│                   ▼                                         │
│            ┌─────────────┐                                  │
│            │ 最终答案: 18│                                  │
│            └─────────────┘                                  │
│                                                              │
│  核心思想: 正确的推理路径更可能收敛到同一答案               │
└──────────────────────────────────────────────────────────────┘
```

```python
import torch
from collections import Counter

def self_consistency_generate(prompt, num_samples=5, max_new_tokens=150):
    """Self-Consistency: 多次采样取多数"""
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    answers = []
    
    for i in range(num_samples):
        torch.manual_seed(42 + i)  # 不同随机种子
        with torch.no_grad():
            outputs = model.generate(
                **inputs,
                max_new_tokens=max_new_tokens,
                temperature=0.7,  # 较高温度增加多样性
                do_sample=True,
                top_p=0.95
            )
        response = tokenizer.decode(outputs[0][inputs['input_ids'].shape[1]:],
                                   skip_special_tokens=True)
        answers.append(response)
        print(f"\n--- 采样 {i+1} ---")
        print(response[:200])
    
    # 提取答案并投票（简化版：实际中用正则提取数字）
    print(f"\n{'='*40}")
    print(f"共 {num_samples} 次采样，取多数投票")
    return answers

# 测试Self-Consistency
math_problem = """Q: 小明有50元，买了一本书12元，一支笔3元，一个笔记本5元。
他还剩多少钱？
A: 让我们一步一步来思考。"""

print("=== Self-Consistency 推理 ===")
results = self_consistency_generate(math_problem, num_samples=3)

# 简化的答案提取和投票
import re
def extract_answer(text):
    """从CoT输出中提取最终数字答案"""
    numbers = re.findall(r'\d+', text)
    return numbers[-1] if numbers else None

answer_votes = [extract_answer(r) for r in results]
answer_votes = [a for a in answer_votes if a]
if answer_votes:
    final = Counter(answer_votes).most_common(1)[0]
    print(f"\n答案投票: {Counter(answer_votes)}")
    print(f"最终答案(多数投票): {final[0]} ({final[1]}票)")
```

## Tree of Thoughts (ToT)

Tree of Thoughts将CoT的线性推理扩展为**树形搜索**：在每个推理步骤生成多个候选思路，评估并选择最有前途的继续展开。

```
┌──────────────────────────────────────────────────────────────┐
│              Tree of Thoughts (ToT) 树形搜索                 │
│                                                              │
│                        ┌──────┐                              │
│                        │ 问题 │                              │
│                        └──┬───┘                              │
│                           │                                 │
│              ┌────────────┼────────────┐                    │
│              ▼            ▼            ▼                     │
│         ┌──────┐     ┌──────┐    ┌──────┐                  │
│         │思路1A│     │思路1B│    │思路1C│  ← 生成多个       │
│         │score:│     │score:│    │score:│    候选思路       │
│         │  0.9 │     │  0.3 │    │  0.7 │                   │
│         └──┬───┘     └──────┘    └──┬───┘                  │
│            │          ✗ 剪枝         │                      │
│     ┌──────┼──────┐                 │                      │
│     ▼      ▼      ▼                 │                      │
│  ┌──────┐┌──────┐┌──────┐          │                      │
│  │思路2A││思路2B││思路2C│   ...    │                      │
│  │  0.85││  0.4 ││  0.8 │          │                      │
│  └──┬───┘└──────┘└──┬───┘          │                      │
│     │      ✗        │               │                      │
│     ▼               ▼               ▼                      │
│  ┌──────┐       ┌──────┐       ┌──────┐                   │
│  │答案A │       │答案B │       │答案C │                   │
│  └──────┘       └──────┘       └──────┘                   │
│                                                              │
│  过程: BFS/DFS搜索 + 自我评估打分 + 回溯                    │
│  优势: 可以探索不同推理路径，发现最优解                      │
│  代价: 需要更多LLM调用（成本N倍）                           │
└──────────────────────────────────────────────────────────────┘
```

```python
# Tree of Thoughts 简化实现
class TreeOfThoughts:
    """简化版ToT求解器"""
    
    def __init__(self, llm_generate_func, max_depth=3, num_branches=3):
        self.generate = llm_generate_func
        self.max_depth = max_depth
        self.num_branches = num_branches
    
    def generate_thoughts(self, problem, current_state, history=""):
        """生成多个候选思路"""
        prompt = f"""问题: {problem}
当前推理过程: {current_state}
历史: {history if history else "无"}

请生成{self.num_branches}个不同的下一步推理思路，每行一个:
1."""
        response = self.generate(prompt, max_new_tokens=100)
        thoughts = [t.strip() for t in response.split('\n') if t.strip()][:self.num_branches]
        return thoughts
    
    def evaluate_state(self, problem, state):
        """评估当前状态的前景 (0-1)"""
        prompt = f"""问题: {problem}
当前推理: {state}

评估当前推理思路是否能正确解决问题。
只回答一个0到1之间的小数，1表示非常有前途:
评分:"""
        response = self.generate(prompt, max_new_tokens=10)
        try:
            import re
            score = float(re.search(r'0?\.\d+|1\.0|1', response).group())
            return score
        except:
            return 0.5
    
    def solve(self, problem):
        """ToT求解"""
        print(f"问题: {problem}\n")
        best_path = []
        
        # BFS搜索
        from collections import deque
        queue = deque([("", 1.0, [])])  # (state, score, path)
        
        for depth in range(self.max_depth):
            print(f"\n--- 深度 {depth+1} ---")
            next_queue = []
            
            while queue and len(next_queue) < self.num_branches:
                state, score, path = queue.popleft()
                if score < 0.3:  # 剪枝
                    continue
                
                # 生成候选思路
                thoughts = self.generate_thoughts(problem, state)
                
                for thought in thoughts:
                    new_state = f"{state} {thought}".strip()
                    thought_score = self.evaluate_state(problem, new_state)
                    new_path = path + [thought]
                    next_queue.append((new_state, thought_score, new_path))
                    print(f"  思路: {thought[:50]}... → 评分: {thought_score:.2f}")
            
            # 按评分排序，保留最好的
            next_queue.sort(key=lambda x: x[1], reverse=True)
            queue = deque(next_queue[:self.num_branches])
        
        if queue:
            best_state, best_score, best_path = queue[0]
            print(f"\n最优路径评分: {best_score:.2f}")
            return best_path
        return []

# 使用ToT解决24点游戏
tot = TreeOfThoughts(llm_generate, max_depth=3, num_branches=3)
problem = "用1, 5, 6, 7四个数字，通过加减乘除得到24"
print("=== Tree of Thoughts 求解 24点 ===")
solution = tot.solve(problem)
print(f"\n推理路径: {' → '.join(solution)}")
```

## ReAct框架

ReAct（Reasoning + Acting）将**推理**和**行动**交替进行：LLM先思考下一步该做什么，然后执行行动（如调用工具），再根据观察结果继续推理。

```
┌──────────────────────────────────────────────────────────────┐
│                  ReAct: 推理 + 行动                           │
│                                                              │
│  ┌─────────┐   ┌──────┐   ┌──────┐   ┌──────┐   ┌────────┐│
│  │ Thought │──→│ Act  │──→│ Observe│──→│ Thought│──→│ Answer ││
│  │ (推理)  │   │(行动)│   │(观察) │   │ (推理) │   │ (答案) ││
│  └─────────┘   └──────┘   └──────┘   └──────┘   └────────┘│
│       ↑                                    │                 │
│       └────────────────────────────────────┘                 │
│                   循环直到得出答案                            │
│                                                              │
│  示例: "2024年诺贝尔文学奖得主是哪国人？"                     │
│                                                              │
│  Thought 1: 我需要先查找2024年诺贝尔文学奖得主               │
│  Act 1: Search("2024 诺贝尔文学奖 得主")                    │
│  Observe 1: 韩江，韩国作家，获得2024年诺贝尔文学奖           │
│                                                              │
│  Thought 2: 韩江是韩国人，我需要���认韩国的首都               │
│  Act 2: Search("韩国 首都")                                 │
│  Observe 2: 韩国的首都是首尔                                 │
│                                                              │
│  Thought 3: 韩江是韩国人，已获得所需信息                     │
│  Act 3: Finish("韩国")                                      │
│                                                              │
│  ✓ 通过工具调用获取外部信息，避免幻觉                        │
└──────────────────────────────────────────────────────────────┘
```

```python
# ReAct框架简化实现
class ReActAgent:
    """简化版ReAct Agent"""
    
    def __init__(self, llm_generate_func, tools=None):
        self.generate = llm_generate_func
        self.tools = tools or {}
    
    def format_prompt(self, question, history):
        """构建ReAct提示"""
        tool_descriptions = "\n".join(
            f"- {name}: {desc}" for name, desc in self.tools.items()
        )
        prompt = f"""你是一个能够使用工具的AI助手。请交替使用Thought（思考）和Action（行动）来解决问题。

可用工具:
{tool_descriptions}

格式:
Thought: [你的思考]
Action: [工具名](参数)

示例:
Thought: 我需要查找信息
Action: search(查询关键词)

问题: {question}

{history}
Thought:"""
        return prompt
    
    def parse_response(self, response):
        """解析LLM输出，提取Thought和Action"""
        import re
        thought = ""
        action = None
        
        # 提取Thought
        thought_match = re.search(r'Thought:\s*(.+?)(?:\nAction:|$)', response, re.DOTALL)
        if thought_match:
            thought = thought_match.group(1).strip()
        
        # 提取Action
        action_match = re.search(r'Action:\s*(\w+)\(([^)]*)\)', response)
        if action_match:
            tool_name = action_match.group(1)
            tool_arg = action_match.group(2).strip().strip("'\"")
            action = (tool_name, tool_arg)
        
        return thought, action
    
    def run(self, question, max_steps=5):
        """执行ReAct循环"""
        history = ""
        
        for step in range(max_steps):
            # 推理: 生成下一步Thought和Action
            prompt = self.format_prompt(question, history)
            response = self.generate(prompt, max_new_tokens=100)
            
            thought, action = self.parse_response(response)
            print(f"\n--- Step {step+1} ---")
            print(f"Thought: {thought}")
            
            if action is None or action[0] == "finish":
                print(f"Final Answer: {thought}")
                return thought
            
            # 行动: 执行工具
            tool_name, tool_arg = action
            print(f"Action: {tool_name}({tool_arg})")
            
            if tool_name in self.tools:
                # 模拟工具执行
                observation = f"[模拟] {tool_name}('{tool_arg}') 的结果"
                print(f"Observation: {observation}")
                history += f"Thought: {thought}\nAction: {tool_name}({tool_arg})\nObservation: {observation}\n"
            else:
                print(f"Observation: 工具 {tool_name} 不存在")
                history += f"Thought: {thought}\nAction: {tool_name}({tool_arg})\nObservation: 工具不存在\n"
        
        return "达到最大步数限制"

# 定义工具
tools = {
    "search": "搜索网络获取信息。参数: 搜索关键词",
    "calculate": "数学计算。参数: 数学表达式",
    "finish": "给出最终答案。参数: 最终答案"
}

agent = ReActAgent(llm_generate, tools)
print("=== ReAct Agent 示例 ===")
result = agent.run("iPhone 15 Pro的价格是多少？加上13%税后多少钱？")
```

## 小结

| 方法 | 核心思想 | 适用场景 | LLM调用次数 | 效果提升 |
|------|------|------|------|------|
| Zero-shot | 不给示例，直接提问 | 简单任务 | 1 | 基线 |
| One-shot | 给1个示例 | 格式明确任务 | 1 | 小幅 |
| Few-shot | 给多个示例 | 分类/抽取 | 1 | 中等 |
| Zero-shot CoT | "一步一步来思考" | 数学/逻辑推理 | 1 | 显著 |
| Few-shot CoT | 示例含推理过程 | 复杂推理 | 1 | 大幅 |
| Self-Consistency | 多次采样+投票 | 有确定答案的推理 | N (5-40) | 进一步提升 |
| Tree of Thoughts | 树形搜索推理 | 探索类问题 | N×M | 最强但最贵 |
| ReAct | 推理+工具调用 | 需外部信息 | N | 解决幻觉 |

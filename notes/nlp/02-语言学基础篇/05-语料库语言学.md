# 语料库语言学

语料库是 NLP 的数据基础——从大规模真实语言数据中发现语言规律。

## 什么是语料库？

```
语料库 (Corpus): 经过系统收集、整理、标注的大规模文本/语音集合

┌────────────────────────────────────────────────────┐
│                语料库的类型                         │
├──────────────┬─────────────────────────────────────┤
│  按规模分     │ 小型(万级) / 中型(百万) / 大型(亿级) │
├──────────────┼─────────────────────────────────────┤
│  按标注分     │ 生语料 / 标注语料 / 树库            │
├──────────────┼─────────────────────────────────────┤
│  按语言分     │ 单语 / 双语(平行) / 多语           │
├──────────────┼─────────────────────────────────────┤
│  按领域分     │ 通用 / 专业 / 特定领域             │
├──────────────┼─────────────────────────────────────┤
│  按模态分     │ 文本 / 语音 / 多模态               │
└──────────────┴─────────────────────────────────────┘
```

## 语料库统计

### Zipf 定律

```
Zipf 定律: 词频与排名成反比

频率 ∝ 1/排名^α

英语中:
  排名第1的词 (the):   频率 ~7%
  排名第2的词 (of):    频率 ~3.5%
  排名第3的词 (and):   频率 ~2.3%
  ...
  排名第100的词:       频率 ~0.07%

含义: 少数词出现极频繁, 大多数词很罕见

┌──────────────────────────────────────────────┐
│  词频分布 (Zipf)                             │
│                                              │
│  频率                                         │
│   │ ■                                        │
│   │ ██                                       │
│   │ ███                                      │
│   │ █████                                    │
│   │ █████████                                │
│   │ ████████████████                         │
│   │ █████████████████████████████████        │
│   └──────────────────────────────────── 排名 │
│    1  10  100  1000  10000  100000          │
│                                              │
│    ←高频少数    低频多数→                     │
└──────────────────────────────────────────────┘
```

```python
import nltk
from nltk.corpus import brown
from collections import Counter
import numpy as np

nltk.download('brown', quiet=True)

# 获取 Brown 语料库的词
words = brown.words()  # ~116万词

# 统计词频
word_freq = Counter(word.lower() for word in words)

# 排序
sorted_freqs = sorted(word_freq.values(), reverse=True)
ranks = range(1, len(sorted_freqs) + 1)

# 验证 Zipf 定律
print("Zipf 定律验证 (Brown Corpus):")
print(f"{'排名':<8} {'词':<12} {'频率':<10} {'理论(1/rank)':<12}")
print("-" * 45)

top_words = [w for w, _ in word_freq.most_common(10)]
total = sum(sorted_freqs)

for rank, word in enumerate(top_words, 1):
    freq = word_freq[word]
    actual_pct = freq / total * 100
    theoretical = sorted_freqs[0] / rank
    print(f"{rank:<8} {word:<12} {actual_pct:<10.2f}% {theoretical:<12.0f}")

# Heap 定律: 词汇量随语料增长
print(f"\n总词数(tokens): {len(words):,}")
print(f"不同词(types): {len(word_freq):,}")
print(f"TTR (Type-Token Ratio): {len(word_freq)/len(words):.4f}")
```

### 中文语料统计特点

```python
import jieba
from collections import Counter

chinese_text = """
自然语言处理是人工智能的重要分支。
自然语言处理研究如何让计算机理解和生成人类语言。
深度学习极大地推动了自然语言处理的发展。

大语言模型展示了惊人的语言理解和生成能力。
中文自然语言处理面临独特的挑战。
中文需要先进行分词处理。
分词的准确性直接影响下游任务的效果。
""" * 100  # 放大语料

# 分词
words = jieba.lcut(chinese_text)
chars = list(chinese_text.replace('\n', ''))

# 统计
word_freq = Counter(words)
char_freq = Counter(chars)

print(f"中文语料统计:")
print(f"  总字数: {len(chars):,}")
print(f" 总词数: {len(words):,}")
print(f" 不同字: {len(char_freq):,}")
print(f" 不同词: {len(word_freq):,}")

print(f"\n高频字 TOP 10:")
for char, freq in char_freq.most_common(10):
    print(f"  '{char}': {freq:,} ({freq/len(chars):.2%})")

print(f"\n高频词 TOP 10:")
for word, freq in word_freq.most_common(10):
    if len(word) > 0:
        print(f"  '{word}': {freq:,} ({freq/len(words):.2%})")
```

## 语料库标注体系

### 标注层次

```
NLP 语料标注层次:

Level 0: 原始文本 (Raw Text)
  "The cat sat on the mat."

Level 1: 分词 (Tokenization)
  [The] [cat] [sat] [on] [the] [mat] [.]

Level 2: 词性 (POS Tagging)
  The/DT cat/NN sat/VBD on/IN the/DT mat/NN ./.

Level 3: 句法分析 (Parsing)
  (S (NP (DT The) (NN cat)) 
     (VP (VBD sat) (PP (IN on) (NP (DT the) (NN mat)))))

Level 4: 命名实体 (NER)
  The cat sat on the mat.
  ─── ─── ───────────────
  O   O   O

Level 5: 语义角色 (SRL)
  [The cat]_Agent [sat]_Predicate [on the mat]_Location

Level 6: 共指/指代 (Coreference)
  "John bought a car. He likes it."
  John=He, car=it

Level 7: 话语/篇章 (Discourse)
  整体修辞结构标注
```

## 构建自己的语料库

```python
import os
import json
from collections import defaultdict

class CorpusBuilder:
    """语料库构建工具"""
    
    def __init__(self, name='my_corpus'):
        self.name = name
        self.documents = []
        self.metadata = []
    
    def add_document(self, doc_id, text, source=None, date=None, 
                     annotations=None):
        """添加文档"""
        self.documents.append({
            'id': doc_id,
            'text': text,
        })
        self.metadata.append({
            'id': doc_id,
            'source': source,
            'date': date,
            'annotations': annotations or {},
            'char_count': len(text),
        })
    
    def add_annotation(self, doc_id, annotation_type, value):
        """添加标注"""
        for meta in self.metadata:
            if meta['id'] == doc_id:
                meta['annotations'][annotation_type] = value
                break
    
    def get_stats(self):
        """统计信息"""
        total_chars = sum(m['char_count'] for m in self.metadata)
        
        # 词统计
        all_words = []
        for doc in self.documents:
            import jieba
            all_words.extend(jieba.lcut(doc['text']))
        
        from collections import Counter
        word_freq = Counter(all_words)
        
        return {
            'documents': len(self.documents),
            'total_chars': total_chars,
            'total_words': len(all_words),
            'unique_words': len(word_freq),
            'avg_doc_length': total_chars / len(self.documents),
            'top_words': word_freq.most_common(20),
        }
    
    def save(self, path):
        """保存语料库"""
        data = {
            'name': self.name,
            'documents': self.documents,
            'metadata': self.metadata,
        }
        with open(path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        print(f"语料库已保存: {path}")
    
    def load(self, path):
        """加载语料库"""
        with open(path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        self.name = data['name']
        self.documents = data['documents']
        self.metadata = data['metadata']
        print(f"语料库已加载: {self.name} ({len(self.documents)} 文档)")

# 使用
corpus = CorpusBuilder('news_corpus')

# 添加文档
corpus.add_document(
    doc_id='001',
    text='苹果公司发布了新一代iPhone，搭载最新的A17芯片。',
    source='科技新闻',
    date='2024-09-01',
    annotations={'category': '科技', 'sentiment': 'neutral'}
)

corpus.add_document(
    doc_id='002', 
    text='中国女排在奥运会决赛中以3比1战胜对手，获得金牌。',
    source='体育新闻',
    date='2024-08-11',
    annotations={'category': '体育', 'sentiment': 'positive'}
)

# 统计
stats = corpus.get_stats()
print(f"\n语料库统计:")
for key, value in stats.items():
    if key != 'top_words':
        print(f"  {key}: {value}")
print(f"  高频词: {stats['top_words'][:5]}")
```

## 评估指标: Kappa 系数

```
标注一致性: 多个标注者的一致程度

Cohen's Kappa (两个标注者):
  κ = (P_o - P_e) / (1 - P_e)
  
  P_o = 观察到的一致率
  P_e = 偶然一致率

  κ > 0.8  → 几乎完美一致
  κ 0.6-0.8 → 高度一致
  κ 0.4-0.6 → 中等一致  
  κ < 0.4  → 一致性差
```

```python
import numpy as np
from sklearn.metrics import cohen_kappa_score

# 两个标注者的标注结果
annotator1 = [1, 1, 0, 1, 0, 0, 1, 0, 1, 0]
annotator2 = [1, 1, 0, 0, 0, 1, 1, 0, 1, 0]

kappa = cohen_kappa_score(annotator1, annotator2)
print(f"Cohen's Kappa: {kappa:.4f}")

if kappa > 0.8:
    print("→ 几乎完美一致")
elif kappa > 0.6:
    print("→ 高度一致")
elif kappa > 0.4:
    print("→ 中等一致")
else:
    print("→ 一致性差，需要重新审视标注标准")
```

## 小结

| 概念 | 说明 | 重要性 |
|------|------|--------|
| **语料库** | 系统收集的语言数据 | NLP的基础 |
| **Zipf定律** | 词频 ∝ 1/排名 | 理解词分布 |
| **标注层次** | 分词→词性→句法→语义 | 多层次理解 |
| **平行语料** | 双语对照 | 机器翻译 |
| **标注一致性** | Kappa系数 | 数据质量 |
| **TTR** | 词汇丰富度 | 语料特征 |

> 没有好的语料库，就没有好的 NLP 系统。数据是 NLP 的第一要素。

# 04 - 文档问答系统

基于RAG的PDF/文档问答系统允许用户直接对文档内容提问，系统自动检索相关段落并生成带来源引用的答案。本章将构建一个完整的文档问答系统，支持PDF解析、智能分块、语义检索和来源标注。

## 1. 项目概述

### 1.1 核心能力

- **多格式支持**：PDF、DOCX、TXT、Markdown
- **智能分块**：递归分块 + 语义边界检测
- **语义检索**：ChromaDB向量数据库
- **来源追溯**：答案附带原文引用和页码

## 2. 文档解析

### 2.1 PDF解析（PyMuPDF）

```python
import fitz  # PyMuPDF
from typing import List, Dict, Tuple

class PDFParser:
    """PDF文档解析器 - 按页提取文本并保留位置信息"""

    def parse(self, filepath: str) -> List[Dict]:
        """解析PDF，返回每页的文本和元数据"""
        doc = fitz.open(filepath)
        pages = []

        for page_num in range(len(doc)):
            page = doc.load_page(page_num)

            # 提取文本
            text = page.get_text("text")

            # 提取段落级别的文本块
            blocks = page.get_text("blocks")  # (x0,y0,x1,y1,text,block_no,type)

            paragraphs = []
            for block in blocks:
                if block[4].strip():  # 非空文本
                    paragraphs.append({
                        "text": block[4].strip(),
                        "bbox": block[:4],  # 边界框
                    })

            pages.append({
                "page_num": page_num + 1,  # 页码从1开始
                "text": text,
                "paragraphs": paragraphs,
                "char_count": len(text),
            })

        doc.close()
        print(f"解析完成: {len(pages)} 页, 总字符数: {sum(p['char_count'] for p in pages)}")
        return pages

    def parse_with_metadata(self, filepath: str) -> List[Document]:
        """解析PDF并生成带元数据的Document列表"""
        pages = self.parse(filepath)
        documents = []

        filename = os.path.basename(filepath)

        for page in pages:
            # 为每个段落创建独立文档块
            for para in page["paragraphs"]:
                if len(para["text"]) < 10:  # 过滤过短段落
                    continue
                doc = Document(
                    page_content=para["text"],
                    metadata={
                        "source": filepath,
                        "filename": filename,
                        "page": page["page_num"],
                        "bbox": para["bbox"],
                    }
                )
                documents.append(doc)

        return documents

# 使用示例
parser = PDFParser()
pdf_docs = parser.parse_with_metadata("attention_is_all_you_need.pdf")
print(f"生成了 {len(pdf_docs)} 个文档块")
```

### 2.2 多格式解析器

```python
class UniversalDocumentParser:
    """通用文档解析器 - 支持PDF/DOCX/TXT/MD"""

    def parse(self, filepath: str) -> List[Document]:
        ext = os.path.splitext(filepath)[1].lower()

        if ext == ".pdf":
            return PDFParser().parse_with_metadata(filepath)
        elif ext in [".docx", ".doc"]:
            return self._parse_docx(filepath)
        elif ext in [".txt", ".md"]:
            return self._parse_text(filepath)
        else:
            raise ValueError(f"不支持的文件格式: {ext}")

    def _parse_docx(self, filepath: str) -> List[Document]:
        """解析Word文档"""
        from docx import Document as DocxDocument

        doc = DocxDocument(filepath)
        documents = []

        for i, para in enumerate(doc.paragraphs):
            text = para.text.strip()
            if len(text) < 10:
                continue
            documents.append(Document(
                page_content=text,
                metadata={
                    "source": filepath,
                    "filename": os.path.basename(filepath),
                    "paragraph": i + 1,
                }
            ))
        return documents

    def _parse_text(self, filepath: str) -> List[Document]:
        """解析纯文本/Markdown"""
        with open(filepath, "r", encoding="utf-8") as f:
            content = f.read()

        # 按段落分割
        paragraphs = content.split("\n\n")
        documents = []

        for i, para in enumerate(paragraphs):
            para = para.strip()
            if len(para) < 10:
                continue
            documents.append(Document(
                page_content=para,
                metadata={
                    "source": filepath,
                    "filename": os.path.basename(filepath),
                    "paragraph": i + 1,
                }
            ))
        return documents
```

## 3. 文档分块策略

### 3.1 递归分块 vs 语义分块

```
递归分块（RecursiveCharacterTextSplitter）:
  基于字符/分隔符的固定大小切分，保证块大小一致
  
  原文: "今天天气很好。我们去公园散步。公园里花开了。"
  
  +----------------------+  +----------------------+
  | 今天天气很好。       |  | 公园里花开了。       |
  | 我们去公园散步。     |  |                      |
  +----------------------+  +----------------------+
         (Chunk 0)               (Chunk 1)
         |_______________| 重叠
  
语义分块（SemanticChunker）:
  基于Embedding相似度判断段落边界，更智能但更慢
  
  原文: "今天天气很好。我们去公园散步。| 公园里花开了。"
                              ^语义边界（话题变化点）
```

### 3.2 对比表格

| 特性 | 递归分块 | 语义分块 |
|------|----------|----------|
| 速度 | 快（线性遍历） | 慢（需要Embedding计算） |
| 块大小 | 均匀 | 不均匀 |
| 语义完整性 | 可能切断语义 | 保留语义完整性 |
| 检索效果 | 好 | 更好 |
| 适用场景 | 通用文档 | 论文、长文档 |

### 3.3 语义分块实现

```python
import numpy as np
from sentence_transformers import SentenceTransformer

class SemanticChunker:
    """基于Embedding相似度的语义分块器"""

    def __init__(self, model_name: str = "BAAI/bge-small-zh-v1.5",
                 similarity_threshold: float = 0.6):
        self.model = SentenceTransformer(model_name)
        self.threshold = similarity_threshold

    def split(self, text: str, min_chunk_size: int = 100) -> List[str]:
        """根据语义相似度进行分块"""
        # 按句子分割
        sentences = self._split_sentences(text)

        if len(sentences) <= 1:
            return [text]

        # 计算句子embedding
        embeddings = self.model.encode(sentences)

        chunks = []
        current_chunk = [sentences[0]]
        current_embedding = embeddings[0]

        for i in range(1, len(sentences)):
            # 计算当前句子与当前块的相似度
            similarity = self._cosine_similarity(
                embeddings[i], current_embedding
            )

            if similarity >= self.threshold:
                # 相似度高，合并到当前块
                current_chunk.append(sentences[i])
                # 更新块的平均embedding
                current_embedding = np.mean(
                    embeddings[len(current_chunk)-len(current_chunk):i+1],
                    axis=0
                )
            else:
                # 相似度低，开始新块
                chunk_text = "".join(current_chunk)
                if len(chunk_text) >= min_chunk_size:
                    chunks.append(chunk_text)
                current_chunk = [sentences[i]]
                current_embedding = embeddings[i]

        # 处理最后一块
        if current_chunk:
            chunk_text = "".join(current_chunk)
            if len(chunk_text) >= 10:
                chunks.append(chunk_text)

        return chunks

    def _split_sentences(self, text: str) -> List[str]:
        """中文句子分割"""
        import re
        # 按中文标点分割
        sentence_endings = r'[。！？!?；;]+'
        parts = re.split(f'({sentence_endings})', text)
        sentences = []
        for i in range(0, len(parts) - 1, 2):
            sentences.append(parts[i] + parts[i+1])
        if len(parts) % 2 == 1 and parts[-1].strip():
            sentences.append(parts[-1])
        return sentences

    def _cosine_similarity(self, a: np.ndarray, b: np.ndarray) -> float:
        return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

# 使用示例
chunker = SemanticChunker(similarity_threshold=0.65)
chunks = chunker.split(long_document_text)
print(f"语义分块: {len(chunks)} 个块")
```

## 4. 向量化存储（ChromaDB）

### 4.1 ChromaDB创建与查询

```python
import chromadb
from chromadb.config import Settings

class DocumentVectorStore:
    """ChromaDB向量存储封装"""

    def __init__(self, collection_name: str = "pdf_qa",
                 embedding_model: str = "BAAI/bge-large-zh-v1.5"):
        self.client = chromadb.PersistentClient(
            path="./chroma_db",
            settings=Settings(anonymized_telemetry=False)
        )

        # 使用sentence-transformer作为embedding函数
        from chromadb.utils import embedding_functions
        self.embedding_fn = embedding_functions.SentenceTransformerEmbeddingFunction(
            model_name=embedding_model,
            device="cuda"
        )

        # 创建或获取集合
        self.collection = self.client.get_or_create_collection(
            name=collection_name,
            embedding_function=self.embedding_fn,
            metadata={"hnsw:space": "cosine"}
        )

    def add_documents(self, documents: List[Document]):
        """批量添加文档到向量库"""
        ids = []
        contents = []
        metadatas = []

        for i, doc in enumerate(documents):
            doc_id = f"{doc.metadata.get('filename', 'unknown')}_{i}"
            ids.append(doc_id)
            contents.append(doc.page_content)

            # 转换metadata为ChromaDB兼容格式
            meta = {}
            for k, v in doc.metadata.items():
                if isinstance(v, (str, int, float, bool)):
                    meta[k] = v
                elif isinstance(v, (list, tuple)):
                    meta[k] = str(v)
            metadatas.append(meta)

        # 分批插入（避免过大）
        batch_size = 100
        for i in range(0, len(ids), batch_size):
            self.collection.add(
                ids=ids[i:i+batch_size],
                documents=contents[i:i+batch_size],
                metadatas=metadatas[i:i+batch_size]
            )

        print(f"已添加 {len(documents)} 个文档块到ChromaDB")

    def search(self, query: str, top_k: int = 5) -> List[Dict]:
        """语义搜索"""
        results = self.collection.query(
            query_texts=[query],
            n_results=top_k,
            include=["documents", "metadatas", "distances"]
        )

        formatted = []
        for i in range(len(results["ids"][0])):
            distance = results["distances"][0][i]
            # Cosine距离转相似度 (0-1)
            similarity = 1.0 - distance / 2.0

            formatted.append({
                "content": results["documents"][0][i],
                "metadata": results["metadatas"][0][i],
                "similarity": round(similarity, 4),
                "page": results["metadatas"][0][i].get("page", "N/A"),
                "source": results["metadatas"][0][i].get("filename", "unknown"),
            })

        return formatted

# 使用示例
vector_store = DocumentVectorStore(collection_name="paper_qa")
vector_store.add_documents(pdf_docs)

results = vector_store.search("什么是Transformer架构", top_k=3)
for r in results:
    print(f"[{r['source']} 第{r['page']}页] (相似度: {r['similarity']})")
    print(f"  {r['content'][:100]}...\n")
```

## 5. 问答流程

### 5.1 完整问答系统

```python
class DocumentQA:
    """完整的文档问答系统"""

    def __init__(self, llm, vector_store: DocumentVectorStore):
        self.llm = llm
        self.vector_store = vector_store

    def ask(self, question: str, top_k: int = 5, return_sources: bool = True) -> Dict:
        """对文档提问"""

        # Step 1: 检索相关段落
        retrieved = self.vector_store.search(question, top_k=top_k)

        # Step 2: 构建上下文（附带来源信息）
        context_parts = []
        for i, r in enumerate(retrieved, 1):
            source_info = f"[来源{i}] {r['source']}"
            if r.get("page"):
                source_info += f" 第{r['page']}页"
            context_parts.append(f"{source_info}\n{r['content']}")

        context = "\n\n---\n\n".join(context_parts)

        # Step 3: Prompt模板
        prompt = self._build_prompt(question, context)

        # Step 4: LLM生成回答
        response = self.llm.invoke(prompt)

        # Step 5: 构建最终响应
        result = {
            "question": question,
            "answer": response.content,
            "sources": [
                {
                    "index": i,
                    "source": r["source"],
                    "page": r.get("page", "N/A"),
                    "content": r["content"][:200],
                    "similarity": r["similarity"],
                }
                for i, r in enumerate(retrieved, 1)
            ]
        }
        return result

    def _build_prompt(self, question: str, context: str) -> str:
        return f"""你是一个文档分析助手。请根据以下文档内容回答用户问题。

## 文档内容
{context}

## 用户问题
{question}

## 回答要求
1. 严格基于提供的文档内容回答，不要编造信息
2. 如果文档中未提及相关信息，请明确说明
3. 回答时引用具体的来源编号，例如 [来源1] 或 [来源3 第5页]
4. 如果涉及多个来源，综合它们的信息
5. 回答要简洁、准确、有条理

## 回答："""
```

## 6. Gradio交互界面

```python
import gradio as gr

def create_qa_interface(qa_system: DocumentQA):
    """创建Gradio交互界面"""

    def answer_question(question: str, history: list):
        """处理单次提问"""
        if not question.strip():
            return "", history

        result = qa_system.ask(question, top_k=5)

        # 格式化回答
        formatted_answer = f"### 回答\n\n{result['answer']}\n\n---\n\n### 参考来源\n\n"

        for s in result["sources"]:
            page_info = f"第{s['page']}页" if s['page'] != "N/A" else ""
            formatted_answer += (
                f"**[{s['index']}] {s['source']}** {page_info}\n"
                f"相似度: {s['similarity']:.2%}\n"
                f"> {s['content']}...\n\n"
            )

        history = history + [(question, formatted_answer)]
        return "", history

    # 创建Gradio界面
    with gr.Blocks(title="文档问答系统", theme=gr.themes.Soft()) as demo:
        gr.Markdown("# 文档问答系统")
        gr.Markdown("上传PDF/文档，直接对文档内容提问，获取带来源引用的回答。")

        with gr.Row():
            with gr.Column(scale=1):
                file_input = gr.File(
                    label="上传文档",
                    file_types=[".pdf", ".docx", ".txt", ".md"],
                    file_count="multiple"
                )
                upload_btn = gr.Button("加载文档", variant="primary")
                upload_status = gr.Textbox(label="状态", interactive=False)

            with gr.Column(scale=2):
                chatbot = gr.Chatbot(
                    label="对话",
                    height=500,
                    show_copy_button=True
                )
                msg = gr.Textbox(
                    label="输入问题",
                    placeholder="请基于文档内容提问...",
                    show_label=False
                )
                clear_btn = gr.Button("清空对话")

        msg.submit(answer_question, [msg, chatbot], [msg, chatbot])
        clear_btn.click(lambda: (None, []), None, [msg, chatbot])

    return demo
```

# 03 - 搜索引擎构建

文档搜索引擎是知识管理系统的核心组件。本章将构建一个混合检索系统，结合传统的BM25精确匹配和现代的语义向量检索，并加入Cross-encoder精排，实现高质量的搜索体验。

---

## 1. 项目概述

### 1.1 系统能力

- **多源文档索引**：支持本地文件、Web爬虫等多种来源
- **混合检索**：BM25稀疏检索 + Dense向量检索，取长补短
- **结果重排序**：Cross-encoder对候选结果精排
- **Web展示**：搜索界面，结果高亮显示

### 1.2 检索流程总览


## 2. 文档索引构建

### 2.1 文档采集与分块

```python
import os
import json
from typing import List, Dict
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.schema import Document

class DocumentIndexer:
    """文档索引构建器"""

    def __init__(self, chunk_size: int = 500, chunk_overlap: int = 50):
        self.splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            separators=["\n\n", "\n", "。", "！", "？", "；", "，", " ", ""]
        )

    def load_from_directory(self, dir_path: str) -> List[Document]:
        """从目录加载所有支持的文档"""
        documents = []
        supported_exts = {".txt", ".md", ".json", ".csv"}

        for root, _, files in os.walk(dir_path):
            for file in files:
                ext = os.path.splitext(file)[1].lower()
                if ext not in supported_exts:
                    continue

                filepath = os.path.join(root, file)
                with open(filepath, "r", encoding="utf-8") as f:
                    content = f.read()

                # 分块
                chunks = self.splitter.split_text(content)
                for i, chunk in enumerate(chunks):
                    doc = Document(
                        page_content=chunk,
                        metadata={
                            "source": filepath,
                            "filename": file,
                            "chunk_id": i,
                            "char_start": i * (self.splitter._chunk_size - self.splitter._chunk_overlap),
                        }
                    )
                    documents.append(doc)

        print(f"从 {dir_path} 加载了 {len(documents)} 个文档块")
        return documents

# 使用示例
indexer = DocumentIndexer(chunk_size=500, chunk_overlap=50)
docs = indexer.load_from_directory("./knowledge_docs/")
```

## 3. BM25稀疏检索

### 3.1 BM25原理

BM25是一种基于词频-逆文档频率的概率检索模型，对短查询效果尤其好。

```
BM25(d, q) = Σ IDF(qi) × [f(qi,d) × (k1+1)] / [f(qi,d) + k1 × (1-b + b × |d|/avgdl)]

其中:
  - f(qi, d) : 词qi在文档d中的词频
  - |d|      : 文档d的长度
  - avgdl    : 所有文档的平均长度
  - k1, b    : 可调参数 (默认 k1=1.5, b=0.75)
  - IDF(qi)  : 词的逆文档频率
```

### 3.2 BM25实现

```python
from rank_bm25 import BM25Okapi
import jieba
import numpy as np
from typing import List, Tuple

class BM25Retriever:
    """BM25检索器"""

    def __init__(self, documents: List[Document]):
        self.documents = documents
        # 对文档进行分词
        self.tokenized_docs = [
            list(jieba.cut(doc.page_content)) for doc in documents
        ]
        self.bm25 = BM25Okapi(self.tokenized_docs, k1=1.5, b=0.75)

    def search(self, query: str, top_k: int = 10) -> List[Tuple[Document, float]]:
        """BM25检索"""
        tokenized_query = list(jieba.cut(query))
        scores = self.bm25.get_scores(tokenized_query)

        # 获取Top-K索引
        top_indices = np.argsort(scores)[::-1][:top_k]

        results = []
        for idx in top_indices:
            if scores[idx] > 0:  # 过滤得分为0的结果
                results.append((self.documents[idx], float(scores[idx])))

        return results

# 使用示例
bm25_retriever = BM25Retriever(docs)
results = bm25_retriever.search("如何配置数据库连接", top_k=5)
for doc, score in results:
    print(f"Score: {score:.4f} | Source: {doc.metadata['filename']}")
    print(f"  Content: {doc.page_content[:80]}...\n")
```

## 4. 稠密向量检索

### 4.1 语义向量检索

```python
from langchain_community.embeddings import HuggingFaceBgeEmbeddings
from langchain_community.vectorstores import FAISS

class DenseRetriever:
    """稠密向量检索器"""

    def __init__(self, documents: List[Document], model_name: str = "BAAI/bge-large-zh-v1.5"):
        self.model = HuggingFaceBgeEmbeddings(
            model_name=model_name,
            model_kwargs={"device": "cuda"},
            encode_kwargs={"normalize_embeddings": True}
        )
        self.vectorstore = FAISS.from_documents(documents, self.model)

    def search(self, query: str, top_k: int = 10) -> List[Tuple[Document, float]]:
        """向量相似度检索"""
        docs_with_scores = self.vectorstore.similarity_search_with_score(
            query, k=top_k
        )
        # FAISS返回的是L2距离，越小越相似；转换为相似度分数
        results = []
        for doc, distance in docs_with_scores:
            similarity = 1.0 / (1.0 + distance)  # 转换为0-1之间的相似度
            results.append((doc, similarity))
        return results

# 使用示例
dense_retriever = DenseRetriever(docs)
results = dense_retriever.search("如何配置数据库连接", top_k=5)
for doc, score in results:
    print(f"Similarity: {score:.4f} | {doc.metadata['filename']}")
```

## 5. 混合检索与融合

### 5.1 RRF (Reciprocal Rank Fusion) 融合算法

```python
def reciprocal_rank_fusion(
    bm25_results: List[Tuple[Document, float]],
    dense_results: List[Tuple[Document, float]],
    k: int = 60,
    top_k: int = 10
) -> List[Tuple[Document, float]]:
    """
    RRF融合算法：合并BM25和稠密检索的结果

    公式: RRF_score(d) = Σ 1/(k + rank_i(d))
    其中 rank_i(d) 是文档d在第i个检索器中的排名
    """
    scores = {}

    # 处理BM25结果
    for rank, (doc, _) in enumerate(bm25_results, 1):
        doc_id = doc.metadata.get("source", "") + str(doc.metadata.get("chunk_id", 0))
        scores[doc_id] = scores.get(doc_id, 0) + 1.0 / (k + rank)

    # 处理稠密检索结果
    for rank, (doc, _) in enumerate(dense_results, 1):
        doc_id = doc.metadata.get("source", "") + str(doc.metadata.get("chunk_id", 0))
        scores[doc_id] = scores.get(doc_id, 0) + 1.0 / (k + rank)

    # 按融合分数排序
    doc_map = {}
    for result_list in [bm25_results, dense_results]:
        for doc, _ in result_list:
            doc_id = doc.metadata.get("source", "") + str(doc.metadata.get("chunk_id", 0))
            doc_map[doc_id] = doc

    sorted_docs = sorted(scores.items(), key=lambda x: x[1], reverse=True)

    results = []
    for doc_id, score in sorted_docs[:top_k]:
        results.append((doc_map[doc_id], score))

    return results
```

### 5.2 混合检索引擎

```python
class HybridSearchEngine:
    """混合搜索引擎：BM25 + Dense + Reranker"""

    def __init__(self, documents: List[Document], use_reranker: bool = True):
        self.bm25 = BM25Retriever(documents)
        self.dense = DenseRetriever(documents)
        self.use_reranker = use_reranker

        if use_reranker:
            from sentence_transformers import CrossEncoder
            self.reranker = CrossEncoder("BAAI/bge-reranker-large")

    def search(self, query: str, top_k: int = 5) -> List[Dict]:
        """执行混合检索"""
        # Step 1: 并行执行BM25和Dense检索
        bm25_results = self.bm25.search(query, top_k=30)
        dense_results = self.dense.search(query, top_k=30)

        # Step 2: RRF融合
        fused_results = reciprocal_rank_fusion(
            bm25_results, dense_results, top_k=20
        )

        # Step 3: Cross-encoder精排
        if self.use_reranker and fused_results:
            docs = [doc.page_content for doc, _ in fused_results]
            pairs = [[query, doc] for doc in docs]
            rerank_scores = self.reranker.predict(pairs)

            # 按精排分数重新排序
            reranked = sorted(
                zip(fused_results, rerank_scores),
                key=lambda x: x[1], reverse=True
            )

            final_results = []
            for (doc, _), rerank_score in reranked[:top_k]:
                final_results.append({
                    "content": doc.page_content,
                    "score": float(rerank_score),
                    "source": doc.metadata.get("source", ""),
                    "filename": doc.metadata.get("filename", ""),
                    "chunk_id": doc.metadata.get("chunk_id", 0),
                })
            return final_results

        # 不使用精排时直接返回融合结果
        final_results = []
        for doc, score in fused_results[:top_k]:
            final_results.append({
                "content": doc.page_content,
                "score": float(score),
                "source": doc.metadata.get("source", ""),
                "filename": doc.metadata.get("filename", ""),
                "chunk_id": doc.metadata.get("chunk_id", 0),
            })
        return final_results

# 使用示例
engine = HybridSearchEngine(docs, use_reranker=True)
results = engine.search("数据库连接配置方法", top_k=5)
for i, r in enumerate(results):
    print(f"[{i+1}] Score: {r['score']:.4f} | {r['filename']}")
    print(f"    {r['content'][:100]}...\n")
```

## 6. 搜索结果展示

### 6.1 高亮显示

```python
import re

def highlight_keywords(text: str, query: str, context_chars: int = 100) -> str:
    """在文本中高亮查询关键词，并截取上下文"""
    # 提取查询中的关键词
    keywords = jieba.lcut(query)
    keywords = [k for k in keywords if len(k) > 1]  # 过滤单字

    # 找关键词出现位置
    positions = []
    for kw in keywords:
        for m in re.finditer(re.escape(kw), text):
            positions.append(m.start())

    if not positions:
        # 未找到关键词，直接截取开头
        snippet = text[:context_chars]
    else:
        # 取第一个匹配位置附近的上下文
        first_pos = positions[0]
        start = max(0, first_pos - context_chars // 2)
        end = min(len(text), first_pos + context_chars // 2)
        snippet = text[start:end]

    # 高亮关键词 (用HTML标记)
    for kw in sorted(keywords, key=len, reverse=True):
        snippet = re.sub(
            f"({re.escape(kw)})",
            r'<mark style="background-color: #FFF176;">\1</mark>',
            snippet
        )

    return snippet

# 使用示例
snippet = highlight_keywords("数据库连接需要配置JDBC URL和用户名密码", "数据库连接")
print(snippet)
```

### 6.2 Web搜索界面（Flask）

```python
from flask import Flask, render_template, request, jsonify

app = Flask(__name__)
engine = HybridSearchEngine(docs, use_reranker=True)

@app.route("/")
def index():
    return render_template("search.html")

@app.route("/search")
def search():
    query = request.args.get("q", "").strip()
    if not query:
        return jsonify({"results": [], "total": 0})

    results = engine.search(query, top_k=10)

    # 添加高亮片段
    for r in results:
        r["snippet"] = highlight_keywords(r["content"], query, context_chars=150)

    return jsonify({
        "results": results,
        "total": len(results),
        "query": query
    })

if __name__ == "__main__":
    app.run(debug=True, port=5000)
```

## 7. Elasticsearch实现（生产级方案）

```python
from elasticsearch import Elasticsearch
from elasticsearch.helpers import bulk

class ESSearchEngine:
    """基于Elasticsearch的生产级搜索引擎"""

    def __init__(self):
        self.es = Elasticsearch(["http://localhost:9200"])
        self.index_name = "docs_search"

    def create_index(self):
        """创建支持BM25和向量检索的索引"""
        mapping = {
            "mappings": {
                "properties": {
                    "content": {
                        "type": "text",
                        "analyzer": "ik_max_word",  # 中文分词器
                        "search_analyzer": "ik_smart"
                    },
                    "content_vector": {
                        "type": "dense_vector",
                        "dims": 1024,
                        "index": True,
                        "similarity": "cosine"
                    },
                    "source": {"type": "keyword"},
                    "filename": {"type": "keyword"},
                    "chunk_id": {"type": "integer"},
                }
            }
        }
        if self.es.indices.exists(index=self.index_name):
            self.es.indices.delete(index=self.index_name)
        self.es.indices.create(index=self.index_name, body=mapping)

    def index_documents(self, documents: List[Document], embeddings_model):
        """批量索引文档"""
        actions = []
        for i, doc in enumerate(documents):
            vector = embeddings_model.embed_query(doc.page_content)
            actions.append({
                "_index": self.index_name,
                "_id": f"doc_{i}",
                "_source": {
                    "content": doc.page_content,
                    "content_vector": vector,
                    "source": doc.metadata.get("source", ""),
                    "filename": doc.metadata.get("filename", ""),
                    "chunk_id": doc.metadata.get("chunk_id", 0),
                }
            })
        bulk(self.es, actions)
        self.es.indices.refresh(index=self.index_name)

    def hybrid_search(self, query: str, query_vector: list, top_k: int = 10):
        """Elasticsearch混合检索：BM25 + KNN"""
        body = {
            "query": {
                "bool": {
                    "should": [
                        {"match": {"content": {"query": query, "boost": 1.0}}},
                        {
                            "knn": {
                                "content_vector": {
                                    "vector": query_vector,
                                    "k": top_k,
                                    "boost": 0.5
                                }
                            }
                        }
                    ]
                }
            },
            "size": top_k
        }
        response = self.es.search(index=self.index_name, body=body)
        return response["hits"]["hits"]
```

## 8. 检索评估

### 8.1 评估指标

```python
def evaluate_search_quality(engine, test_queries, ground_truth):
    """评估搜索质量：NDCG@k 和 MRR"""
    from sklearn.metrics import ndcg_score

    ndcg_values = []
    mrr_values = []

    for query, relevant_docs in zip(test_queries, ground_truth):
        results = engine.search(query, top_k=10)
        result_ids = [f"{r['filename']}_{r['chunk_id']}" for r in results]

        # NDCG@10
        relevance = [1.0 if rid in relevant_docs else 0.0 for rid in result_ids]
        ideal = sorted(relevance, reverse=True)

        if sum(ideal) > 0:
            ndcg_values.append(ndcg_score([relevance], [ideal]))

        # MRR
        for rank, rid in enumerate(result_ids, 1):
            if rid in relevant_docs:
                mrr_values.append(1.0 / rank)
                break
        else:
            mrr_values.append(0.0)

    print("=" * 40)
    print(f"NDCG@10: {np.mean(ndcg_values):.4f}")
    print(f"MRR:      {np.mean(mrr_values):.4f}")
    print("=" * 40)
```

| 方法 | NDCG@10 | MRR | 延迟(ms) | 内存 |

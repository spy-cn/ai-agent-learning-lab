# 02 - 智能客服机器人

基于RAG（Retrieval-Augmented Generation）的智能客服系统是当前大模型应用的热门方向。它将企业知识库与大语言模型结合，既能保证回答的专业性和准确性，又能提供自然流畅的对话体验。

---

## 1. 项目概述

### 1.1 核心能力

- **知识库问答**：基于企业FAQ和产品文档回答用户问题
- **意图识别**：自动判断咨询、投诉、闲聊等意图
- **多轮对话**：支持上下文记忆的连续对话
- **来源追溯**：回答附带知识来源引用

## 4. 检索流程详解

### 4.1 两阶段检索与重排序

```python
from sentence_transformers import CrossEncoder

# 加载交叉编码器用于精排
cross_encoder = CrossEncoder("BAAI/bge-reranker-large")

def two_stage_retrieval(query: str, initial_k: int = 20, final_k: int = 5):
    """
    两阶段检索：
    1. 粗排：FAISS向量检索，返回Top-K候选
    2. 精排：Cross-encoder对候选重新打分排序
    """
    # Stage 1: 粗排 - 向量检索
    initial_docs = vectorstore.similarity_search(query, k=initial_k)

    # Stage 2: 精排 - Cross-encoder重新打分
    pairs = [[query, doc.page_content] for doc in initial_docs]
    scores = cross_encoder.predict(pairs)

    # 按精排分数重新排序
    ranked = sorted(
        zip(initial_docs, scores),
        key=lambda x: x[1],
        reverse=True
    )

    # 返回Top-K
    return [{"content": doc.page_content, "score": float(score),
             "metadata": doc.metadata}
            for doc, score in ranked[:final_k]]
```

## 5. 意图识别

### 5.1 意图分类器

```python
from typing import Literal

IntentType = Literal["咨询", "投诉", "闲聊", "售后", "订单查询"]

# 基于LLM的意图识别Prompt
INTENT_PROMPT = """你是一个客服意图识别助手。请根据用户输入，判断其意图类别。

意图类别：咨询、投诉、闲聊、售后、订单查询

用户输入：{user_input}

请直接回答意图类别（仅一个词）。"""

def identify_intent(user_input: str, llm) -> IntentType:
    """使用LLM识别用户意图"""
    prompt = INTENT_PROMPT.format(user_input=user_input)
    response = llm.invoke(prompt).content.strip()

    valid_intents = {"咨询", "投诉", "闲聊", "售后", "订单查询"}
    if response in valid_intents:
        return response
    return "闲聊"  # 默认意图
```

## 6. LLM回答生成

### 6.1 Prompt模板设计

```python
RAG_PROMPT_TEMPLATE = """你是一个专业的客服助手。请根据以下知识库内容回答用户问题。

## 知识库内容
{context}

## 对话历史
{chat_history}

## 用户问题
{question}

## 回答要求
1. 基于知识库内容回答，不要编造信息
2. 如果知识库中没有相关信息，诚实告知用户
3. 回答要简洁、友好、专业
4. 可以适当引导用户提供更多信息

## 回答："""

def build_context(retrieved_docs: list) -> str:
    """构建上下文"""
    contexts = []
    for i, doc in enumerate(retrieved_docs, 1):
        context_item = f"[来源{i}] {doc['content']}"
        contexts.append(context_item)
    return "\n\n".join(contexts)

def generate_answer(question: str, chat_history: str, llm):
    """生成最终回答"""
    # 检索相关文档
    retrieved = two_stage_retrieval(question, initial_k=20, final_k=5)
    context = build_context(retrieved)

    # 构建Prompt
    prompt = RAG_PROMPT_TEMPLATE.format(
        context=context,
        chat_history=chat_history,
        question=question
    )

    # 调用LLM生成
    response = llm.invoke(prompt)

    return {
        "answer": response.content,
        "sources": [
            {"id": r["metadata"].get("id", ""), "score": round(r["score"], 4)}
            for r in retrieved
        ]
    }
```

### 6.2 多轮对话管理

```python
from collections import defaultdict

class ConversationManager:
    """多轮对话管理器"""

    def __init__(self, max_history: int = 5):
        self.sessions = defaultdict(list)  # session_id -> [(role, content), ...]
        self.max_history = max_history

    def add_message(self, session_id: str, role: str, content: str):
        """添加一条消息到对话历史"""
        self.sessions[session_id].append((role, content))
        # 保留最近N轮
        if len(self.sessions[session_id]) > self.max_history * 2:
            self.sessions[session_id] = self.sessions[session_id][-self.max_history * 2:]

    def get_history(self, session_id: str) -> str:
        """获取格式化的对话历史"""
        history = self.sessions.get(session_id, [])
        if not history:
            return "无历史对话"

        formatted = []
        for role, content in history:
            role_name = "用户" if role == "user" else "助手"
            formatted.append(f"{role_name}: {content}")
        return "\n".join(formatted)

    def clear_session(self, session_id: str):
        """清除会话"""
        if session_id in self.sessions:
            del self.sessions[session_id]

# 使用示例
conv_mgr = ConversationManager(max_history=5)
conv_mgr.add_message("user_123", "user", "怎么退货？")
conv_mgr.add_message("user_123", "assistant", "您可以在订单页面申请退货...")
```

## 7. 完整系统集成（LangChain实现）

```python
from langchain.chains import RetrievalQA
from langchain_community.llms import ChatGLM
from langchain.prompts import PromptTemplate

class CustomerServiceBot:
    """端到端智能客服机器人"""

    def __init__(self, llm, vectorstore, cross_encoder):
        self.llm = llm
        self.vectorstore = vectorstore
        self.cross_encoder = cross_encoder
        self.conv_mgr = ConversationManager()

    def chat(self, session_id: str, user_message: str) -> dict:
        """处理单轮对话"""

        # Step 1: 意图识别
        intent = identify_intent(user_message, self.llm)

        # Step 2: 根据意图选择处理策略
        if intent == "闲聊":
            answer = self._handle_chitchat(user_message)

        elif intent == "投诉":
            answer = self._handle_complaint(user_message)

        else:
            # 咨询/售后/订单查询 -> RAG处理
            answer = self._handle_rag(session_id, user_message)

        # Step 3: 更新对话历史
        self.conv_mgr.add_message(session_id, "user", user_message)
        self.conv_mgr.add_message(session_id, "assistant", answer["answer"])

        return answer

    def _handle_rag(self, session_id: str, question: str):
        """RAG处理逻辑"""
        history = self.conv_mgr.get_history(session_id)
        return generate_answer(question, history, self.llm)

    def _handle_chitchat(self, message: str):
        """闲聊处理"""
        prompt = f"请友好地回复用户的闲聊：{message}"
        response = self.llm.invoke(prompt)
        return {"answer": response.content, "sources": []}

    def _handle_complaint(self, message: str):
        """投诉处理：安抚 + 转接人工"""
        response = ("非常抱歉给您带来了不好的体验。"
                    "我已经记录了您的问题，客服专员会在24小时内与您联系。"
                    "请问还有其他需要帮助的吗？")
        return {"answer": response, "sources": []}
```

```python
def evaluate_retrieval(test_queries: list, ground_truth_ids: list):
    """评估检索效果"""
    recall_scores = []
    mrr_scores = []

    for query, gt_id in zip(test_queries, ground_truth_ids):
        results = search_similar(query, k=5)
        retrieved_ids = [r["metadata"].get("id") for r in results]

        # Recall@5
        recall = 1.0 if gt_id in retrieved_ids else 0.0
        recall_scores.append(recall)

        # MRR
        try:
            rank = retrieved_ids.index(gt_id) + 1
            mrr_scores.append(1.0 / rank)
        except ValueError:
            mrr_scores.append(0.0)

    print(f"Recall@5: {np.mean(recall_scores):.4f}")
    print(f"MRR: {np.mean(mrr_scores):.4f}")
```

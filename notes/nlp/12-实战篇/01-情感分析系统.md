# 01 - 情感分析系统

情感分析（Sentiment Analysis）是NLP中最经典的应用之一，广泛应用于电商评价分析、舆情监控、用户反馈处理等场景。本篇将带领你从零构建一个完整的中文情感分析系统，涵盖数据处理、模型训练、API部署和前端展示全流程。

---

## 1. 项目概述

我们构建的系统具备以下核心能力：

- **三分类情感判断**：正面（Positive）、负面（Negative）、中性（Neutral）
- **多领域适配**：支持外卖、电影、电商评论
- **BERT微调模型**：基于 `bert-base-chinese` 进行Fine-tuning
- **高可用部署**：FastAPI + Docker + Web界面


### 2.2 数据清洗与预处理

```python
import re
import jieba
from typing import List, Tuple

def clean_text(text: str) -> str:
    """清洗文本：去除特殊字符、URL、多余空格"""
    # 去除URL
    text = re.sub(r'https?://\S+', '', text)
    # 去除HTML标签
    text = re.sub(r'<[^>]+>', '', text)
    # 去除特殊字符（保留中英文和标点）
    text = re.sub(r'[^\u4e00-\u9fa5a-zA-Z0-9，。！？、；：""''（）\s]', '', text)
    # 合并多余空格
    text = re.sub(r'\s+', ' ', text).strip()
    return text

def preprocess_data(data: List[dict]) -> List[dict]:
    """完整预处理流水线"""
    label_map = {"positive": 0, "negative": 1, "neutral": 2}
    processed = []
    for item in data:
        text = clean_text(item["text"])
        if len(text) < 2:  # 过滤过短文本
            continue
        processed.append({
            "text": text,
            "label": label_map[item["label"]]
        })
    return processed
```

### 2.3 Tokenization

```python
from transformers import AutoTokenizer

tokenizer = AutoTokenizer.from_pretrained("bert-base-chinese")

def tokenize_function(examples):
    """将文本转换为模型输入格式"""
```

## 3. 模型方案对比

### 3.1 传统方法：TF-IDF + 分类器

### 3.2 BERT微调（推荐方案）

```python
from transformers import (
    AutoModelForSequenceClassification,
    TrainingArguments,
    Trainer,
    DataCollatorWithPadding
)

# 加载预训练模型，微调为三分类
model = AutoModelForSequenceClassification.from_pretrained(
    "bert-base-chinese",
    num_labels=3,
    id2label={0: "positive", 1: "negative", 2: "neutral"},
    label2id={"positive": 0, "negative": 1, "neutral": 2}
)
```

### 3.3 方案对比

| 方法 | 优点 | 缺点 | 准确率（典型） |
|------|------|------|----------------|
| TF-IDF + SVM | 训练快速，可解释性强 | 忽略语序，泛化差 | ~82% |
| TextCNN | 卷积捕捉n-gram | 长距离依赖弱 | ~86% |
| LSTM/BiLSTM | 处理序列信息 | 训练慢，并行差 | ~88% |
| BERT微调 | **语义理解强，精度最高** | 资源消耗大 | **~93%** |

```python
import torch
from torch.utils.data import Dataset
from datasets import load_dataset, DatasetDict
import json

class SentimentDataset(Dataset):
    """自定义PyTorch数据集"""
    def __init__(self, texts, labels, tokenizer, max_length=128):
        self.texts = texts
        self.labels = labels
        self.tokenizer = tokenizer
        self.max_length = max_length

    def __len__(self):
        return len(self.texts)

    def __getitem__(self, idx):
        text = str(self.texts[idx])
        label = self.labels[idx]

        encoding = self.tokenizer(
            text,
            truncation=True,
            padding="max_length",
            max_length=self.max_length,
            return_tensors="pt"
        )

        return {
            "input_ids": encoding["input_ids"].flatten(),
            "attention_mask": encoding["attention_mask"].flatten(),
            "labels": torch.tensor(label, dtype=torch.long)
        }
```

## 5. 模型部署：FastAPI服务

### 5.1 API服务代码

```python
# app.py
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from transformers import pipeline
import uvicorn

app = FastAPI(title="情感分析API", version="1.0.0")

# 加载模型为推理Pipeline
sentiment_pipeline = pipeline(
    "sentiment-analysis",
    model="./models/sentiment_best",
    tokenizer="./models/sentiment_best",
    device=0  # GPU: 0, CPU: -1
)

class SentimentRequest(BaseModel):
    text: str
    return_probs: bool = False

class SentimentResponse(BaseModel):
    text: str
    sentiment: str
    confidence: float
    probabilities: dict = None

@app.post("/predict", response_model=SentimentResponse)
async def predict_sentiment(request: SentimentRequest):
    """预测单条文本的情感"""
    try:
        result = sentiment_pipeline(request.text)[0]
        response = SentimentResponse(
            text=request.text,
            sentiment=result["label"],
            confidence=round(result["score"], 4)
        )
        return response
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/predict_batch")
async def predict_batch(texts: list[str]):
    """批量预测"""
    results = sentiment_pipeline(texts)
    return [
        {"text": t, "sentiment": r["label"], "confidence": round(r["score"], 4)}
        for t, r in zip(texts, results)
    ]

@app.get("/health")
async def health_check():
    return {"status": "healthy", "model": "bert-base-chinese-sentiment"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

## 7. 性能评估

### 7.1 评估指标

```python
# 加载测试集进行评估
import time

def benchmark_model(pipeline, test_texts, batch_size=32):
    """性能基准测试"""
    start_time = time.time()

    total = len(test_texts)
    results = []

    for i in range(0, total, batch_size):
        batch = test_texts[i:i+batch_size]
        results.extend(pipeline(batch))

    elapsed = time.time() - start_time
    throughput = total / elapsed

    print(f"总样本数: {total}")
    print(f"总耗时: {elapsed:.2f}s")
    print(f"吞吐量: {throughput:.1f} 样本/秒")

    return results
```

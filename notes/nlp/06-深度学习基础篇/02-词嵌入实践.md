# 词嵌入实践

上一篇介绍了 PyTorch 的基础和 `nn.Embedding` 的用法。在实际 NLP 任务中，我们很少从零开始训练词向量——**预训练词向量**（Word2Vec、GloVe、FastText 等）包含从海量语料中学到的丰富语义信息，能显著提升模型效果。本篇将系统讲解如何加载、使用、微调预训练词向量，并通过 TensorBoard 进行可视化。

---

## 一、nn.Embedding 的创建与使用

### 1.1 基本创建方式

```python
import torch
import torch.nn as nn

# ===== 方式1：随机初始化（从零训练） =====
# 权重从 N(0, 1) 随机采样
emb1 = nn.Embedding(num_embeddings=10000, embedding_dim=300)
print(f"随机初始化 均值={emb1.weight.mean():.4f}, 标准差={emb1.weight.std():.4f}")

# ===== 方式2：指定 padding_idx（推荐） =====
# padding_idx 对应的向量始终为0，且不被梯度更新
emb2 = nn.Embedding(10000, 300, padding_idx=0)
print(f"padding_idx=0 的向量: {emb2.weight[0][:5]}")  # 全0

# ===== 方式3：用预训练权重初始化 =====
pretrained_weights = torch.randn(10000, 300) * 0.1  # 模拟预训练向量
emb3 = nn.Embedding.from_pretrained(pretrained_weights, freeze=False)
print(f"from_pretrained 形状: {emb3.weight.shape}")
```

### 1.2 三种创建方式对比

```
┌─────────────────┬──���───────────────────────┬────────────────────┐
│      方式       │         说明             │     适用场景       │
├─────────────────┼──────────────────────────┼────────────────────┤
│ 随机初始化       │ 从 N(0,1) 采样           │ 数据量大、从头训练 │
│ padding_idx=0   │ 指定索引向量恒为0        │ 处理变长序列必备   │
│ from_pretrained │ 用预训练权重初始化        │ 迁移学习、数据量小 │
└─────────────────┴──────────────────────────┴────────────────────┘
```

### 1.3 查表操作详解

```python
import torch
import torch.nn as nn

emb = nn.Embedding(10, 4, padding_idx=0)

# ===== 单句查表 =====
seq = torch.tensor([1, 3, 5, 0, 0])  # 最后两个是 padding
vec = emb(seq)
print(f"单句: 输入{seq.shape} → 输出{vec.shape}")  # (5,) → (5, 4)
print(f"padding位置向量: {vec[-1]}")  # 全0

# ===== 批量查表 =====
batch = torch.tensor([
    [1, 2, 3, 0],
    [4, 5, 6, 7],
])
batch_vec = emb(batch)
print(f"\n批量: 输入{batch.shape} → 输出{batch_vec.shape}")  # (2,4) → (2,4,4)

# ===== 梯度传播：只有查过的行有梯度 =====
batch.clamp_(min=1)  # 去掉 padding
out = emb(batch).sum()
out.backward()
grad_nonzero = (emb.weight.grad.sum(dim=1) != 0).sum().item()
print(f"\n有梯度的词向量数: {grad_nonzero} / 10")
```

## 二、加载预训练 Word2Vec / GloVe

### 2.1 词向量格式

```
GloVe / Word2Vec 文本格式（每行一个词）:

  词 v1 v2 v3 ... v300
  词 v1 v2 v3 ... v300
  ...

示例 (glove.6B.50d.txt, 前5行):
  the 0.418 0.24968 -0.41242 0.1217 0.34527 ...
  , -0.013139 0.23684 -0.16899 0.40951 0.24126 ...
  . 0.15164 0.30177 -0.16763 0.17684 0.31719 ...
  of 0.70853 0.57088 -0.47160 0.18008 0.64434 ...
  to 0.68047 -0.03926 0.30186 -0.17792 0.42962 ...

第一行通常是 "400000 50"（词数 维度），但 GloVe 没有表头
```

### 2.2 加载 GloVe 向量

```python
import numpy as np
import torch
import torch.nn as nn

def load_glove(path, vocab_limit=None):
    """加载 GloVe 词向量文件
    path: glove.6B.50d.txt 路径
    vocab_limit: 只加载前 N 个词（测试用）
    返回: word2idx, embedding_matrix
    """
    word2idx = {'<pad>': 0, '<unk>': 1}
    vectors = [[0.0] * 50, [0.0] * 50]  # pad 和 unk 用零向量

    with open(path, 'r', encoding='utf-8') as f:
        for i, line in enumerate(f):
            if vocab_limit and i >= vocab_limit:
                break
            parts = line.rstrip().split(' ')
            word = parts[0]
            vec = np.array(parts[1:], dtype=np.float32)
            word2idx[word] = len(word2idx)
            vectors.append(vec)

    embedding_matrix = np.vstack(vectors)
    return word2idx, embedding_matrix

# ===== 实际使用（需下载 glove 文件） =====
# word2idx, emb_matrix = load_glove('glove.6B.50d.txt', vocab_limit=50000)
# print(f"加载 {len(word2idx)} 个词, 维度 {emb_matrix.shape[1]}")

# ===== 这里用随机数据模拟 =====
print("=== 模拟加载预训练向量 ===")
np.random.seed(42)
emb_matrix = np.random.randn(100, 50).astype(np.float32) * 0.1
word2idx = {f'word_{i}': i for i in range(100)}
emb_matrix[0] = 0  # pad 为零向量
print(f"词表大小: {len(word2idx)}, 嵌入矩阵形状: {emb_matrix.shape}")
```

### 2.3 加载 gensim 训练的 Word2Vec

```python
# 假设已用 gensim 训练好 Word2Vec 模型
# from gensim.models import Word2Vec
# w2v_model = Word2Vec.load('my_w2v.model')

def build_embedding_matrix_from_gensim(w2v_model, vocab, embed_dim):
    """根据任务词表，从 gensim 模型中提取对应向量"""
    matrix = np.zeros((len(vocab), embed_dim))
    hits, misses = 0, 0
    for word, idx in vocab.items():
        if word in w2v_model.wv:
            matrix[idx] = w2v_model.wv[word]
            hits += 1
        else:
            # 词不在预训练模型中，随机初始化
            matrix[idx] = np.random.randn(embed_dim) * 0.1
            misses += 1
    print(f"命中: {hits} / {len(vocab)}, 未命中: {misses}")
    return matrix

# 模拟示例
print("=== 从 gensim 模型提取词向量 ===")
class FakeW2V:
    def __init__(self):
        self.wv = {f'word_{i}': np.random.randn(50) * 0.1 for i in range(80)}
fake_model = FakeW2V()
matrix = build_embedding_matrix_from_gensim(fake_model, word2idx, embed_dim=50)
print(f"嵌入矩阵形状: {matrix.shape}")
```

### 2.4 用预训练向量初始化 nn.Embedding

```python
import torch.nn as nn

# ===== 关键步骤：from_pretrained =====
# 将 numpy 矩阵转为 Tensor，再用 from_pretrained 初始化
weight_tensor = torch.FloatTensor(matrix)
embedding = nn.Embedding.from_pretrained(weight_tensor, padding_idx=0, freeze=False)
# freeze=False 表示允许在下游任务中微调
# freeze=True  表示冻结，训练时不更新

print(f"Embedding 形状: {embedding.weight.shape}")
print(f"参数量: {embedding.weight.numel()}")

# 验证：查表结果应等于原始矩阵对应行
idx = torch.tensor([5, 10, 15])
lookup = embedding(idx).detach().numpy()
expected = matrix[[5, 10, 15]]
print(f"加载是否正确: {np.allclose(lookup, expected)}")
```

## 三、词嵌入层在模型中的位置

### 3.1 典型 NLP 模型架构

```
典型的文本分类模型:

  输入: 词索引序列  (batch, seq_len)
    │
    │ ┌─────────────────────────────────┐
    │ │  Embedding Layer (词嵌入层)     │ ← 把索引变成稠密向量
    │ │  权重: (vocab_size, embed_dim)  │    是整个模型第一个"可学习"组件
    │ └────────────┬────────────────────┘
    │              │
    │              ▼
    │    (batch, seq_len, embed_dim)
    │              │
    │              │ ┌────────────────────┐
    │              │ │ Encoder Layer      │ ← 提取语义特征
    │              │ │ (RNN/CNN/Attention)│    可有多层
    │              │ └─────────┬──────────┘
    │              │           │
    │              │           ▼
    │              │   (batch, hidden_dim)
    │              │           │
    │              │           │ ┌──────────────┐
    │              │           │ │ Classifier   │ ← 分类/回归输出
    │              │           │ │ (Linear)     │
    │              │           │ └──────┬───────┘
    │              │           │        │
    └──────────────┴───────────┴────────▼
                              logits (batch, num_classes)
```

### 3.2 Embedding 是模型的"底层地基"

```
Embedding 的作用：
  ┌─────────────────────────────────────────────────────┐
  │  1. 把离散符号（词）变成连续向量                     │
  │  2. 向量空间中，相似词的距离近                       │
  │  3. 向量可参与梯度下降，实现端到端学习               │
  │  4. 预训练向量把"通用语言知识"迁移到下游任务         │
  └─────────────────────────────────────────────────────┘

下游模型的"金字塔":
                    ┌──────────────┐
                    │  分类/生成    │
                    └──────┬───────┘
                    ┌──────┴───────┐
                    │ Transformer  │
                    └──────┬───────┘
                    ┌──────┴───────┐
                    │  RNN / CNN   │
                    └──────┬───────┘
                    ┌──────┴───────┐
                    │  Embedding   │  ← 地基：所有上层都依赖它
                    └──────────────┘
```

## 四、词嵌入微调 vs 冻结

### 4.1 两种策略对比

```
策略1: 冻结 (freeze=True)
  - 训练时不更新 Embedding 权重
  - 适合: 数据量小、预训练质量高、任务简单
  - 优势: 训练快、防过拟合、显存占用小

策略2: 微调 (freeze=False)
  - 训练时更新 Embedding 权重
  - 适合: 数据量大、领域专业词汇多、任务复杂
  - 优势: 向量更贴合下游任务；风险: 可能过拟合

┌──────────────┬────────────────────┬────────────────────┐
│    策略      │      冻结           │      微调          │
├──────────────┼────────────────────┼────────────────────┤
│ 参数更新     │ 否                 │ 是                 │
│ 训练速度     │ 快                 │ 慢                 │
│ 显存占用     │ 小                 │ 大                 │
│ 数据需求     │ 小数据即可         │ 需较大数据         │
│ 典型场景     │ 通用领域迁移       │ 专业领域适配       │
└──────────────┴────────────────────┴────────────────────┘
```

### 4.2 代码实现

```python
import torch
import torch.nn as nn
import numpy as np

# ===== 策略1：完全冻结 =====
matrix = np.random.randn(1000, 100).astype(np.float32)
emb_freeze = nn.Embedding.from_pretrained(
    torch.FloatTensor(matrix), padding_idx=0, freeze=True
)
print(f"冻结 - requires_grad: {emb_freeze.weight.requires_grad}")  # False

# ===== 策略2：完全微调 =====
emb_finetune = nn.Embedding.from_pretrained(
    torch.FloatTensor(matrix), padding_idx=0, freeze=False
)
print(f"微调 - requires_grad: {emb_finetune.weight.requires_grad}")  # True

# ===== 策略3：部分微调（差分学习率，进阶） =====
class TextModel(nn.Module):
    def __init__(self, matrix, hidden_dim, num_classes):
        super().__init__()
        self.embedding = nn.Embedding.from_pretrained(
            torch.FloatTensor(matrix), padding_idx=0, freeze=False
        )
        self.encoder = nn.Linear(matrix.shape[1], hidden_dim)
        self.fc = nn.Linear(hidden_dim, num_classes)

    def forward(self, x):
        emb = self.embedding(x).mean(dim=1)
        h = torch.relu(self.encoder(emb))
        return self.fc(h)

model = TextModel(matrix, hidden_dim=64, num_classes=3)

# 给 Embedding 和上层设置不同的学习率
optimizer = torch.optim.Adam([
    {'params': model.embedding.parameters(), 'lr': 0.0001},  # 小lr微调
    {'params': model.encoder.parameters(), 'lr': 0.001},
    {'params': model.fc.parameters(), 'lr': 0.001},
])
print(f"\n参数组学习率: {[g['lr'] for g in optimizer.param_groups]}")

# ===== 动态切换：先冻结预热，再解冻微调 =====
print("\n=== 动态切换策略 ===")
model.embedding.weight.requires_grad = False  # 先冻结
print(f"阶段1（预热）- requires_grad: {model.embedding.weight.requires_grad}")
# ... 训练若干轮 ...
model.embedding.weight.requires_grad = True   # 解冻微调
print(f"阶段2（微调）- requires_grad: {model.embedding.weight.requires_grad}")
```

## 五、可视化词嵌入（TensorBoard Embedding Projector）

### 5.1 可视化原理

```
TensorBoard Embedding Projector:

  原始词向量 (300维)              投影到2D/3D (PCA 或 t-SNE)
  ┌──────────────────┐           ┌───────────────────────┐
  │ word_1 [v1..v300]│           │      ·狗              │
  │ word_2 [v1..v300]│   PCA/    │   ·猫    ·动物        │
  │ word_3 [v1..v300]│  t-SNE    │      ·鱼              │
  │ ...              │  ─────→   │                       │
  │ word_N [v1..v300]│           │ ·汽车  ·飞机  ·火车   │
  └──────────────────┘           └───────────────────────┘

  功能:
  - 搜索特定词，查看其邻居（最近义词）
  - 观察词簇（同义词聚集在一起）
  - 发现语义异常（离群点）
```

### 5.2 代码实现

```python
import torch
import torch.nn as nn
from torch.utils.tensorboard import SummaryWriter
import numpy as np

# ===== 准备模拟数据 =====
# 假设这是训练好的词嵌入
vocab_size = 100
embed_dim = 50
embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=0)

# 词标签（对应每个索引）
words = [f'词_{i}' for i in range(vocab_size)]

# ===== 写入 TensorBoard =====
writer = SummaryWriter(log_dir='./runs/embedding_demo')

# 添加 embedding 可视化
writer.add_embedding(
    mat=embedding.weight.data,    # (vocab_size, embed_dim)
    metadata=words,               # 每行的标签（词）
    global_step=0,
)
writer.close()
print("已写入 TensorBoard 日志到 ./runs/embedding_demo")

# ===== 启动查看 =====
# 命令行运行: tensorboard --logdir=./runs
# 浏览器打开: http://localhost:6006
# 选择 "Projector" 标签页
```

### 5.3 可视化训练过程

```python
import torch
import torch.nn as nn
from torch.utils.tensorboard import SummaryWriter

# 训练过程中定期记录 embedding 变化
writer = SummaryWriter('./runs/embedding_training')

# 模拟一个训练过程
vocab_size, embed_dim = 50, 10
embedding = nn.Embedding(vocab_size, embed_dim)
optimizer = torch.optim.Adam(embedding.parameters(), lr=0.1)
words = [f'词_{i}' for i in range(vocab_size)]

for step in range(10):
    # 模拟损失：让相近索引的词向量接近
    idx = torch.randint(0, vocab_size, (8,))
    vec = embedding(idx)
    target = embedding(idx.roll(1))  # 相邻索引为目标
    loss = ((vec - target) ** 2).mean()

    optimizer.zero_grad()
    loss.backward()
    optimizer.step()

    # 每5步记录一次 embedding
    if step % 5 == 0:
        writer.add_embedding(
            mat=embedding.weight.data,
            metadata=words,
            global_step=step,
        )
        print(f"Step {step}: loss={loss.item():.4f}")

writer.close()
print("\n训练完成，可用 tensorboard --logdir=./runs/embedding_training 查看")
```

## 六、完整示例：带预训练词嵌入的文本分类

```python
import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader
from collections import Counter
import numpy as np

torch.manual_seed(42)
np.random.seed(42)

# ===== 1. 数据准备 =====
texts = [
    "这部 电影 真的 太 好看 了", "画面 精美 值得 一看",
    "演员 演技 棒 剧情棒", "推荐 大家 都 去 看",
    "这部 电影 真的 太 烂 了", "剧情 差 演员 演技 差",
    "浪费 时间 无聊", "不 推荐 太 差 了",
] * 10
labels = [1, 1, 1, 1, 0, 0, 0, 0] * 10

# ===== 2. 构建词表 =====
counter = Counter()
for t in texts:
    counter.update(t.split())
vocab = {'<pad>': 0, '<unk>': 1}
for w, _ in counter.most_common():
    vocab[w] = len(vocab)

# ===== 3. 模拟预训练词向量 =====
embed_dim = 50
print(f"词表大小: {len(vocab)}, 嵌入维度: {embed_dim}")
# 实际场景：从 GloVe/Word2Vec 文件加载，这里随机模拟
pretrained = np.random.randn(len(vocab), embed_dim).astype(np.float32) * 0.1
pretrained[0] = 0  # pad 为零向量

# ===== 4. 模型定义（带预训练词嵌入） =====
class TextClassifier(nn.Module):
    def __init__(self, pretrained_weights, hidden_dim, num_classes, freeze=False):
        super().__init__()
        self.embedding = nn.Embedding.from_pretrained(
            torch.FloatTensor(pretrained_weights),
            padding_idx=0,
            freeze=freeze,  # 是否冻结词嵌入
        )
        embed_dim = pretrained_weights.shape[1]
        self.fc1 = nn.Linear(embed_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, num_classes)
        self.dropout = nn.Dropout(0.3)

    def forward(self, x):
        mask = (x != 0).float().unsqueeze(-1)
        emb = self.embedding(x) * mask
        pooled = emb.sum(dim=1) / mask.sum(dim=1).clamp(min=1)
        h = torch.relu(self.fc1(pooled))
        h = self.dropout(h)
        return self.fc2(h)

# ===== 5. 数据加载 =====
max_len = 10
def text_to_seq(text):
    idx = [vocab.get(w, 1) for w in text.split()][:max_len]
    return idx + [0] * (max_len - len(idx))

class DS(Dataset):
    def __init__(self, texts, labels):
        self.x = torch.tensor([text_to_seq(t) for t in texts])
        self.y = torch.tensor(labels)
    def __len__(self): return len(self.y)
    def __getitem__(self, i): return self.x[i], self.y[i]

loader = DataLoader(DS(texts, labels), batch_size=16, shuffle=True)

# ===== 6. 训练对比：冻结 vs 微调 =====
def train_and_eval(freeze, name):
    torch.manual_seed(42)
    model = TextClassifier(pretrained, hidden_dim=32, num_classes=2, freeze=freeze)
    opt = torch.optim.Adam(filter(lambda p: p.requires_grad, model.parameters()), lr=0.005)
    crit = nn.CrossEntropyLoss()
    for epoch in range(20):
        total_loss, correct, total = 0, 0, 0
        for bx, by in loader:
            opt.zero_grad()
            logits = model(bx)
            loss = crit(logits, by)
            loss.backward()
            opt.step()
            total_loss += loss.item() * len(by)
            correct += (logits.argmax(1) == by).sum().item()
            total += len(by)
        if epoch % 5 == 0:
            print(f"  [{name}] Epoch {epoch}: loss={total_loss/total:.4f}, acc={correct/total:.4f}")
    return model

print("=== 冻结词嵌入 ===")
train_and_eval(freeze=True, name="freeze")

print("\n=== 微调词嵌入 ===")
train_and_eval(freeze=False, name="finetune")
```

## 小结

| 要点 | 内容 |
|------|------|
| nn.Embedding | 词索引→稠密向量的查表层，NLP 模型第一层 |
| 创建方式 | 随机初始化 / `padding_idx` / `from_pretrained` |
| 预训练来源 | GloVe（`.txt`格式）、Word2Vec（gensim）、FastText |
| 加载步骤 | 读文件 → 对齐词表 → 命中/未命中统计 → `from_pretrained` |
| 在模型位置 | 输入层之后、Encoder 之前，是模型的"地基" |
| 冻结策略 | 数据少/通用任务→冻结；数据多/专业领域→微调 |
| 差分学习率 | 给 Embedding 设更小 lr，防止破坏预训练知识 |
| 可视化 | `SummaryWriter.add_embedding()` + TensorBoard Projector |
| 投影方法 | PCA（线性降维）/ t-SNE（非线性，保局部结构） |

# 工具定义与绑定

工具（Tool）让 LLM 超越纯文本对话——能够搜索信息、执行代码、查询数据库、调用 API。本篇讲解 LangGraph 中工具的定义和使用。

---

## 什么是工具调用

```
用户: "北京明天天气怎么样？"
         │
         ▼
    ┌──────────┐
    │   LLM    │ → "我需要调用天气查询工具"
    │  推理    │ → tool_call: {name: "get_weather", args: {city: "北京"}}
    └──────────┘
         │
         ▼
    ┌──────────────┐
    │ 工具执行      │ → get_weather("北京") → "晴，25°C"
    └──────────────┘
         │
         ▼
    ┌──────────┐
    │   LLM    │ → "北京明天天气晴朗，气温25度"
    │  总结    │
    └──────────┘
```

## 工具定义方式

### 方式一：@tool 装饰器（推荐）

```python
from langchain_core.tools import tool

@tool
def search_weather(city: str) -> str:
    """查询指定城市的天气

    Args:
        city: 城市名称，如"北京"、"上海"
    """
    # 实际实现
    weathers = {"北京": "晴 25°C", "上海": "多云 28°C"}
    return weathers.get(city, f"暂无{city}的天气数据")

@tool
def calculate(expression: str) -> str:
    """计算数学表达式

    Args:
        expression: 数学表达式，如 "2+3*4"
    """
    try:
        result = eval(expression)
        return f"结果: {result}"
    except:
        return "无法计算"

@tool
def search_web(query: str, max_results: int = 5) -> str:
    """搜索互联网获取信息

    Args:
        query: 搜索关键词
        max_results: 最大结果数（默认5）
    """
    # 调用搜索API
    results = search_api(query, max_results)
    return "\n".join(results)
```

### @tool 的关键要素

```python
@tool
def my_tool(param1: str, param2: int) -> str:
    """工具的简短描述（LLM 会看到这个！）

    详细的参数说明也在 docstring 中。

    Args:
        param1: 第一个参数的描述
        param2: 第二个参数的描述
    """
    # 实现
    pass
```

| 要素 | 说明 | 重要性 |
|------|------|--------|
| 函数名 | 工具名称 | ⭐⭐⭐⭐⭐ |
| docstring | LLM 用来理解工具功能 | ⭐⭐⭐⭐⭐ |
| 参数类型注解 | 告诉 LLM 参数类型 | ⭐⭐⭐⭐⭐ |
| 参数描述 | 在 docstring 中 | ⭐⭐⭐⭐ |
| 返回类型 | 输出类型 | ⭐⭐⭐ |

### 方式二：StructuredTool

```python
from langchain_core.tools import StructuredTool
from pydantic import BaseModel, Field

class SearchInput(BaseModel):
    query: str = Field(description="搜索关键词")
    max_results: int = Field(default=5, description="最大结果数")

def search_func(query: str, max_results: int = 5) -> str:
    return search_api(query, max_results)

search_tool = StructuredTool.from_function(
    func=search_func,
    name="search",
    description="搜索互联网",
    args_schema=SearchInput
)
```

### 方式三：直接 Tool 类

```python
from langchain_core.tools import Tool

simple_tool = Tool(
    name="get_time",
    description="获取当前时间",
    func=lambda: str(datetime.now())
)
```

## 绑定工具到 LLM

```python
from langchain_openai import ChatOpenAI

llm = ChatOpenAI(model="gpt-4o")

# 定义工具
tools = [search_weather, calculate, search_web]

# 绑定
llm_with_tools = llm.bind_tools(tools)

# 现在 LLM 可以决定调用哪些工具
response = llm_with_tools.invoke("北京天气怎么样？")
print(response.tool_calls)
# [{'name': 'search_weather', 'args': {'city': '北京'}, 'id': 'xxx'}]
```

## 工具调用消息类型

```python
from langchain_core.messages import (
    HumanMessage,
    AIMessage,
    ToolMessage
)

# 1. 用户消息
human_msg = HumanMessage("北京天气怎么样？")

# 2. AI 决定调用工具（AIMessage 带 tool_calls）
ai_msg = llm_with_tools.invoke([human_msg])
# ai_msg.tool_calls = [{"name": "search_weather", "args": {"city": "北京"}, "id": "call_xxx"}]

# 3. 执行工具，返回 ToolMessage
tool_result = search_weather.invoke({"city": "北京"})
tool_msg = ToolMessage(
    content=tool_result,
    tool_call_id=ai_msg.tool_calls[0]["id"]  # 必须关联到对应的 tool_call
)

# 4. 将工具结果发回 LLM，生成最终回答
final_response = llm_with_tools.invoke([human_msg, ai_msg, tool_msg])
print(final_response.content)  # "北京今天晴天，气温25度"
```

### 消息流

```
HumanMessage("北京天气？")
        │
        ▼
   LLM 推理
        │
        ▼
AIMessage(tool_calls=[{name: "search_weather", args: {city: "北京"}}])
        │
        ▼
   执行工具
        │
        ▼
ToolMessage(content="晴 25°C", tool_call_id="call_xxx")
        │
        ▼
   LLM 总结
        │
        ▼
AIMessage("北京今天晴天，气温25度")
```

## 在 LangGraph 中使用工具

### 完整的 Tool-Calling Agent

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_core.tools import tool
from langchain_core.messages import ToolMessage
from langchain_openai import ChatOpenAI


# === 1. 定义工具 ===
@tool
def search(query: str) -> str:
    """搜索互联网信息"""
    return f"搜索结果: {query}的相关信息..."

@tool
def calculate(expression: str) -> str:
    """计算数学表达式"""
    try:
        return str(eval(expression))
    except:
        return "无法计算"

tools = [search, calculate]
llm = ChatOpenAI(model="gpt-4o")
llm_with_tools = llm.bind_tools(tools)


# === 2. 定义状态 ===
class State(TypedDict):
    messages: Annotated[list, add_messages]


# === 3. 定义节点 ===
def agent(state: State) -> dict:
    """LLM 推理节点"""
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response]}

def tool_executor(state: State) -> dict:
    """工具执行节点"""
    last_message = state["messages"][-1]
    results = []

    for tool_call in last_message.tool_calls:
        # 根据工具名执行
        tool_name = tool_call["name"]
        tool_args = tool_call["args"]

        # 查找工具
        for t in tools:
            if t.name == tool_name:
                result = t.invoke(tool_args)
                break
        else:
            result = f"未知工具: {tool_name}"

        results.append(ToolMessage(
            content=str(result),
            tool_call_id=tool_call["id"]
        ))

    return {"messages": results}


# === 4. 路由 ===
def should_use_tools(state: State) -> str:
    last_msg = state["messages"][-1]
    if last_msg.tool_calls:
        return "tools"
    return END


# === 5. 构建图 ===
builder = StateGraph(State)
builder.add_node("agent", agent)
builder.add_node("tools", tool_executor)

builder.add_edge(START, "agent")
builder.add_conditional_edges("agent", should_use_tools, {
    "tools": "tools",
    END: END
})
builder.add_edge("tools", "agent")  # 工具结果回到 agent

graph = builder.compile()


# === 6. 运行 ===
result = graph.invoke({
    "messages": [("user", "搜索一下LangGraph，然后算一下123*456")]
})

for msg in result["messages"]:
    print(f"[{msg.type}]: {msg.content[:100] if msg.content else '(tool_calls)'}")
```

## 使用 ToolExecutor（简化版）

```python
from langgraph.prebuilt import ToolNode

# 用预构建的 ToolNode 代替手动工具执行
tool_node = ToolNode(tools)

builder.add_node("tools", tool_node)
```

或使用完整的预构建 Agent：

```python
from langgraph.prebuilt import create_react_agent

# 一行创建 ReAct Agent
agent = create_react_agent(llm, tools)

result = agent.invoke({
    "messages": [("user", "北京天气怎么样？")]
})
```

## 工具最佳实践

### 1. 清晰的描述

```python
# ❌ 不好
@tool
def search(q: str) -> str:
    """搜索"""
    ...

# ✅ 好
@tool
def search_products(query: str, category: str = "all") -> str:
    """在商品库中搜索商品

    根据关键词搜索商品，可按类目筛选。

    Args:
        query: 搜索关键词，如"红色连衣裙"
        category: 商品类目，如"服装"、"电子产品"，默认全部
    """
    ...
```

### 2. 错误处理

```python
@tool
def divide(a: float, b: float) -> str:
    """除法计算"""
    if b == 0:
        return "错误: 除数不能为零"
    return str(a / b)
```

### 3. 返回结构化数据

```python
@tool
def get_stock_price(symbol: str) -> str:
    """获取股票价格"""
    data = fetch_stock(symbol)
    # 返回结构化的文本
    return f"""
    股票: {data['name']} ({symbol})
    当前价: ¥{data['price']}
    涨跌: {data['change']} ({data['change_percent']}%)
    成交量: {data['volume']}
    """
```

## 小结

| 要点 | 说明 |
|------|------|
| @tool 装饰器 | 最简单的工具定义方式 |
| bind_tools | 将工具绑定到 LLM |
| ToolMessage | 工具执行结果的消息类型 |
| tool_calls | AI消息中的工具调用列表 |
| ToolNode | 预构建的工具执行节点 |
| create_react_agent | 预构建的完整Agent |

## 搜索工具

### Tavily Search（推荐）

```python
# 安装: pip install langchain-tavily
from langchain_tavily import TavilySearch

search = TavilySearch(
    max_results=5,
    api_key="tvly-your-key"  # 或设置 TAVILY_API_KEY 环境变量
)

# 直接使用
results = search.invoke("LangGraph最新版本")
# 作为工具绑定到 Agent
tools = [search]
llm_with_tools = llm.bind_tools(tools)
```

### DuckDuckGo（免费）

```python
# 安装: pip install duckduckgo-search
from langchain_community.tools import DuckDuckGoSearchRun

search = DuckDuckGoSearchRun()
results = search.invoke("Python异步编程")
```

### Wikipedia

```python
from langchain_community.tools import WikipediaQueryRun
from langchain_community.utilities import WikipediaAPIWrapper

wikipedia = WikipediaQueryRun(api_wrapper=WikipediaAPIWrapper(top_k_results=2))
result = wikipedia.invoke("量子计算")
```

## 代码执行工具

### Python REPL

```python
from langchain_experimental.tools import PythonREPLTool

python_tool = PythonREPLTool()

# 可以执行任意 Python 代码
result = python_tool.invoke("print(2**10)")
result = python_tool.invoke("import numpy as np; print(np.mean([1,2,3]))")

# 绑定到 Agent
tools = [python_tool]
```

### 安全的代码执行

```python
from langchain_experimental.tools import ShellTool

shell = ShellTool()
# ⚠️ 危险！仅用于可信环境
```

## 数据库工具

### SQL Database

```python
# 安装: pip install langchain-community sqlalchemy
from langchain_community.utilities import SQLDatabase
from langchain_community.tools import QuerySQLDataBaseTool

# 连接数据库
db = SQLDatabase.from_uri("sqlite:///mydb.db")

# 查询工具
query_tool = QuerySQLDataBaseTool(db=db)
result = query_tool.invoke("SELECT * FROM users LIMIT 5")
```

### 完整的 SQL Agent 工具集

```python
from langchain_community.tools import (
    QuerySQLDataBaseTool,
    InfoSQLDatabaseTool,
    ListSQLDatabaseTool,
    QuerySQLCheckerTool
)

db = SQLDatabase.from_uri("postgresql://user:pass@localhost/db")

list_tool = ListSQLDatabaseTool(db=db)          # 列出所有表
schema_tool = InfoSQLDatabaseTool(db=db)        # 获取表结构
execute_tool = QuerySQLDataBaseTool(db=db)      # 执行SQL
checker_tool = QuerySQLCheckerTool(db=db)       # 检查SQL语法

sql_tools = [list_tool, schema_tool, execute_tool, checker_tool]
```

## 向量检索工具

```python
from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain.tools import retriever_tool

# 创建向量存储
vector_store = Chroma(
    embedding_function=OpenAIEmbeddings(),
    collection_name="documents"
)

# 创建检索工具
retrieval_tool = retriever_tool.create_retriever_tool(
    retriever=vector_store.as_retriever(),
    name="search_documents",
    description="在知识库中搜索相关文档"
)
```

## 组合多种工具

```python
"""
组合多种工具的全能 Agent
"""
from langchain_openai import ChatOpenAI
from langchain_tavily import TavilySearch
from langchain_experimental.tools import PythonREPLTool
from langchain_community.tools import WikipediaQueryRun, WikipediaAPIWrapper
from langgraph.prebuilt import create_react_agent

# 收集工具
tools = [
    TavilySearch(max_results=3),                    # 网络搜索
    PythonREPLTool(),                                # Python执行
    WikipediaQueryRun(api_wrapper=WikipediaAPIWrapper()),  # 维基百科
]

llm = ChatOpenAI(model="gpt-4o")

# 创建全能 Agent
agent = create_react_agent(llm, tools)

# 测试
result = agent.invoke({
    "messages": [("user", "搜索LangGraph最新信息，计算2^20，并查一下图论的历史")]
})
```

## 工具选择策略

### 按需提供

```python
# ❌ 不好：一次提供所有工具
all_tools = [search, calc, code, db, file, wiki, ...]
agent = create_react_agent(llm, all_tools)
# 太多工具会让 LLM 困惑

# ✅ 好：根据任务选择工具
def select_tools(task_type: str) -> list:
    if task_type == "research":
        return [web_search, wiki_search, doc_retriever]
    elif task_type == "coding":
        return [python_repl, file_read, file_write]
    elif task_type == "analysis":
        return [sql_query, python_repl, chart_generator]
```

### 动态工具选择

```python
def dynamic_tool_node(state: State) -> dict:
    """根据对话动态选择工具"""
    last_msg = state["messages"][-1].content.lower()

    if any(kw in last_msg for kw in ["搜索", "查询", "最新"]):
        relevant_tools = [web_search, wiki_search]
    elif any(kw in last_msg for kw in ["计算", "代码", "运行"]):
        relevant_tools = [python_repl, calculator]
    else:
        relevant_tools = [general_qa]

    # 只绑定相关工具
    state["available_tools"] = relevant_tools
    return state
```

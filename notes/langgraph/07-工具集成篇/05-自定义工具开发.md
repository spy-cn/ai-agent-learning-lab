# 自定义工具开发

当预构建工具不够用时，你需要开发自己的工具。本篇讲解复杂工具、异步工具、工具链的设计模式。

---

## 工具设计原则

```
┌──────────────────────────────────────────────┐
│              好工具的特征                       │
├──────────────────────────────────────────────┤
│ 1. 功能单一明确（一个工具做一件事）              │
│ 2. 描述清晰（LLM 能理解何时使用）               │
│ 3. 参数精简（只保留必要参数）                   │
│ 4. 健壮（错误时返回有意义的错误信息）             │
│ 5. 快速（避免长时间阻塞）                      │
│ 6. 安全（不暴露敏感信息）                      │
└──────────────────────────────────────────────┘
```

## 复杂工具示例

### 数据库查询工具

```python
from langchain_core.tools import tool
from pydantic import BaseModel, Field
import sqlite3

class QueryInput(BaseModel):
    table: str = Field(description="表名: users, orders, products")
    conditions: str = Field(default="", description="查询条件，如 'age > 25'")
    limit: int = Field(default=10, description="返回行数")

@tool(args_schema=QueryInput)
def query_database(table: str, conditions: str = "", limit: int = 10) -> str:
    """查询数据库获取数据

    支持的表: users（用户）, orders（订单）, products（商品）
    """
    # 安全检查：白名单
    allowed_tables = {"users", "orders", "products"}
    if table not in allowed_tables:
        return f"错误: 不允许查询表 '{table}'"

    conn = sqlite3.connect("app.db")
    cursor = conn.cursor()

    try:
        sql = f"SELECT * FROM {table}"
        if conditions:
            sql += f" WHERE {conditions}"
        sql += f" LIMIT {limit}"

        cursor.execute(sql)
        columns = [desc[0] for desc in cursor.description]
        rows = cursor.fetchall()

        # 格式化结果
        result = f"查询到 {len(rows)} 条记录:\n"
        result += " | ".join(columns) + "\n"
        result += "-" * 50 + "\n"
        for row in rows[:5]:  # 只预览前5条
            result += " | ".join(str(v) for v in row) + "\n"

        return result

    except Exception as e:
        return f"查询出错: {str(e)}"
    finally:
        conn.close()
```

### 文件处理工具

```python
@tool
def analyze_document(file_path: str, analysis_type: str = "summary") -> str:
    """分析文档内容

    Args:
        file_path: 文件路径
        analysis_type: 分析类型 - summary（摘要）, keywords（关键词）, sentiment（情感）
    """
    import os

    # 安全检查
    if not os.path.exists(file_path):
        return f"错误: 文件不存在 '{file_path}'"

    # 读取文件
    _, ext = os.path.splitext(file_path)
    try:
        if ext == ".txt":
            with open(file_path, "r", encoding="utf-8") as f:
                content = f.read()
        elif ext == ".pdf":
            # 需要 PyPDF2
            from PyPDF2 import PdfReader
            reader = PdfReader(file_path)
            content = "\n".join(page.extract_text() for page in reader.pages)
        else:
            return f"不支持的文件类型: {ext}"

        # 分析
        if analysis_type == "summary":
            summary = llm.invoke(f"总结以下内容（200字内）:\n{content[:2000]}")
            return summary.content
        elif analysis_type == "keywords":
            keywords = llm.invoke(f"提取5个关键词:\n{content[:2000]}")
            return keywords.content
        elif analysis_type == "sentiment":
            sentiment = llm.invoke(f"分析情感倾向:\n{content[:2000]}")
            return sentiment.content
        else:
            return f"未知分析类型: {analysis_type}"

    except Exception as e:
        return f"分析出错: {str(e)}"
```

### API 调用工具

```python
import requests

@tool
def call_api(method: str, url: str, params: str = "", body: str = "") -> str:
    """调用 REST API

    Args:
        method: HTTP 方法 (GET/POST/PUT/DELETE)
        url: API 地址
        params: 查询参数 JSON (可选)
        body: 请求体 JSON (可选)
    """
    import json

    try:
        kwargs = {"timeout": 10}

        if params:
            kwargs["params"] = json.loads(params)

        if body:
            kwargs["json"] = json.loads(body)

        response = requests.request(method.upper(), url, **kwargs)

        return f"""
        状态码: {response.status_code}
        响应: {response.text[:500]}
        """

    except json.JSONDecodeError:
        return "错误: params 或 body 不是有效的 JSON"
    except requests.Timeout:
        return "错误: 请求超时"
    except Exception as e:
        return f"错误: {str(e)}"
```

## 异步工具

```python
import aiohttp
import asyncio

@tool
async def async_web_scraper(url: str, selector: str = "p") -> str:
    """异步抓取网页内容

    Args:
        url: 网页地址
        selector: CSS选择器，默认抓取所有段落
    """
    try:
        async with aiohttp.ClientSession() as session:
            async with session.get(url, timeout=10) as response:
                if response.status != 200:
                    return f"HTTP {response.status}"

                html = await response.text()

                # 简单解析
                from bs4 import BeautifulSoup
                soup = BeautifulSoup(html, "html.parser")
                elements = soup.select(selector)

                texts = [e.get_text(strip=True) for e in elements[:10]]
                return "\n".join(texts) if texts else "未找到匹配内容"

    except asyncio.TimeoutError:
        return "抓取超时"
    except Exception as e:
        return f"抓取出错: {str(e)}"
```

## 带状态的工具

```python
from langchain_core.tools import BaseTool
from pydantic import BaseModel, Field
from typing import Optional

class ShoppingCartTool(BaseTool):
    """购物车工具（有状态）"""

    name: str = "shopping_cart"
    description: str = "管理购物车: add（添加）, remove（移除）, view（查看）, checkout（结算）"

    cart: dict = {}  # 状态

    def _run(self, action: str, product_id: str = "", quantity: int = 1) -> str:
        if action == "add":
            self.cart[product_id] = self.cart.get(product_id, 0) + quantity
            return f"已添加 {product_id} x{quantity}，当前购物车: {self.cart}"

        elif action == "remove":
            if product_id in self.cart:
                del self.cart[product_id]
                return f"已移除 {product_id}，当前购物车: {self.cart}"
            return f"购物车中没有 {product_id}"

        elif action == "view":
            if not self.cart:
                return "购物车为空"
            items = "\n".join(f"  - {k}: {v}件" for k, v in self.cart.items())
            return f"购物车:\n{items}"

        elif action == "checkout":
            total = sum(prices.get(pid, 0) * qty for pid, qty in self.cart.items())
            self.cart.clear()
            return f"结算完成，总计: ¥{total}"

        return f"未知操作: {action}"

    async def _arun(self, action: str, **kwargs) -> str:
        return self._run(action, **kwargs)


cart_tool = ShoppingCartTool()
```

## 工具的输入验证

```python
from pydantic import BaseModel, Field, field_validator

class EmailInput(BaseModel):
    to: str = Field(description="收件人邮箱")
    subject: str = Field(description="邮件主题")
    body: str = Field(description="邮件内容")

    @field_validator("to")
    @classmethod
    def validate_email(cls, v):
        if "@" not in v:
            raise ValueError("无效的邮箱地址")
        return v

    @field_validator("subject")
    @classmethod
    def validate_subject(cls, v):
        if len(v) > 100:
            raise ValueError("主题太长（最多100字符）")
        return v

@tool(args_schema=EmailInput)
def send_email(to: str, subject: str, body: str) -> str:
    """发送邮件"""
    # 实际发送逻辑
    return f"邮件已发送至 {to}"
```

### 4. 超时控制

```python
import signal

class TimeoutError(Exception):
    pass

def with_timeout(seconds):
    def handler(signum, frame):
        raise TimeoutError("工具执行超时")
    def decorator(func):
        def wrapper(*args, **kwargs):
            signal.signal(signal.SIGALRM, handler)
            signal.alarm(seconds)
            try:
                return func(*args, **kwargs)
            finally:
                signal.alarm(0)
        return wrapper
    return decorator

@tool
```

## 小结

| 要点 | 说明 |
|------|------|
| 功能单一 | 每个工具只做一件事 |
| 描述清晰 | LLM 依赖描述做决策 |
| 参数验证 | 使用 Pydantic |
| 错误处理 | 返回有意义的错误信息 |
| 安全限制 | 白名单、路径限制、超时 |
| 异步工具 | async def + await |
| 有状态工具 | 继承 BaseTool |
| 工具链 | 多个工具组合使用 |

## 优化一：减少 LLM 调用

### 合并请求

```python
# ❌ 多次调用
def slow_node(state):
    title = llm.invoke("生成标题")      # 2s
    summary = llm.invoke("生成摘要")    # 2s
    tags = llm.invoke("生成标签")       # 2s
    # 总计 6s

# ✅ 一次调用
def fast_node(state):
    response = llm.invoke("""
    生成标题、摘要和3个标签，用JSON返回:
    {"title": "...", "summary": "...", "tags": [...]}
    """)
    # 总计 2s
    import json
    result = json.loads(response.content)
    return result
```

### 使用更小的模型

```python
from langchain_openai import ChatOpenAI

# 简单任务用小模型
fast_llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

# 复杂任务用大模型
smart_llm = ChatOpenAI(model="gpt-4o", temperature=0)

def smart_node(state):
    # 简单路由用小模型
    intent = fast_llm.invoke(f"分类意图: {state['message']}")
    if intent.content == "简单":
        return fast_llm.invoke(...)
    else:
        return smart_llm.invoke(...)
```

### 缓存 LLM 响应


## 优化三：批处理

### 批量 LLM 调用

```python
# ❌ 逐个调用
def process_slow(items: list[str]):
    results = []
    for item in items:
        result = llm.invoke(f"处理: {item}")
        results.append(result)
    # N 个项目 = N 次调用

# ✅ 批量调用
def process_fast(items: list[str]):
    # 一次处理多个
    batch_prompt = "\n".join(f"{i}. {item}" for i, item in enumerate(items))
    response = llm.invoke(f"逐行处理以下内容:\n{batch_prompt}")
    # 1 次调用
```

### 批量嵌入

```python
from langchain_openai import OpenAIEmbeddings

embeddings = OpenAIEmbeddings()

# ❌ 逐个嵌入
vectors = [embeddings.embed_query(text) for text in texts]  # N 次API调用

# ✅ 批量嵌入
vectors = embeddings.embed_documents(texts)  # 1 次API调用
```

## 优化四：缓存策略

### 多级缓存

```python
import hashlib
import json
from functools import lru_cache

class MultiLevelCache:
    """多级缓存"""

    def __init__(self):
        self.l1 = {}  # 内存缓存（最快）
        self.l2 = None  # Redis（次快）

    def _key(self, prompt: str, model: str) -> str:
        content = f"{model}:{prompt}"
        return hashlib.sha256(content.encode()).hexdigest()

    def get(self, prompt: str, model: str):
        key = self._key(prompt, model)

        # L1: 内存
        if key in self.l1:
            return self.l1[key]

        # L2: Redis
        if self.l2:
            val = self.l2.get(key)
            if val:
                self.l1[key] = val  # 回填 L1
                return val

        return None

    def set(self, prompt: str, model: str, value: str, ttl: int = 3600):
        key = self._key(prompt, model)
        self.l1[key] = value
        if self.l2:
            self.l2.setex(key, ttl, value)

cache = MultiLevelCache()
```

## 优化五：流式输出

### 减少用户感知延迟

```python
# ❌ 等全部完成才返回
result = graph.invoke(input)
return result["messages"][-1].content  # 用户等了10秒才看到任何东西

# ✅ 流式返回
async def stream_response():
    async for msg, metadata in graph.astream(input, stream_mode="messages"):
        if msg.content:
            yield msg.content  # 用户立即看到逐字输出

# 用户感知延迟: 从 10s → 0.5s
```

## 成本控制

### Token 监控

```python
class TokenTracker:
    """Token 使用追踪"""

    def __init__(self):
        self.usage = {"prompt": 0, "completion": 0, "total_cost": 0}

    def track(self, response):
        """记录一次 LLM 调用的 token 使用"""
        usage = response.response_metadata.get("token_usage", {})
        self.usage["prompt"] += usage.get("prompt_tokens", 0)
        self.usage["completion"] += usage.get("completion_tokens", 0)

        # 计算成本（以 GPT-4o 为例）
        cost = (
            usage.get("prompt_tokens", 0) * 0.000005 +      # $5/M input
            usage.get("completion_tokens", 0) * 0.000015     # $15/M output
        )
        self.usage["total_cost"] += cost

    def get_usage(self):
        return self.usage

token_tracker = TokenTracker()

# 在节点中使用
def llm_node(state):
    response = llm.invoke(...)
    token_tracker.track(response)
    return ...
```

### 预算控制

```python
class BudgetController:
    """预算控制"""

    def __init__(self, daily_budget: float = 10.0):
        self.daily_budget = daily_budget
        self.spent = 0.0

    def can_spend(self, estimated_cost: float) -> bool:
        if self.spent + estimated_cost > self.daily_budget:
            return False
        self.spent += estimated_cost
        return True

    def estimate_cost(self, prompt_tokens: int, max_output: int = 500) -> float:
        return (
            prompt_tokens * 0.000005 +
            max_output * 0.000015
        )

budget = BudgetController(daily_budget=10.0)

def budgeted_llm_call(state):
    estimated = budget.estimate_cost(
        prompt_tokens=len(state["messages"]) * 100
    )
    if not budget.can_spend(estimated):
        return {"answer": "今日预算已用完"}

    response = llm.invoke(...)
    return {"answer": response.content}
```

# 持久化与状态存储

生产环境中，状态持久化是可靠性的基石。本篇深入讲解各种存储方案、数据管理和容灾策略。

---

## 持久化需求

```
生产环境的状态管理挑战:

1. 多用户并发 → 需要线程安全的状态存储
2. 长时间对话 → 状态可能很大
3. 服务重启 → 状态不能丢失
4. 水平扩展 → 多实例共享状态
5. 历史回溯 → 需要保存状态版本
6. 数据合规 → 敏感数据需要加密
```

---

## 存储方案对比

| 方案 | 持久性 | 性能 | 扩展性 | 成本 | 适用场景 |
|------|--------|------|--------|------|----------|

## PostgreSQL（生产推荐）

### 安装

```bash
pip install langgraph-checkpoint-postgres psycopg[binary]
```

### 配置

```python
from langgraph.checkpoint.postgres import PostgresSaver

DB_URI = "postgresql://user:password@localhost:5432/langgraph"

# 方式一：上下文管理器（推荐）
with PostgresSaver.from_conn_string(DB_URI) as checkpointer:
    checkpointer.setup()  # 创建必要的表和索引
    graph = builder.compile(checkpointer=checkpointer)

# 方式二：连接池（生产推荐）
from psycopg_pool import ConnectionPool

pool = ConnectionPool(DB_URI, max_size=20)
checkpointer = PostgresSaver(pool)
checkpointer.setup()
graph = builder.compile(checkpointer=checkpointer)
```

### 数据库表结构

LangGraph 会自动创建以下表：

```sql

-- 写入记录表

    thread_id TEXT NOT NULL,

```

### 状态清理

```python
import psycopg
from datetime import datetime, timedelta

async def cleanup_old_checkpoints(db_uri: str, days: int = 30):
    """清理超过N天的历史检查点"""
    cutoff = datetime.now() - timedelta(days=days)

    async with await psycopg.AsyncConnection.connect(db_uri) as conn:
        async with conn.cursor() as cur:
            await cur.execute(
                "DELETE FROM checkpoints WHERE created_at < %s",
                (cutoff,)
            )
            await conn.commit()
            print(f"已清理 {cur.rowcount} 条旧记录")
```

## 数据生命周期管理

### 状态过期

```python
from datetime import datetime, timedelta

class ExpiringCheckpointer:
    """带过期时间的 Checkpointer 包装"""

    def __init__(self, base_checkpointer, ttl_hours=24):
        self.base = base_checkpointer
        self.ttl = timedelta(hours=ttl_hours)

    def cleanup_expired(self):
        """清理过期状态"""
        cutoff = datetime.now() - self.ttl
        # 实现清理逻辑...

    def get(self, config):
        # 检查是否过期
        state = self.base.get(config)
        if state and state.metadata.get("created_at"):
            created = datetime.fromisoformat(state.metadata["created_at"])
            if datetime.now() - created > self.ttl:
                return None  # 已过期
        return state
```

### 分级存储

```python
class TieredCheckpointer:
    """分级存储：热数据在 Redis，冷数据在 PostgreSQL"""

    def __init__(self, redis, postgres):
        self.redis = redis      # 热存储
        self.postgres = postgres # 冷存储

    def get(self, config):
        # 先查 Redis（快）
        state = self.redis.get(config)
        if state:
            return state

        # 再查 PostgreSQL（慢但全）
        state = self.postgres.get(config)
        if state:
            # 回填 Redis
            self.redis.put(config, state)
        return state

    def put(self, config, state):
        # 双写
        self.redis.put(config, state)
        self.postgres.put(config, state)
```

## 状态迁移与备份

### 导出状态

```python
import json

def export_thread_state(graph, thread_id: str) -> dict:
    """导出指定线程的完整状态"""
    config = {"configurable": {"thread_id": thread_id}}

    state = graph.get_state(config)
    history = list(graph.get_state_history(config))

    return {
        "thread_id": thread_id,
        "current_state": state.values,
        "next": state.next,
        "history": [
            {
                "step": s.metadata.get("step"),
                "values": s.values,
                "next": s.next,
            }
            for s in history
        ],
        "exported_at": datetime.now().isoformat()
    }

# 导出为 JSON
data = export_thread_state(graph, "user-alice-001")
with open("alice_state.json", "w") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)
```

## 加密与安全

### 敏感数据加密

```python
from cryptography.fernet import Fernet

class EncryptedCheckpointer:
    """加密的 Checkpointer"""

    def __init__(self, base_checkpointer, encryption_key: bytes):
        self.base = base_checkpointer
        self.cipher = Fernet(encryption_key)

    def _encrypt(self, data: bytes) -> bytes:
        return self.cipher.encrypt(data)

    def _decrypt(self, data: bytes) -> bytes:
        return self.cipher.decrypt(data)

    def put(self, config, state):
        # 加密后存储
        encrypted = self._encrypt(serialize(state))
        self.base.put(config, encrypted)

    def get(self, config):
        encrypted = self.base.get(config)
        if encrypted:
            return deserialize(self._decrypt(encrypted))
        return None
```

### PII 脱敏

```python
import re

def sanitize_state(state: dict) -> dict:
    """脱敏状态中的敏感信息"""
    sanitized = {}

    for key, value in state.items():
        if isinstance(value, str):
            # 脱敏手机号
            value = re.sub(r'\d{11}', '[PHONE]', value)
            # 脱敏邮箱
            value = re.sub(r'\S+@\S+', '[EMAIL]', value)
            # 脱敏身份证
            value = re.sub(r'\d{17}[\dXx]', '[ID]', value)

        sanitized[key] = value

    return sanitized
```

## 监控与指标

```python
import time
from collections import defaultdict

class MonitoredCheckpointer:
    """带监控的 Checkpointer"""

    def __init__(self, base):
        self.base = base
        self.metrics = defaultdict(list)

    def put(self, config, state):
        start = time.time()
        self.base.put(config, state)
        elapsed = time.time() - start
        self.metrics["put_time"].append(elapsed)

    def get(self, config):
        start = time.time()
        result = self.base.get(config)
        elapsed = time.time() - start
        self.metrics["get_time"].append(elapsed)
        return result

    def get_stats(self):
        return {
            "put_avg": sum(self.metrics["put_time"]) / max(len(self.metrics["put_time"]), 1),
            "get_avg": sum(self.metrics["get_time"]) / max(len(self.metrics["get_time"]), 1),
            "total_operations": len(self.metrics["put_time"]) + len(self.metrics["get_time"]),
        }
```

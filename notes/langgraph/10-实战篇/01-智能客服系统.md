# 实战：智能客服系统

构建一个完整的智能客服 Agent——支持多轮对话、知识库检索、人工转接、情绪感知。

---

## 系统架构

```
用户消息 → [安全检查] → [意图分类] → [路由]
                                         │
                    ┌────────────────────┼────────────────────┐
                    │                    │                    │
                    ▼                    ▼                    ▼
              [FAQ Agent]         [工单 Agent]          [人工转接]
              (知识库RAG)          (创建工单)            (HITL)
                    │                    │                    │
                    └────────────────────┼────────────────────┘
                                         │
                                         ▼
                                   [回复用户]
```

---

## 完整实现

```python
"""
智能客服系统
"""
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langgraph.checkpoint.memory import MemorySaver
from langchain_openai import ChatOpenAI
from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage
from pydantic import BaseModel
from typing import Literal
import operator

# === 初始化 ===
llm = ChatOpenAI(model="gpt-4o", temperature=0.3)
embeddings = OpenAIEmbeddings()

# 模拟知识库（实际用 Chroma 加载文档）
faq_db = [
    {"q": "退货政策", "a": "7天内无理由退货，15天内质量问题可换货。"},
    {"q": "配送时间", "a": "一线城市24小时达，其他城市2-3天。"},
    {"q": "支付方式", "a": "支持微信、支付宝、银行卡、货到付款。"},
]


class CustomerServiceState(TypedDict):
    messages: Annotated[list, add_messages]
    user_id: str
    intent: str  # faq, complaint, human_agent
    emotion: str
    faq_answer: str
    ticket_id: str
    escalated: bool


# === 1. 意图分类 ===
class IntentClassification(BaseModel):
    intent: Literal["faq", "complaint", "human_agent"]
    emotion: Literal["happy", "neutral", "frustrated", "angry"]

def classify_intent(state: CustomerServiceState) -> dict:
    classifier = llm.with_structured_output(IntentClassification)
    result = classifier.invoke(
        f"分析用户的意图和情绪:\n"
        f"intents: faq(常见问题), complaint(投诉), human_agent(要求人工)\n"
        f"用户消息: {state['messages'][-1].content}"
    )
    return {"intent": result.intent, "emotion": result.emotion}

def route_intent(state: CustomerServiceState) -> str:
    return state["intent"]


# === 2. FAQ Agent ===
def faq_agent(state: CustomerServiceState) -> dict:
    """基于知识库回答常见问题"""
    user_msg = state["messages"][-1].content

    # 简单匹配（实际用向量检索）
    answer = "暂无相关信息"
    for item in faq_db:
        if item["q"] in user_msg:
            answer = item["a"]
            break

    # 如果没找到，用 LLM 回答
    if answer == "暂无相关信息":
        response = llm.invoke(
            f"你是客服。友好地回答问题，如果不确定就说会转人工:\n{user_msg}"
        )
        answer = response.content

    return {
        "faq_answer": answer,
        "messages": [AIMessage(answer)]
    }


# === 3. 投诉处理 ===
def complaint_agent(state: CustomerServiceState) -> dict:
    """处理投诉"""
    # 如果情绪激动，优先安抚
    if state["emotion"] in ["frustrated", "angry"]:
        response = llm.invoke(
            f"用户情绪{state['emotion']}，请先道歉安抚，然后记录投诉内容:\n"
            f"用户消息: {state['messages'][-1].content}"
        )
    else:
        response = llm.invoke(
            f"记录用户的投诉并提供解决方案:\n{state['messages'][-1].content}"
        )

    ticket_id = f"TK-{hash(state['user_id']) % 10000:04d}"

    return {
        "ticket_id": ticket_id,
        "messages": [AIMessage(
            f"{response.content}\n\n您的工单号: {ticket_id}，我们会尽快处理。"
        )]
    }


# === 4. 人工转接 ===
def human_handoff(state: CustomerServiceState) -> dict:
    """人工转接"""
    return {
        "escalated": True,
        "messages": [AIMessage(
            "我正在为您转接人工客服，请稍候...\n"
            "预计等待时间: 约3分钟。"
        )]
    }


# === 构建图 ===
builder = StateGraph(CustomerServiceState)
builder.add_node("classify", classify_intent)
builder.add_node("faq", faq_agent)
builder.add_node("complaint", complaint_agent)
builder.add_node("human", human_handoff)

builder.add_edge(START, "classify")
builder.add_conditional_edges("classify", route_intent, {
    "faq": "faq",
    "complaint": "complaint",
    "human_agent": "human",
})
builder.add_edge("faq", END)
builder.add_edge("complaint", END)
builder.add_edge("human", END)

# 编译
cs_agent = builder.compile(checkpointer=MemorySaver())


# === 测试 ===
if __name__ == "__main__":
    config = {"configurable": {"thread_id": "user-001"}}
    state_init = {
        "messages": [],
        "user_id": "user-001",
        "intent": "", "emotion": "",
        "faq_answer": "", "ticket_id": "", "escalated": False
    }

    # 测试1: FAQ
    result = cs_agent.invoke({
        **state_init,
        "messages": [("user", "你们的退货政策是什么？")]
    }, config=config)
    print("FAQ测试:", result["messages"][-1].content)

    # 测试2: 投诉（带情绪）
    result = cs_agent.invoke({
        **state_init,
        "messages": [("user", "我等了5天还没收到货！太生气了！")]
    }, config={"configurable": {"thread_id": "user-002"}})
    print("\n投诉测试:", result["messages"][-1].content)
```

# 实战：代码生成助手

构建一个能生成代码、执行验证、自动修复的 Agent。

---

## 架构

```
用户需求 → [理解需求] → [生成代码] → [执行测试]
                                        │
                                  ┌─────┴─────┐
                                  │           │
                                通过        失败
                                  │           │
                                  ▼           ▼
                              [返回用户]  [分析错误→修复→重试]
```

---

## 实现

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.tools import tool
from langchain_core.messages import AIMessage
import operator

llm = ChatOpenAI(model="gpt-4o", temperature=0)


class CodeState(TypedDict):
    messages: Annotated[list, add_messages]
    requirement: str
    generated_code: str
    test_code: str
    test_result: str
    test_passed: bool
    error_message: str
    attempts: int
    final_code: str


@tool
def execute_python(code: str) -> str:
    """执行Python代码并返回输出"""
    import subprocess
    import tempfile
    import os

    with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
        f.write(code)
        temp_path = f.name

    try:
        result = subprocess.run(
            ['python', temp_path],
            capture_output=True,
            text=True,
            timeout=10
        )
        output = result.stdout
        if result.returncode != 0:
            output += f"\nERROR:\n{result.stderr}"
        return output
    except subprocess.TimeoutExpired:
        return "ERROR: 执行超时"
    finally:
        os.unlink(temp_path)


def understand_requirement(state: CodeState) -> dict:
    """理解并结构化需求"""
    response = llm.invoke(
        f"分析以下编程需求，提取关键信息:\n{state['requirement']}\n\n"
        f"返回: 功能描述、输入输出、边界条件"
    )
    return {"messages": [response]}

def generate_code(state: CodeState) -> dict:
    """生成代码"""
    attempts = state.get("attempts", 0)
    prev_error = state.get("error_message", "")

    prompt = f"根据需求生成Python代码:\n{state['requirement']}\n\n"

    if prev_error:
        prompt += f"""
        之前的代码有错误，请修复:
        代码: {state.get('generated_code', '')}
        错误: {prev_error}
        """

    prompt += "\n只返回代码，不要解释。用```python包裹。"

    response = llm.invoke(prompt)
    code = response.content

    # 提取代码块
    if "```python" in code:
        code = code.split("```python")[1].split("```")[0].strip()
    elif "```" in code:
        code = code.split("```")[1].split("```")[0].strip()

    return {"generated_code": code, "attempts": attempts + 1}

def test_code(state: CodeState) -> dict:
    """执行测试"""
    code = state["generated_code"]

    # 添加测试输出
    test_code = code + "\n\n# 测试\nprint('=== 测试开始 ===')\nprint('OK')\n"

    result = execute_python.invoke(test_code)

    passed = "ERROR" not in result

    return {
        "test_result": result,
        "test_passed": passed,
        "error_message": "" if passed else result
    }

def should_retry(state: CodeState) -> str:
    if state["test_passed"]:
        return "success"
    elif state["attempts"] >= 3:
        return "give_up"
    else:
        return "retry"

def finalize(state: CodeState) -> dict:
    return {"final_code": state["generated_code"]}


# 构建
builder = StateGraph(CodeState)
builder.add_node("understand", understand_requirement)
builder.add_node("generate", generate_code)
builder.add_node("test", test_code)
builder.add_node("finalize", finalize)

builder.add_edge(START, "understand")
builder.add_edge("understand", "generate")
builder.add_edge("generate", "test")
builder.add_conditional_edges("test", should_retry, {
    "success": "finalize",
    "retry": "generate",    # 回到生成（带错误信息）
    "give_up": "finalize",
})
builder.add_edge("finalize", END)

code_agent = builder.compile()
```

---

## 使用

```python
result = code_agent.invoke({
    "requirement": "写一个函数，输入一个列表，返回其中所有偶数的平方",
    "messages": [],
    "generated_code": "", "test_code": "", "test_result": "",
    "test_passed": False, "error_message": "", "attempts": 0,
    "final_code": ""
})

print("最终代码:")
print(result["final_code"])
print(f"\n尝试次数: {result['attempts']}")
```

---

## 小结

| 要点 | 说明 |
|------|------|
| 需求理解 | LLM 分析结构化需求 |
| 代码生成 | 带错误反馈的迭代生成 |
| 自动测试 | 执行代码验证 |
| 错误修复 | 根据错误消息重试 |
| 尝试限制 | 最多3次防止无限循环 |

# 异步流式输出

流式输出大幅改善用户体验——让用户立即看到响应，而不是等待完整结果。

---

## 流式模式详解

### 四种 stream_mode

```python
# 1. values: 每步的完整状态
for event in graph.stream(input, stream_mode="values"):
    print(event)

# 2. updates: 每步的增量更新
for event in graph.stream(input, stream_mode="updates"):
    print(event)

# 3. messages: Token 级流式
for msg, meta in graph.stream(input, stream_mode="messages"):
    print(msg.content, end="")

# 4. debug: 调试信息
for event in graph.stream(input, stream_mode="debug"):
    print(event)
```

## Server-Sent Events (SSE)

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
import json

app = FastAPI()

@app.post("/chat/stream")
async def stream_chat(request: ChatRequest):
    """SSE 流式接口"""
    config = {"configurable": {"thread_id": request.thread_id}}

    async def generate():
        # 发送开始事件
        yield f"data: {json.dumps({'type': 'start'})}\n\n"

        # 流式输出
        async for msg, metadata in graph.astream(
            {"messages": [("user", request.message)]},
            config=config,
            stream_mode="messages"
        ):
            if msg.content:
                chunk = json.dumps({
                    "type": "token",
                    "content": msg.content
                })
                yield f"data: {chunk}\n\n"

        # 发送结束事件
        yield f"data: {json.dumps({'type': 'done'})}\n\n"

    return StreamingResponse(generate(), media_type="text/event-stream")
```

## WebSocket 实时通信

```python
from fastapi import FastAPI, WebSocket

app = FastAPI()

@app.websocket("/ws/chat")
async def websocket_chat(ws: WebSocket):
    await ws.accept()

    while True:
        # 接收用户消息
        data = await ws.receive_json()
        config = {"configurable": {"thread_id": data["thread_id"]}}

        # 流式响应
        async for msg, metadata in graph.astream(
            {"messages": [("user", data["message"])]},
            config=config,
            stream_mode="messages"
        ):
            if msg.content:
                await ws.send_json({
                    "type": "token",
                    "content": msg.content
                })

        await ws.send_json({"type": "done"})
```

## 前端集成示例

```javascript
// SSE 客户端
const eventSource = new EventSource('/chat/stream');

eventSource.onmessage = (event) => {
    const data = JSON.parse(event.data);

    switch(data.type) {
        case 'start':
            console.log('开始生成...');
            break;
        case 'token':
            // 逐字追加到页面
            document.getElementById('response').textContent += data.content;
            break;
        case 'done':
            console.log('完成');
            break;
    }
};
```

## 流式 + 状态更新

```python
async def stream_with_state(config):
    """同时流式 token 和状态更新"""
    async for event in graph.astream(
        input,
        config=config,
        stream_mode=["messages", "updates"]
    ):
        mode, data = event

        if mode == "messages":
            msg, meta = data
            if msg.content:
                yield {"type": "token", "content": msg.content}

        elif mode == "updates":
            for node, update in data.items():
                yield {"type": "node_complete", "node": node}
```

## 小结

| 要点 | 说明 |
|------|------|
| stream_mode | values/updates/messages/debug |
| SSE | HTTP 流式（简单） |
| WebSocket | 双向实时（强大） |
| Token 流式 | 逐字输出提升体验 |
| 多模式组合 | 同时输出 token 和状态 |

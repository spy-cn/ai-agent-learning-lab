## 多轮对话管理

### 对话状态追踪

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph.message import add_messages
import operator

class ConversationState(TypedDict):
    messages: Annotated[list, add_messages]
    # 对话状态
    stage: str  # greeting, information_gathering, solution, closing
    collected_info: dict  # 已收集的信息
    pending_questions: list  # 待问的问题
    user_intent: str
    sentiment: str  # positive, neutral, negative
```

### ��阶段对话

```python
def detect_stage(state: ConversationState) -> dict:
    """识别对话阶段"""
    msg_count = len(state["messages"])
    info = state.get("collected_info", {})

    if msg_count <= 1:
        stage = "greeting"
    elif len(info) < 3:
        stage = "information_gathering"
    elif state.get("solution_provided"):
        stage = "closing"
    else:
        stage = "solution"

    return {"stage": stage}

def route_by_stage(state: ConversationState) -> str:
    return state["stage"]

def greeting_handler(state: ConversationState) -> dict:
    response = llm.invoke("打招呼并询问用户需要什么帮助")
    return {"messages": [response]}

def info_gathering_handler(state: ConversationState) -> dict:
    """收集必要信息"""
    required_fields = ["name", "issue_type", "priority"]

    missing = [f for f in required_fields if f not in state["collected_info"]]

    if missing:
        # 主动提问
        response = llm.invoke(f"询问用户的: {missing[0]}")
        return {"messages": [response]}
    else:
        return {"stage": "solution"}

def solution_handler(state: ConversationState) -> dict:
    """提供解决方案"""
    info = state["collected_info"]
    response = llm.invoke(f"基于以下信息提供解决方案: {info}")
    return {"messages": [response]}

def closing_handler(state: ConversationState) -> dict:
    response = llm.invoke("礼貌结束对话")
    return {"messages": [response]}
```

## 信息提取

### 从对话中提取结构化信息

```python
from pydantic import BaseModel

class ExtractedInfo(BaseModel):
    name: str = ""
    issue_type: str = ""
    priority: str = "normal"
    product: str = ""

def extract_info(state: ConversationState) -> dict:
    """从用户消息中提取信息"""
    extractor = llm.with_structured_output(ExtractedInfo)
    result = extractor.invoke(
        f"从以下消息中提取信息:\n{state['messages'][-1].content}"
    )

    # 合并已有信息
    existing = state.get("collected_info", {})
    new_info = result.model_dump()
    # 只更新非空字段
    for k, v in new_info.items():
        if v:
            existing[k] = v

    return {"collected_info": existing}
```

## 主动提问

```python
def smart_questioning(state: ConversationState) -> dict:
    """根据上下文主动提问"""
    info = state.get("collected_info", {})

    # 定义信息需求模板
    QUESTIONS = {
        "name": "请问您怎么称呼？",
        "product": "您使用的是哪款产品？",
        "issue_type": "您遇到的是什么类型的问题？",
        "severity": "这个问题对您的影响有多大？",
        "duration": "这个问题出现多久了？",
    }

    # 找出缺失信息
    missing = [k for k, q in QUESTIONS.items() if k not in info]

    if missing:
        next_question = QUESTIONS[missing[0]]
        response = llm.invoke(f"用友好的语气问: {next_question}")
        return {"messages": [response]}

    return {}  # 信息完整，不需要提问
```

## 情感感知

```python
class EmotionAnalysis(BaseModel):
    emotion: str  # happy, frustrated, angry, confused, neutral
    urgency: str  # low, medium, high

def detect_emotion(state: ConversationState) -> dict:
    """检测用户情绪"""
    analyzer = llm.with_structured_output(EmotionAnalysis)
    result = analyzer.invoke(
        f"分析用户消息的情绪和紧迫度:\n{state['messages'][-1].content}"
    )

    return {"emotion": result.emotion, "urgency": result.urgency}

def adjust_response_by_emotion(state: ConversationState) -> dict:
    """根据情绪调整回复风格"""
    emotion = state.get("emotion", "neutral")

    if emotion == "frustrated":
        prompt = "用户感到沮丧，请用同理心和耐心回复"
    elif emotion == "angry":
        prompt = "用户生气了，请先道歉再解决问题"
    elif emotion == "confused":
        prompt = "用户困惑，请用简单清晰的语言解释"
    else:
        prompt = "正常回复"

    response = llm.invoke(f"{prompt}\n用户消息: {state['messages'][-1].content}")
    return {"messages": [response]}
```

## 对话体验最佳实践

### 1. 上下文摘要

```python
def manage_context_window(state: ConversationState) -> dict:
    """长对话时摘要上下文"""
    messages = state["messages"]

    if len(messages) > 20:
        # 摘要早期对话
        summary = llm.invoke(f"总结对话要点:\n{format(messages[:15])}")
        recent = messages[15:]

        # 用摘要替代早期消息
        from langchain_core.messages import SystemMessage
        return {"messages": [SystemMessage(f"对话摘要: {summary.content}")] + recent}

    return {}
```

### 2. 对话修复

```python
def detect_misunderstanding(state: ConversationState) -> dict:
    """检测是否误解了用户"""
    last_user_msg = state["messages"][-1].content.lower()

    repair_signals = ["不是", "不对", "我说的是", "重新", "误解"]

    if any(sig in last_user_msg for sig in repair_signals):
        return {"needs_repair": True}
    return {"needs_repair": False}

def repair_conversation(state: ConversationState) -> dict:
    """对话修复"""
    response = llm.invoke(
        f"用户似乎对回答不满意。请重新理解需求:\n"
        f"完整对话: {state['messages'][-4:]}"
    )
    return {"messages": [response]}
```

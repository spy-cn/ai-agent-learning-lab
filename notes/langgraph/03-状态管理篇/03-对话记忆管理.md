# 对话记忆管理

记忆（Memory）是智能 Agent 的核心能力之一。本篇深入讲解 LangGraph 中的短期记忆、长期记忆、消息窗口管理。

## 短期记忆：消息管理

### 最基本的记忆：add_messages

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph.message import add_messages

class State(TypedDict):
    messages: Annotated[list, add_messages]  # 自动追加所有消息
```

这是最简单的记忆方式——所有对话消息都保存在列表中。

### 问题：消息无限增长

```python
# 对话100轮后
state["messages"] = [msg1, msg2, ..., msg200]  # 200条消息！

# 全部传给 LLM → 超出 token 限制！
llm.invoke(state["messages"])  # TokenLimitError
```

## 消息窗口管理

### 策略一：保留最近 N 条

```python
from langchain_core.messages import trim_messages

def trim_messages_node(state: State) -> dict:
    """裁剪消息，只保留最近的 N 条"""
    trimmed = trim_messages(
        state["messages"],
        max_tokens=4000,         # 最大 token 数
        strategy="last",         # 保留最新的
        token_counter=llm,       # 用 LLM 的 tokenizer 计数
        start_on="human",        # 确保以 human 消息开始
    )
    return {"messages": trimmed}
```

### 策略二：在节点中手动管理

```python
def manage_context(state: State) -> dict:
    """在节点中控制传入 LLM 的消息"""
    messages = state["messages"]

    # 保留 System + 最近 N 轮对话
    system_msgs = [m for m in messages if m.type == "system"]
    conversation = [m for m in messages if m.type != "system"]

    # 只取最近 10 条
    recent = conversation[-10:]

    # 合并
    context = system_msgs + recent

    # 调用 LLM 时只传裁剪后的消息
    response = llm.invoke(context)

    return {"messages": [response]}
```

### 策略三：使用 MessagesFilter

```python
from langchain_core.messages import message_generator

def smart_filter(state: State) -> dict:
    messages = state["messages"]

    # 始终保留第一条 system message
    system = messages[0] if messages[0].type == "system" else None

    # 保留最近 5 轮（10条消息）
    recent = messages[-10:] if len(messages) > 10 else messages

    # 如果有工具调用消息，确保工具调用和工具结果成对出现
    filtered = ensure_tool_pairs(recent)

    if system:
        filtered = [system] + filtered

    response = llm.invoke(filtered)
    return {"messages": [response]}
```

## 消息摘要：压缩长对话

当对话很长时，将早期消息**摘要**为一条 System 消息：

```python
def summarize_if_needed(state: State) -> dict:
    """如果消息太多，摘要早期对话"""
    messages = state["messages"]

    if len(messages) <= 20:
        return {}  # 不需要摘要

    # 分离系统消息和对话
    system_msgs = [m for m in messages if m.type == "system"]
    old_msgs = messages[1:-6]   # 除了 system 和最近3轮
    recent_msgs = messages[-6:] # 最近3轮

    # 生成摘要
    summary_prompt = f"请简洁地总结以下对话的关键信息:\n\n{format_messages(old_msgs)}"
    summary = llm.invoke([
        SystemMessage("你是对话摘要助手"),
        HumanMessage(summary_prompt)
    ])

    # 用摘要替代旧消息
    new_system = SystemMessage(
        f"{system_msgs[0].content}\n\n"
        f"## 对话历史摘要\n{summary.content}"
    )

    # 更新消息列表（替换，不是追加）
    return {
        "messages": [new_system] + recent_msgs,
        "summary": summary.content
    }
```

### 图解消息摘要

```
压缩前 (20条消息):
[system, msg1, msg2, ..., msg14, msg15...msg20]

                    │ summarize_if_needed()
                    ▼

压缩后 (8条消息):
[system+摘要, msg15, msg16, ..., msg20]
     ↑
   旧的14条消息被压缩成1条
```

## 多用户记忆隔离

### 使用 thread_id 隔离

```python
memory = MemorySaver()
graph = builder.compile(checkpointer=memory)

async def chat(user_id: str, message: str):
    config = {"configurable": {"thread_id": f"user-{user_id}"}}
    result = await graph.ainvoke(
        {"messages": [HumanMessage(message)]},
        config=config
    )
    return result["messages"][-1].content

# 用户 A 和 B 的记忆完全独立
await chat("alice", "我喜欢蓝色")
await chat("bob", "我喜欢红色")

await chat("alice", "我喜欢什么颜色？")  # → "蓝色"
await chat("bob", "我喜欢什么颜色？")    # → "红色"
```

## 长期记忆：跨会话持久化

### 方案一：存入数据库

```python
import sqlite3

class LongTermMemory:
    def __init__(self, db_path="memory.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS user_facts (
                user_id TEXT,
                fact_key TEXT,
                fact_value TEXT,
                PRIMARY KEY (user_id, fact_key)
            )
        """)

    def save_fact(self, user_id: str, key: str, value: str):
        self.conn.execute(
            "INSERT OR REPLACE INTO user_facts VALUES (?, ?, ?)",
            (user_id, key, value)
        )
        self.conn.commit()

    def get_facts(self, user_id: str) -> dict:
        cursor = self.conn.execute(
            "SELECT fact_key, fact_value FROM user_facts WHERE user_id = ?",
            (user_id,)
        )
        return dict(cursor.fetchall())

memory_store = LongTermMemory()


# 在节点中使用长期记忆
def load_user_context(state: State, config) -> dict:
    user_id = config["configurable"].get("user_id", "default")
    facts = memory_store.get_facts(user_id)

    if facts:
        context = "已知用户信息: " + "; ".join(f"{k}: {v}" for k, v in facts.items())
        return {"context": context}
    return {}


def extract_and_save_facts(state: State, config) -> dict:
    """从对话中提取用户信息并保存"""
    user_id = config["configurable"].get("user_id", "default")

    # 用 LLM 提取关键信息
    extraction_prompt = f"""
    从以下对话中提取用户的个人信息（如姓名、偏好、需求等）。
    如果没有可提取的信息，返回空 JSON。

    对话: {state["messages"][-2:]}

    返回 JSON 格式: {{"key": "value"}}
    """

    result = llm.invoke(extraction_prompt)

    import json
    try:
        facts = json.loads(result.content)
        for key, value in facts.items():
            memory_store.save_fact(user_id, key, value)
    except:
        pass

    return {}
```

### 方案二：使用向量数据库

```python
from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings

# 向量存储
vector_store = Chroma(
    embedding_function=OpenAIEmbeddings(),
    collection_name="user_memories"
)

def save_memory(user_id: str, content: str):
    """将记忆存入向量数据库"""
    vector_store.add_texts(
        texts=[content],
        metadatas=[{"user_id": user_id, "type": "memory"}]
    )

def retrieve_memories(user_id: str, query: str, k: int = 3) -> list:
    """检索相关记忆"""
    results = vector_store.similarity_search(
        query,
        k=k,
        filter={"user_id": user_id}
    )
    return [doc.page_content for doc in results]


# 在节点中使用
def load_relevant_memories(state: State, config) -> dict:
    user_id = config["configurable"]["user_id"]
    query = state["messages"][-1].content

    memories = retrieve_memories(user_id, query)

    if memories:
        context = "相关记忆:\n" + "\n".join(f"- {m}" for m in memories)
        return {"context": context}
    return {}
```

## 完整示例：带多级记忆的 Agent

```python
"""
多级记忆 Agent:
- 短期: Checkpoint（当前对话）
- 长期: 向量数据库（跨会话事实）
"""
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langgraph.checkpoint.memory import MemorySaver
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, SystemMessage, AIMessage


class State(TypedDict):
    messages: Annotated[list, add_messages]
    user_id: str
    long_term_context: str
    summary: str


llm = ChatOpenAI(model="gpt-4o")
memories = {}  # 简化：用字典代替数据库


def load_memory(state: State) -> dict:
    """加载长期记忆"""
    user_id = state["user_id"]
    user_memories = memories.get(user_id, [])

    if user_memories:
        context = "关于用户的信息:\n" + "\n".join(f"- {m}" for m in user_memories)
        return {"long_term_context": context}
    return {"long_term_context": ""}


def chat(state: State) -> dict:
    """对话节点"""
    messages = list(state["messages"])

    # 注入长期记忆
    if state.get("long_term_context"):
        messages.insert(1, SystemMessage(state["long_term_context"]))

    response = llm.invoke(messages)
    return {"messages": [response]}


def save_memory(state: State) -> dict:
    """提取并保存长期记忆"""
    user_msg = state["messages"][-2].content if len(state["messages"]) >= 2 else ""
    ai_msg = state["messages"][-1].content

    # 简单规则提取（实际用 LLM）
    memory_prompt = f"从以下对话中提取值得记住的用户信息（如名字、偏好），如果没有返回'无':\n用户: {user_msg}\nAI: {ai_msg}"
    extraction = llm.invoke(memory_prompt)

    if extraction.content.strip() != "无":
        user_id = state["user_id"]
        if user_id not in memories:
            memories[user_id] = []
        memories[user_id].append(extraction.content)

    return {}


# 构建图
builder = StateGraph(State)
builder.add_node("load_memory", load_memory)
builder.add_node("chat", chat)
builder.add_node("save_memory", save_memory)

builder.add_edge(START, "load_memory")
builder.add_edge("load_memory", "chat")
builder.add_edge("chat", "save_memory")
builder.add_edge("save_memory", END)

graph = builder.compile(checkpointer=MemorySaver())
```

```python
def should_summarize(state: State) -> str:
```

### 4. 隐私保护

```python
def sanitize_memory(content: str) -> str:
    """在保存记忆前脱敏"""
    import re
    # 脱敏手机号
    content = re.sub(r'\d{11}', '[PHONE]', content)
    # 脱敏身份证
    content = re.sub(r'\d{17}[\dXx]', '[ID]', content)
    return content
```

## 小结

| 记忆类型 | 存储 | 生命周期 | 用途 |
|----------|------|----------|------|
| 当前对话 | Checkpoint | 单次会话 | 上下文连续 |
| 会话摘要 | 数据库 | 跨会话 | 长对话压缩 |
| 用户画像 | 向量库 | 永久 | 个性化 |
| 知识库 | 向量库 | 永久 | RAG检索 |

| 管理策略 | 说明 |
|----------|------|
| 消息窗口 | 只保留最近 N 条 |
| 摘要压缩 | 将旧消息摘要为一条 |
| 分层管理 | 短期→中期→长期 |
| thread_id 隔离 | 不同用户/对话独立 |

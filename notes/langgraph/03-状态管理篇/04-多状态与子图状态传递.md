# 多状态与子图状态传递

随着应用复杂度增加，单一 State 可能变得臃肿。本篇讲解如何通过多状态设计、子图状态隔离、状态映射等技术管理复杂场景。

---

## 为什么需要多状态

### 单一状态的问题

```python
# 所有东西都塞进一个 State
class MegaState(TypedDict):
    # 对话相关
    messages: Annotated[list, add_messages]
    # RAG 相关
    question: str
    retrieved_docs: list
    doc_scores: list
    # 代码生成相关
    code_request: str
    generated_code: str
    test_results: dict
    # 用户信息
    user_id: str
    user_preferences: dict
    # 分析相关
    data_query: str
    chart_data: list
    insights: list
    # ... 越来越多
```

问题：
- 🔴 单个节点需要处理过多字段
- 🔴 不同模块耦合在一起
- 🔴 难以独立测试和复用
- 🔴 团队协作困难

---

## 解决方案：子图状态隔离

### 核心思想

```
主图 State（共享字段）
    │
    ├── RAG 子图（有自己的 State）
    │   ├── question
    │   ├── docs
    │   └── answer
    │
    ├── 代码生成子图（有自己的 State）
    │   ├── request
    │   ├── code
    │   └── tests
    │
    └── 数据分析子图（有自己的 State）
        ├── query
        ├── data
        └── chart
```

---

## 子图状态定义

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph.message import add_messages
import operator

# === 主图状态（共享字段） ===
class MainState(TypedDict):
    messages: Annotated[list, add_messages]    # 对话消息（共享）
    user_id: str                                # 用户ID（共享）
    current_task: str                           # 当前任务（共享）

# === RAG 子图状态 ===
class RAGState(TypedDict):
    messages: Annotated[list, add_messages]    # 对话消息
    question: str
    retrieved_docs: Annotated[list, operator.add]
    answer: str

# === 代码生成子图状态 ===
class CodeState(TypedDict):
    messages: Annotated[list, add_messages]
    request: str
    generated_code: str
    test_passed: bool
```

## 子图的构建与使用

### 定义子图

```python
def build_rag_subgraph():
    """构建 RAG 子图"""
    builder = StateGraph(RAGState)

    def retrieve(state: RAGState) -> dict:
        docs = vector_store.search(state["question"])
        return {"retrieved_docs": docs}

    def generate(state: RAGState) -> dict:
        context = "\n".join(state["retrieved_docs"])
        response = llm.invoke([
            SystemMessage(f"基于以下内容回答:\n{context}"),
            HumanMessage(state["question"])
        ])
        return {"answer": response.content, "messages": [response]}

    builder.add_node("retrieve", retrieve)
    builder.add_node("generate", generate)
    builder.add_edge(START, "retrieve")
    builder.add_edge("retrieve", "generate")
    builder.add_edge("generate", END)

    return builder.compile()
```

### 在主图中使用子图

```python
# 构建主图
main_builder = StateGraph(MainState)

# 添加子图作为节点
rag_graph = build_rag_subgraph()
main_builder.add_node("rag_agent", rag_graph)
```

### 状态映射问题

当子图的 State 与主图不完全匹配时，需要**状态映射**：

```python
# 主图有 messages 和 user_id
# 子图需要 messages 和 question

# 方式一：添加适配节点
def adapt_to_rag(state: MainState) -> dict:
    """将主图状态适配为子图输入"""
    return {
        "question": state["messages"][-1].content  # 从消息中提取问题
    }

main_builder.add_node("adapt_rag", adapt_to_rag)
main_builder.add_edge("adapt_rag", "rag_agent")
```

## 共享 Schema（推荐方式）

让子图和主图**共享相同的 State 类型**，避免映射：

```python
# 所有图使用同一个 State
class GlobalState(TypedDict):
    messages: Annotated[list, add_messages]
    question: str
    docs: Annotated[list, operator.add]
    answer: str
    code: str
    user_id: str

# 子图也使用 GlobalState
def build_rag_subgraph():
    builder = StateGraph(GlobalState)
    # 子图只操作自己关心的字段
    def retrieve(state: GlobalState) -> dict:
        return {"docs": search(state["question"])}
    ...

# 主图
main_builder = StateGraph(GlobalState)
main_builder.add_node("rag", build_rag_subgraph().compile())
```

### 共享 Schema 的优缺点

| 优点 | 缺点 |
|------|------|
| 无需状态映射 | 状态对象可能较大 |
| 数据自由流动 | 模块边界不清晰 |
| 实现简单 | 子图可访问所有字段 |

## 独立 Schema + 显式映射

当需要强隔离时，使用独立 Schema 并显式映射：

```python
from langgraph.graph import StateGraph, START, END

class MainState(TypedDict):
    messages: Annotated[list, add_messages]
    user_id: str

class AnalysisState(TypedDict):
    query: str
    data: list
    result: str

# 构建独立子图
def build_analysis_subgraph():
    builder = StateGraph(AnalysisState)

    def fetch_data(state: AnalysisState) -> dict:
        data = db.query(state["query"])
        return {"data": data}

    def analyze(state: AnalysisState) -> dict:
        result = analyze_data(state["data"])
        return {"result": result}

    builder.add_node("fetch", fetch_data)
    builder.add_node("analyze", analyze)
    builder.add_edge(START, "fetch")
    builder.add_edge("fetch", "analyze")
    builder.add_edge("analyze", END)

    return builder.compile()

# 在主图中包装子图
def analysis_wrapper(state: MainState) -> dict:
    """包装函数：处理状态映射"""
    subgraph = build_analysis_subgraph()

    # 主图 → 子图：提取问题作为 query
    sub_input = {
        "query": state["messages"][-1].content,
        "data": [],
        "result": ""
    }

    # 执行子图
    sub_result = subgraph.invoke(sub_input)

    # 子图 → 主图：将结果转为消息
    return {
        "messages": [AIMessage(sub_result["result"])]
    }

main_builder.add_node("analyze", analysis_wrapper)
```

### 图解状态映射

```
主图 State                     子图 State
┌────────────────┐             ┌────────────────┐
│ messages: [...]│ ──映射──►   │ query: "..."   │
│ user_id: "123" │             │ data: []       │
└────────────────┘             │ result: ""     │
                               └────────────────┘
                                       │
                                   子图执行
                                       │
                                       ▼
                               ┌────────────────┐
                               │ result: "分析结果"│
                               └────────────────┘
                                       │
                               ◄──映射──
                                       │
                                       ▼
主图 State 更新:
  messages: [..., AIMessage("分析结果")]
```

## 多子图协作

```python
"""
主图调度多个子图
"""
class MainState(TypedDict):
    messages: Annotated[list, add_messages]
    task_type: str
    result: str

def classify(state: MainState) -> dict:
    msg = state["messages"][-1].content
    if "搜索" in msg or "查询" in msg:
        task_type = "rag"
    elif "代码" in msg or "编程" in msg:
        task_type = "code"
    elif "数据" in msg or "分析" in msg:
        task_type = "analysis"
    else:
        task_type = "chat"
    return {"task_type": task_type}

def route(state: MainState) -> str:
    return state["task_type"]

# 各子图的包装函数
def rag_wrapper(state: MainState) -> dict:
    result = rag_subgraph.invoke({"question": state["messages"][-1].content})
    return {"messages": [AIMessage(result["answer"])]}

def code_wrapper(state: MainState) -> dict:
    result = code_subgraph.invoke({"request": state["messages"][-1].content})
    return {"messages": [AIMessage(result["code"])]}

def analysis_wrapper(state: MainState) -> dict:
    result = analysis_subgraph.invoke({"query": state["messages"][-1].content})
    return {"messages": [AIMessage(result["result"])]}

def chat_wrapper(state: MainState) -> dict:
    response = llm.invoke(state["messages"])
    return {"messages": [response]}

# 构建主图
builder = StateGraph(MainState)
builder.add_node("classify", classify)
builder.add_node("rag", rag_wrapper)
builder.add_node("code", code_wrapper)
builder.add_node("analysis", analysis_wrapper)
builder.add_node("chat", chat_wrapper)

builder.add_edge(START, "classify")
builder.add_conditional_edges("classify", route, {
    "rag": "rag",
    "code": "code",
    "analysis": "analysis",
    "chat": "chat",
})
for node in ["rag", "code", "analysis", "chat"]:
    builder.add_edge(node, END)

graph = builder.compile()
```

### 架构图

```
                    ┌──► rag ──────► END
                    │
START → classify ───┼──► code ─────► END
                    │
                    ├──► analysis ─► END
                    │
                    └──► chat ─────► END
```

## 状态传递最佳实践

### 1. 明确输入输出契约

```python
class RAGState(TypedDict):
    """
    RAG 子图状态

    输入:
        - question: 用户问题（必需）
        - messages: 对话历史（可选）

    输出:
        - answer: 生成的答案
        - sources: 引用来源
    """
    question: str
    messages: Annotated[list, add_messages]
    retrieved_docs: Annotated[list, operator.add]
    answer: str
    sources: list
```

### 2. 保持子图独立可测试

```python
# 子图可以独立运行和测试
rag_graph = build_rag_subgraph()

# 不需要主图就能测试
result = rag_graph.invoke({
    "question": "什么是LangGraph？",
    "messages": [],
    "retrieved_docs": [],
    "answer": ""
})
assert "answer" in result
```

## 小结

| 要点 | 说明 |
|------|------|
| 多状态设计 | 不同模块用不同的 State 类型 |
| 共享 Schema | 子图和主图用同一个 State（简单） |
| 独立 Schema + 映射 | 子图有自己的 State（强隔离） |
| 包装函数 | 处理主图 ↔ 子图的状态转换 |
| 子图组合 | 主图调度多个子图协作 |
| 可测试性 | 子图应能独立运行和测试 |

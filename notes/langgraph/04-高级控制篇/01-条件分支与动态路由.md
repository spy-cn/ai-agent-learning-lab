# 条件分支与动态路由

条件分支是 LangGraph 最强大的特性——它让图可以根据当前状态动态决定执行路径，实现复杂的决策逻辑。

## 路由函数设计模式

### 模式一：基于关键字

```python
def route_by_keyword(state: State) -> str:
    """基于关键词匹配路由"""
    msg = state["messages"][-1].content.lower()

    keywords_map = {
        "weather": ["天气", "温度", "下雨"],
        "news": ["新闻", "头条", "热点"],
        "code": ["代码", "编程", "函数"],
        "math": ["计算", "等于", "加", "减"],
    }

    for dest, keywords in keywords_map.items():
        if any(kw in msg for kw in keywords):
            return dest

    return "general"

graph.add_conditional_edges("classifier", route_by_keyword, {
    "weather": "weather_node",
    "news": "news_node",
    "code": "code_node",
    "math": "math_node",
    "general": "general_node",
})
```

### 模式二：基于 LLM 结构化输出

```python
from pydantic import BaseModel
from langchain_openai import ChatOpenAI

class RouteDecision(BaseModel):
    """路由决策模型"""
    destination: str = Field(description="目标节点名称")
    reason: str = Field(description="选择原因")

def route_by_llm(state: State) -> str:
    """使用 LLM 做路由决策"""
    structured_llm = ChatOpenAI(model="gpt-4o").with_structured_output(RouteDecision)

    decision = structured_llm.invoke(
        f"分析用户意图，选择最合适的处理节点。\n\n"
        f"可选节点: search, calculate, translate, summarize, general\n"
        f"用户消息: {state['messages'][-1].content}"
    )

    print(f"路由决策: {decision.destination} (原因: {decision.reason})")
    return decision.destination
```

### 模式三：基于状态标志

```python
def route_by_flags(state: State) -> str:
    """根据布尔标志路由"""
    if not state.get("retrieved_docs"):
        return "retrieve"          # 还没检索
    elif state.get("needs_review"):
        return "review"            # 需要审核
    elif state.get("confidence", 0) < 0.5:
        return "enhance_query"     # 置信度低，改写查询
    elif state.get("draft_ready"):
        return "generate"          # 可以生成
    else:
        return END
```

### 模式四：多级路由

```python
def multi_level_router(state: State) -> str:
    """多级路由：先分类大类，再分小类"""
    msg = state["messages"][-1].content

    # 第一级：大类别
    if any(kw in msg for kw in ["技术", "代码", "编程"]):
        # 第二级：技术子类
        if "python" in msg.lower():
            return "python_expert"
        elif "javascript" in msg.lower():
            return "js_expert"
        else:
            return "general_tech"
    elif any(kw in msg for kw in ["生活", "美食", "旅游"]):
        if "美食" in msg:
            return "food_expert"
        elif "旅游" in msg:
            return "travel_expert"
        else:
            return "lifestyle"
    else:
        return "general"
```

## 动态目标：Send API

当目标节点和参数都是动态的时，使用 `Send`：

```python
from langgraph.constants import Send

def dynamic_dispatch(state: State) -> list[Send]:
    """根据任务类型动态分发"""
    sends = []

    for task in state.get("pending_tasks", []):
        if task["type"] == "search":
            sends.append(Send("search_agent", {"query": task["content"]}))
        elif task["type"] == "calc":
            sends.append(Send("calculator", {"expression": task["content"]}))

    return sends if sends else [Send("idle", {})]
```

## 复杂路由场景

### 场景一：自适应 RAG

```python
def adaptive_rag_route(state: State) -> str:
    """自适应 RAG 路由"""
    question = state["question"]

    # 判断是否需要检索
    if is_simple_question(question):
        return "direct_answer"      # 简单问题直接回答

    # 判断查询质量
    quality = assess_query_quality(question)
    if quality < 0.3:
        return "rewrite_query"      # 查询质量差，先改写

    # 判断需要的检索类型
    if needs_web_search(question):
        return "web_search"         # 需要联网搜索
    else:
        return "local_retrieve"     # 本地知识库检索

# 构建图
builder.add_conditional_edges("router", adaptive_rag_route, {
    "direct_answer": "generate",
    "rewrite_query": "rewrite",
    "web_search": "web_search",
    "local_retrieve": "retrieve",
})
```

### 场景二：多 Agent 协调

```python
def supervisor_route(state: State) -> str:
    """Supervisor 决定下一个 Agent"""
    task = state["current_task"]
    agents_status = state.get("agent_status", {})

    # 跳过正在忙的 Agent
    if task["type"] == "research" and not agents_status.get("researcher_busy"):
        return "researcher"
    elif task["type"] == "writing" and not agents_status.get("writer_busy"):
        return "writer"
    elif task["type"] == "review" and not agents_status.get("reviewer_busy"):
        return "reviewer"
    else:
        return "wait"  # 所有 Agent 都忙
```

### 场景三：错误恢复路由

```python
def error_aware_route(state: State) -> str:
    """带错误恢复的路由"""
    error = state.get("error")

    if error:
        retry_count = state.get("retry_count", 0)
        if retry_count < 3:
            return "retry"          # 重试
        else:
            return "fallback"       # 降级处理

    if state.get("success"):
        return "output"             # 成功，输出结果

    return "process"                # 正常处理
```

## 条件边 + 循环 = Agent

经典 Agent 循环：

```python
# === ReAct Agent 的路由 ===
def should_continue(state: State) -> str:
    """Agent 是否继续"""
    last_message = state["messages"][-1]

    # 如果 LLM 要求调用工具
    if hasattr(last_message, "tool_calls") and last_message.tool_calls:
        return "tools"

    # 否则结束（LLM 已经给出最终答案）
    return END

# 构建
builder.add_edge(START, "agent")
builder.add_conditional_edges("agent", should_continue, {
    "tools": "tools",
    END: END
})
builder.add_edge("tools", "agent")  # 工具→回到 agent
```

```
       ┌──────────────────┐
       │                  │
       ▼                  │
   ┌────────┐      ┌───────┐
   │ agent  │ ───► │ tools │
   │ (LLM)  │      │(执行) │
   └────────┘      └───────┘
       │
       ▼ (无工具调用)
      END
```

def debug_router(state: State) -> str:

### 记录路由历史

```python
def logging_router(state: State) -> str:
    decision = actual_routing_logic(state)

    # 记录到状态
    history = state.get("route_history", [])
    history.append({
        "step": len(history),
        "from": "current",
        "to": decision,
        "reason": state.get("last_reason", "")
    })

    # 注意：这里返回 decision，同时 state 更新在节点中处理
    return decision
```

## 性能考虑

### 路由函数应轻量

```python
# ❌ 不好：路由中做重操作
def slow_router(state):
    docs = vector_store.search(state["question"])  # ❌ 向量检索太慢
    if len(docs) > 0:
        return "generate"
    return "rewrite"

# ✅ 好：路由只做判断
def fast_router(state):
    if state.get("has_docs"):  # 只检查标志
        return "generate"
    return "rewrite"
```

### LLM 路由的成本

```python
# LLM 路由每次都消耗 token
# 如果路由频繁，考虑:
# 1. 缓存常见路由
# 2. 使用更小的模型做路由
# 3. 尽量用规则路由代替 LLM 路由
```

## 小结

| 路由策略 | 适用场景 | 复杂度 |
|----------|----------|--------|
| 关键词匹配 | 简单意图分类 | 低 |
| LLM 结构化输出 | 复杂语义理解 | 中 |
| 状态标志 | 基于流程控制 | 低 |
| 多级路由 | 层次化决策 | 中 |
| Send 动态分发 | 并行任务分派 | 中 |

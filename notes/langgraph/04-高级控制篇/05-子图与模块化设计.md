# 子图与模块化设计

随着应用复杂度增长，把所有逻辑放在一个大图中会变得难以维护。子图（Subgraph）是 LangGraph 的模块化方案——将复杂图拆分为可复用的独立模块。

---

## 为什么需要子图

### 单体图的问题

```
一个巨大的图:
START → classify → retrieve → rerank → generate → review → revise → format → END
                 ↓
            web_search → parse → clean
                 ↓
            code_gen → test → fix

问题:
- 🔴 节点太多，难以理解
- 🔴 无法独立测试各模块
- 🔴 无法复用
- 🔴 团队协作困难
```

### 子图的解决方案

```
主图（调度层）:
START → [RAG子图] → [Review子图] → END

RAG子图（独立模块）:
retrieve → rerank → generate

Review子图（独立模块）:
check → feedback → revise
```

---

## 子图的两种模式

### 模式一：共享 State

子图和主图使用**相同的 State 类型**：

```python
class GlobalState(TypedDict):
    messages: Annotated[list, add_messages]
    question: str
    docs: list
    answer: str

# 子图使用同一个 GlobalState
def build_rag_subgraph():
    builder = StateGraph(GlobalState)

    def retrieve(state: GlobalState) -> dict:
        docs = search(state["question"])
        return {"docs": docs}

    def generate(state: GlobalState) -> dict:
        answer = llm.invoke(f"基于 {state['docs']} 回答 {state['question']}")
        return {"answer": answer.content}

    builder.add_node("retrieve", retrieve)
    builder.add_node("generate", generate)
    builder.add_edge(START, "retrieve")
    builder.add_edge("retrieve", "generate")
    builder.add_edge("generate", END)

    return builder.compile()

# 主图
main_builder = StateGraph(GlobalState)
main_builder.add_node("rag", build_rag_subgraph())  # 直接添加
```

特点：
- ✅ 简单直接，无需状态转换
- ✅ 数据自由流转
- ❌ 子图可访问所有字段
- ❌ 模块边界不清晰

### 模式二：独立 State + 状态映射

子图有自己的 State，主图通过**包装函数**转换：

```python
class MainState(TypedDict):
    messages: Annotated[list, add_messages]
    question: str
    answer: str

class RAGState(TypedDict):
    query: str
    docs: list
    result: str

# 子图（独立 State）
def build_rag_subgraph():
    builder = StateGraph(RAGState)

    def retrieve(state: RAGState) -> dict:
        docs = search(state["query"])
        return {"docs": docs}

    def generate(state: RAGState) -> dict:
        result = llm.invoke(f"基于 {state['docs']}")
        return {"result": result.content}

    builder.add_node("retrieve", retrieve)
    builder.add_node("generate", generate)
    builder.add_edge(START, "retrieve")
    builder.add_edge("retrieve", "generate")
    builder.add_edge("generate", END)

    return builder.compile()

# 包装函数：处理状态映射
def rag_wrapper(state: MainState) -> dict:
    rag_graph = build_rag_subgraph()

    # MainState → RAGState
    rag_input = {
        "query": state["question"],
        "docs": [],
        "result": ""
    }

    # 执行子图
    rag_result = rag_graph.invoke(rag_input)

    # RAGState → MainState
    return {
        "answer": rag_result["result"],
        "messages": [AIMessage(rag_result["result"])]
    }

# 主图
main_builder = StateGraph(MainState)
main_builder.add_node("rag", rag_wrapper)
```

特点：
- ✅ 强隔离，子图不依赖主图
- ✅ 可独立测试
- ✅ 可复用
- ❌ 需要写映射代码

## 子图设计模式

### 模式一：管道子图

```
RAG 子图内部:
retrieve → rerank → generate
```

```python
def build_rag_pipeline():
    builder = StateGraph(RAGState)

    builder.add_node("retrieve", retrieve_node)
    builder.add_node("rerank", rerank_node)
    builder.add_node("generate", generate_node)

    builder.add_edge(START, "retrieve")
    builder.add_edge("retrieve", "rerank")
    builder.add_edge("rerank", "generate")
    builder.add_edge("generate", END)

    return builder.compile()
```

### 模式二：带条件路由的子图

```python
def build_adaptive_rag():
    builder = StateGraph(RAGState)

    def route_query(state) -> str:
        if is_simple(state["query"]):
            return "direct"
        return "retrieve"

    builder.add_node("retrieve", retrieve_node)
    builder.add_node("direct", direct_answer_node)
    builder.add_node("generate", generate_node)

    builder.add_conditional_edges(START, route_query)
    builder.add_edge("retrieve", "generate")
    builder.add_edge("direct", "generate")
    builder.add_edge("generate", END)

    return builder.compile()
```

### 模式三：带循环的子图

```python
def build_self_correcting_rag():
    builder = StateGraph(RAGState)

    def quality_check(state) -> str:
        if state.get("quality_score", 0) < 0.7:
            return "rewrite"
        return END

    builder.add_node("retrieve", retrieve_node)
    builder.add_node("generate", generate_node)
    builder.add_node("check", check_node)
    builder.add_node("rewrite", rewrite_node)

    builder.add_edge(START, "retrieve")
    builder.add_edge("retrieve", "generate")
    builder.add_edge("generate", "check")
    builder.add_conditional_edges("check", quality_check)
    builder.add_edge("rewrite", "retrieve")  # 循环

    return builder.compile()
```

## 子图嵌套

子图可以包含子图，形成层级结构：

```python
def build_code_agent():
    """代码 Agent 子图"""
    builder = StateGraph(CodeState)
    builder.add_node("plan", plan_code)
    builder.add_node("generate", generate_code)
    builder.add_node("test", run_tests)
    builder.add_node("fix", fix_code)

    builder.add_edge(START, "plan")
    builder.add_edge("plan", "generate")
    builder.add_edge("generate", "test")
    builder.add_conditional_edges("test", lambda s: "fix" if not s["pass"] else END)
    builder.add_edge("fix", "generate")

    return builder.compile()

def build_full_system():
    """主图"""
    builder = StateGraph(MainState)

    # 嵌套子图
    rag = build_rag_subgraph()
    code = build_code_agent()

    builder.add_node("classify", classify_intent)
    builder.add_node("rag", rag)           # 子图作为节点
    builder.add_node("code", code)         # 子图作为节点

    builder.add_conditional_edges("classify", route_intent, {
        "rag": "rag",
        "code": "code",
    })
    builder.add_edge("rag", END)
    builder.add_edge("code", END)

    return builder.compile()
```

### 嵌套结构图

```
主图:
  START → classify ──┬──► [RAG子图] ──► END
                    │     ├ retrieve
                    │     ├ generate
                    │     └ ...
                    │
                    └──► [Code子图] ──► END
                          ├ plan
                          ├ generate
                          ├ test
                          └ fix
```

## 可复用子图工厂

```python
class RAGGraphFactory:
    """可配置的 RAG 子图工厂"""

    def __init__(self, llm, retriever, config=None):
        self.llm = llm
        self.retriever = retriever
        self.config = config or {}

    def build(self):
        """构建 RAG 子图"""
        builder = StateGraph(RAGState)

        builder.add_node("retrieve", self._retrieve)
        builder.add_node("generate", self._generate)

        if self.config.get("enable_rerank"):
            builder.add_node("rerank", self._rerank)
            builder.add_edge("retrieve", "rerank")
            builder.add_edge("rerank", "generate")
        else:
            builder.add_edge("retrieve", "generate")

        builder.add_edge(START, "retrieve")
        builder.add_edge("generate", END)

        return builder.compile()

    def _retrieve(self, state):
        docs = self.retriever.search(state["query"])
        return {"docs": docs}

    def _rerank(self, state):
        reranked = rerank_docs(state["docs"], state["query"])
        return {"docs": reranked}

    def _generate(self, state):
        context = "\n".join(state["docs"])
        response = self.llm.invoke(f"Context: {context}\nQ: {state['query']}")
        return {"result": response.content}


# 使用工厂
rag_factory = RAGGraphFactory(
    llm=ChatOpenAI(model="gpt-4o"),
    retriever=my_vector_store,
    config={"enable_rerank": True}
)

rag_graph = rag_factory.build()
```

## 子图的测试

### 独立测试

```python
import pytest

class TestRAGSubgraph:
    def setup_method(self):
        self.rag = build_rag_subgraph()

    def test_retrieve_and_generate(self):
        result = self.rag.invoke({
            "query": "什么是LangGraph",
            "docs": [],
            "result": ""
        })
        assert "result" in result
        assert len(result["result"]) > 0

    def test_empty_query(self):
        result = self.rag.invoke({
            "query": "",
            "docs": [],
            "result": ""
        })
        # 应该优雅处理空查询
        assert "result" in result

    def test_with_existing_docs(self):
        result = self.rag.invoke({
            "query": "总结一下",
            "docs": ["LangGraph是图式框架", "支持循环和分支"],
            "result": ""
        })
        assert "LangGraph" in result["result"]
```

## 子图最佳实践

### 1. 单一职责

```python
# ✅ 好：每个子图负责一件事
rag_subgraph = build_rag()        # 只做检索+生成
review_subgraph = build_review()  # 只做审查+修改

# ❌ 不好：一个子图做太多事
mega_subgraph = build_everything()  # 检索+生成+审查+发送+...
```

### 2. 清晰的输入输出

```python
class RAGState(TypedDict):
    """
    RAG 子图状态

    输入字段:
        query: 检索查询（必需）

    输出字段:
        result: 生成的答案
        sources: 引用的文档来源
    """
    query: str
    docs: list          # 内部字段
    result: str         # 输出
    sources: list       # 输出
```

### 3. 配置化

```python
def build_rag(config: dict):
    """通过配置控制子图行为"""
    builder = StateGraph(RAGState)

    # 基础节点
    builder.add_node("retrieve", retrieve)

    # 可选的 rerank
    if config.get("rerank"):
        builder.add_node("rerank", rerank)
        builder.add_edge("retrieve", "rerank")
        next_after_rerank = "rerank"
    else:
        next_after_rerank = "retrieve"

    # 可选的查询改写
    if config.get("query_rewrite"):
        builder.add_node("rewrite", rewrite)
        builder.add_edge(START, "rewrite")
        builder.add_edge("rewrite", next_after_rerank)
    else:
        builder.add_edge(START, next_after_rerank)

    return builder.compile()
```

## 小结

| 要点 | 说明 |
|------|------|
| 子图的作用 | 模块化、复用、独立测试 |
| 共享 State | 简单但耦合 |
| 独立 State | 强隔离但需映射 |
| 包装函数 | 处理主图↔子图状态转换 |
| 工厂模式 | 参数化构建子图 |
| 单一职责 | 每个子图只做一件事 |

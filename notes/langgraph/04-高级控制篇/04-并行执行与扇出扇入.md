# 并行执行与扇出扇入

并行执行让多个节点同时运行，大幅提升复杂任务的效率。本篇讲解 Map-Reduce 模式、Send API、扇出扇入等技术。

---

## 并行的两种方式

### 方式一：多出边（静态并行）

```python
# 一个节点有多条出边 → 并行执行
builder.add_edge("split", "task_a")
builder.add_edge("split", "task_b")
builder.add_edge("split", "task_c")
```

```
         ┌──► task_a ──┐
split ───┼──► task_b ──┼──► merge
         └──► task_c ──┘
```

### 方式二：Send API（动态并行）

```python
from langgraph.constants import Send

def dynamic_fan_out(state):
    return [Send("worker", {"item": item}) for item in state["items"]]
```

## 静态并行：多出边

### 基本 Map-Reduce

```python
class State(TypedDict):
    query: str
    web_results: str
    db_results: str
    api_results: str
    final_answer: str

# 并行的数据源查询
def search_web(state: State) -> dict:
    results = web_search(state["query"])
    return {"web_results": results}

def search_db(state: State) -> dict:
    results = db_query(state["query"])
    return {"db_results": results}

def call_api(state: State) -> dict:
    results = api_call(state["query"])
    return {"api_results": results}

# 汇总节点
def merge_results(state: State) -> dict:
    combined = f"Web: {state['web_results']}\nDB: {state['db_results']}\nAPI: {state['api_results']}"
    answer = llm.invoke(f"基于以下信息回答: {combined}")
    return {"final_answer": answer.content}

# 构建
builder.add_node("web", search_web)
builder.add_node("db", search_db)
builder.add_node("api", call_api)
builder.add_node("merge", merge_results)

builder.add_edge(START, "web")  # 三条入边 = 三个并行起点
builder.add_edge(START, "db")
builder.add_edge(START, "api")

builder.add_edge("web", "merge")  # 三条入边 = 等待全部完成
builder.add_edge("db", "merge")
builder.add_edge("api", "merge")

builder.add_edge("merge", END)
```

### 执行图解

```
              ┌──► web ──┐
START ────────┼──► db  ──┼──► merge ──► END
              └──► api ──┘
              (并行)      (等全部完成)
```

> ⚠️ `merge` 节点会等待 `web`、`db`、`api` **全部完成后**才执行。

## 动态并行：Send API

### 基本用法

```python
from langgraph.constants import Send

class MainState(TypedDict):
    tasks: list[str]
    results: Annotated[list[str], operator.add]  # 追加合并

class WorkerState(TypedDict):
    task: str
    result: str

def distribute(state: MainState) -> list[Send]:
    """动态分发任务"""
    return [
        Send("worker", {"task": task})
        for task in state["tasks"]
    ]

def worker(state: WorkerState) -> dict:
    """处理单个任务"""
    result = process(state["task"])
    return {"result": result}  # 注意：返回的是 worker 的 state

def aggregate(state: MainState) -> dict:
    """汇总"""
    return {"final": "\n".join(state["results"])}

# 构建
builder = StateGraph(MainState)
builder.add_node("worker", worker)
builder.add_node("aggregate", aggregate)

builder.add_conditional_edges(START, distribute)
builder.add_edge("worker", "aggregate")
builder.add_edge("aggregate", END)

graph = builder.compile()

# 运行
result = graph.invoke({"tasks": ["task1", "task2", "task3"], "results": []})
# worker 会并行执行3次
```

### Send vs 多出边

| 特性 | 多出边 | Send API |
|------|--------|----------|
| 并行数量 | 固定 | 动态 |
| 节点输入 | 共享主 state | 每个有独立输入 |
| 适合场景 | 已知数量的并行 | 运行时决定数量 |
| 代码复杂度 | 低 | 中 |

## 实战：并行文档处理

```python
"""
并行处理多个文档，每个文档独立处理
"""
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.constants import Send
import operator

class MainState(TypedDict):
    documents: list[dict]     # 原始文档列表
    summaries: Annotated[list[str], operator.add]    # 摘要结果
    keywords: Annotated[list[str], operator.add]     # 关键词
    analysis: str

class WorkerState(TypedDict):
    document: dict
    summary: str
    keywords: list[str]

llm = ChatOpenAI(model="gpt-4o")


def fan_out(state: MainState) -> list[Send]:
    """分发每个文档给 worker"""
    return [
        Send("process_doc", {"document": doc})
        for doc in state["documents"]
    ]


def process_doc(state: WorkerState) -> dict:
    """处理单个文档（会被并行调用）"""
    doc = state["document"]
    content = doc.get("content", "")

    # 生成摘要
    summary_response = llm.invoke(f"用一句话总结: {content[:500]}")

    # 提取关键词
    keywords_response = llm.invoke(f"提取3个关键词: {content[:500]}")

    return {
        "summary": summary_response.content,
        "keywords": keywords_response.content.split(", "),
    }


def synthesize(state: MainState) -> dict:
    """综合所有结果"""
    all_summaries = "\n".join(f"- {s}" for s in state["summaries"])
    all_keywords = ", ".join(state["keywords"])

    analysis = llm.invoke(
        f"基于以下摘要和关键词生成综合分析:\n摘要:\n{all_summaries}\n关键词: {all_keywords}"
    )

    return {"analysis": analysis.content}


# 构建
builder = StateGraph(MainState)
builder.add_node("process_doc", process_doc)
builder.add_node("synthesize", synthesize)

builder.add_conditional_edges(START, fan_out)
builder.add_edge("process_doc", "synthesize")
builder.add_edge("synthesize", END)

graph = builder.compile()

# 运行
result = graph.invoke({
    "documents": [
        {"id": 1, "content": "LangGraph是一个图式框架..."},
        {"id": 2, "content": "RAG是检索增强生成..."},
        {"id": 3, "content": "多Agent系统可以..."},
    ],
    "summaries": [],
    "keywords": [],
    "analysis": ""
})
```

### 执行流程

```
                ┌──► process_doc(doc1) ──┐
   fan_out ─────┼──► process_doc(doc2) ──┼──► synthesize → END
                └──► process_doc(doc3) ──┘
                (3个并行)              (等3个都完成)
```

## 扇入（Fan-In）模式

### 超时处理

```python
import asyncio

async def fan_in_with_timeout(state: State) -> dict:
    """扇入时设置超时"""
    try:
        # 等待所有并行任务，最多10秒
        results = await asyncio.wait_for(
            gather_all_tasks(state),
            timeout=10.0
        )
        return {"results": results}
    except asyncio.TimeoutError:
        return {"results": [], "error": "部分任务超时"}
```

### 部分结果聚合

```python
def flexible_merge(state: State) -> dict:
    """不需要全部完成，有结果就用"""
    results = {}

    # 检查哪些并行任务完成了
    if state.get("web_results"):
        results["web"] = state["web_results"]
    if state.get("db_results"):
        results["db"] = state["db_results"]
    if state.get("api_results"):
        results["api"] = state["api_results"]

    if not results:
        return {"answer": "所有数据源都失败了"}

    # 用可用的结果生成答案
    answer = llm.invoke(f"基于部分数据回答: {results}")
    return {"answer": answer.content}
```

## 并行 + 循环

```python
"""
每轮迭代中并行处理
"""
def parallel_research(state: State) -> list[Send]:
    """并行研究多个子主题"""
    topics = decompose_topic(state["main_topic"])
    return [Send("researcher", {"topic": t}) for t in topics]

def researcher(state: WorkerState) -> dict:
    """单个研究任务"""
    findings = search_and_summarize(state["topic"])
    return {"findings": [findings]}  # 追加到主 state

def evaluate_completeness(state: State) -> str:
    """评估是否需要更多研究"""
    if len(state["findings"]) < 5 and state.get("iteration", 0) < 3:
        return "research_more"
    return "compile"

# 构建
builder.add_conditional_edges(START, parallel_research)
builder.add_edge("researcher", "evaluate")
builder.add_conditional_edges("evaluate", evaluate_completeness, {
    "research_more": "research_more_dispatcher",
    "compile": "compile_report",
})
```

## 性能优化

### 1. 避免不必要的串行

```python
# ❌ 串行：慢
builder.add_edge(START, "fetch_a")
builder.add_edge("fetch_a", "fetch_b")
builder.add_edge("fetch_b", "fetch_c")

# ✅ 并行：快
builder.add_edge(START, "fetch_a")
builder.add_edge(START, "fetch_b")
builder.add_edge(START, "fetch_c")
```

### 2. 合并 LLM 调用

```python
# ❌ 多次串行 LLM 调用
def slow_node(state):
    title = llm.invoke("生成标题")     # 等2秒
    summary = llm.invoke("生成摘要")   # 再等2秒
    tags = llm.invoke("生成标签")      # 再等2秒
    # 总计 6 秒

# ✅ 一次 LLM 调用
def fast_node(state):
    response = llm.invoke("""
    生成标题、摘要和3个标签，用JSON格式返回:
    {"title": "...", "summary": "...", "tags": ["...", "..."]}
    """)
    # 总计 2 秒
```

### 3. 并行度控制

```python
# 控制并行数量，避免 API 限流
MAX_PARALLEL = 5

def batched_fan_out(state: State) -> list[Send]:
    """分批发送"""
    tasks = state["tasks"]
    batches = [tasks[i:i+MAX_PARALLEL] for i in range(0, len(tasks), MAX_PARALLEL)]

    # 只发第一批（简化示例）
    return [Send("worker", {"task": t}) for t in batches[0]]
```

## 小结

| 模式 | 语法 | 适用场景 |
|------|------|----------|
| 静态并行 | 多条 add_edge | 已知数量的并行 |
| 动态并行 | Send API | 运行时决定数量 |
| 扇入 | 多条入边到同一节点 | 等待全部完成 |
| Map-Reduce | Send + 聚合节点 | 批量处理同质任务 |

| 要点 | 说明 |
|------|------|
| Send | 动态分发并行任务 |
| operator.add | 并行结果的合并 reducer |
| 等待机制 | 多入边节点等所有来源完成 |
| 性能 | 并行能显著减少总耗时 |

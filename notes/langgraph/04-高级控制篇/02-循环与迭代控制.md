# 循环与迭代控制

循环是 Agent 区别于简单 Chain 的关键能力。本篇讲解如何在 LangGraph 中实现自循环、反馈环、迭代优化等模式。

---

## 为什么需要循环

### 线性流程的局限

```
[用户提问] → [检索] → [生成] → [输出]

问题：
- 如果检索结果不好怎么办？→ 不能回头重新检索
- 如果生成的答案质量差？→ 不能修改后再生成
- 如果需要多轮工具调用？→ 不能循环执行
```

### 循环解决什么

```
[用户提问] → [检索] → [评估] ──不好──→ [改写查询] → [重新检索]
                           │                              │
                           └───好──────────────────────→ [生成]
                                                          │
                                                     [质量检查]
                                                          │
                                              不满意 ──→ [修改] → [重新生成]
                                                          │
                                              满意 ──→ [输出]
```

## 循环的基本模式

### 模式一：Agent 工具循环

最经典的循环——LLM 决定是否调用工具：

```python
def agent(state: State) -> dict:
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response]}

def should_continue(state: State) -> str:
    last_msg = state["messages"][-1]
    if last_msg.tool_calls:
        return "tools"
    return END

builder.add_edge(START, "agent")
builder.add_conditional_edges("agent", should_continue, {
    "tools": "tools",
    END: END
})
builder.add_edge("tools", "agent")  # ← 关键：回到 agent 形成循环
```

```
START → agent ←─── tools
          │            │
          │  (有工具调用)│
          └────────────┘
          │
          │ (无工具调用)
          ▼
         END
```

### 模式二：质量迭代

```python
def generate(state: State) -> dict:
    draft = llm.invoke("生成答案...")
    return {"draft": draft.content, "iteration": state.get("iteration", 0) + 1}

def evaluate(state: State) -> dict:
    score = score_answer(state["draft"])
    return {"quality_score": score}

def should_refine(state: State) -> str:
    if state["quality_score"] < 0.7 and state.get("iteration", 0) < 3:
        return "refine"    # 质量不够，继续迭代
    return "output"        # 质量够了或超过迭代次数

def refine(state: State) -> dict:
    improved = llm.invoke(f"改进这个答案: {state['draft']}")
    return {"draft": improved.content}

# 构建循环
builder.add_edge(START, "generate")
builder.add_edge("generate", "evaluate")
builder.add_conditional_edges("evaluate", should_refine, {
    "refine": "refine",
    "output": "output",
})
builder.add_edge("refine", "evaluate")  # ← 重新评估
```

```
START → generate → evaluate ←─── refine
                     │               │
                     │  (质量不够)    │
                     └───────────────┘
                     │
                     │ (质量够了)
                     ▼
                   output → END
```

### 模式三：检索-重试循环

```python
def retrieve(state: State) -> dict:
    docs = vector_store.search(state["query"])
    return {"docs": docs, "search_count": state.get("search_count", 0) + 1}

def assess_docs(state: State) -> dict:
    quality = assess_retrieval_quality(state["docs"])
    return {"retrieval_quality": quality}

def route_after_retrieval(state: State) -> str:
    if state["retrieval_quality"] < 0.5 and state["search_count"] < 3:
        return "rewrite"     # 文档质量差，改写查询
    return "generate"        # 文档够好

def rewrite_query(state: State) -> dict:
    new_query = llm.invoke(f"改写查询以获得更好结果: {state['query']}")
    return {"query": new_query.content}

builder.add_edge(START, "retrieve")
builder.add_edge("retrieve", "assess_docs")
builder.add_conditional_edges("assess_docs", route_after_retrieval, {
    "rewrite": "rewrite_query",
    "generate": "generate",
})
builder.add_edge("rewrite_query", "retrieve")  # ← 重新检索
```

## 递归上限控制

### 默认限制

```python
# LangGraph 默认最多 25 步
# 超过会抛出 GraphRecursionError

# 调整上限
result = graph.invoke(
    input,
    config={"recursion_limit": 50}
)
```

### 安全的循环设计

```python
def should_continue(state: State) -> str:
    """带安全检查的循环控制"""
    # 安全阀1: 步数检查
    step = state.get("step_count", 0)
    if step >= 10:
        return "force_end"

    # 安全阀2: 错误检查
    if state.get("error_count", 0) >= 3:
        return "error_handler"

    # 安全阀3: 超时检查
    if state.get("elapsed_time", 0) > 60:  # 60秒
        return "timeout_handler"

    # 正常条件
    if state.get("task_complete"):
        return END

    return "continue"
```

## 循环中的状态累积

### 迭代改进模式

```python
class State(TypedDict):
    messages: Annotated[list, add_messages]
    draft: str
    feedback_history: Annotated[list, operator.add]  # 反馈累积
    iteration: int

def generate_or_refine(state: State) -> dict:
    iteration = state.get("iteration", 0)

    if iteration == 0:
        # 第一次生成
        prompt = f"写一篇关于{state['topic']}的文章"
    else:
        # 基于反馈改进
        feedback = "\n".join(state["feedback_history"])
        prompt = f"""
        改进以下文章:
        {state['draft']}

        之前几轮的反馈:
        {feedback}
        """

    response = llm.invoke(prompt)
    return {"draft": response.content, "iteration": iteration + 1}

def critique(state: State) -> dict:
    feedback = llm.invoke(f"评价这个文章的不足: {state['draft']}")
    return {
        "feedback_history": [feedback.content],  # 追加到历史
        "latest_feedback": feedback.content
    }

def should_continue(state: State) -> str:
    if state["iteration"] >= 3:  # 最多迭代3次
        return END
    return "refine"
```

## 并行循环

多个独立的循环可以并行运行：

```python
from langgraph.constants import Send

def parallel_tasks(state: State) -> list[Send]:
    """分发多个并行任务"""
    return [
        Send("worker", {"task": task})
        for task in state["tasks"]
    ]

# 每个 worker 有自己的执行循环
def worker_should_continue(state) -> str:
    if state.get("done"):
        return END
    return "process"
```

## 完整示例：自我改进的研究 Agent

```python
"""
研究 Agent: 搜索→评估→补充→生成→审查→修改
"""
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
import operator

class ResearchState(TypedDict):
    messages: Annotated[list, add_messages]
    topic: str
    search_results: Annotated[list, operator.add]
    outline: str
    draft: str
    review_feedback: str
    iteration: int
    final_report: str


def search(state: ResearchState) -> dict:
    results = search_web(state["topic"])
    return {
        "search_results": results,
        "iteration": state.get("iteration", 0) + 1
    }

def create_outline(state: ResearchState) -> dict:
    context = "\n".join(state["search_results"][:5])
    outline = llm.invoke(f"基于以下信息创建大纲:\n{context}\n主题: {state['topic']}")
    return {"outline": outline.content}

def write_draft(state: ResearchState) -> dict:
    draft = llm.invoke(f"按大纲写文章:\n大纲: {state['outline']}\n资料: {state['search_results']}")
    return {"draft": draft.content}

def review(state: ResearchState) -> dict:
    feedback = llm.invoke(f"审查这篇文章的质量:\n{state['draft']}")
    return {"review_feedback": feedback.content}

def should_revise(state: ResearchState) -> str:
    if "满意" in state["review_feedback"] or state["iteration"] >= 3:
        return "finalize"
    return "revise"

def revise(state: ResearchState) -> dict:
    revised = llm.invoke(f"""
    根据反馈修改文章:
    原文: {state['draft']}
    反馈: {state['review_feedback']}
    """)
    return {"draft": revised.content}

def finalize(state: ResearchState) -> dict:
    return {"final_report": state["draft"]}


# 构建图
builder = StateGraph(ResearchState)
builder.add_node("search", search)
builder.add_node("outline", create_outline)
builder.add_node("write", write_draft)
builder.add_node("review", review)
builder.add_node("revise", revise)
builder.add_node("finalize", finalize)

builder.add_edge(START, "search")
builder.add_edge("search", "outline")
builder.add_edge("outline", "write")
builder.add_edge("write", "review")
builder.add_conditional_edges("review", should_revise, {
    "revise": "revise",
    "finalize": "finalize",
})
builder.add_edge("revise", "review")  # ← 修改后重新审查
builder.add_edge("finalize", END)

graph = builder.compile()
```

图结构：

```
START → search → outline → write → review ←─── revise
                                        │             │
                                        │   (需修改)   │
                                        └─────────────┘
                                        │
                                        │ (满意)
                                        ▼
                                     finalize → END
```

## 循环控制最佳实践

| 实践 | 说明 |
|------|------|
| 设置最大迭代 | 防止无限循环 |
| 累积反馈 | 每轮改进都记录 |
| 质量评估 | 有客观的退出标准 |
| 状态监控 | 记录每轮的关键指标 |
| 超时保护 | 设置时间上限 |
| 错误恢复 | 循环出错时降级处理 |

---

## 小结

| 循环模式 | 典型场景 | 控制方式 |
|----------|----------|----------|
| Agent工具循环 | LLM自主调用工具 | tool_calls 是否为空 |
| 质量迭代 | 逐步改进输出 | 质量评分 + 最大次数 |
| 检索重试 | 改写查询重新检索 | 文档质量评估 |
| 审查-修改 | 写作/代码生成 | 审查反馈 |
| 并行循环 | 多任务独立处理 | Send API |

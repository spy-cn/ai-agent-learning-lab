# 图的编译与运行

本篇深入讲解 StateGraph 的编译过程、运行方式、流式输出和各种运行配置。

---

## 编译（Compile）

### 基本编译

```python
from langgraph.graph import StateGraph, START, END

builder = StateGraph(State)
# ... 添加节点和边 ...

# 基本编译
graph = builder.compile()
```

### 编译做了什么

```
编译前（builder）                    编译后（compiled graph）
┌──────────────────┐               ┌──────────────────────┐
│ 节点定义列表      │               │ Pregel 执行器         │
│ 边定义列表        │  ────────►   │ 拓扑排序              │
│ 状态 Schema      │   compile()  │ Reducer 映射          │
│ （尚未可执行）    │               │ Checkpointer 接入点    │
└──────────────────┘               │ 中断点配置            │
                                    │ （可执行）             │
                                    └──────────────────────┘
```

### 编译参数

```python
graph = builder.compile(
    checkpointer=MemorySaver(),       # 状态持久化
    interrupt_before=["approval"],    # 在指定节点前暂停
    interrupt_after=["draft"],        # 在指定节点后暂停
)
```

## 运行方式

### 1. invoke（同步执行）

最常用的方式——执行完整图，返回最终结果：

```python
result = graph.invoke({"messages": [("user", "你好")]})
```

特点：
- 同步阻塞，等待执行完成
- 返回最终完整状态
- 适合不需要中间结果的场景

### 2. stream（流式执行）

逐步输出中间结果：

```python
for event in graph.stream({"messages": [("user", "你好")]}):
    print(event)
```

### 3. ainvoke / astream（异步执行）

```python
result = await graph.ainvoke({"messages": [("user", "你好")]})

async for event in graph.astream({"messages": [("user", "你好")]}):
    print(event)
```

---

## stream 的四种模式

LangGraph 的 stream 支持四种输出模式：

### 模式一：values（默认）

输出每一步的**完整状态**：

```python
for event in graph.stream(input, stream_mode="values"):
    print(event)
# 输出:
# {"messages": [HumanMessage("你好")]}                    ← 初始状态
# {"messages": [HumanMessage("你好"), AIMessage("你好！")]}  ← 第一步后
# {"messages": [..., AIMessage("我能帮你什么？")]}          ← 第二步后
```

### 模式二：updates

输出每一步的**状态更新**（增量）：

```python
for event in graph.stream(input, stream_mode="updates"):
    print(event)
# 输出:
# {"node_name": {"messages": [AIMessage("你好！")]}}  ← 只输出新增部分
# {"node_name": {"messages": [AIMessage("...")]}}
```

### 模式三：debug

输出详细的调试信息：

```python
for event in graph.stream(input, stream_mode="debug"):
    print(event)
# 输出:
# {"type": "task", "payload": {"node": "agent", ...}}
# {"type": "task_result", "payload": {"node": "agent", ...}}
```

### 模式四：messages（Token 级流式）

输出 LLM 生成的每一个 token：

```python
for msg, metadata in graph.stream(input, stream_mode="messages"):
    if msg.content:
        print(msg.content, end="", flush=True)
# 逐字输出: 你 好 ！ 我 是 A I 助 手 ...
```

### 多模式组合

```python
for event in graph.stream(
    input,
    stream_mode=["updates", "messages"]  # 同时输出两种模式
):
    mode, data = event
    if mode == "updates":
        print(f"状态更新: {data}")
    elif mode == "messages":
        print(data[0].content, end="")
```

## 运行配置（Config）

### 基本 Config

```python
config = {
    "configurable": {
        "thread_id": "conversation-1",     # 对话线程ID
    },
    "recursion_limit": 50,                  # 最大递归步数
}

result = graph.invoke(input, config=config)
```

### thread_id（对话隔离）

```python
# 不同的 thread_id 代表不同的对话
config_a = {"configurable": {"thread_id": "user-alice"}}
config_b = {"configurable": {"thread_id": "user-bob"}}

# Alice 的对话
graph.invoke({"messages": [("user", "我是Alice")]}, config=config_a)

# Bob 的对话（与 Alice 独立）
graph.invoke({"messages": [("user", "我是Bob")]}, config=config_b)

# Alice 继续对话
graph.invoke({"messages": [("user", "我叫什么？")]}, config=config_a)
# AI 知道是 Alice（因为同一个 thread_id）
```

### recursion_limit（循环上限）

```python
# 默认 25 步
# 如果 Agent 循环较多，可以调大
result = graph.invoke(
    input,
    config={"recursion_limit": 100}
)
```

> ⚠️ 超过限制会抛出 `GraphRecursionError`。

### 自定义参数传递

```python
config = {
    "configurable": {
        "thread_id": "conv-1",
        "model": "gpt-4o",           # 自定义参数
        "temperature": 0.3,
        "max_tokens": 2000,
    }
}

# 在节点函数中读取
def my_node(state, config):
    model = config["configurable"]["model"]
    temp = config["configurable"]["temperature"]
    llm = ChatOpenAI(model=model, temperature=temp)
    ...
```

## 中断与恢复

### interrupt_before

```python
graph = builder.compile(
    checkpointer=MemorySaver(),
    interrupt_before=["approval_node"]
)

config = {"configurable": {"thread_id": "1"}}

# 执行到 approval_node 前会暂停
result = graph.invoke(input, config=config)

# 此时图处于暂停状态
# 可以检查当前状态
state = graph.get_state(config)
print(state.values)    # 当前状态值
print(state.next)      # ['approval_node']  下一步要执行的

# 恢复执行（传 None 表示继续）
result = graph.invoke(None, config=config)
```

### interrupt_after

```python
graph = builder.compile(
    checkpointer=MemorySaver(),
    interrupt_after=["generate_draft"]
)

# 执行完 generate_draft 后暂停
result = graph.invoke(input, config=config)

# 可以查看草稿
state = graph.get_state(config)
draft = state.values.get("draft")
print(f"草稿: {draft}")

# 满意后继续
result = graph.invoke(None, config=config)
```

### 更新状态后恢复

```python
# 暂停后，可以手动修改状态
graph.update_state(config, {"draft": "这是修改后的草稿"})

# 然后继续执行
result = graph.invoke(None, config=config)
```

## 状态管理 API

### get_state

```python
# 获取当前状态
state = graph.get_state(config)

print(state.values)       # 完整状态字典
print(state.next)         # 下一个要执行的节点列表
print(state.config)       # 当前配置
print(state.metadata)     # 元数据（步数等）
```

### get_state_history

```python
# 获取所有历史状态
for snapshot in graph.get_state_history(config):
    print(f"Step {snapshot.metadata['step']}: {snapshot.next}")
    print(f"  State: {list(snapshot.values.keys())}")
```

### update_state

```python
# 手动更新状态
graph.update_state(
    config,
    {"messages": [HumanMessage("手动添加的消息")]}
)

# 也可以模拟某个节点的输出
graph.update_state(
    config,
    {"answer": "这是手动设置的答案"},
    as_node="generate_node"  # 模拟 generate_node 的输出
)
```

## 完整示例：带记忆和中断的对话

```python
"""
完整的对话 Agent：带记忆 + 中断审批
"""
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langgraph.checkpoint.memory import MemorySaver
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage

class State(TypedDict):
    messages: Annotated[list, add_messages]
    draft: str
    approved: bool

llm = ChatOpenAI(model="gpt-4o")

def generate_draft(state: State) -> dict:
    response = llm.invoke(
        state["messages"] + [HumanMessage("请生成一个回复草稿")]
    )
    return {"draft": response.content}

def review_and_approve(state: State) -> dict:
    # 在实际应用中，这里可能是人工审核
    # 我们先自动通过
    return {"approved": True}

def final_output(state: State) -> dict:
    return {"messages": [HumanMessage(state["draft"])]}

# 构建图
builder = StateGraph(State)
builder.add_node("draft", generate_draft)
builder.add_node("review", review_and_approve)
builder.add_node("output", final_output)

builder.add_edge(START, "draft")
builder.add_edge("draft", "review")
builder.add_edge("review", "output")
builder.add_edge("output", END)

# 编译（带 checkpointer 和中断点）
graph = builder.compile(
    checkpointer=MemorySaver(),
    interrupt_before=["review"]  # 在审核前暂停
)

# 运行
config = {"configurable": {"thread_id": "demo-1"}}

print("=== 第一次调用（到 review 前暂停）===")
result = graph.invoke(
    {"messages": [HumanMessage("写一封商务邮件给客户")]},
    config=config
)
state = graph.get_state(config)
print(f"草稿: {state.values.get('draft', 'N/A')[:100]}...")
print(f"下一步: {state.next}")  # ['review']

print("\n=== 恢复执行 ===")
result = graph.invoke(None, config=config)
print(f"最终消息数: {len(result['messages'])}")
```

## 流式输出完整示例

```python
"""
多种流式模式演示
"""

# === 模式一：updates（看每步更新） ===
print("=== updates 模式 ===")
for event in graph.stream(input, config=config, stream_mode="updates"):
    for node_name, update in event.items():
        print(f"[{node_name}] 更新了: {list(update.keys())}")

# === 模式二：messages（Token 流式） ===
print("\n=== messages 模式 ===")
for msg, metadata in graph.stream(input, config=config, stream_mode="messages"):
    if msg.content:
        print(msg.content, end="", flush=True)
print()  # 换行

# === 模式三：组合模式 ===
print("\n=== 组合模式 ===")
for event in graph.stream(
    input, config=config,
    stream_mode=["updates", "messages"]
):
    mode, data = event
    if mode == "updates":
        for node, update in data.items():
            print(f"\n[{node}] 完成")
    elif mode == "messages":
        if data[0].content:
            print(data[0].content, end="", flush=True)
```

## 小结

| 要点 | 说明 |
|------|------|
| 编译 | `builder.compile(checkpointer=, interrupt_before=)` |
| 同步执行 | `graph.invoke(input, config)` |
| 流式执行 | `graph.stream(input, config, stream_mode=)` |
| 异步执行 | `await graph.ainvoke()` / `async for in graph.astream()` |
| 四种 stream 模式 | values / updates / debug / messages |
| thread_id | 对话线程隔离 |
| recursion_limit | 循环上限（默认25） |
| 中断恢复 | `interrupt_before` + `invoke(None, config)` |
| 获取状态 | `graph.get_state(config)` |
| 更新状态 | `graph.update_state(config, values)` |

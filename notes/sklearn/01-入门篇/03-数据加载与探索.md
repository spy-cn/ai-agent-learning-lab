# 03 - 数据加载与探索

## sklearn 内置数据集

sklearn 提供了丰富的练习数据集，无需下载，一行代码即可加载。

### 数据集类型

```python
from sklearn import datasets

# 三种加载方式：
# 1. Toy datasets（小数据集，直接内置）
iris = datasets.load_iris()              # 返回 Bunch 对象

# 2. Real-world datasets（需下载）
housing = datasets.fetch_california_housing()  # 自动下载

# 3. 生成数据（用于测试和学习）
X, y = datasets.make_classification(n_samples=100, n_features=4)
```

### Bunch 对象结构

```python
from sklearn.datasets import load_iris

iris = load_iris()
print(type(iris))  # <class 'sklearn.utils._bunch.Bunch'>

# Bunch 就像一个字典，但可以用 . 访问属性
print(iris.keys())

# 输出：
# dict_keys(['data', 'target', 'frame', 'target_names',
#            'DESCR', 'feature_names', 'filename'])
```

| 属性 | 说明 | 示例 |
|------|------|------|
| `data` | 特征矩阵 X | 形状 (150, 4) |
| `target` | 标签向量 y | 形状 (150,) |
| `feature_names` | 特征名称 | ['sepal length', 'sepal width', ...] |
| `target_names` | 类别名称 | ['setosa', 'versicolor', 'virginica'] |
| `DESCR` | 数据集描述文档 | 一段详细的文字说明 |

## 实战：加载并探索 Iris 数据集

```python
from sklearn.datasets import load_iris
import pandas as pd
import numpy as np

# 加载数据
iris = load_iris()

# 转为 DataFrame 更方便探索
df = pd.DataFrame(iris.data, columns=iris.feature_names)
df['target'] = iris.target
df['species'] = df['target'].map({
    0: 'setosa',
    1: 'versicolor',
    2: 'virginica'
})

print(df.head())
```

输出：
```
   sepal length (cm)  sepal width (cm)  petal length (cm)  petal width (cm)  target    species
0                5.1               3.5                1.4               0.2       0     setosa
1                4.9               3.0                1.4               0.2       0     setosa
2                4.7               3.2                1.3               0.2       0     setosa
3                4.6               3.1                1.5               0.2       0     setosa
4                5.0               3.6                1.4               0.2       0     setosa
```

### 基本统计信息

```python
# 数据基本信息
```

## 数据可视化探索

```python
import matplotlib.pyplot as plt

# 设置中文字体（可选）
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

fig, axes = plt.subplots(2, 2, figsize=(12, 10))

features = iris.feature_names
colors = ['red', 'green', 'blue']

for idx, feature in enumerate(features):
    ax = axes[idx // 2][idx % 2]
    for target, color in zip([0, 1, 2], colors):
        subset = df[df['target'] == target]
        ax.hist(subset[feature], bins=15, alpha=0.6,
                color=color, label=iris.target_names[target])
    ax.set_xlabel(feature)
    ax.set_ylabel('频数')
    ax.legend()

plt.tight_layout()
plt.savefig('iris_distribution.png', dpi=100)
plt.show()
```

### 散点图矩阵（更直观）

```python
from pandas.plotting import scatter_matrix

scatter_matrix(df[features], c=df['target'], figsize=(12, 12),
               marker='o', hist_kwds={'bins': 20}, s=60, alpha=.8)
plt.suptitle('Iris 散点图矩阵', fontsize=16)
plt.show()
```

> 💡 从散点图矩阵可以一眼看出哪些特征对分类最有区分度。比如 `petal length` vs `petal width` 能很好地分开三类花。

## 生成模拟数据

学习和测试算法时，可以生成特定分布的数据：

```python
from sklearn.datasets import make_classification, make_regression, make_blobs
import matplotlib.pyplot as plt

# 1. 分类数据
X, y = make_classification(
    n_samples=500,       # 样本数
    n_features=2,        # 特征数（设为2方便可视化）
    n_informative=2,     # 有信息的特征数
    n_redundant=0,       # 冗余特征数
    n_clusters_per_class=1,
    n_classes=3,         # 类别数
    random_state=42
)

plt.figure(figsize=(6, 5))
plt.scatter(X[:, 0], X[:, 1], c=y, cmap='viridis', s=50, alpha=0.7)
plt.title('生成的分类数据')
plt.colorbar(label='类别')
plt.show()

# 2. 回归数据
X, y = make_regression(
    n_samples=200,
    n_features=1,        # 单特征
    noise=10,            # 加入噪声
    random_state=42
)
plt.scatter(X, y, s=10)
plt.title('生成的回归数据')
plt.show()

# 3. 聚类数据（团块状）
X, y = make_blobs(
    n_samples=300,
    centers=4,           # 4个簇
    cluster_std=0.8,     # 簇的标准差
    random_state=42
)
plt.scatter(X[:, 0], X[:, 1], c=y, cmap='viridis', s=50)
plt.title('生成的聚类数据')
plt.show()
```

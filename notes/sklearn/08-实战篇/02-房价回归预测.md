## 完整流程

### 数据加载与 EDA

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('house_prices.csv')
print(f"数据形状: {df.shape}")

# 目标变量分布
plt.figure(figsize=(10, 4))
plt.subplot(1, 2, 1)
sns.histplot(df['SalePrice'], kde=True)
plt.title('房价分布（右偏）')

# 对数变换
plt.subplot(1, 2, 2)
sns.histplot(np.log1p(df['SalePrice']), kde=True, color='green')
plt.title('对数变换后（更正态）')
plt.tight_layout()
plt.show()

# 用对数变换后的目标
y = np.log1p(df['SalePrice'])
```

### 特征工程

```python
# 缺失值处理
missing = df.isnull().sum()
missing = missing[missing > 0].sort_values(ascending=False)
print(missing.head(10))

# 区分特征类型
numeric_features = df.select_dtypes(include=[np.number]).columns.drop('SalePrice').tolist()
categorical_features = df.select_dtypes(exclude=[np.number]).columns.tolist()
```

### Pipeline

```python
from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.impute import SimpleImputer
from sklearn.ensemble import GradientBoostingRegressor

preprocessor = ColumnTransformer([
    ('num', Pipeline([
        ('imputer', SimpleImputer(strategy='median')),
        ('scaler', StandardScaler())
    ]), numeric_features),
    ('cat', Pipeline([
        ('imputer', SimpleImputer(strategy='constant', fill_value='Missing')),
        ('onehot', OneHotEncoder(handle_unknown='ignore', sparse_output=False))
    ]), categorical_features)
])

pipe = Pipeline([
    ('preprocessor', preprocessor),
    ('model', GradientBoostingRegressor(random_state=42))
])
```

### 评估

```python
from sklearn.model_selection import cross_val_score
from sklearn.metrics import mean_squared_error

scores = cross_val_score(pipe, df.drop('SalePrice', axis=1), y,
                         cv=5, scoring='neg_root_mean_squared_error')
print(f"RMSE: {-scores.mean():.4f} ± {scores.std():.4f}")
```

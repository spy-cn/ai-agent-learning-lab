## 完整项目模板代码

```python
"""
端到端 ML 项目模板
"""

# ===== 1. 配置 =====
import yaml
import numpy as np
import pandas as pd
from pathlib import Path

# 固定随机种子，保证可复现
SEED = 42
np.random.seed(SEED)

# ===== 2. 数据加载 =====
def load_data(path):
    """加载数据"""
    df = pd.read_csv(path)
    print(f"数据形状: {df.shape}")
    return df

# ===== 3. 数据划分 =====
from sklearn.model_selection import train_test_split

def split_data(df, target, test_size=0.2):
    """划分训练/测试集"""
    X = df.drop(target, axis=1)
    y = df[target]
    return train_test_split(X, y, test_size=test_size,
                           random_state=SEED, stratify=y)

# ===== 4. Pipeline 构建 =====
from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.impute import SimpleImputer

def build_pipeline(model):
    """构建完整 Pipeline"""
    numeric_features = X_train.select_dtypes(include=[np.number]).columns
    categorical_features = X_train.select_dtypes(exclude=[np.number]).columns

    preprocessor = ColumnTransformer([
        ('num', Pipeline([
            ('imputer', SimpleImputer(strategy='median')),
            ('scaler', StandardScaler())
        ]), numeric_features),
        ('cat', Pipeline([
            ('imputer', SimpleImputer(strategy='most_frequent')),
            ('onehot', OneHotEncoder(handle_unknown='ignore'))
        ]), categorical_features)
    ])

    return Pipeline([
        ('preprocessor', preprocessor),
        ('model', model)
    ])

# ===== 5. 训练 =====
from sklearn.model_selection import cross_val_score

def train_and_evaluate(pipe, X_train, y_train, X_test, y_test):
    """训练并评估"""
    pipe.fit(X_train, y_train)
    train_score = pipe.score(X_train, y_train)
    test_score = pipe.score(X_test, y_test)
    cv_scores = cross_val_score(pipe, X_train, y_train, cv=5)

    print(f"训练集: {train_score:.4f}")
    print(f"测试集: {test_score:.4f}")
    print(f"CV:     {cv_scores.mean():.4f} ± {cv_scores.std():.4f}")

    return pipe

# ===== 6. 调参 =====
from sklearn.model_selection import GridSearchCV

def tune_hyperparameters(pipe, param_grid, X_train, y_train):
    """网格搜索调参"""
    grid = GridSearchCV(pipe, param_grid, cv=5, n_jobs=-1, verbose=1)
    grid.fit(X_train, y_train)
    print(f"最佳参数: {grid.best_params_}")
    print(f"最佳分数: {grid.best_score_:.4f}")
    return grid.best_estimator_

# ===== 7. 保存 =====
import joblib

def save_model(model, path):
    """保存模型"""
    joblib.dump(model, path)
    print(f"模型已保存: {path}")

# ===== 8. 主流程 =====
if __name__ == '__main__':
    df = load_data('data/raw/data.csv')
    X_train, X_test, y_train, y_test = split_data(df, 'target')

    pipe = build_pipeline(RandomForestClassifier(random_state=SEED))
    pipe = train_and_evaluate(pipe, X_train, y_train, X_test, y_test)

    # 调参
    param_grid = {'model__n_estimators': [100, 200], 'model__max_depth': [10, 20]}
    best_model = tune_hyperparameters(pipe, param_grid, X_train, y_train)

    save_model(best_model, 'models/best_model.joblib')
```

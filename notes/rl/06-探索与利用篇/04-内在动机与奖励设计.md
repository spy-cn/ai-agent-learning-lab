# 内在动机与奖励设计

内在动机（Intrinsic Motivation）是生物系统的一个特征——出于内在驱动而非外部奖励去探索世界。在RL中，内在动机通过**内在奖励**来促进探索和学习。

## 外在奖励 vs 内在奖励

```
┌──────────────────────────────────────────────────┐
│                  奖励类型对比                      │
├──────────────────┬───────────────────────────────┤
│   外在奖励        │      内在奖励                  │
│   (Extrinsic)    │   (Intrinsic)                 │
├──────────────────┼───────────────────────────────┤
│ 环境提供          │ 智能体自己产生                 │
│ 稀疏              │ 密集                          │
│ 任务相关          │ 与具体任务无关                 │
│ 固定              │ 随学习动态变化                 │
│ 例: 游戏得分      │ 例: 好奇心、掌控感             │
└──────────────────┴───────────────────────────────┘

总奖励: r_total = r_extrinsic + β · r_intrinsic
```

## 奖励设计

### 奖励塑造（Reward Shaping）

修改奖励函数不改变最优策略的方法：

$$F(s, a, s') = \gamma \Phi(s') - \Phi(s)$$

其中 Φ(s) 是势函数。新的奖励 $r' = r + F$ 不改变最优策略。

```python
def potential_based_shaping(reward, state, next_state, gamma, phi):
    """基于势函数的奖励塑造（保证不改变最优策略）"""
    shaping = gamma * phi(next_state) - phi(state)
    return reward + shaping

# 示例: 用距离目标的距离作为势函数
def distance_potential(state, goal):
    """距离目标的负距离作为势函数"""
    return -np.linalg.norm(state - goal)

# 这样塑造后的奖励更密集，加速学习
```

### 常见奖励设计模式

```python
# === 1. 稀疏→密集 ===

# 稀疏: 只有到达目标才有奖励
def sparse_reward(state, goal):
    return 1.0 if np.linalg.norm(state - goal) < 0.1 else 0.0

# 密集: 用距离作为连续奖励
def dense_reward(state, goal):
    return -np.linalg.norm(state - goal)

# 潜在问题: 可能导致"绕圈"行为（一直接近但不到达）


# === 2. 时间惩罚 ===

def time_penalty_reward(reward, step_penalty=-0.01):
    """每步小惩罚，鼓励快速到达"""
    return reward + step_penalty


# === 3. 存活奖励 ===

def survival_reward(reward, alive_bonus=0.1):
    """每存活一步给小奖励"""
    return reward + alive_bonus


# === 4. 动作成本 ===

def action_penalty_reward(reward, action, penalty=0.01):
    """动作越大，惩罚越大（鼓励节能）"""
    return reward - penalty * np.sum(action**2)


# === 5. 组合奖励 ===

def composite_reward(env_reward, state, action, next_state, goal):
    """组合多种奖励"""
    distance_reward = -np.linalg.norm(next_state - goal)
    time_penalty = -0.01
    action_cost = -0.001 * np.sum(action**2)
    goal_bonus = 10.0 if np.linalg.norm(next_state - goal) < 0.1 else 0.0
    
    return env_reward + 0.1 * distance_reward + time_penalty + action_cost + goal_bonus
```

## 内在奖励的计算方法

### 1. 预测误差（Predictive Error）

```python
class PredictionErrorReward:
    """前向模型预测误差作为内在奖励"""
    
    def __init__(self, state_dim, action_dim):
        self.forward_model = ForwardModel(state_dim, action_dim)
    
    def get_reward(self, state, action, next_state):
        """预测误差越大，内在奖励越高"""
        pred_next = self.forward_model(state, action)
        error = F.mse_loss(pred_next, next_state, reduction='none').sum(-1)
        return error
```

### 2. 信息增益（Information Gain）

```python
class InformationGainReward:
    """信息增益: 选择能减少不确定性的动作"""
    
    def __init__(self, state_dim, n_models=5):
        self.models = [ForwardModel(state_dim) for _ in range(n_models)]
    
    def get_reward(self, state, action, next_state):
        """模型预测的不一致程度"""
        predictions = []
        for model in self.models:
            pred = model(state, action)
            predictions.append(pred)
        
        predictions = torch.stack(predictions)
        # 模型间分歧越大 → 不确定性越大 → 信息增益越大
        disagreement = predictions.var(dim=0).mean()
        return disagreement
```

### 3. 赋权（Empowerment）

```python
class EmpowermentReward:
    """赋权: 智能体对未来状态的控制能力"""
    
    def __init__(self, state_dim, action_dim):
        # 逆向模型: 给定(s, s')，预测a
        self.inverse_model = InverseModel(state_dim, action_dim)
    
    def get_reward(self, state, next_state):
        """
        如果从s到s'有多个不同的动作路径
        → 智能体对s'有高控制力 → 高赋权
        如果只有一种动作能达到s' → 低赋权
        """
        # 简化: 用逆向模型的熵作为赋权的近似
        action_logits = self.inverse_model(state, next_state)
        action_probs = F.softmax(action_logits, dim=-1)
        entropy = -(action_probs * torch.log(action_probs + 1e-8)).sum(-1)
        return entropy
```

### 4. 变分信息最大化

```python
class VariationalInformationReward(nn.Module):
    """基于变分推断的信息奖励"""
    
    def __init__(self, state_dim, action_dim, latent_dim=32):
        super().__init__()
        # 编码器: q(z|s,a,s')
        self.encoder = nn.Sequential(
            nn.Linear(state_dim*2 + action_dim, 128),
            nn.ReLU(),
            nn.Linear(128, latent_dim * 2)  # μ和σ
        )
        # 解码器: p(s'|s,a,z)
        self.decoder = nn.Sequential(
            nn.Linear(state_dim + action_dim + latent_dim, 128),
            nn.ReLU(),
            nn.Linear(128, state_dim)
        )
    
    def get_reward(self, state, action, next_state):
        """互信息的变分下界"""
        # encode
        enc_input = torch.cat([state, action, next_state], dim=-1)
        enc_out = self.encoder(enc_input)
        mu, log_var = enc_out.chunk(2, dim=-1)
        
        # 重参数化采样
        std = (0.5 * log_var).exp()
        eps = torch.randn_like(std)
        z = mu + eps * std
        
        # decode
        dec_input = torch.cat([state, action, z], dim=-1)
        pred_next = self.decoder(dec_input)
        
        # 重建损失
        reconst_loss = F.mse_loss(pred_next, next_state, reduction='none').sum(-1)
        
        # KL散度
        kl_loss = -0.5 * (1 + log_var - mu.pow(2) - log_var.exp()).sum(-1)
        
        # 互信息下界 = -reconst_loss - kl_loss
        mi_lower_bound = -reconst_loss - 0.1 * kl_loss
        
        return mi_lower_bound
```

## 奖励缩放与归一化

```python
class RewardNormalizer:
    """在线奖励归一化"""
    
    def __init__(self, gamma=0.99):
        self.gamma = gamma
        self.mean = 0.0
        self.var = 1.0
        self.count = 1e-4
    
    def update(self, reward):
        """更新统计量"""
        self.count += 1
        delta = reward - self.mean
        self.mean += delta / self.count
        self.var += delta * (reward - self.mean)
    
    def normalize(self, reward):
        """归一化奖励"""
        std = np.sqrt(self.var / self.count + 1e-8)
        return (reward - self.mean) / std


# 内在奖励和外在奖励分别归一化
class DualRewardNormalizer:
    """分别归一化内在和外在奖励"""
    
    def __init__(self):
        self.ext_normalizer = RewardNormalizer()
        self.int_normalizer = RewardNormalizer()
    
    def normalize(self, ext_reward, int_reward, int_coef=0.1):
        self.ext_normalizer.update(ext_reward)
        self.int_normalizer.update(int_reward)
        
        norm_ext = self.ext_normalizer.normalize(ext_reward)
        norm_int = self.int_normalizer.normalize(int_reward)
        
        return norm_ext + int_coef * norm_int
```

## 内在奖励的衰减

```python
class IntrinsicRewardScheduler:
    """内在奖励随时间衰减"""
    
    def __init__(self, start_coef=1.0, end_coef=0.01, decay_steps=100000):
        self.start_coef = start_coef
        self.end_coef = end_coef
        self.decay_steps = decay_steps
        self.current_step = 0
    
    def step(self):
        self.current_step += 1
        # 线性衰减
        coef = self.start_coef - (self.start_coef - self.end_coef) * min(
            self.current_step / self.decay_steps, 1.0
        )
        return coef
```

## 实际案例

### 例1: 机器人导航

```python
def navigation_reward(state, action, next_state, goal, obstacles):
    """机器人导航的奖励设计"""
    reward = 0.0
    
    # 1. 到达目标的大奖励
    dist_to_goal = np.linalg.norm(next_state[:2] - goal)
    if dist_to_goal < 0.3:
        reward += 10.0
    
    # 2. 接近目标的小奖励（密集化）
    prev_dist = np.linalg.norm(state[:2] - goal)
    new_dist = dist_to_goal
    reward += (prev_dist - new_dist) * 1.0  # 进步奖励
    
    # 3. 碰撞惩罚
    for obs in obstacles:
        if np.linalg.norm(next_state[:2] - obs) < 0.2:
            reward -= 5.0
    
    # 4. 时间惩罚
    reward -= 0.01
    
    # 5. 内在奖励（好奇心）
    # + 0.1 * curiosity_model(next_state)
    
    return reward
```

### 例2: RLHF中的奖励模型

```python
class RLHFRewardModel(nn.Module):
    """人类反馈强化学习的奖励模型"""
    
    def __init__(self, state_dim, hidden_dim=256):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1)
        )
    
    def forward(self, state):
        return self.net(state)
    
    def train_from_preferences(self, preferred_states, less_preferred_states):
        """从人类偏好数据训练"""
        # Bradley-Terry模型
        r_preferred = self.net(preferred_states)
        r_less = self.net(less_preferred_states)
        
        # P(preferred > less) = sigmoid(r_pref - r_less)
        loss = -F.logsigmoid(r_preferred - r_less).mean()
        
        return loss
```

## 奖励设计的常见陷阱

### 1. 奖励骇客（Reward Hacking）

```python
# 智能体找到了获取高奖励的"漏洞"，但不是预期行为

# 示例: 赛车游戏
# 目标: 完成赛道
# 奖励: 基于速度
# 问题: 智能体在一个地方转圈获得高速度奖励

# 解决方案:
# - 奖励设计要考虑目标
# - 添加进度约束
# - 使用多维奖励
```

### 2. 稀疏奖励导致无法学习

```python
# 解决方案:
# 1. 奖励塑造（密集化）
# 2. 内在奖励（好奇心）
# 3. 课程学习（由易到难）
# 4. 模仿学习（提供示范）
```

## 小结

| 奖励类型 | 来源 | 用途 |
|----------|------|------|
| **外在奖励** | 环境 | 任务目标 |
| **预测误差** | 前向模型 | 好奇心探索 |
| **信息增益** | 模型不确定性 | 减少不确定 |
| **赋权** | 逆向模型 | 控制能力 |
| **奖励塑造** | 势函数 | 加速学习 |

# 基于计数的探索

基于计数的探索是最直觉的探索方法：**访问次数少的状态更值得探索**。

## 核心思想

```
好奇心 = 1 / sqrt(N(s))

N(s): 状态s被访问的次数

新状态 N(s)=0:  好奇心=∞  → 强烈探索
常见状态 N(s)=100: 好奇心=0.1 → 很少再去
```

## 理论基础

对于表格型MDP，UCB公式：

$$r_{int}(s) = \frac{c}{\sqrt{N(s)}}$$

理论保证regret为 $O(\sqrt{|S||A|T})$，优于ε-greedy。

## 挑战：连续/高维状态

```
问题: 状态空间太大，几乎不会访问同一状态两次

FrozenLake(16状态):   可行 ✓
CartPole(连续):        不可行 ✗
Atari(图像):           完全不可行 ✗

解决方案:
1. 状态离散化/哈希
2. 密度模型估计访问频率
3. 哈希连续状态
```

## 方法一：伪计数（Pseudo-Count）

用密度模型近似计数：

```python
import numpy as np
import torch
import torch.nn as nn

class CTMDensityModel:
    """Context Tree Switching密度模型"""
    
    def __init__(self, n_symbols):
        self.n_symbols = n_symbols
        self.counts = np.ones(n_symbols)  # 初始为1（拉普拉斯平滑）
        self.total = n_symbols
    
    def update(self, symbol):
        self.counts[symbol] += 1
        self.total += 1
    
    def get_density(self, symbol):
        return self.counts[symbol] / self.total
    
    def get_pseudo_count(self, symbol):
        """伪计数 = 密度 × total"""
        density = self.get_density(symbol)
        # 如果见过这个状态，density > 0
        # 伪计数 ≈ density × total
        return density * self.total


# 更通用的高维密度模型
class NeuralDensityModel(nn.Module):
    """神经网络密度模型"""
    
    def __init__(self, state_dim, hidden_dim=256):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, state_dim * 2)  # 输出均值和方差
        )
    
    def forward(self, state):
        out = self.net(state)
        mean, log_var = out.chunk(2, dim=-1)
        return mean, log_var.exp()
    
    def log_density(self, state):
        """计算log密度"""
        mean, var = self.forward(state)
        log_density = -0.5 * ((state - mean) ** 2 / var + torch.log(2 * np.pi * var))
        return log_density.sum(dim=-1)
```

### 伪计数公式

从密度模型ρ(s)推导伪计数：

$$N(s) = \frac{\rho(s) \cdot (1 - \rho'(s))}{\rho'(s) - \rho(s)}$$

其中 ρ'(s) 是见过s之后的密度。

```python
def pseudo_count(density_before, density_after):
    """从密度变化推导伪计数"""
    if density_after <= density_before:
        return float('inf')
    
    N = density_before * (1 - density_after) / (density_after - density_before)
    return N

# 内在奖励
def count_based_reward(N_s):
    """基于伪计数的内在奖励"""
    return 1.0 / np.sqrt(N_s + 1e-8)
```

## 方法二：状态哈希

```python
import hashlib

class HashBasedCount:
    """基于哈希的状态计数"""
    
    def __init__(self, hash_size=1024):
        self.hash_size = hash_size
        self.counts = np.zeros(hash_size)
    
    def get_hash(self, state):
        """将状态哈希到固定大小"""
        if isinstance(state, np.ndarray):
            state_bytes = state.tobytes()
        else:
            state_bytes = str(state).encode()
        
        h = int(hashlib.md5(state_bytes).hexdigest(), 16) % self.hash_size
        return h
    
    def get_count(self, state):
        """获取状态访问次数"""
        h = self.get_hash(state)
        return self.counts[h]
    
    def update(self, state):
        """更新计数"""
        h = self.get_hash(state)
        self.counts[h] += 1
    
    def get_intrinsic_reward(self, state):
        """基于计数的内在奖励"""
        count = self.get_count(state)
        return 1.0 / np.sqrt(count + 1)


# 局部敏感哈希（LSH）— 适合连续状态
class LSHCount:
    """Locality-Sensitive Hashing for continuous states"""
    
    def __init__(self, state_dim, n_hashes=8, hash_dim=64):
        self.n_hashes = n_hashes
        # 随机投影矩阵
        self.projections = [np.random.randn(hash_dim, state_dim) 
                           for _ in range(n_hashes)]
        self.counts = [{} for _ in range(n_hashes)]
    
    def get_hash(self, state):
        """多哈希函数"""
        hashes = []
        for proj, count_table in zip(self.projections, self.counts):
            projected = proj @ state
            binary = (projected > 0).astype(int)
            h = tuple(binary)
            hashes.append((h, count_table))
        return hashes
    
    def update(self, state):
        for h, count_table in self.get_hash(state):
            count_table[h] = count_table.get(h, 0) + 1
    
    def get_intrinsic_reward(self, state):
        """平均多个哈希的计数"""
        total_reward = 0
        for h, count_table in self.get_hash(state):
            count = count_table.get(h, 0)
            total_reward += 1.0 / np.sqrt(count + 1)
        return total_reward / self.n_hashes
```

## 方法三：SimHash

```python
class SimHash:
    """SimHash: 简单高效的连续状态哈希"""
    
    def __init__(self, state_dim, n_bits=8):
        self.n_bits = n_bits
        # 随机超平面
        self.hyperplanes = np.random.randn(n_bits, state_dim)
        self.counts = {}
    
    def hash(self, state):
        """SimHash: 用超平面将状态映射到位串"""
        projections = self.hyperplanes @ state
        bits = (projections > 0).astype(int)
        return tuple(bits)
    
    def get_count(self, state):
        h = self.hash(state)
        return self.counts.get(h, 0)
    
    def update(self, state):
        h = self.hash(state)
        self.counts[h] = self.counts.get(h, 0) + 1
    
    def get_reward(self, state):
        count = self.get_count(state)
        return 1.0 / np.sqrt(count + 1)
```

## 完整的计数探索Agent

```python
class CountBasedExplorationAgent:
    """完整的基于计数的探索Agent"""
    
    def __init__(self, state_dim, action_dim, 
                 count_method='simhash', int_reward_coef=0.1):
        self.int_reward_coef = int_reward_coef
        
        # 基础RL算法（如DQN）
        self.dqn = DQN(state_dim, action_dim)
        
        # 计数器
        if count_method == 'simhash':
            self.counter = SimHash(state_dim, n_bits=8)
        elif count_method == 'lsh':
            self.counter = LSHCount(state_dim)
        else:
            self.counter = HashBasedCount()
    
    def act(self, state):
        return self.dqn.act(state)
    
    def step(self, state, action, reward, next_state, done):
        # 1. 计算内在奖励
        intrinsic_reward = self.counter.get_reward(next_state)
        
        # 2. 更新计数
        self.counter.update(next_state)
        
        # 3. 总奖励
        total_reward = reward + self.int_reward_coef * intrinsic_reward
        
        # 4. 存入经验回放
        self.dqn.remember(state, action, total_reward, next_state, done)
        
        # 5. 训练DQN
        self.dqn.learn()
```

## 小结

| 方法 | 适用场景 | 效果 |
|------|----------|------|
| **精确计数** | 表格型MDP | 最优 |
| **SimHash** | 连续状态 | 好 |
| **LSH** | 高维状态 | 好 |
| **伪计数** | 密度模型 | 理论优雅 |
| **自编码器** | 图像状态 | 中等 |

基于计数的方法提供了理论清晰的探索框架，但在高维连续空间中需要近似。

# 信息论基础

信息论中的概念在强化学习中广泛应用，特别是在策略优化、探索和最大熵RL中。

## 熵（Entropy）

### 定义

衡量随机变量不确定性的度量：

$$H(X) = -\sum_x P(x) \log P(x) = E_{x \sim P}\left[-\log P(x)\right]$$

```python
import numpy as np

def entropy(probs, base=np.e):
    """计算离散分布的熵"""
    probs = np.array(probs)
    probs = probs[probs > 0]  # 排除0概率项
    log_probs = np.log(probs) / np.log(base)
    return -np.sum(probs * log_probs)

# 不同分布的熵
print("=== 离散分布的熵 ===")
print(f"确定性分布 [1,0,0,0]: H = {entropy([1,0,0,0]):.4f}")  # 0
print(f"偏斜分布 [0.7,0.2,0.1,0]: H = {entropy([0.7,0.2,0.1,0]):.4f}")
print(f"均匀分布 [0.25]*4: H = {entropy([0.25]*4):.4f}")  # ln(4)=1.386 最大
```

### 熵的性质

| 性质 | 说明 |
|------|------|
| **非负性** | H(X) ≥ 0 |
| **确定性最小** | 确定性分布熵为0 |
| **均匀分布最大** | 给定|X|，均匀分布熵最大 = log\|X\| |
| **链式法则** | H(X,Y) = H(X) + H(Y\|X) |

### 连续随机变量的熵（微分熵）

$$H(X) = -\int_{-\infty}^{\infty} f(x) \log f(x) dx$$

```python
# 正态分布的微分熵
# H(N(μ, σ²)) = 0.5 * log(2πeσ²) = 0.5 + 0.5*log(2π) + log(σ)

def gaussian_entropy(sigma):
    """高斯分布的微分熵"""
    return 0.5 + 0.5 * np.log(2 * np.pi) + np.log(sigma)

print("\n=== 高斯分布的微分熵 ===")
for sigma in [0.1, 0.5, 1.0, 2.0]:
    print(f"σ={sigma}: H = {gaussian_entropy(sigma):.4f}")
# σ越大，熵越大（越不确定）
```

### RL中的应用：最大熵RL

SAC（Soft Actor-Critic）使用最大熵目标：

$$J_{maxent} = E\left[\sum_t \gamma^t (r_t + \alpha H(\pi(\cdot|s_t)))\right]$$

```python
# 最大熵RL：奖励 = 环境奖励 + α × 策略熵
# 鼓励策略保持随机性，促进探索

def soft_reward(env_reward, action_probs, alpha=0.2):
    """加入熵奖励的软奖励"""
    policy_entropy = entropy(action_probs)
    return env_reward + alpha * policy_entropy
```

## KL散度（Kullback-Leibler Divergence）

衡量两个概率分布的差异：

$$D_{KL}(P \| Q) = \sum_x P(x) \log \frac{P(x)}{Q(x)} = E_{x \sim P}\left[\log \frac{P(x)}{Q(x)}\right]$$

### 性质

| 性质 | 说明 |
|------|------|
| **非负性** | $D_{KL}(P \| Q) \geq 0$ |
| **非对称性** | $D_{KL}(P \| Q) \neq D_{KL}(Q \| P)$ |
| **等于0** | 当且仅当 P = Q |

```python
def kl_divergence(p, q):
    """KL散度 D_KL(P || Q)"""
    p = np.array(p)
    q = np.array(q)
    mask = p > 0
    return np.sum(p[mask] * np.log(p[mask] / q[mask]))

P = [0.5, 0.3, 0.2]
Q1 = [0.4, 0.4, 0.2]  # 与P相近
Q2 = [0.8, 0.1, 0.1]  # 与P差异大

print(f"D_KL(P||Q1) = {kl_divergence(P, Q1):.6f}")
print(f"D_KL(P||Q2) = {kl_divergence(P, Q2):.6f}")
# Q2与P差异更大 → KL散度更大
```

### RL中的应用

#### 1. TRPO的信任域约束

$$D_{KL}(\pi_{old} \| \pi_{new}) \leq \delta$$

限制新旧策略的差异，保证稳定更新。

#### 2. PPO的惩罚项

```python
def ppo_loss(ratio, advantage, beta=0.01):
    """PPO的损失函数（含KL惩罚）"""
    policy_loss = -np.mean(ratio * advantage)
    kl_penalty = beta * kl_divergence(old_policy, new_policy)
    return policy_loss + kl_penalty
```

#### 3. VAE/World Model中的ELBO

```python
# 变分推断中
# log P(x) ≥ E_Q[log P(x|z)] - D_KL(Q(z|x) || P(z))
#                    └─────────────────────────────┘
#                              ELBO
```

## 交叉熵（Cross-Entropy）

$$H(P, Q) = -\sum_x P(x) \log Q(x) = H(P) + D_{KL}(P \| Q)$$

```python
def cross_entropy(p, q):
    """交叉熵 H(P,Q)"""
    p = np.array(p)
    q = np.array(q)
    mask = p > 0
    return -np.sum(p[mask] * np.log(q[mask]))

# RL中的应用：策略梯度的损失函数
# Loss = -E[log π(a|s) · A(s,a)]
#      = Cross-Entropy(behavior, target) 加权
```

## 互信息（Mutual Information）

衡量两个随机变量的相关性：

$$I(X; Y) = H(X) - H(X|Y) = H(Y) - H(Y|X) = D_{KL}(P(X,Y) \| P(X)P(Y))$$

```python
def mutual_information(joint, marginal_x, marginal_y):
    """互信息 I(X;Y)"""
    mi = 0.0
    for i in range(len(joint)):
        for j in range(len(joint[0])):
            if joint[i][j] > 0:
                mi += joint[i][j] * np.log(
                    joint[i][j] / (marginal_x[i] * marginal_y[j])
                )
    return mi

# RL中的应用：
# 1. 信息瓶颈（Information Bottleneck）
# 2. 表征学习——最大化状态表征与未来回报的互信息
# 3. 好奇心探索——最大化前后状态表征的互信息
```

## JS散度（Jensen-Shannon Divergence）

KL散度的对称化版本：

$$D_{JS}(P \| Q) = \frac{1}{2} D_{KL}(P \| M) + \frac{1}{2} D_{KL}(Q \| M)$$

其中 $M = \frac{1}{2}(P + Q)$。

```python
def js_divergence(p, q):
    """JS散度（对称、有界）"""
    p = np.array(p)
    q = np.array(q)
    m = 0.5 * (p + q)
    return 0.5 * kl_divergence(p, m) + 0.5 * kl_divergence(q, m)

# 性质：
# 1. 对称：D_JS(P||Q) = D_JS(Q||P)
# 2. 有界：0 ≤ D_JS ≤ log(2)
# 3. 三角不等式成立
```

## PyTorch中的分布操作

```python
import torch
import torch.distributions as dist

# === 离散分布 ===
# Categorical（用于离散动作）
logits = torch.randn(4)  # 4个动作的logits
cat_dist = dist.Categorical(logits=logits)

# 采样
action = cat_dist.sample()
print(f"采样动作: {action.item()}")

# 对数概率
log_prob = cat_dist.log_prob(action)
print(f"log π(a): {log_prob.item():.4f}")

# 熵
entropy = cat_dist.entropy()
print(f"熵: {entropy.item():.4f}")

# === 连续分布 ===
# Normal（用于连续动作）
mean = torch.zeros(2)  # 2维动作均值
std = torch.ones(2)
normal_dist = dist.Normal(mean, std)

# 采样
action = normal_dist.sample()
print(f"\n连续动作: {action}")

# 对数概率（多维需要求和）
log_prob = normal_dist.log_prob(action).sum()
print(f"log π(a): {log_prob.item():.4f}")

# 熵
entropy = normal_dist.entropy().sum()
print(f"熵: {entropy.item():.4f}")
```

## 实战：计算策略的熵

```python
class PolicyEntropyCalculator:
    """计算神经网络策略的熵"""
    
    def __init__(self, n_actions):
        self.n_actions = n_actions
    
    def discrete_entropy(self, logits):
        """离散动作策略的熵"""
        probs = torch.softmax(logits, dim=-1)
        log_probs = torch.log_softmax(logits, dim=-1)
        entropy = -(probs * log_probs).sum(dim=-1)
        return entropy
    
    def continuous_entropy(self, mean, std):
        """高斯策略的熵"""
        # H(N(μ,σ)) = 0.5 + 0.5*log(2π) + log(σ)
        entropy = 0.5 + 0.5 * np.log(2 * np.pi) + torch.log(std)
        return entropy.sum(dim=-1)  # 多维动作求和
    
    def sample_with_log_prob(self, logits):
        """采样并计算log概率"""
        dist = torch.distributions.Categorical(logits=logits)
        action = dist.sample()
        log_prob = dist.log_prob(action)
        entropy = dist.entropy()
        return action, log_prob, entropy


# 使用示例
calculator = PolicyEntropyCalculator(n_actions=4)

# 模拟策略输出
batch_size = 32
state_dim = 10
logits = torch.randn(batch_size, 4)  # 策略网络输出

# 计算每个状态的策略熵
entropies = calculator.discrete_entropy(logits)
print(f"平均策略熵: {entropies.mean().item():.4f}")
print(f"熵范围: [{entropies.min().item():.4f}, {entropies.max().item():.4f}]")

# 训练中监控熵的变化
# 健康的训练：熵应该先高后低（先探索后利用）
# 熵降得太快：过早收敛，可能陷入局部最优
# 熵降得太慢：探索过多，学习效率低
```

## 信息论在RL中的应用总结

| 概念 | 应用场景 |
|------|----------|
| **熵 H(π)** | 最大熵RL（SAC）、探索鼓励 |
| **KL散度** | TRPO信任域、PPO约束 |
| **交叉熵** | 策略梯度损失 |
| **互信息** | 表征学习、好奇心探索 |
| **JS散度** | GAN式训练、分布匹配 |
| **微分熵** | 连续动作策略分析 |

## 小结

信息论为RL提供了：
1. **量化不确定性**的工具（熵）
2. **衡量分布差异**的度量（KL散度）
3. **设计目标函数**的原理（最大熵）
4. **约束策略更新**的方法（信任域）

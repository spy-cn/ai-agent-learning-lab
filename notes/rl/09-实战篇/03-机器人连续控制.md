# 机器人连续控制

MuJoCo 和 PyBullet 环境是连续控制 RL 的标准基准——从简单的 Pendulum 到复杂的人形机器人。

## 连续控制环境概览

```
MuJoCo 连续控制环境
├── 简单 (2~6维动作)
│   ├── Pendulum-v1: 倒立摆 (1维连续动作)
│   ├── Reacher-v4: 机械臂到达 (2维)
│   └── Swimmer-v4: 游泳者 (2维)
├── 中等 (3~8维动作)
│   ├── Hopper-v4: 单腿跳跃 (3维)
│   ├── Walker2d-v4: 双足行走 (6维)
│   └── HalfCheetah-v4: 半猎豹奔跑 (6维)
└── 困难 (8+维动作)
    ├── Ant-v4: 蚂蚁运动 (8维)
    ├── Humanoid-v4: 人形机器人 (17维)
    └── ShadowHand: 灵巧手操控 (20维)
```

```
HalfCheetah (半猎豹)
                ╱ head
         ┌─────╱
    ─────┤    ╱
    thigh┤   ╱  ← torso
    ─────┤  ╱

        ●     ●  ← feet

目标: 尽快向前奔跑 (最大化速度)
```

## 方法选择

| 环境难度 | 推荐算法 | 训练步数 | 预期性能 |
|----------|----------|----------|----------|
| Pendulum | SAC / PPO | 50K | ~-150 |
| Hopper | PPO / SAC | 1M | ~3000 |
| HalfCheetah | SAC / PPO | 1~3M | ~10000 |
| Ant | SAC | 3M | ~5000 |
| Humanoid | PPO / SAC | 5~10M | ~1000 |

## 1. Pendulum: 连续控制入门

```python
import gymnasium as gym
import numpy as np
import torch
import torch.nn as nn

env = gym.make('Pendulum-v1')
print(f"状态空间: {env.observation_space}")  # Box(-inf, inf, (3,))
print(f"动作空间: {env.action_space}")        # Box(-2, 2, (1,))

# Pendulum 状态: [cos(θ), sin(θ), 角速度]
# 动作: 关节力矩 ∈ [-2, 2]
# 奖励: -(θ² + 0.1*θ̇² + 0.001*action²)
# 目标: 保持杆子直立向上
```

### SAC 解决 Pendulum

```python
from stable_baselines3 import SAC

env = gym.make('Pendulum-v1')

model = SAC(
    "MlpPolicy", 
    env,
    learning_rate=3e-4,
    buffer_size=100000,
    batch_size=256,
    gamma=0.99,
    tau=0.005,          # 软更新系数
    ent_coef='auto',     # 自动温度调节
    verbose=1,
    tensorboard_log="./logs/pendulum/"
)

model.learn(total_timesteps=20000)

# 评估
from stable_baselines3.common.evaluation import evaluate_policy
mean_reward, std_reward = evaluate_policy(model, env, n_eval_episodes=20)
print(f"SAC Pendulum: {mean_reward:.1f} ± {std_reward:.1f}")

env.close()
```

## 2. HalfCheetah: 中等难度

```python
from stable_baselines3 import SAC, PPO, TD3
from stable_baselines3.common.vec_env import VecNormalize, DummyVecEnv
from stable_baselines3.common.callbacks import EvalCallback, CheckpointCallback

def train_halfcheetah(algo='sac', total_timesteps=1_000_000):
    """训练 HalfCheetah"""
    
    # 创建训练和评估环境
    train_env = DummyVecEnv([lambda: gym.make('HalfCheetah-v4')])
    eval_env = DummyVecEnv([lambda: gym.make('HalfCheetah-v4')])
    
    # VecNormalize: 归一化观测和奖励（对 MuJoCo 很重要！）
    train_env = VecNormalize(train_env, norm_obs=True, norm_reward=True)
    eval_env = VecNormalize(eval_env, norm_obs=True, norm_reward=False, 
                            training=False)
    
    # 同步归一化统计量
    eval_env.obs_rms = train_env.obs_rms
    
    # 选择算法
    if algo == 'sac':
        model = SAC("MlpPolicy", train_env, verbose=1,
                    learning_rate=3e-4, buffer_size=1_000_000,
                    batch_size=256, gamma=0.99, tau=0.005,
                    tensorboard_log=f"./logs/halfcheetah_{algo}/")
    elif algo == 'ppo':
        model = PPO("MlpPolicy", train_env, verbose=1,
                    learning_rate=3e-4, n_steps=2048, batch_size=64,
                    n_epochs=10, gamma=0.99, gae_lambda=0.95,
                    clip_range=0.2,
                    tensorboard_log=f"./logs/halfcheetah_{algo}/")
    elif algo == 'td3':
        model = TD3("MlpPolicy", train_env, verbose=1,
                    learning_rate=3e-4, buffer_size=1_000_000,
                    batch_size=256, gamma=0.99, tau=0.005,
                    tensorboard_log=f"./logs/halfcheetah_{algo}/")
    
    # 回调
    eval_callback = EvalCallback(
        eval_env, best_model_save_path=f"./models/halfcheetah_{algo}/",
        log_path=f"./logs/halfcheetah_{algo}/", eval_freq=10000,
        deterministic=True
    )
    
    checkpoint_callback = CheckpointCallback(
        save_freq=50000, save_path=f"./checkpoints/halfcheetah_{algo}/"
    )
    
    # 训练
    model.learn(total_timesteps=total_timesteps, 
                callback=[eval_callback, checkpoint_callback])
    
    # 保存归一化统计量
    train_env.save(f"vecnormalize_halfcheetah_{algo}.pkl")
    
    return model, train_env

model, norm_env = train_halfcheetah(algo='sac', total_timesteps=500_000)
```

## 3. 算法对比实验

```python
import matplotlib.pyplot as plt

def compare_algorithms(env_name='HalfCheetah-v4', 
                       algos=['sac', 'ppo', 'td3'],
                       total_timesteps=300_000):
    """对比不同算法在同一环境上的表现"""
    
    results = {}
    
    for algo in algos:
        print(f"\n{'='*50}")
        print(f"训练 {algo} on {env_name}")
        print(f"{'='*50}")
        
        env = DummyVecEnv([lambda: gym.make(env_name)])
        env = VecNormalize(env, norm_obs=True, norm_reward=True)
        
        if algo == 'sac':
            model = SAC("MlpPolicy", env, verbose=0)
        elif algo == 'ppo':
            model = PPO("MlpPolicy", env, verbose=0)
        elif algo == 'td3':
            model = TD3("MlpPolicy", env, verbose=0)
        
        # 记录训练奖励
        rewards = []
        
        class RewardLogger:
            def __init__(self):
                self.rewards = []
            
            def __call__(self, locals, globals):
                if 'info' in locals:
                    ep_info = locals.get('ep_info_buffer', [])
                    for info in ep_info:
                        if 'r' in info:
                            self.rewards.append(info['r'])
                return True
        
        logger = RewardLogger()
        
        from stable_baselines3.common.callbacks import CallbackList, BaseCallback
        
        class RewardCallback(BaseCallback):
            def __init__(self):
                super().__init__()
                self.rewards = []
            
            def _on_step(self):
                infos = self.locals.get('infos', [])
                for info in infos:
                    if 'episode' in info:
                        self.rewards.append(info['episode']['r'])
                return True
        
        cb = RewardCallback()
        model.learn(total_timesteps=total_timesteps, callback=cb)
        results[algo] = cb.rewards
        
        env.close()
    
    # 绘制对比图
    fig, ax = plt.subplots(figsize=(12, 6))
    for algo, rewards in results.items():
        if len(rewards) > 10:
            window = min(50, len(rewards) // 5)
            moving_avg = np.convolve(rewards, np.ones(window)/window, mode='valid')
            ax.plot(moving_avg, label=algo.upper(), linewidth=2)
    
    ax.set_xlabel('Episode')
    ax.set_ylabel('Episode Reward')
    ax.set_title(f'Algorithm Comparison on {env_name}')
    ax.legend()
    ax.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig(f'algo_comparison_{env_name}.png', dpi=150)
    plt.show()
    
    return results

results = compare_algorithms(total_timesteps=200_000)
```

## 5. 灵巧手操作（高维控制）

```python
# Shadow Hand 灵巧手: 物体重定向任务
# 动作空间: 20维
# 观测空间: ~60维 (关节角度 + 物体位姿)

def train_shadow_hand():
    """训练 ShadowHand 灵巧手"""
    try:
        env = gym.make('ShadowHandBlock-v1')
        
        model = SAC(
            "MlpPolicy", env,
            learning_rate=3e-4,
            buffer_size=2_000_000,
            batch_size=512,
            gamma=0.95,
            tau=0.005,
            ent_coef='auto',
            verbose=1,
            tensorboard_log="./logs/shadow_hand/"
        )
        
        model.learn(total_timesteps=5_000_000)
        model.save("sac_shadow_hand")
        
        env.close()
    except Exception as e:
        print(f"需要安装 isaac gym 或 mujoco_maze: {e}")

train_shadow_hand()
```

## 训练技巧

```
连续控制最佳实践:
┌───────────────────────────────────────────────────┐
│                                                   │
│  1. 观测归一化是关键                               │
│     → VecNormalize 或 RunningMeanStd             │
│     → 特别重要于 PPO                             │
│                                                   │
│  2. 动作归一化到 [-1, 1]                          │
│     → 大多数环境已内置                            │
│                                                   │
│  3. 奖励缩放                                      │
│     → 如果奖励量级很大，需缩放                     │
│                                                   │
│  4. 算法选择                                      │
│     → 样本效率优先: SAC                          │
│     → 稳定性优先: PPO                            │
│     → 高精度控制: TD3                            │
│                                                   │
│  5. 并行环境                                      │
│     → PPO: 8~16 个并行环境                       │
│     → SAC/TD3: 1 个即可（off-policy）            │
│                                                   │
│  6. 大网络适用场景                                │
│     → 高维观测 (>50维): [256, 256]              │
│     → 低维观测: [64, 64] 或 [128, 128]          │
│                                                   │
│  7. 超参数敏感度                                  │
│     SAC > TD3 > PPO (从高到低)                   │
│                                                   │
└───────────────────────────────────────────────────┘
```

## 小结

- 连续控制是 RL 在机器人学中的核心应用
- **MuJoCo/PyBullet** 是标准基准环境
- **SAC** 是当前最强通用连续控制算法
- **观测归一化** 对所有算法都至关重要
- 从 Pendulum → Hopper → HalfCheetah → Humanoid 逐步进阶

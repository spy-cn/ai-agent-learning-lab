# 端到端 RL 项目模板

一个完整的 RL 项目从问题定义到部署上线涉及多个环节。本章提供一个标准化的项目模板。

## 项目生命周期

```
RL 项目完整流程:

┌─────────┐   ┌─────────┐   ┌─────────┐   ┌─────────┐
│ 问题定义 │ → │ 环境建模 │ → │ 算法选择 │ → │ 基线实验 │
└─────────┘   └─────────┘   └─────────┘   └─────────┘
                                                ↓
┌─────────┐   ┌─────────┐   ┌─────────┐   ┌─────────┐
│ 部署上线 │ ← │ 模型评估 │ ← │ 调优迭代 │ ← │ 超参搜索 │
└─────────┘   └─────────┘   └─────────┘   └─────────┘
```

# src/utils/config.py
import yaml
from dataclasses import dataclass, field
from typing import Dict, Any
import copy

def load_config(path: str) -> dict:
    """加载 YAML 配置"""
    with open(path, 'r') as f:
        config = yaml.safe_load(f)
    return config

def merge_config(base: dict, override: dict) -> dict:
    """递归合并配置"""
    result = copy.deepcopy(base)
    for key, value in override.items():
        if key in result and isinstance(result[key], dict) and isinstance(value, dict):
            result[key] = merge_config(result[key], value)
        else:
            result[key] = value
    return result

# 使用
config = load_config('config/default.yaml')
print(f"算法: {config['algorithm']['name']}")
print(f"学习率: {config['algorithm']['dqn']['learning_rate']}")

## 2. 日志系统

```python
# src/utils/logger.py
import os
import json
import numpy as np
from datetime import datetime
from torch.utils.tensorboard import SummaryWriter

class Logger:
    """统一日志管理"""
    
    def __init__(self, log_dir="logs", experiment_name="exp", 
                 use_tensorboard=True, use_wandb=False):
        self.log_dir = os.path.join(log_dir, experiment_name)
        os.makedirs(self.log_dir, exist_ok=True)
        
        self.use_tensorboard = use_tensorboard
        self.use_wandb = use_wandb
        
        if use_tensorboard:
            self.tb_writer = SummaryWriter(self.log_dir)
        
        if use_wandb:
            import wandb
            self.wandb = wandb.init(
                project="rl_project",
                name=experiment_name,
                dir=self.log_dir
            )
        
        self.metrics = {}
        self.start_time = datetime.now()
        
        # 保存实验配置
        self.config_path = os.path.join(self.log_dir, "config.json")
    
    def log_config(self, config: dict):
        """保存配置"""
        with open(self.config_path, 'w') as f:
            json.dump(config, f, indent=2, default=str)
    
    def log_metric(self, key: str, value: float, step: int):
        """记录标量指标"""
        if key not in self.metrics:
            self.metrics[key] = []
        self.metrics[key].append((step, value))
        
        if self.use_tensorboard:
            self.tb_writer.add_scalar(key, value, step)
        
        if self.use_wandb:
            self.wandb.log({key: value}, step=step)
    
    def log_histogram(self, key: str, values, step: int):
        """记录直方图"""
        if self.use_tensorboard:
            self.tb_writer.add_histogram(key, values, step)
    
    def log_video(self, tag: str, frames: np.ndarray, step: int, fps=30):
        """记录视频"""
        if self.use_tensorboard:
            self.tb_writer.add_video(tag, frames, step, fps=fps)
    
    def print_summary(self, step: int):
        """打印摘要"""
        elapsed = (datetime.now() - self.start_time).total_seconds()
        print(f"\n{'='*60}")
        print(f"Step: {step} | Elapsed: {elapsed:.0f}s")
        for key, values in self.metrics.items():
            if values:
                recent = [v for _, v in values[-10:]]
                print(f"  {key}: {np.mean(recent):.4f} (avg last 10)")
        print(f"{'='*60}\n")
    
    def close(self):
        if self.use_tensorboard:
            self.tb_writer.close()
        if self.use_wandb:
            self.wandb.finish()

logger = Logger(experiment_name="cartpole_dqn_v1")
logger.log_config({"algo": "DQN", "lr": 1e-3})
```

## 3. 完整训练脚本

```python
# scripts/train.py
"""
端到端 RL 训练脚本
用法: python train.py --config config/default.yaml --algo dqn
"""
import argparse
import yaml
import gymnasium as gym
import numpy as np
import torch
import random
import os

def set_seed(seed: int):
    """设置随机种子"""
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)

def parse_args():
    parser = argparse.ArgumentParser()
    parser.add_argument("--config", type=str, default="config/default.yaml")
    parser.add_argument("--algo", type=str, default=None, 
                        help="Override algorithm")
    parser.add_argument("--env", type=str, default=None,
                        help="Override environment")
    parser.add_argument("--timesteps", type=int, default=None,
                        help="Override total timesteps")
    parser.add_argument("--seed", type=int, default=None)
    parser.add_argument("--name", type=str, default="run1")
    return parser.parse_args()

def main():
    args = parse_args()
    
    # 加载配置
    with open(args.config) as f:
        config = yaml.safe_load(f)
    
    # 覆盖参数
    if args.algo:
        config['algorithm']['name'] = args.algo
    if args.env:
        config['env']['name'] = args.env
    if args.timesteps:
        config['training']['total_timesteps'] = args.timesteps
    if args.seed:
        config['project']['seed'] = args.seed
    
    algo_name = config['algorithm']['name']
    env_name = config['env']['name']
    seed = config['project']['seed']
    total_timesteps = config['training']['total_timesteps']
    
    print(f"Algorithm: {algo_name}")
    print(f"Environment: {env_name}")
    print(f"Seed: {seed}")
    print(f"Total timesteps: {total_timesteps}")
    
    # 设置随机种子
    set_seed(seed)
    
    # 创建环境
    env = gym.make(env_name)
    
    # 创建智能体
    if algo_name == 'dqn':
        from stable_baselines3 import DQN
        algo_config = config['algorithm']['dqn']
        model = DQN("MlpPolicy", env, verbose=1, 
                    tensorboard_log=f"logs/{args.name}/", **algo_config)
    elif algo_name == 'ppo':
        from stable_baselines3 import PPO
        algo_config = config['algorithm']['ppo']
        model = PPO("MlpPolicy", env, verbose=1,
                    tensorboard_log=f"logs/{args.name}/", **algo_config)
    elif algo_name == 'sac':
        from stable_baselines3 import SAC
        algo_config = config['algorithm']['sac']
        model = SAC("MlpPolicy", env, verbose=1,
                    tensorboard_log=f"logs/{args.name}/", **algo_config)
    else:
        raise ValueError(f"Unknown algorithm: {algo_name}")
    
    # 回调
    from stable_baselines3.common.callbacks import (
        EvalCallback, CheckpointCallback, CallbackList
    )
    
    eval_env = gym.make(env_name)
    callbacks = [
        EvalCallback(eval_env, best_model_save_path=f"checkpoints/{args.name}/",
                     log_path=f"logs/{args.name}/", eval_freq=5000),
        CheckpointCallback(save_freq=10000, 
                          save_path=f"checkpoints/{args.name}/"),
    ]
    
    # 训练
    model.learn(total_timesteps=total_timesteps, callback=callbacks)
    
    # 保存最终模型
    model.save(f"checkpoints/{args.name}/final_model")
    print(f"Model saved to checkpoints/{args.name}/final_model")
    
    env.close()
    eval_env.close()

if __name__ == "__main__":
    main()
```

## 4. 评估脚本

```python
# scripts/evaluate.py
"""
评估训练好的模型
用法: python evaluate.py --model checkpoints/run1/best_model --env CartPole-v1
"""
import argparse
import gymnasium as gym
import numpy as np
import torch

def evaluate(model_path, env_name, n_episodes=20, deterministic=True, 
             render=False):
    """全面评估模型"""
    from stable_baselines3 import DQN, PPO, SAC, TD3
    
    # 自动检测算法类型
    # 这里简化处理，实际可以从路径或配置读取
    model = None
    for AlgoClass in [DQN, PPO, SAC, TD3]:
        try:
            model = AlgoClass.load(model_path)
            print(f"Loaded as {AlgoClass.__name__}")
            break
        except:
            continue
    
    if model is None:
        raise ValueError("Could not load model")
    
    env = gym.make(env_name, 
                   render_mode='human' if render else None)
    
    all_rewards = []
    all_lengths = []
    
    for ep in range(n_episodes):
        obs, info = env.reset()
        total_reward = 0
        steps = 0
        done = False
        
        while not done:
            action, _ = model.predict(obs, deterministic=deterministic)
            obs, reward, terminated, truncated, info = env.step(action)
            total_reward += reward
            steps += 1
            done = terminated or truncated
        
        all_rewards.append(total_reward)
        all_lengths.append(steps)
        print(f"Episode {ep+1}: Reward={total_reward:.2f}, Length={steps}")
    
    # 统计
    print(f"\n{'='*40}")
    print(f"Episodes: {n_episodes}")
    print(f"Reward: {np.mean(all_rewards):.2f} ± {np.std(all_rewards):.2f}")
    print(f"  Min: {np.min(all_rewards):.2f}")
    print(f"  Max: {np.max(all_rewards):.2f}")
    print(f"  Median: {np.median(all_rewards):.2f}")
    print(f"Length: {np.mean(all_lengths):.1f} ± {np.std(all_lengths):.1f}")
    print(f"{'='*40}")
    
    env.close()
    return all_rewards

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--model", type=str, required=True)
    parser.add_argument("--env", type=str, required=True)
    parser.add_argument("--episodes", type=int, default=20)
    parser.add_argument("--render", action="store_true")
    args = parser.parse_args()
    
    evaluate(args.model, args.env, args.episodes, render=args.render)
```

## 5. 超参数搜索

```python
# scripts/sweep.py
"""
超参数搜索
用法: python sweep.py --config config/sweep.yaml
"""
import optuna
import yaml
import gymnasium as gym
from stable_baselines3 import PPO
from stable_baselines3.common.evaluation import evaluate_policy

def objective(trial: optuna.Trial, env_name="CartPole-v1"):
    """Optuna 目标函数"""
    
    # 定义搜索空间
    learning_rate = trial.suggest_float('learning_rate', 1e-5, 1e-2, log=True)
    n_steps = trial.suggest_categorical('n_steps', [128, 256, 512, 1024, 2048])
    batch_size = trial.suggest_categorical('batch_size', [32, 64, 128, 256])
    gamma = trial.suggest_float('gamma', 0.9, 0.999)
    gae_lambda = trial.suggest_float('gae_lambda', 0.8, 1.0)
    clip_range = trial.suggest_float('clip_range', 0.1, 0.3)
    n_epochs = trial.suggest_int('n_epochs', 3, 10)
    
    # 创建环境和模型
    env = gym.make(env_name)
    
    model = PPO(
        "MlpPolicy", env,
        learning_rate=learning_rate,
        n_steps=n_steps,
        batch_size=batch_size,
        gamma=gamma,
        gae_lambda=gae_lambda,
        clip_range=clip_range,
        n_epochs=n_epochs,
        verbose=0
    )
    
    # 训练
    model.learn(total_timesteps=50000)
    
    # 评估
    mean_reward, _ = evaluate_policy(model, env, n_eval_episodes=10)
    
    env.close()
    return mean_reward

def run_sweep(n_trials=50):
    """运行超参搜索"""
    study = optuna.create_study(direction='maximize')
    study.optimize(objective, n_trials=n_trials)
    
    print(f"\nBest trial:")
    print(f"  Value: {study.best_trial.value}")
    print(f"  Params:")
    for key, value in study.best_trial.params.items():
        print(f"    {key}: {value}")
    
    return study

# 运行
# study = run_sweep(n_trials=20)
```

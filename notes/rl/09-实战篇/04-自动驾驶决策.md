# 自动驾驶决策

强化学习在自动驾驶领域有多种应用场景——从车道保持到路口决策，从泊车到轨迹规划。

## RL 在自动驾驶中的应用层次

```
自动驾驶系统层级与 RL 应用:

┌─────────────────────────────────────────┐
│           架构层 (宏观决策)              │  ← 路线选择
│  - 全局路径规划 (导航)                   │     RL: 路网级决策
├───────────────��─────────────────────────┤
│           行为层 (中观决策)              │  ← 变道/超车
│  - 车道保持 / 变道 / 超车               │     RL: 马尔可夫决策
│  - 路口通行 (左转/右转/直行)            │
│  - 交叉口让行                           │
├─────────────────────────────────────────┤
│           运动层 (微观控制)              │  ← 加减速
│  - 速度控制 (ACC自适应巡航)             │     RL: 连续控制
│  - 横向控制 (转向角)                    │
│  - 轨迹跟踪 (MPC替代)                   │
├─────────────────────────────────────────┤
│           感知层 (信号处理)              │  ← 目标检测
│  - 目标检测/跟踪                         │     RL: 主动感知
│  - 语义分割                              │
│  - 深度估计                              │
└─────────────────────────────────────────┘
```

## 1. 车道保持：简易 highway-env

```python
# pip install highway-env
import gymnasium as gym
import highway_env
import numpy as np

# === 高速公路变道环境 ===
env = gym.make('highway-v0')

# 配置环境
env.configure({
    "observation": {
        "type": "Kinematics",
        "vehicles_count": 5,       # 观测周围5辆车
        "features": ["presence", "x", "y", "vx", "vy"],
        "features_range": {
            "x": [-100, 100],
            "y": [-50, 50],
            "vx": [-20, 20],
            "vy": [-20, 20]
        },
        "absolute": False,          # 相对坐标
        "order": "sorted"           # 按距离排序
    },
    "action": {
        "type": "DiscreteMetaAction",  # 高级语义动作
        "longitudinal": True,
        "lateral": True
    },
    "lanes_count": 4,              # 4车道
    "vehicles_count": 50,          # 交通流密度
    "duration": 40,                # 每回合40秒
    "collision_reward": -1.0,      # 碰撞惩罚
    "right_lane_reward": 0.1,      # 右侧车道奖励
    "high_speed_reward": 0.4,      # 高速奖励
    "lane_change_reward": 0.0,     # 变道无奖励
})

print(f"观测空间: {env.observation_space}")
print(f"动作空间: {env.action_space}")
# 动作: 0=左变道, 1=空闲, 2=右变道, 3=加速, 4=减速
```

### DQN 训练 highway

```python
from stable_baselines3 import DQN
from stable_baselines3.common.vec_env import DummyVecEnv

env = gym.make('highway-v0')
env = DummyVecEnv([lambda: env])

model = DQN(
    "MlpPolicy",
    env,
    learning_rate=5e-4,
    buffer_size=50000,
    learning_starts=1000,
    batch_size=32,
    gamma=0.8,
    train_freq=1,
    target_update_interval=50,
    exploration_fraction=0.1,
    verbose=1,
    tensorboard_log="./logs/highway/"
)

model.learn(total_timesteps=20000)
model.save("dqn_highway")

# 测试
obs = env.reset()
for step in range(100):
    action, _ = model.predict(obs, deterministic=True)
    obs, reward, done, info = env.step(action)
    if done:
        obs = env.reset()
        print(f"Crash at step {step}, reward: {reward}")

env.close()
```

## 3. 泊车任务

```python
# === 泊车环境 ===
env = gym.make('parking-v0')

env.configure({
    "observation": {
        "type": "KinematicsGoal",
        "features": ['x', 'y', 'vx', 'vy', 'cos_h', 'sin_h'],
        "scales": [100, 100, 5, 5, 1, 1],
        "normalize": True
    },
    "action": {
        "type": "ContinuousAction",  # 连续控制: 转向+油门
    },
})

# 泊车是连续控制问题 → SAC
from stable_baselines3 import SAC, HerReplayBuffer

model = SAC(
    "MlpPolicy",
    env,
    replay_buffer_class=HerReplayBuffer,  # Hindsight Experience Replay
    replay_buffer_kwargs=dict(
        n_sampled_goal=4,
        goal_selection_strategy="future",  # 用未来状态作为目标
    ),
    learning_rate=1e-3,
    buffer_size=1_000_000,
    batch_size=256,
    gamma=0.95,
    verbose=1,
    tensorboard_log="./logs/parking/"
)

model.learn(total_timesteps=100_000)
model.save("sac_parking_her")
```

## 4. 自定义驾驶环境

```python
class DrivingEnv(gym.Env):
    """自定义驾驶环境: 直道巡航"""
    
    def __init__(self):
        super().__init__()
        
        # 状态: [自车速度, 前车距离, 前车速度, 车道偏移]
        self.observation_space = gym.spaces.Box(
            low=np.array([0, 0, 0, -10]),
            high=np.array([50, 200, 50, 10]),
            dtype=np.float32
        )
        
        # 动作: [加速度, 转向角]
        self.action_space = gym.spaces.Box(
            low=np.array([-3, -0.5]),
            high=np.array([3, 0.5]),
            dtype=np.float32
        )
        
        self.dt = 0.1  # 时间步长
        self.max_steps = 1000
        self.reset()
    
    def reset(self, seed=None):
        self.speed = 20.0        # 初始速度
        self.front_dist = 50.0   # 前车距离
        self.front_speed = 25.0  # 前车速度
        self.lane_offset = 0.0   # 车道偏移
        self.steps = 0
        return self._get_obs(), {}
    
    def _get_obs(self):
        return np.array([
            self.speed, self.front_dist, self.front_speed, self.lane_offset
        ], dtype=np.float32)
    
    def step(self, action):
        accel, steer = action
        
        # 更新物理状态
        self.speed = np.clip(self.speed + accel * self.dt, 0, 50)
        self.lane_offset = np.clip(self.lane_offset + steer * self.dt * 10, 
                                    -10, 10)
        
        # 前车随机减速/加速
        self.front_speed = np.clip(
            self.front_speed + np.random.randn() * 2, 10, 40
        )
        
        # 更新前车距离
        rel_speed = self.speed - self.front_speed
        self.front_dist += -rel_speed * self.dt
        
        # 计算奖励
        reward = self._compute_reward()
        
        # 终止条件
        terminated = self.front_dist < 5  # 碰撞
        truncated = self.steps >= self.max_steps
        
        self.steps += 1
        return self._get_obs(), reward, terminated, truncated, {}
    
    def _compute_reward(self):
        """奖励设计"""
        # 1. 速度奖励 (鼓励保持目标速度)
        target_speed = 25.0
        speed_reward = -abs(self.speed - target_speed) * 0.1
        
        # 2. 安全奖励
        safety_reward = 0
        if self.front_dist < 20:
            safety_reward = -(20 - self.front_dist) * 0.5
        if self.front_dist < 5:
            safety_reward -= 10  # 碰撞重罚
        
        # 3. 车道保持
        lane_reward = -abs(self.lane_offset) * 0.1
        
        # 4. 时间惩罚 (鼓励高效)
        time_penalty = -0.01
        
        return speed_reward + safety_reward + lane_reward + time_penalty

# 训练自定义环境
env = DrivingEnv()
model = SAC("MlpPolicy", env, verbose=1)
model.learn(total_timesteps=50000)
```

## 安全 RL

```python
# 自动驾驶中安全至关重要 → Constrained RL

class SafeDrivingWrapper(gym.Wrapper):
    """安全驾驶包装器"""
    
    def __init__(self, env, safety_distance=10):
        super().__init__(env)
        self.safety_distance = safety_distance
        self.cost = 0
    
    def step(self, action):
        obs, reward, terminated, truncated, info = self.env.step(action)
        
        # 计算成本 (违反安全约束)
        front_dist = obs[1]
        if front_dist < self.safety_distance:
            self.cost = (self.safety_distance - front_dist) / self.safety_distance
        else:
            self.cost = 0
        
        info['cost'] = self.cost
        return obs, reward, terminated, truncated, info

# 使用 CPO (Constrained Policy Optimization) 或 RCPO 等安全 RL 算法
```

## 挑战与展望

```
自动驾驶 RL 挑战:
┌───────────────────────────────────────────────┐
│                                               │
│  1. 安全性保证                                │
│     → Constrained RL / Safe RL               │
│     → 形式化验证                              │
│                                               │
│  2. Sim-to-Real 迁移                          │
│     → 域随机化 (Domain Randomization)        │
│     → 域适应 (Domain Adaptation)             │
│                                               │
│  3. 多目标优化                                │
│     → 安全 vs 速度 vs 舒适度                 │
│     → 帕累托最优                             │
│                                               │
│  4. 长尾场景                                  │
│     → 极端天气/事故/施工                     │
│     → 课程学习                              │
│                                               │
│  5. 实时性                                    │
│     → 推理延迟 < 100ms                       │
│     → 模型压缩/量化                         │
│                                               │
└───────────────────────────────────────────────┘
```

## 小结

- RL 在自动驾驶中主要用于 **行为决策** 层
- **highway-env** 是入门最佳实践环境
- 安全是核心挑战 → 需要 **Constrained RL**
- **Sim-to-Real** 是从研究到产品的关键鸿沟
- 实际系统通常采用 **RL + 规则** 的混合架构

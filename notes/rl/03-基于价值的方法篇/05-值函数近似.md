# 值函数近似（Value Function Approximation）

当状态空间很大或连续时，表格方法不可行，需要用**函数近似**来表示值函数。

## 为什么需要函数近似？

```
问题: 表格方法无法处理大状态空间

FrozenLake:     16个状态      → Q表 [16×4] = 64个条目     ✓ 可行
CartPole:       连续4维状态    → 无限个状态               ✗ 表格不可行
Atari:          84×84×4图像   → 2^(84*84*4) ≈ 10^84674   ✗ 完全不可行

解决方案: 用函数近似 Q(s,a) ≈ Q(s,a;θ)
         θ是函数的参数（如神经网络的权重）
```

## 函数近似的类型

| 类型 | 表达式 | 示例 |
|------|--------|------|
| **线性近似** | $Q(s,a) = \theta^T \phi(s,a)$ | 特征+线性回归 |
| **神经网络** | $Q(s,a) = f_\theta(s,a)$ | DQN |
| **决策树** | | 非参数方法 |
| **核方法** | | RKHS理论 |

## 特征表示

### 线性近似的特征工程

```python
import numpy as np

# 方法1: 独热编码（one-hot）
def one_hot_features(state, n_states):
    return np.eye(n_states)[state]

# 方法2: 多项式特征
def polynomial_features(state, degree=2):
    """生成多项式特征"""
    features = [1.0]  # 偏置项
    features.extend(state)  # 一次项
    for i in range(len(state)):
        for j in range(i, len(state)):
            features.append(state[i] * state[j])  # 二次项
    return np.array(features)

# 方法3: 径向基函数（RBF）
def rbf_features(state, centers, sigma=1.0):
    """径向基函数特征"""
    features = []
    for c in centers:
        dist = np.linalg.norm(np.array(state) - c)
        features.append(np.exp(-dist**2 / (2 * sigma**2)))
    return np.array(features)

# 方法4: 分桶（Tile Coding / Coarse Coding）
def tile_coding(state, n_tilings=8, bins=8):
    """Tile Coding: 将连续状态分桶"""
    features = np.zeros(n_tilings * bins)
    for i in range(n_tilings):
        offset = i / n_tilings  # 每个tiling的偏移
        binned = int((state[0] + offset) * bins) % bins
        features[i * bins + binned] = 1.0
    return features

# 示例: CartPole状态
state = [0.05, 0.5, -0.1, 0.3]  # [位置, 速度, 角度, 角速度]
poly_feat = polynomial_features(state, degree=2)
print(f"多项式特征维度: {len(poly_feat)}")  # 1 + 4 + 10 = 15
```

## 线性值函数近似

$$\hat{Q}(s, a; \mathbf{w}) = \mathbf{w}^T \phi(s, a)$$

### 梯度下降更新

$$\mathbf{w} \leftarrow \mathbf{w} + \alpha [target - \hat{Q}(s,a;\mathbf{w})] \phi(s,a)$$

```python
class LinearQFunction:
    """线性Q函数近似"""
    
    def __init__(self, n_features, n_actions, lr=0.01):
        self.n_features = n_features
        self.n_actions = n_actions
        self.lr = lr
        # 每个动作一组权重
        self.w = np.zeros((n_actions, n_features))
    
    def __call__(self, state_features):
        """计算所有动作的Q值"""
        return self.w @ state_features
    
    def update(self, state_features, action, target):
        """梯度下降更新"""
        prediction = self.w[action] @ state_features
        error = target - prediction
        self.w[action] += self.lr * error * state_features
    
    def best_action(self, state_features):
        return np.argmax(self(state_features))


# 线性半梯度SARSA
def linear_sarsa(env, n_episodes=10000, gamma=0.99, alpha=0.01, epsilon=0.1):
    """线性函数近似的SARSA"""
    n_actions = env.action_space.n
    
    # 假设用多项式特征
    def get_features(state):
        return polynomial_features(state, degree=2)
    
    n_features = len(get_features([0]*4))
    q_func = LinearQFunction(n_features, n_actions, lr=alpha)
    
    for episode in range(n_episodes):
        state, _ = env.reset()
        features = get_features(state)
        
        # ε-greedy选择动作
        if np.random.random() < epsilon:
            action = np.random.randint(n_actions)
        else:
            action = q_func.best_action(features)
        
        done = False
        while not done:
            # 执行动作
            next_state, reward, term, trunc, _ = env.step(action)
            next_features = get_features(next_state)
            done = term or trunc
            
            # 选择下一个动作
            if np.random.random() < epsilon:
                next_action = np.random.randint(n_actions)
            else:
                next_action = q_func.best_action(next_features)
            
            # SARSA更新
            if done:
                target = reward
            else:
                target = reward + gamma * q_func(next_features)[next_action]
            
            q_func.update(features, action, target)
            
            features = next_features
            action = next_action
    
    return q_func
```

## 神经网络值函数近似

```python
import torch
import torch.nn as nn

class NeuralQFunction(nn.Module):
    """神经网络Q函数"""
    
    def __init__(self, state_dim, action_dim, hidden_dims=[128, 128]):
        super().__init__()
        
        layers = []
        prev_dim = state_dim
        for hidden_dim in hidden_dims:
            layers.append(nn.Linear(prev_dim, hidden_dim))
            layers.append(nn.ReLU())
            prev_dim = hidden_dim
        layers.append(nn.Linear(prev_dim, action_dim))
        
        self.net = nn.Sequential(*layers)
    
    def forward(self, state):
        return self.net(state)
    
    def get_q_values(self, state):
        """获取所有动作的Q值"""
        with torch.no_grad():
            state_t = torch.FloatTensor(state).unsqueeze(0)
            q_values = self.net(state_t)
        return q_values.numpy()[0]
    
    def best_action(self, state):
        q_values = self.get_q_values(state)
        return np.argmax(q_values)


# 梯度下降训练
def update_nn_q(q_net, optimizer, state, action, target, gamma=0.99):
    """神经网络Q函数的更新"""
    state_t = torch.FloatTensor(state).unsqueeze(0)
    target_t = torch.FloatTensor([target])
    
    # 预测Q值
    q_values = q_net(state_t)
    q_a = q_values[0, action]
    
    # MSE损失
    loss = nn.MSELoss()(q_a, target_t)
    
    # 反向传播
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()
    
    return loss.item()
```

## 收敛性分析

### 致命三要素（Deadly Triad）

函数近似 + 自举 + 离线策略 = **训练不稳定/发散**

```
┌──────────────────────────────────────┐
│         致命三要素                     │
│                                      │
│    函数近似      自举        离线策略  │
│   (神经网络)   (bootstrap)  (off-policy)│
│       \           |           /       │
│        \          |          /        │
│         ┌─────────┼─────────┐         │
│         │  不稳定/发散风险  │         │
│         └───────────────────┘         │
└──────────────────────────────────────┘

DQN用以下技术缓解：
1. 经验回放（打破数据相关性）
2. 目标网络（稳定bootstrap目标）
3. 梯度裁剪（防止梯度爆炸）
```

### 各种方法的收敛性

| 方法 | 表格 | 线性 | 非线性 |
|------|------|------|--------|
| MC | ✓ | ✓ | 大多✓ |
| TD(0) On-policy | ✓ | ✓ | ✗ |
| Q-Learning | ✓ | ✗ | ✗ |

**关键结论**：非线性函数近似 + off-policy 不保证收敛，需要技巧。

## 半梯度方法

只用部分梯度（忽略目标对θ的依赖）：

```python
def semi_gradient_td_zero(q_net, optimizer, state, action, reward, 
                          next_state, done, gamma=0.99):
    """半梯度TD(0)
    
    完整梯度: ∇[target - Q(s,a;θ)]²
              = 2(target - Q)·∇Q - 2(target-Q)·∇target
    
    半梯度:   只用 ∇Q，忽略 ∇target
              → 更简单，但偏差换稳定
    """
    state_t = torch.FloatTensor(state).unsqueeze(0)
    next_state_t = torch.FloatTensor(next_state).unsqueeze(0)
    
    # 当前Q值
    q_values = q_net(state_t)
    q_sa = q_values[0, action]
    
    # TD目标（stop gradient → 半梯度）
    with torch.no_grad():
        if done:
            td_target = reward
        else:
            next_q = q_net(next_state_t).max()
            td_target = reward + gamma * next_q
    
    # 损失 = (TD目标 - 当前Q)²
    loss = nn.MSELoss()(q_sa, td_target)
    
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()
    
    return loss.item()
```

## 平均奖励设定（Average Reward）

对于连续任务（无终止），使用平均奖励：

$$\rho(\pi) = \lim_{T \to \infty} \frac{1}{T} E\left[\sum_{t=0}^{T-1} R_{t+1}\right]$$

差分回报：

$$G_t = \sum_{k=0}^{\infty} (R_{t+k+1} - \rho^\pi)$$

```python
class AverageRewardDQN:
    """平均奖励DQN（连续任务）"""
    
    def __init__(self, ...):
        self.average_reward = 0.0  # 维护平均奖励估计
        self.lr_r = 0.01  # 平均奖励学习率
        ...
    
    def compute_td_target(self, reward, next_q, done):
        # 差分TD目标: r - ρ̄ + max Q(s')
        td_target = reward - self.average_reward + next_q.max()
        return td_target
    
    def update_average_reward(self, reward):
        # 增量更新平均奖励
        self.average_reward += self.lr_r * (reward - self.average_reward)
```

## 实际技巧与最佳实践

### 1. 状态归一化

```python
class RunningMeanStd:
    """在线均值/方差估计（用于状态归一化）"""
    
    def __init__(self, shape):
        self.mean = np.zeros(shape, dtype=np.float64)
        self.var = np.ones(shape, dtype=np.float64)
        self.count = 1e-4
    
    def update(self, x):
        batch_mean = np.mean(x, axis=0)
        batch_var = np.var(x, axis=0)
        batch_count = x.shape[0]
        self.update_from_moments(batch_mean, batch_var, batch_count)
    
    def update_from_moments(self, batch_mean, batch_var, batch_count):
        delta = batch_mean - self.mean
        tot_count = self.count + batch_count
        
        new_mean = self.mean + delta * batch_count / tot_count
        m_a = self.var * self.count
        m_b = batch_var * batch_count
        M2 = m_a + m_b + np.square(delta) * self.count * batch_count / tot_count
        new_var = M2 / tot_count
        
        self.mean = new_mean
        self.var = new_var
        self.count = tot_count
    
    def normalize(self, x):
        return (x - self.mean) / np.sqrt(self.var + 1e-8)


# 在训练中应用
class NormalizedEnv:
    """带状态归一化的环境包装器"""
    def __init__(self, env):
        self.env = env
        self.obs_rms = RunningMeanStd(shape=env.observation_space.shape)
    
    def reset(self):
        obs, _ = self.env.reset()
        self.obs_rms.update(np.array([obs]))
        return self.obs_rms.normalize(obs), _
    
    def step(self, action):
        obs, reward, term, trunc, info = self.env.step(action)
        self.obs_rms.update(np.array([obs]))
        return self.obs_rms.normalize(obs), reward, term, trunc, info
```

## 小结

| 概念 | 说明 |
|------|------|
| **表格 → 函数近似** | 处理大/连续状态空间 |
| **线性近似** | 简单、有理论保证 |
| **神经网络** | 强大表达力，但不稳定 |
| **半梯度** | 忽略目标依赖，简化计算 |
| **致命三要素** | 近似+自举+离线=不稳定 |
| **DQN技巧** | 回放+目标网络+裁剪 |

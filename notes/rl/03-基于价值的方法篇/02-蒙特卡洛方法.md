# 蒙特卡洛方法（Monte Carlo Methods）

蒙特卡洛方法通过**完整的回合经验**来学习值函数，不需要环境模型。

## 核心思想

```
动态规划: 需要知道P(s'|s,a)和R(s,a,s')
蒙特卡洛: 不需要！只需要经验样本（完整回合）

MC估计 V(s) ≈ 多次经过s后的回报平均值

     回合1: s → ... → 终止  G₁ = 10
     回合2: s → ... → 终止  G₂ = 8
     回合3: s → ... → 终止  G₃ = 12
     
     V(s) ≈ (10 + 8 + 12) / 3 = 10
```

## 特点

| 特性 | 说明 |
|------|------|
| **无模型** | 不需要P和R |
| **回合制** | 必须等回合结束才能更新 |
| **无偏估计** | 回报是真实值的无偏估计 |
| **高方差** | 不同回合的回报差异可能很大 |

## 蒙特卡洛预测

估计给定策略的V^π(s)。

### 首次访问MC（First-Visit MC）

只统计每个回合中状态s第一次出现时的回报：

```python
import numpy as np
import gymnasium as gym

def first_visit_mc_prediction(env, policy, n_episodes=10000, gamma=0.99):
    """首次访问蒙特卡洛预测"""
    V = np.zeros(env.observation_space.n)
    returns = {s: [] for s in range(env.observation_space.n)}
    
    for episode in range(n_episodes):
        # 1. 生成完整回合
        trajectory = []
        state, _ = env.reset()
        done = False
        
        while not done:
            action = policy(state)
            next_state, reward, term, trunc, _ = env.step(action)
            trajectory.append((state, action, reward))
            state = next_state
            done = term or trunc
        
        # 2. 计算回报并更新V
        G = 0
        visited = set()
        
        for t in reversed(range(len(trajectory))):
            state, action, reward = trajectory[t]
            G = gamma * G + reward
            
            if state not in visited:  # 首次访问
                returns[state].append(G)
                V[state] = np.mean(returns[state])
                visited.add(state)
    
    return V
```

### 每次访问MC（Every-Visit MC）

统计状态s每次出现时的回报：

```python
def every_visit_mc_prediction(env, policy, n_episodes=10000, gamma=0.99):
    """每次访问蒙特卡洛预测"""
    V = np.zeros(env.observation_space.n)
    returns_sum = np.zeros(env.observation_space.n)
    returns_count = np.zeros(env.observation_space.n)
    
    for episode in range(n_episodes):
        trajectory = []
        state, _ = env.reset()
        done = False
        
        while not done:
            action = policy(state)
            next_state, reward, term, trunc, _ = env.step(action)
            trajectory.append((state, action, reward))
            state = next_state
            done = term or trunc
        
        G = 0
        for t in reversed(range(len(trajectory))):
            state, action, reward = trajectory[t]
            G = gamma * G + reward
            
            # 每次访问都更新（不检查是否首次）
            returns_sum[state] += G
            returns_count[state] += 1
            V[state] = returns_sum[state] / returns_count[state]
    
    return V
```

### 增量更新（避免存储所有returns）

```python
def mc_prediction_incremental(env, policy, n_episodes=10000, gamma=0.99):
    """增量更新版蒙特卡洛预测"""
    V = np.zeros(env.observation_space.n)
    N = np.zeros(env.observation_space.n)  # 访问次数
    
    for episode in range(n_episodes):
        trajectory = []
        state, _ = env.reset()
        done = False
        
        while not done:
            action = policy(state)
            next_state, reward, term, trunc, _ = env.step(action)
            trajectory.append((state, action, reward))
            state = next_state
            done = term or trunc
        
        G = 0
        visited = set()
        
        for t in reversed(range(len(trajectory))):
            state, action, reward = trajectory[t]
            G = gamma * G + reward
            
            if state not in visited:
                N[state] += 1
                alpha = 1.0 / N[state]  # 1/n 步长
                V[state] += alpha * (G - V[state])
                visited.add(state)
    
    return V
```

## 蒙特卡洛控制

### On-Policy MC

使用当前策略收集数据并改进当前策略。

#### ε-greedy 探索

```python
def get_epsilon_greedy_action(Q, state, n_actions, epsilon):
    """ε-greedy策略"""
    if np.random.random() < epsilon:
        return np.random.randint(n_actions)  # 探索
    else:
        return np.argmax(Q[state])  # 利用


def on_policy_mc_control(env, n_episodes=10000, gamma=0.99, epsilon=0.1):
    """On-Policy MC控制算法"""
    n_states = env.observation_space.n
    n_actions = env.action_space.n
    
    Q = np.zeros((n_states, n_actions))
    returns_sum = np.zeros((n_states, n_actions))
    returns_count = np.zeros((n_states, n_actions))
    
    for episode in range(n_episodes):
        # 1. 用ε-greedy生成回合
        trajectory = []
        state, _ = env.reset()
        done = False
        
        while not done:
            action = get_epsilon_greedy_action(Q, state, n_actions, epsilon)
            next_state, reward, term, trunc, _ = env.step(action)
            trajectory.append((state, action, reward))
            state = next_state
            done = term or trunc
        
        # 2. MC更新Q
        G = 0
        visited = set()
        
        for t in reversed(range(len(trajectory))):
            s, a, r = trajectory[t]
            G = gamma * G + r
            
            if (s, a) not in visited:
                returns_sum[s, a] += G
                returns_count[s, a] += 1
                Q[s, a] = returns_sum[s, a] / returns_count[s, a]
                visited.add((s, a))
    
    # 提取最终策略
    policy = np.argmax(Q, axis=1)
    
    return Q, policy
```

### Off-Policy MC

用行为策略b收集数据，评估和改进目标策略π。

#### 重要采样

```python
def off_policy_mc_control(env, n_episodes=50000, gamma=0.99, epsilon=0.1):
    """Off-Policy MC控制（重要采样）"""
    n_states = env.observation_space.n
    n_actions = env.action_space.n
    
    Q = np.zeros((n_states, n_actions))
    C = np.zeros((n_states, n_actions))  # 累积权重
    
    # 目标策略：贪心
    target_policy = np.argmax(Q, axis=1)
    
    for episode in range(n_episodes):
        # 行为策略：ε-greedy（用于探索）
        trajectory = []
        state, _ = env.reset()
        done = False
        
        while not done:
            action = get_epsilon_greedy_action(Q, state, n_actions, epsilon)
            next_state, reward, term, trunc, _ = env.step(action)
            trajectory.append((state, action, reward))
            state = next_state
            done = term or trunc
        
        # 重要采样更新
        G = 0
        W = 1.0  # 重要采样权重
        
        for t in reversed(range(len(trajectory))):
            s, a, r = trajectory[t]
            G = gamma * G + r
            
            C[s, a] += W
            Q[s, a] += (W / C[s, a]) * (G - Q[s, a])
            
            # 目标策略是贪心的
            target_policy[s] = np.argmax(Q[s])
            
            if a != target_policy[s]:
                break  # 重要采样权重为0
            
            # 更新权重
            # W *= P_target(a|s) / P_behavior(a|s)
            # 贪心策略：P_target = 1
            # ε-greedy行为策略：P_behavior = ε/|A| + (1-ε) if argmax
            n_actions = env.action_space.n
            if a == np.argmax(Q[s]):
                behavior_prob = epsilon / n_actions + (1 - epsilon)
            else:
                behavior_prob = epsilon / n_actions
            W *= 1.0 / behavior_prob
    
    return Q, target_policy
```

## On-Policy vs Off-Policy

| 特性 | On-Policy | Off-Policy |
|------|-----------|------------|
| **数据策略** | 用π收集数据 | 用b收集，学习π |
| **探索** | ε-greedy直接探索 | 行为策略负责探索 |
| **样本效率** | 较低（数据被丢弃） | 较高（可复用） |
| **方差** | 较低 | 较高（重要采样） |
| **实现** | 简单 | 复杂 |

## 蒙特卡洛的优缺点

### 优点

1. **无需环境模型**
2. **无偏估计**
3. **适合回合制任务**
4. **不受马尔可夫假设限制**（可以用完整历史）

### 缺点

1. **必须等回合结束**——不能在线学习
2. **高方差**——回报波动大
3. **回合太长时收敛慢**
4. **连续任务不适用**（没有明确终止）

## 实战：Blackjack环境

```python
"""
用MC解决21点（Blackjack）游戏
经典MC应用示例
"""
import numpy as np
import gymnasium as gym

def mc_blackjack():
    env = gym.make("Blackjack-v1")
    
    n_states = (32, 11, 2)  # (玩家点数, 庄家明牌, 有无可用A)
    n_actions = 2  # 要牌(1)或停牌(0)
    
    Q = np.zeros(n_states + (n_actions,))
    N = np.zeros(n_states + (n_actions,))
    
    def policy(observation, epsilon=0.1):
        """ε-greedy策略"""
        player_sum, dealer_card, usable_ace = observation
        if player_sum < 12:
            return 1  # 必须要牌
        if np.random.random() < epsilon:
            return np.random.randint(n_actions)
        return np.argmax(Q[player_sum, dealer_card, int(usable_ace)])
    
    for episode in range(500000):
        # 生成回合
        trajectory = []
        obs, _ = env.reset()
        done = False
        
        while not done:
            action = policy(obs)
            next_obs, reward, term, trunc, _ = env.step(action)
            trajectory.append((obs, action, reward))
            obs = next_obs
            done = term or trunc
        
        # MC更新（首次访问）
        G = 0
        visited = set()
        
        for obs, action, reward in reversed(trajectory):
            G += reward  # γ=1（Blackjack是短回合）
            player, dealer, ace = obs
            idx = (player, dealer, int(ace), action)
            
            if idx not in visited:
                N[idx] += 1
                Q[idx] += (G - Q[idx]) / N[idx]
                visited.add(idx)
    
    # 可视化状态值函数
    V_no_ace = np.max(Q[:, :, 0, :], axis=-1)
    V_ace = np.max(Q[:, :, 1, :], axis=-1)
    
    print("最优策略训练完成!")
    print(f"无可用A时平均V: {V_no_ace[12:22].mean():.4f}")
    print(f"有可用A时平均V: {V_ace[12:22].mean():.4f}")
    
    env.close()

# mc_blackjack()  # 取消注释运行
```

## 小结

| 方法 | 特点 |
|------|------|
| **首次访问MC** | 只统计状态首次出现的回报 |
| **每次访问MC** | 统计状态每次出现的回报 |
| **On-Policy MC** | 用相同策略收集和改进 |
| **Off-Policy MC** | 用行为策略收集，目标策略改进 |
| **重要采样** | Off-policy的核心技术 |

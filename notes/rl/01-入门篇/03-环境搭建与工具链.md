## 核心库详解

### 1. Gymnasium — RL环境标准

Gymnasium（原OpenAI Gym）是RL环境的标准接口：

```python
import gymnasium as gym

# 创建环境
env = gym.make("CartPole-v1")

# 基本API
observation, info = env.reset(seed=42)

for step in range(1000):
    action = env.action_space.sample()  # 随机动作
    observation, reward, terminated, truncated, info = env.step(action)
    
    if terminated or truncated:
        observation, info = env.reset()

env.close()
```

**核心API说明**：

| 方法 | 说明 |
|------|------|
| `env.reset()` | 重置环境，返回初始观测 |
| `env.step(action)` | 执行动作，返回(obs, reward, terminated, truncated, info) |
| `env.render()` | 渲染环境画面 |
| `env.close()` | 关闭环境 |

**step()返回值详解（Gymnasium新格式）**：

```python
observation, reward, terminated, truncated, info = env.step(action)

# observation: 新状态
# reward: 即时奖励
# terminated: 是否因自然终止条件结束（如输了/赢了）
# truncated: 是否因人为截断结束（如超过最大步数）
# info: 额外信息（调试用）

# 注意：旧版Gym中是 done = terminated or truncated
```

### 2. Stable-Baselines3 — RL算法库

经过严格验证的RL算法实现：

```python
from stable_baselines3 import PPO
from stable_baselines3.common.evaluation import evaluate_policy

# 创建模型
model = PPO("MlpPolicy", env, verbose=1)

# 训练
model.learn(total_timesteps=100_000)

# 评估
mean_reward, std_reward = evaluate_policy(model, env, n_eval_episodes=10)
print(f"平均奖励: {mean_reward:.2f} ± {std_reward:.2f}")

# 保存/加载
model.save("ppo_cartpole")
model = PPO.load("ppo_cartpole")
```

### 3. PyTorch — 深度学习框架

```python
import torch
import torch.nn as nn
import torch.optim as optim

# 检查GPU
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
print(f"使用设备: {device}")

# 示例：简单的Q网络
class QNetwork(nn.Module):
    def __init__(self, state_dim, action_dim, hidden_dim=128):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, action_dim)
        )
    
    def forward(self, x):
        return self.net(x)

# 使用
q_net = QNetwork(state_dim=4, action_dim=2).to(device)
state = torch.randn(32, 4).to(device)  # batch=32
q_values = q_net(state)  # [32, 2]
```

## 完整环境验证

```python
"""
环境验证脚本：确认所有核心库正常工作
"""
import sys
import gymnasium as gym
import numpy as np
import torch

def check_versions():
    """检查版本"""
    print(f"Python: {sys.version}")
    print(f"PyTorch: {torch.__version__}")
    print(f"CUDA可用: {torch.cuda.is_available()}")
    print(f"Gymnasium: {gym.__version__}")
    
    try:
        import stable_baselines3 as sb3
        print(f"Stable-Baselines3: {sb3.__version__}")
    except ImportError:
        print("Stable-Baselines3: 未安装")

def test_cartpole():
    """测试CartPole环境"""
    env = gym.make("CartPole-v1")
    obs, info = env.reset(seed=42)
    
    total_reward = 0
    for step in range(100):
        action = env.action_space.sample()
        obs, reward, terminated, truncated, info = env.step(action)
        total_reward += reward
        
        if terminated or truncated:
            break
    
    print(f"\nCartPole随机策略: {total_reward:.0f}步")
    env.close()

def test_pytorch():
    """测试PyTorch"""
    x = torch.randn(3, 4)
    y = torch.randn(4, 5)
    z = x @ y
    print(f"\nPyTorch矩阵乘法测试: {z.shape}")

if __name__ == "__main__":
    check_versions()
    test_cartpole()
    test_pytorch()
    print("\n✅ 环境验证通过！")
```

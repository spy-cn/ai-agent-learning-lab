# 第一个RL程序：Q-Learning解决FrozenLake

## FrozenLake问题

FrozenLake是一个网格世界环境：
- 智能体从起点(S)走到终点(G)
- 避免掉入冰窟(F=安全, H=危险)
- 动作：上/下/左/右

```
┌───────┬───────┬───────┬───────┐
│   S   │   F   │   F   │   F   │     S: Start (起点)
├───────┼───────┼───────┼───────┤     F: Frozen (安全冰面)
│   F   │   H   │   F   │   F   │     H: Hole (冰窟, 掉入死亡)
├───────┼───────┼───────┼───────┤     G: Goal (终点)
│   F   │   F   │   F   │   H   │
├───────┼───────┼───────┼───────┤
│   H   │   F   │   F   │   G   │
└───────┴───────┴───────┴───────┘

奖励：到达G得+1，其他为0
```

## Q-Learning算法

Q-Learning是最经典的无模型RL算法，学习一个Q表来指导决策。

### 算法原理

```
Q表 Q(s, a): 状态-动作价值表

更新规则（TD学习）:
Q(s, a) ← Q(s, a) + α [r + γ max_a' Q(s', a') - Q(s, a)]
                          └──────────────────────────┘
                              TD目标（更好的估计）
```

### 代码实现

```python
"""
Q-Learning解决FrozenLake
从零实现，不依赖任何RL库
"""
import numpy as np
import gymnasium as gym
import matplotlib.pyplot as plt

class QLearningAgent:
    """Q-Learning智能体"""
    
    def __init__(self, n_states, n_actions, 
                 lr=0.1, gamma=0.99, epsilon=1.0,
                 epsilon_min=0.01, epsilon_decay=0.995):
        self.n_states = n_states
        self.n_actions = n_actions
        
        # 超参数
        self.lr = lr              # 学习率α
        self.gamma = gamma        # 折扣因子γ
        self.epsilon = epsilon    # 探索率ε
        self.epsilon_min = epsilon_min
        self.epsilon_decay = epsilon_decay
        
        # Q表：n_states × n_actions
        self.q_table = np.zeros((n_states, n_actions))
    
    def act(self, state, training=True):
        """ε-greedy选择动作"""
        if training and np.random.random() < self.epsilon:
            return np.random.randint(self.n_actions)  # 探索
        
        return np.argmax(self.q_table[state])  # 利用
    
    def learn(self, state, action, reward, next_state, done):
        """Q-Learning更新"""
        # 当前Q值
        current_q = self.q_table[state, action]
        
        # TD目标
        if done:
            target = reward  # 终止状态没有未来奖励
        else:
            target = reward + self.gamma * np.max(self.q_table[next_state])
        
        # Q-Learning更新
        self.q_table[state, action] += self.lr * (target - current_q)
    
    def decay_epsilon(self):
        """衰减探索率"""
        self.epsilon = max(self.epsilon_min, self.epsilon * self.epsilon_decay)


def train(env, agent, n_episodes=10000):
    """训练Q-Learning"""
    rewards = []
    
    for episode in range(n_episodes):
        state, _ = env.reset()
        episode_reward = 0
        done = False
        
        while not done:
            # 1. 选择动作
            action = agent.act(state)
            
            # 2. 执行动作
            next_state, reward, terminated, truncated, _ = env.step(action)
            done = terminated or truncated
            
            # 3. 学习更新
            agent.learn(state, action, reward, next_state, done)
            
            state = next_state
            episode_reward += reward
        
        # 衰减探索率
        agent.decay_epsilon()
        rewards.append(episode_reward)
        
        # 打印进度
        if (episode + 1) % 1000 == 0:
            avg_reward = np.mean(rewards[-1000:])
            print(f"Episode {episode+1}/{n_episodes}, "
                  f"平均奖励: {avg_reward:.3f}, "
                  f"ε: {agent.epsilon:.3f}")
    
    return rewards


def evaluate(env, agent, n_episodes=100):
    """评估训练后的策略"""
    rewards = []
    for episode in range(n_episodes):
        state, _ = env.reset()
        episode_reward = 0
        done = False
        
        while not done:
            action = agent.act(state, training=False)  # 不探索
            next_state, reward, terminated, truncated, _ = env.step(action)
            state = next_state
            episode_reward += reward
            done = terminated or truncated
        
        rewards.append(episode_reward)
    
    print(f"\n评估结果（{n_episodes}回合）:")
    print(f"  成功率: {np.mean(rewards):.2%}")
    print(f"  平均奖励: {np.mean(rewards):.2f}")


def visualize_q_table(agent):
    """可视化Q表"""
    print("\nQ表（学到的策略）:")
    print(f"{'State':>6} | {'←':>6} | {'↓':>6} | {'→':>6} | {'↑':>6} | 最优")
    print("-" * 50)
    
    actions = ['←', '↓', '→', '↑']
    for s in range(agent.n_states):
        q_vals = agent.q_table[s]
        best = actions[np.argmax(q_vals)]
        q_str = " | ".join([f"{v:6.3f}" for v in q_vals])
        print(f"{s:6d} | {q_str} | {best}")


def plot_rewards(rewards, window=100):
    """绘制奖励曲线"""
    plt.figure(figsize=(10, 5))
    plt.plot(rewards, alpha=0.3, color='blue', label='每回合奖励')
    
    # 滑动平均
    if len(rewards) >= window:
        moving_avg = np.convolve(rewards, np.ones(window)/window, mode='valid')
        plt.plot(range(window-1, len(rewards)), moving_avg, 
                color='red', linewidth=2, label=f'{window}回合滑动平均')
    
    plt.xlabel('Episode')
    plt.ylabel('Reward')
    plt.title('Q-Learning Training Progress on FrozenLake')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.savefig('training_progress.png', dpi=150, bbox_inches='tight')
    plt.show()


# ========== 主程序 ==========
if __name__ == "__main__":
    # 创建环境（is_slippery=False使环境确定性，更容易学习）
    env = gym.make("FrozenLake-v1", desc=None, map_name="4x4", 
                   is_slippery=False)
    
    # 创建智能体
    agent = QLearningAgent(
        n_states=env.observation_space.n,   # 16个状态
        n_actions=env.action_space.n,       # 4个动作
        lr=0.1,
        gamma=0.99,
        epsilon=1.0,
        epsilon_decay=0.995
    )
    
    print("=" * 60)
    print("Q-Learning训练FrozenLake")
    print("=" * 60)
    
    # 训练
    rewards = train(env, agent, n_episodes=10000)
    
    # 评估
    evaluate(env, agent, n_episodes=100)
    
    # 查看Q表
    visualize_q_table(agent)
    
    # 绘图
    plot_rewards(rewards)
    
    env.close()
```

## Q表分析

训练后，Q表会学出一条从S到G的最优路径：

```
学到的最优策略:
┌───────┬───────┬───────┬───────┐
│   S→  │   F→  │   F→  │   F↓  │
├───────┼───────┼───────┼───────┤
│   F↓  │   H   │   F↓  │   F↓  │
├───────┼───────┼───────┼───────┤
│   F→  │   F→  │   F↓  │   H   │
├───────┼───────┼───────┼───────┤
│   H   │   F→  │   F→  │   G   │
└───────┴───────┴───────┴───────┘

最优路径: S → 右 → 右 → 下 → 下 → 下 → 右 → 右 → G
```

## 关键超参数详解

| 参数 | 默认值 | 作用 | 调参建议 |
|------|--------|------|----------|
| `lr` (学习率α) | 0.1 | Q值更新步长 | 太大→不稳定；太小→收敛慢 |
| `gamma` (折扣γ) | 0.99 | 未来奖励的重要性 | 0.9~0.99常用 |
| `epsilon` (探索率ε) | 1.0→0.01 | 初始探索率 | 开始高，逐渐降低 |
| `epsilon_decay` | 0.995 | ε衰减速率 | 控制探索到利用的转变速度 |

## 与随机策略对比

```python
def random_baseline(env, n_episodes=1000):
    """随机策略基线"""
    rewards = []
    for _ in range(n_episodes):
        state, _ = env.reset()
        done = False
        ep_reward = 0
        while not done:
            action = env.action_space.sample()  # 纯随机
            state, reward, term, trunc, _ = env.step(action)
            ep_reward += reward
            done = term or trunc
        rewards.append(ep_reward)
    return np.mean(rewards)

# 随机策略成功率约 1.5%
# Q-Learning成功率 100%
```

## 从表格到深度RL

Q-Learning用表格存储Q值，但：
- 状态空间大时不可行（如Atari游戏有2^(84*84*3)个状态）
- 需要用**函数近似**（神经网络）替代表格 → **DQN**

```
Q-Learning (表格)          DQN (神经网络)
                                   
  Q表                              Q网络 Q(s,a;θ)
┌──────────┐                    ┌──────────────┐
│State 0:  │                    │  状态s →      │
│ [0.1,...]│                    │  神经网络 →    │
│State 1:  │                    │  Q(s,a)值     │
│ [0.3,...]│                    └──────────────┘
│ ...      │                    
└──────────┘                    可以处理连续状态空间
```

| 知识点 | 说明 |
|--------|------|
| **Q-Learning** | 无模型、off-policy的TD控制算法 |
| **Q表** | 存储每个状态-动作的价值 |
| **ε-greedy** | 平衡探索与利用的经典策略 |

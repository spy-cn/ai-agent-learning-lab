# 独立Q学习（Independent Q-Learning）

独立Q学习（IQL）是最简单的MARL方法：每个智能体独立地运行标准RL算法，忽略其他智能体的存在。

## 核心思想

```
IQL的哲学:
  把其他智能体当作环境的一部分
  每个智能体独立学习自己的策略

智能体1: 环境(含其他智能体) → Q₁(s,a) → 策略π₁
智能体2: 环境(含其他智能体) → Q₂(s,a) → 策略π₂
...
```

```
┌──────────────────────────────────────────────┐
│              IQL 架构                         │
│                                              │
│  ┌──────────┐  ┌──────────┐  ┌──────────┐  │
│  │ Agent 1  │  │ Agent 2  │  │ Agent N  │  │
│  │ DQN #1   │  │ DQN #2   │  │ DQN #N   │  │
│  └────┬─────┘  └────┬─────┘  └────┬─────┘  │
│       │              │              │        │
│       ▼              ▼              ▼        │
│  ┌──────────────────────────────────────┐   │
│  │            共享环境                    │   │
│  └──────────────────────────────────────┘   │
│                                              │
│  每个智能体:                                  │
│  - 独立的网络和经验回放                       │
│  - 把其他智能体视为环境的部分                 │
│  - 标准Q-Learning更新                        │
└──────────────────────────────────────────────┘
```

## 实现

```python
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np
from collections import deque
import random

class IndependentQLearner:
    """每个智能体一个独立的Q网络"""
    
    def __init__(self, n_agents, state_dim, action_dim, 
                 lr=1e-3, gamma=0.99, epsilon=1.0,
                 epsilon_min=0.05, epsilon_decay=0.995):
        self.n_agents = n_agents
        self.gamma = gamma
        
        # 每个智能体独立的网络和buffer
        self.q_nets = [QNetwork(state_dim, action_dim) for _ in range(n_agents)]
        self.target_nets = [QNetwork(state_dim, action_dim) for _ in range(n_agents)]
        
        for i in range(n_agents):
            self.target_nets[i].load_state_dict(self.q_nets[i].state_dict())
        
        self.optimizers = [
            optim.Adam(self.q_nets[i].parameters(), lr=lr) 
            for i in range(n_agents)
        ]
        
        self.buffers = [ReplayBuffer(50000) for _ in range(n_agents)]
        
        # 独立的探索率
        self.epsilons = [epsilon] * n_agents
        self.epsilon_min = epsilon_min
        self.epsilon_decay = epsilon_decay
    
    def act(self, observations, training=True):
        """每个智能体独立选动作"""
        actions = []
        for i in range(self.n_agents):
            if training and np.random.random() < self.epsilons[i]:
                action = np.random.randint(self.q_nets[i].action_dim)
            else:
                with torch.no_grad():
                    obs = torch.FloatTensor(observations[i]).unsqueeze(0)
                    q = self.q_nets[i](obs)
                    action = q.argmax().item()
            actions.append(action)
        return actions
    
    def update(self):
        """每个智能体独立更新"""
        losses = []
        for i in range(self.n_agents):
            if len(self.buffers[i]) < 64:
                losses.append(0)
                continue
            
            loss = self._update_agent(i)
            losses.append(loss)
        
        return losses
    
    def _update_agent(self, agent_id):
        """更新单个智能体"""
        buffer = self.buffers[agent_id]
        q_net = self.q_nets[agent_id]
        target_net = self.target_nets[agent_id]
        optimizer = self.optimizers[agent_id]
        
        # 采样
        states, actions, rewards, next_states, dones = buffer.sample(64)
        states = torch.FloatTensor(states)
        actions = torch.LongTensor(actions)
        rewards = torch.FloatTensor(rewards)
        next_states = torch.FloatTensor(next_states)
        dones = torch.FloatTensor(dones)
        
        # Q-Learning更新
        current_q = q_net(states).gather(1, actions.unsqueeze(1)).squeeze(1)
        
        with torch.no_grad():
            next_q = target_net(next_states).max(dim=1)[0]
            target_q = rewards + self.gamma * next_q * (1 - dones)
        
        loss = nn.MSELoss()(current_q, target_q)
        
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        
        return loss.item()
    
    def decay_epsilon(self):
        for i in range(self.n_agents):
            self.epsilons[i] = max(self.epsilon_min, 
                                   self.epsilons[i] * self.epsilon_decay)


class QNetwork(nn.Module):
    def __init__(self, state_dim, action_dim, hidden_dim=128):
        super().__init__()
        self.action_dim = action_dim
        self.net = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, action_dim)
        )
    
    def forward(self, x):
        return self.net(x)


class ReplayBuffer:
    def __init__(self, capacity):
        self.buffer = deque(maxlen=capacity)
    
    def push(self, state, action, reward, next_state, done):
        self.buffer.append((state, action, reward, next_state, done))
    
    def sample(self, batch_size):
        batch = random.sample(self.buffer, batch_size)
        states, actions, rewards, next_states, dones = zip(*batch)
        return (np.array(states), np.array(actions), 
                np.array(rewards, dtype=np.float32),
                np.array(next_states), np.array(dones, dtype=np.float32))
    
    def __len__(self):
        return len(self.buffer)
```

## 参数共享版本

当智能体同类时，共享网络参数：

```python
class SharedIndependentQLearner:
    """参数共享的IQL"""
    
    def __init__(self, n_agents, state_dim, action_dim, lr=1e-3):
        self.n_agents = n_agents
        
        # 单个网络服务所有智能体
        self.q_net = QNetwork(state_dim, action_dim)
        self.target_net = QNetwork(state_dim, action_dim)
        self.target_net.load_state_dict(self.q_net.state_dict())
        
        self.optimizer = optim.Adam(self.q_net.parameters(), lr=lr)
        
        # 共享经验回放（所有智能体的数据混合）
        self.buffer = ReplayBuffer(100000)
    
    def act(self, observations):
        """所有智能体用同一个网络"""
        actions = []
        for obs in observations:
            with torch.no_grad():
                q = self.q_net(torch.FloatTensor(obs).unsqueeze(0))
                actions.append(q.argmax().item())
        return actions
    
    def update(self):
        """用所有智能体的数据一起训练"""
        if len(self.buffer) < 256:
            return
        
        states, actions, rewards, next_states, dones = self.buffer.sample(256)
        
        # 标准Q-Learning更新
        current_q = self.q_net(states).gather(1, actions.unsqueeze(1)).squeeze(1)
        
        with torch.no_grad():
            next_q = self.target_net(next_states).max(dim=1)[0]
            target_q = rewards + self.gamma * next_q * (1 - dones)
        
        loss = nn.MSELoss()(current_q, target_q)
        
        self.optimizer.zero_grad()
        loss.backward()
        self.optimizer.step()
```

## IQL的问题：非平稳性

```
问题: 其他智能体也在学习，环境在不断变化

时间线:
  t=0: Agent2随机策略 → Agent1容易适应
  t=100: Agent2学会了一些 → Agent1需要重新适应
  t=1000: Agent2策略又变了 → Agent1又要调整

从Agent1的视角:
  Q(s,a,s')的转移概率在变化（因为s'依赖于Agent2）
  → 经验回放中的旧数据不可靠
  → 学习不稳定
```

## 缓解非平稳性

### 1. 经验回放重置

```python
class FreshBufferIQL:
    """用小buffer保证数据新鲜"""
    
    def __init__(self, buffer_size=5000):
        # 比标准DQN更小的buffer
        # 旧数据快速被丢弃
        self.buffer = deque(maxlen=buffer_size)
```

### 2. 对手建模

```python
class OpponentModeling:
    """建模其他智能体的策略"""
    
    def __init__(self, n_opponents, obs_dim, action_dim):
        # 预测其他智能体的动作
        self.opponent_models = [
            nn.Sequential(
                nn.Linear(obs_dim, 64),
                nn.ReLU(),
                nn.Linear(64, action_dim)
            )
            for _ in range(n_opponents)
        ]
    
    def predict_opponent_action(self, opponent_obs, opponent_id):
        """预测对手动作"""
        logits = self.opponent_models[opponent_id](opponent_obs)
        return torch.softmax(logits, dim=-1)
    
    def update_opponent_model(self, opponent_obs, opponent_action, opponent_id):
        """用观测到的对手数据更新模型"""
        pred = self.predict_opponent_action(opponent_obs, opponent_id)
        loss = nn.CrossEntropyLoss()(pred, opponent_action)
        return loss
```

## IQL的效果

```
尽管有理论问题，IQL在实践中经常出奇地有效:

原因:
1. 深度RL本身就不稳定，IQL的不稳定性只是增加了一些
2. 经验回放的随机采样部分缓解了非平稳性
3. ε-greedy探索提供了一定鲁棒性
4. 参数共享使学习更稳定

适用场景:
- 同类智能体（可以参数共享）
- 不是严格竞争的环境
- 不需要精确建模对手
```

## 实战：合作导航

```python
"""
多智能体合作导航: 多个智能体需要覆盖所有地标
"""
import gymnasium as gym

def train_cooperative_navigation():
    from pettingzoo.mpe import simple_spread_v3
    
    env = simple_spread_v3.env(N=3, local_ratio=0.5, max_cycles=25)
    env.reset()
    
    n_agents = 3
    obs_dim = 18  # 每个智能体的观测维度
    action_dim = 5  # 离散动作
    
    # 用参数共享IQL
    agent = SharedIndependentQLearner(n_agents, obs_dim, action_dim)
    
    n_episodes = 10000
    
    for episode in range(n_episodes):
        env.reset()
        
        for agent_id in env.agent_iter():
            obs, reward, termination, truncation, info = env.last()
            
            if termination or truncation:
                action = None
            else:
                action = agent.act([obs])[0]
            
            env.step(action)
        
        # 训练
        agent.update()
```

## 小结

| 特点 | 说明 |
|------|------|
| **简单** | 直接复用单智能体算法 |
| **非平稳** | 其他智能体在变化 |
| **参数共享** | 同类智能体可共享网络 |
| **实践中有效** | 尽管理论有问题 |
| **对手建模** | 缓解非平稳性的方法 |

IQL是MARL的起点，简单但有效。更复杂的方法（MADDPG、MAPPO）在此基础上改进。

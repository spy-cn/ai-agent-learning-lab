# 自博弈与种群方法

自博弈（Self-Play）是训练超级AI（AlphaGo、OpenAI Five）的核心技术——让智能体与自己或过去的版本对弈。

## 为什么需要自博弈？

```
传统RL: 需要对手提供训练信号
问题: 
  - 弱对手: 学不到东西
  - 强对手: 完全被碾压，学不到东西
  - 固定对手: 只能学到打这个对手的策略

自博弈: 智能体既是学习者又是自己的对手
优势:
  - 对手水平始终匹配（渐进提高）
  - 无限训练数据
  - 可以超越人类水平
```

## 经典案例

| 系统 | 方法 | 成就 |
|------|------|------|
| **TD-Gammon** | Self-play + TD(λ) | 西洋双陆棋世界级 |
| **AlphaGo** | Self-play + MCTS | 击败围棋世界冠军 |
| **AlphaZero** | 纯Self-play | 围棋/象棋/将棋统一 |
| **OpenAI Five** | Self-play + PPO | Dota2 5v5 |
| **AlphaStar** | League Training | 星际争霸2大师级 |
| **Pluribus** | Self-play | 6人德州扑克 |

## 1. 纯自博弈

```python
import numpy as np
import torch
import copy

class NaiveSelfPlay:
    """纯自博弈: 最新版本互相对打"""
    
    def __init__(self, env, agent_class):
        self.env = env
        self.agent = agent_class()  # 当前最好的智能体
    
    def train(self, n_iterations=10000):
        for iteration in range(n_iterations):
            # 当前智能体自己和自己对打
            trajectory = self.play_game(self.agent, self.agent)
            
            # 从双方视角学习
            self.agent.update(trajectory['player1'])
            self.agent.update(trajectory['player2'])
            
            if (iteration + 1) % 100 == 0:
                win_rate = self.evaluate()
                print(f"Iteration {iteration+1}, Win Rate: {win_rate:.2%}")
    
    def play_game(self, player1, player2):
        """两个智能体对打一局"""
        state = self.env.reset()
        trajectories = {'player1': [], 'player2': []}
        done = False
        
        while not done:
            # 玩家1的回合
            action1 = player1.act(state)
            state, reward, done, _ = self.env.step(action1)
            trajectories['player1'].append((state, action1, reward))
            
            if done:
                break
            
            # 玩家2的回合
            action2 = player2.act(state)
            state, reward, done, _ = self.env.step(action2)
            trajectories['player2'].append((state, action2, reward))
        
        return trajectories
```

### 纯自博弈的问题：循环

```
问题: 策略循环
  版本A打败版本B
  版本B'专门打败版本A
  版本A'专门打败版本B'
  → 可能不收敛，策略在循环

石头-剪刀-布效应:
  学会"石头" → 对手学会"布" → 学会"剪刀" → 对手学会"石头" → ...
```

## 2. 历史采样自博弈

```python
class FictitiousSelfPlay:
    """虚拟自博弈: 对打历史版本的混合"""
    
    def __init__(self, env, agent_class, buffer_size=100):
        self.env = env
        self.agent_pool = []  # 历史版本池
        self.buffer_size = buffer_size
    
    def train(self, n_iterations=10000):
        for iteration in range(n_iterations):
            # 当前版本 vs 随机历史版本
            opponent = self.sample_opponent()  # 从池中采样
            
            trajectory = self.play_game(self.agent, opponent)
            self.agent.update(trajectory['player1'])
            
            # 定期保存当前版本
            if iteration % 50 == 0:
                self.save_version()
    
    def sample_opponent(self):
        """从历史版本池中采样"""
        if len(self.agent_pool) == 0:
            return self.agent  # 没有历史，用自己
        
        # 均匀采样或按胜率采样
        return np.random.choice(self.agent_pool)
    
    def save_version(self):
        """保存当前版本到池中"""
        snapshot = copy.deepcopy(self.agent)
        self.agent_pool.append(snapshot)
        
        if len(self.agent_pool) > self.buffer_size:
            self.agent_pool.pop(0)  # 移除最老的
```

```python
class AlphaZeroSelfPlay:
    """AlphaZero的自博弈 + MCTS"""
    
    def __init__(self, game, model_class):
        self.game = game
        self.model = model_class()
        self.mcts = MCTS(self.model)
    
    def self_play_game(self, n_simulations=800):
        """用MCTS进行一局自博弈"""
        state = self.game.reset()
        game_data = []
        done = False
        
        while not done:
            # 用MCTS搜索
            action, policy = self.mcts.search(state, n_simulations)
            
            # 记录数据
            game_data.append({
                'state': state,
                'policy': policy,  # MCTS的搜索概率
                'player': state.current_player
            })
            
            state, reward, done = self.game.step(action)
        
        # 从终局结果标注价值
        for data in game_data:
            # 如果这个玩家最终赢了，value=1; 输了, value=-1
            winner = state.winner
            data['value'] = 1.0 if data['player'] == winner else -1.0
        
        return game_data
    
    def train(self, n_iterations=1000):
        for iteration in range(n_iterations):
            # 1. 自博弈产生数据
            games = [self.self_play_game() for _ in range(100)]
            
            # 2. 训练网络
            for game_data in games:
                self.model.train_step(game_data)
            
            # 3. 评估新模型
            if self.should_update_main():
                win_rate = self.evaluate_against_previous()
                if win_rate > 0.55:  # 超过55%胜率才更新
                    self.update_best_model()
```

```python
class LeagueTraining:
    """AlphaStar的联盟训练框架"""
    
    def __init__(self, n_agents=30):
        # 三类智能体
        self.main_agents = []      # 主选手
        self.league_exploiters = [] # 联盟利用者
        self.main_exploiters = []   # 主选手利用者
        
        for i in range(n_agents // 3):
            self.main_agents.append(Agent(f"main_{i}"))
            self.league_exploiters.append(Agent(f"league_{i}"))
            self.main_exploiters.append(Agent(f"exploiter_{i}"))
        
        # 所有历史版本
        self.frozen_pool = []  # 冻结的过去版本
    
    def train(self):
        while not self.converged():
            # 随机选一个智能体训练
            agent = self.select_agent()
            
            # PFSP: 优先虚拟自博弈选择对手
            opponent = self.pfsp_select_opponent(agent)
            
            # 对打
            result = self.match(agent, opponent)
            
            # 更新
            agent.update(result)
            
            # 检查是否需要冻结
            if agent.is_exploitable():
                self.frozen_pool.append(agent.snapshot())
    
    def pfsp_select_opponent(self, agent):
        """优先虚拟自博弈: 根据胜率选对手"""
        # 偏向选势均力敌的对手
        # 而不是总是选最弱的
        win_rates = self.compute_win_rates(agent, self.frozen_pool)
        
        # 用softmax采样（偏好50%胜率的对手）
        scores = -np.abs(win_rates - 0.5)  # 越接近50%越好
        probs = np.exp(scores / 0.1)
        probs /= probs.sum()
        
        return np.random.choice(self.frozen_pool, p=probs)
```

## 5. 种群训练（PBT）

```python
class PopulationBasedTraining:
    """种群训练: 多个智能体共同进化"""
    
    def __init__(self, n_agents=10):
        self.population = [
            {
                'agent': Agent(hyperparams=self.sample_hyperparams()),
                'score': 0,
            }
            for _ in range(n_agents)
        ]
    
    def train_generation(self):
        # 1. 每个智能体独立训练
        for member in self.population:
            member['agent'].train(n_steps=1000)
        
        # 2. 内部对战评估
        for i, member in enumerate(self.population):
            member['score'] = self.evaluate(member['agent'])
        
        # 3. 选择 + 变异
        new_population = []
        
        # 保留最好的
        sorted_pop = sorted(self.population, key=lambda x: x['score'], reverse=True)
        new_population.extend(sorted_pop[:len(self.population)//4])
        
        # 变异和复制
        for _ in range(len(self.population) - len(new_population)):
            parent = self.tournament_select()
            child = self.mutate(parent)
            new_population.append(child)
        
        self.population = new_population
    
    def mutate(self, parent):
        """变异超参数"""
        child_hyperparams = parent['agent'].hyperparams.copy()
        
        # 随机调整学习率等
        for key in ['lr', 'gamma', 'clip_eps']:
            if np.random.random() < 0.3:
                child_hyperparams[key] *= np.random.uniform(0.8, 1.2)
        
        return {
            'agent': Agent(hyperparams=child_hyperparams),
            'score': 0,
        }
```

## 自博弈的挑战

### 1. 策略循环

```python
# 解决方案:
# - 历史版本池（fictitious play）
# - 足够大的种群
# - Meta-solver选择对手
```

### 2. 局部最优

```python
# 只和自己打可能陷入局部最优
# 比如只学会一种战术

# 解决方案:
# - 多样性鼓励
# - 对抗利用者
# - 课程学习
```

### 3. 非传递性

```
A > B, B > C, 但 C > A（非传递性）

如果策略空间有非传递性:
  训练A → B打败A → C打败B → A打败C → 循环

需要: 种群方法处理非传递性
```

## 小结

| 方法 | 代表 | 适用 |
|------|------|------|
| **纯自博弈** | TD-Gammon | 简单游戏 |
| **历史采样** | AlphaGo Zero | 棋类 |
| **联盟训练** | AlphaStar | 复杂RTS |
| **种群训练** | OpenAI Five | 大规模 |
| **PFSP** | AlphaStar | 非传递性 |

自博弈是训练超越人类水平AI的核心技术，但需要精心设计防止策略循环和局部最优。
